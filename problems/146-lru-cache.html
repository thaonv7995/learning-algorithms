LRU Cache

O(1) O(capacity) 47.4% AC
Mô tả
Thiết kế cấu trúc LRU Cache: get(key) trả value hoặc -1; put(key,value) cập nhật/thêm; khi đầy thì loại key ít dùng nhất. Mọi thao tác O(1).
Ví dụ
Ví dụ 1
In["LRUCache","put","put","get","put","get","put","get","get","get"] [[2],[1,1],[2,2],[1],[3,3],[2],[4,4],[1],[3],[4]]
Out[null,null,null,1,null,-1,null,-1,3,4]
Ý tưởng
Hash map key→node + doubly linked list theo thứ tự dùng gần nhất (head=MRU, tail=LRU). get/put đưa node lên đầu; put vượt capacity thì xóa tail.

/* Hash map + doubly linked list — xem C++ */
class LRUCache:
    class Node:
        __slots__ = ('key', 'val', 'prev', 'next')
        def __init__(self, key=0, val=0):
            self.key, self.val = key, val
            self.prev = self.next = None

    def __init__(self, capacity: int):
        self.cap = capacity
        self.mp = {}
        self.head = self.tail = LRUCache.Node()
        self.head.next = self.tail
        self.tail.prev = self.head

    def _remove(self, node):
        node.prev.next = node.next
        node.next.prev = node.prev

    def _add_front(self, node):
        node.next = self.head.next
        node.prev = self.head
        self.head.next.prev = node
        self.head.next = node

    def get(self, key: int) -> int:
        if key not in self.mp:
            return -1
        node = self.mp[key]
        self._remove(node)
        self._add_front(node)
        return node.val

    def put(self, key: int, value: int) -> None:
        if key in self.mp:
            node = self.mp[key]
            node.val = value
            self._remove(node)
            self._add_front(node)
        else:
            if len(self.mp) == self.cap:
                lru = self.tail.prev
                self._remove(lru)
                del self.mp[lru.key]
            node = LRUCache.Node(key, value)
            self.mp[key] = node
            self._add_front(node)
class LRUCache {
    struct Node { int key, val; Node *prev, *next; };
    int cap;
    unordered_map<int, Node*> mp;
    Node *head, *tail;
    void remove(Node* node) {
        node->prev->next = node->next;
        node->next->prev = node->prev;
    }
    void addFront(Node* node) {
        node->next = head->next;
        node->prev = head;
        head->next->prev = node;
        head->next = node;
    }
public:
    LRUCache(int capacity) : cap(capacity) {
        head = new Node{0, 0, nullptr, nullptr};
        tail = new Node{0, 0, nullptr, nullptr};
        head->next = tail;
        tail->prev = head;
    }
    int get(int key) {
        if (!mp.count(key)) return -1;
        Node* node = mp[key];
        remove(node);
        addFront(node);
        return node->val;
    }
    void put(int key, int value) {
        if (mp.count(key)) {
            Node* node = mp[key];
            node->val = value;
            remove(node);
            addFront(node);
        } else {
            if ((int)mp.size() == cap) {
                Node* lru = tail->prev;
                remove(lru);
                mp.erase(lru->key);
                delete lru;
            }
            Node* node = new Node{key, value, nullptr, nullptr};
            mp[key] = node;
            addFront(node);
        }
    }
};
