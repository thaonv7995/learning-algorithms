Minimum Distance Between BST Nodes

O(n) O(h) 61.3% AC
Mô tả
Cho BST. Trả khoảng cách nhỏ nhất giữa giá trị hai nút bất kỳ (|a-b| min, a!=b).
Ví dụ
Ví dụ 1
Inroot = [4,2,6,1,3]
Out1
Ví dụ 2
Inroot = [1,0,48,null,null,null,49]
Out1
Ý tưởng
Inorder: so sánh node hiện tại với predecessor. Min diff = min(prev diff) along inorder.

class Solution:
    def minDiffInBST(self, root: Optional[TreeNode]) -> int:
        self.prev = None
        self.ans = 10**9
        def dfs(n):
            if not n: return
            dfs(n.left)
            if self.prev is not None:
                self.ans = min(self.ans, n.val - self.prev)
            self.prev = n.val
            dfs(n.right)
        dfs(root)
        return self.ans
class Solution {
    TreeNode* prev=nullptr;
    int ans=INT_MAX;
    void inorder(TreeNode* n){
        if(!n) return;
        inorder(n->left);
        if(prev) ans=min(ans,n->val-prev->val);
        prev=n;
        inorder(n->right);
    }
public:
    int minDiffInBST(TreeNode* root){ inorder(root); return ans; }
};
