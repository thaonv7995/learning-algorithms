Minimum Path Sum

O(M × N) O(N) 68.2% AC
Mô tả
Lưới m×n chứa chi phí từng ô. Tìm tổng chi phí nhỏ nhất từ góc trên-trái đến dưới-phải (chỉ ↓ hoặc →).

/* minPathSum — DP 1D min; xem C++ */
class Solution:
    def minPathSum(self, g: List[List[int]]) -> int:
        m, n = len(g), len(g[0])
        dp = g[0][:]
        for j in range(1, n): dp[j] += dp[j-1]
        for i in range(1, m):
            dp[0] += g[i][0]
            for j in range(1, n):
                dp[j] = g[i][j] + min(dp[j], dp[j-1])
        return dp[-1]
class Solution {
public:
    int minPathSum(vector<vector<int>>& g) {
        int m=g.size(), n=g[0].size();
        vector<int> dp(g[0].begin(), g[0].end());
        for(int j=1;j<n;j++) dp[j]+=dp[j-1];
        for(int i=1;i<m;i++){
            dp[0]+=g[i][0];
            for(int j=1;j<n;j++) dp[j]=g[i][j]+min(dp[j],dp[j-1]);
        }
        return dp[n-1];
    }
};
