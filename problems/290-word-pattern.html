Word Pattern

O(n) O(n) 44.1% AC
Mô tả
Cho pattern (chữ cái) và chuỗi s (từ cách nhau bởi space). Kiểm tra tồn tại song ánh pattern→word và word→pattern.
Ví dụ
Ví dụ 1
Inpattern = "abba", s = "dog cat cat dog"
Outtrue
Ví dụ 2
Inpattern = "abba", s = "dog cat cat fish"
Outfalse
Ví dụ 3
Inpattern = "aaaa", s = "dog cat cat dog"
Outfalse
Ý tưởng
Split s; nếu len khác return false; hai map char↔word; mỗi cặp phải khớp map hiện có.

bool wordPattern(char* pattern, char* s) {
    char* words[256]; int n = 0;
    char* tok = strtok(s, " ");
    while (tok) { words[n++] = tok; tok = strtok(NULL, " "); }
    int plen = (int)strlen(pattern);
    if (n != plen) return false;
    char p2w[256][64] = {{0}}; char w2p[256][2] = {{0}};
    for (int i = 0; i < plen; i++) {
        char c = pattern[i];
        int idx = (unsigned char)c;
        if (p2w[idx][0] && strcmp(p2w[idx], words[i])) return false;
        int widx = words[i][0];
        if (w2p[widx][0] && w2p[widx][0] != c) return false;
        strcpy(p2w[idx], words[i]);
        w2p[widx][0] = c;
    }
    return true;
}
class Solution:
    def wordPattern(self, pattern: str, s: str) -> bool:
        words = s.split()
        if len(words) != len(pattern):
            return False
        p2w, w2p = {}, {}
        for ch, w in zip(pattern, words):
            if p2w.get(ch, w) != w or w2p.get(w, ch) != ch:
                return False
            p2w[ch] = w
            w2p[w] = ch
        return True
class Solution {
public:
    bool wordPattern(string pattern, string s) {
        vector<string> words;
        stringstream ss(s);
        string w;
        while (ss >> w) words.push_back(w);
        if (words.size() != pattern.size()) return false;
        unordered_map<char, string> p2w;
        unordered_map<string, char> w2p;
        for (int i = 0; i < (int)pattern.size(); i++) {
            char c = pattern[i];
            if (p2w.count(c) && p2w[c] != words[i]) return false;
            if (w2p.count(words[i]) && w2p[words[i]] != c) return false;
            p2w[c] = words[i];
            w2p[words[i]] = c;
        }
        return true;
    }
};
