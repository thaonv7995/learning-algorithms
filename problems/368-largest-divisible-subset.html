Largest Divisible Subset

O(n²) O(n) 49.6% AC
Mô tả
Cho mảng nums số nguyên dương distinct. Trả subset lớn nhất sao mọi cặp phần tử có một chia hết cho cái kia.
Ví dụ
Ví dụ 1
Innums = [1,2,3]
Out[1,2]
Ví dụ 2
Innums = [1,2,4,8]
Out[1,2,4,8]
Ý tưởng
Sort nums; dp[i]=max size subset ending i; dp[i]=max(dp[j]+1) nếu nums[i]%nums[j]==0. Trace parent để rebuild.

int* largestDivisibleSubset(int* nums, int n, int* retSize) {
    /* sort + dp O(n^2); return longest chain */
    for (int i=0;i<n;i++) for (int j=i+1;j<n;j++) if(nums[i]>nums[j]){int t=nums[i];nums[i]=nums[j];nums[j]=t;}
    int* dp=malloc(n*4); int* prev=malloc(n*4);
    for(int i=0;i<n;i++){dp[i]=1;prev[i]=-1;}
    int best=1,idx=0;
    for(int i=0;i<n;i++) for(int j=0;j<i;j++) if(nums[i]%nums[j]==0&&dp[j]+1>dp[i]){dp[i]=dp[j]+1;prev[i]=j;}
    for(int i=0;i<n;i++) if(dp[i]>best){best=dp[i];idx=i;}
    int* ans=malloc(best*4); *retSize=0;
    while(idx!=-1){ans[(*retSize)++]=nums[idx]; idx=prev[idx];}
    free(dp); free(prev); return ans;
}
class Solution:
    def largestDivisibleSubset(self, nums: List[int]) -> List[int]:
        nums.sort()
        n = len(nums)
        dp, prev = [1]*n, [-1]*n
        best, idx = 1, 0
        for i in range(n):
            for j in range(i):
                if nums[i] % nums[j] == 0 and dp[j] + 1 > dp[i]:
                    dp[i] = dp[j] + 1; prev[i] = j
            if dp[i] > best: best, idx = dp[i], i
        ans = []
        while idx != -1: ans.append(nums[idx]); idx = prev[idx]
        return ans
class Solution {
public:
    vector<int> largestDivisibleSubset(vector<int>& nums) {
        sort(nums.begin(), nums.end());
        int n = nums.size();
        vector<int> dp(n,1), prev(n,-1);
        int best = 1, idx = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < i; j++)
                if (nums[i] % nums[j] == 0 && dp[j]+1 > dp[i]) { dp[i]=dp[j]+1; prev[i]=j; }
            if (dp[i] > best) { best = dp[i]; idx = i; }
        }
        vector<int> ans;
        while (idx != -1) { ans.push_back(nums[idx]); idx = prev[idx]; }
        return ans;
    }
};
