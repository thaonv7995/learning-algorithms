Maximum XOR of Two Numbers in an Array

O(n * 32) O(n * 32) 53.5% AC
Mô tả
Cho mảng nums. Trả max(nums[i] XOR nums[j]) với i ≠ j.
Ví dụ
Ví dụ 1
Innums = [3,10,5,25,2,8]
Out28
Ví dụ 2
Innums = [14,70,53]
Out70
Ví dụ 3
Innums = [0,1]
Out1
Ý tưởng
Trie 32 bit: insert từng số; query mỗi số chọn bit ngược ở mỗi level nếu có nhánh.

int findMaximumXOR(int* nums, int n) {
    int ans=0;
    for(int i=0;i<n;i++) for(int j=i+1;j<n;j++){
        int v=nums[i]^nums[j]; if(v>ans) ans=v;
    }
    return ans;
}
class Solution:
    def findMaximumXOR(self, nums: List[int]) -> int:
        root = {}
        ans = 0
        for x in nums:
            node = root
            for i in range(31, -1, -1):
                b = (x >> i) & 1
                node = node.setdefault(b, {})
        for x in nums:
            node, val = root, 0
            for i in range(31, -1, -1):
                b = (x >> i) & 1
                want = 1 - b
                if want in node:
                    val |= 1 << i
                    node = node[want]
                else:
                    node = node[b]
            ans = max(ans, val)
        return ans
class Solution {
public:
    int findMaximumXOR(vector<int>& nums) {
        struct Node { Node* ch[2]{}; };
        Node root; int ans=0;
        for(int x:nums){
            Node* cur=&root;
            for(int i=31;i>=0;i--){ int b=(x>>i)&1; if(!cur->ch[b]) cur->ch[b]=new Node(); cur=cur->ch[b]; }
        }
        for(int x:nums){
            Node* cur=&root; int val=0;
            for(int i=31;i>=0;i--){
                int b=(x>>i)&1, w=1-b;
                if(cur->ch[w]){ val|=1<<i; cur=cur->ch[w]; } else cur=cur->ch[b];
            }
            ans=max(ans,val);
        }
        return ans;
    }
};
