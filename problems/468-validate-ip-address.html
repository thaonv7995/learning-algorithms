Validate IP Address

O(n) O(1) 28.3% AC
Mô tả
Validate IP queryIP: trả "IPv4", "IPv6", hoặc "Neither". IPv4: 4 octet 0-255 không leading zero; IPv6: 8 hextet hex 1-4 chars.
Ví dụ
Ví dụ 1
InqueryIP = "172.16.254.1"
Out"IPv4"
Ví dụ 2
InqueryIP = "2001:0db8:85a3:0:0:8A2E:0370:7334"
Out"IPv6"
Ví dụ 3
InqueryIP = "256.256.256.256"
Out"Neither"
Ý tưởng
Split by . hoặc :; validate parts count, digits, range IPv4; hex length 1-4 IPv6.

char* validIPAddress(char* q) {
    int dots=0, cols=0; for(char* p=q;*p;p++){ if(*p=='.') dots++; if(*p==':') cols++; }
    if(dots==3) return strdup("IPv4");
    if(cols==7) return strdup("IPv6");
    return strdup("Neither");
}
class Solution:
    def validIPAddress(self, queryIP: str) -> str:
        if "." in queryIP:
            parts = queryIP.split(".")
            if len(parts) != 4: return "Neither"
            for p in parts:
                if not p or (len(p) > 1 and p[0] == "0") or not p.isdigit() or int(p) > 255:
                    return "Neither"
            return "IPv4"
        if ":" in queryIP:
            parts = queryIP.split(":")
            if len(parts) != 8: return "Neither"
            hexd = set("0123456789abcdefABCDEF")
            for p in parts:
                if not p or len(p) > 4 or any(c not in hexd for c in p):
                    return "Neither"
            return "IPv6"
        return "Neither"
class Solution {
public:
    string validIPAddress(string q) {
        auto ipv4=[&](){
            vector<string> p; string s;
            for(char c:q){ if(c=='.'){ p.push_back(s); s=""; } else s+=c; }
            p.push_back(s);
            if(p.size()!=4) return false;
            for(string& x:p){
                if(x.empty()||(x.size()>1&&x[0]=='0')||x.size()>3) return false;
                for(char c:x) if(!isdigit(c)) return false;
                if(stoi(x)>255) return false;
            }
            return true;
        };
        if(q.find('.')!=string::npos) return ipv4()? "IPv4":"Neither";
        if(q.find(':')!=string::npos){
            vector<string> p; string s;
            for(char c:q){ if(c==':'){ p.push_back(s); s=""; } else s+=c; }
            p.push_back(s);
            if(p.size()!=8) return "Neither";
            for(string& x:p){
                if(x.empty()||x.size()>4) return "Neither";
                for(char c:x) if(!isxdigit(c)) return "Neither";
            }
            return "IPv6";
        }
        return "Neither";
    }
};
