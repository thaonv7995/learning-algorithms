Erect the Fence

O(n log n) O(n) 53% AC
Mô tả
Cho tập điểm 2D. Trả các điểm trên convex hull (fence) theo thứ tự ngược chiều kim đồng hồ.
Ví dụ
Ví dụ 1
Inpoints = [[0,0],[0,1],[1,1],[1,0]]
Out[[0,0],[1,0],[1,1],[0,1]]
Ví dụ 2
Inpoints = [[1,1],[2,2],[3,3]]
Out[[1,1],[2,2],[3,3]]
Ý tưởng
Andrew monotone chain: sort points, build lower và upper hull, loại duplicate cuối.

/* Convex hull Andrew — xem C++ */
class Solution:
    def outerTrees(self, points: List[List[int]]) -> List[List[int]]:
        pts = sorted(set(map(tuple, points)))
        if len(pts) <= 1:
            return [list(p) for p in pts]
        def cross(o,a,b):
            return (a[0]-o[0])*(b[1]-o[1])-(a[1]-o[1])*(b[0]-o[0])
        lower=[]
        for p in pts:
            while len(lower)>=2 and cross(lower[-2],lower[-1],p)<0: lower.pop()
            lower.append(p)
        upper=[]
        for p in reversed(pts):
            while len(upper)>=2 and cross(upper[-2],upper[-1],p)<0: upper.pop()
            upper.append(p)
        return [list(p) for p in lower[:-1]+upper[:-1]]
class Solution {
    long long cross(vector<int>& o, vector<int>& a, vector<int>& b) {
        return (long long)(a[0]-o[0])*(b[1]-o[1]) - (long long)(a[1]-o[1])*(b[0]-o[0]);
    }
public:
    vector<vector<int>> outerTrees(vector<vector<int>>& points) {
        sort(points.begin(), points.end());
        vector<vector<int>> lower, upper;
        for (auto& p : points) {
            while (lower.size()>=2 && cross(lower[lower.size()-2], lower.back(), p) < 0) lower.pop_back();
            lower.push_back(p);
        }
        for (int i=points.size()-1;i>=0;i--) {
            auto& p = points[i];
            while (upper.size()>=2 && cross(upper[upper.size()-2], upper.back(), p) < 0) upper.pop_back();
            upper.push_back(p);
        }
        lower.pop_back(); upper.pop_back();
        lower.insert(lower.end(), upper.begin(), upper.end());
        return lower;
    }
};
