Beautiful Arrangement II

O(n) O(1) 61.1% AC
Mô tả
Tạo permutation 1..n độ dài n với |diff| liên tiếp khác nhau max (beautiful arrangement II).
Ví dụ
Ví dụ 1
Inn=3,k=1
Out[1,2,3]
Ví dụ 2
Inn=3,k=2
Out[1,3,2]
Ý tưởng
Two pointers l=1,r=n; alternate pick l,r để diff giảm dần.

class Solution:
    def constructArray(self, n: int, k: int) -> List[int]:
        ans=[]; l,r=1,n
        while l<=r:
            if k>0:
                if k%2: ans.append(l); l+=1
                else: ans.append(r); r-=1
                k-=1
            else:
                ans.append(l); l+=1
        return ans
class Solution {
public:
    vector<int> constructArray(int n, int k) {
        vector<int> ans; int l=1,r=n;
        while(l<=r){
            if(k>0){ if(k&1) ans.push_back(l++); else ans.push_back(r--); k--; }
            else ans.push_back(l++);
        }
        return ans;
    }
};
