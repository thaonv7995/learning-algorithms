Partition Labels

O(n) O(1) 81.9% AC
Mô tả
Cho chuỗi s chữ thường. Chia thành nhiều phần sao mỗi chữ cái xuất hiện tối đa một phần. Trả độ dài từng phần (max partition).
Ví dụ
Ví dụ 1
Ins = "ababcbacadefegdehijhklij"
Out[9,7,8]
Ví dụ 2
Ins = "eccbbbbdec"
Out[10]
Ý tưởng
Ghi last index mỗi ký tự. Scan i: end = max(end, last[s[i]]). Khi i==end cắt partition.

int* partitionLabels(char* s, int* retSize) {
    int last[26]; int n=strlen(s);
    for(int i=0;i<26;i++) last[i]=-1;
    for(int i=0;i<n;i++) last[s[i]-'a']=i;
    int* ans=malloc(n*4); *retSize=0; int st=0, ed=0;
    for(int i=0;i<n;i++){ ed=ed>last[s[i]-'a']?ed:last[s[i]-'a']; if(i==ed){ ans[(*retSize)++]=ed-st+1; st=i+1; } }
    return ans;
}
class Solution:
    def partitionLabels(self, s: str) -> List[int]:
        last = {c: i for i, c in enumerate(s)}
        ans, start, end = [], 0, 0
        for i, c in enumerate(s):
            end = max(end, last[c])
            if i == end:
                ans.append(end - start + 1); start = i + 1
        return ans
class Solution {
public:
    vector<int> partitionLabels(string s) {
        int last[26];
        for(int i=0;i<(int)s.size();i++) last[s[i]-'a']=i;
        vector<int> ans; int st=0, ed=0;
        for(int i=0;i<(int)s.size();i++){
            ed=max(ed,last[s[i]-'a']);
            if(i==ed){ ans.push_back(ed-st+1); st=i+1; }
        }
        return ans;
    }
};
