Elimination Game

O(log n) O(1) 46.4% AC
Mô tả
List 1..n; lần lượt xóa mọi số thứ 2 từ trái, rồi phải, trái,... Trả số cuối cùng còn lại.
Ví dụ
Ví dụ 1
Inn = 9
Out6
Ví dụ 2
Inn = 1
Out1
Ý tưởng
Recursion/math: f(n, leftToRight): n==1 return 1; half=f(n/2, !dir); n chẵn return half; n lẻ return dir ? half*2 : half*2+1.
Mã nguồn
solution.c C99 
static int f(int n, int left) {
    if (n == 1) return 1;
    int half = f(n/2, !left);
    if (n % 2 == 0) return left ? half : 2*half;
    return left ? half : 2*half+1;
}
int lastRemaining(int n) { return f(n, 1); }
class Solution:
    def lastRemaining(self, n: int) -> int:
        def f(n, left):
            if n == 1: return 1
            half = f(n // 2, not left)
            if n % 2 == 0:
                return half if left else 2 * half
            return half if left else 2 * half + 1
        return f(n, True)
class Solution {
    int f(int n, bool left) {
        if (n == 1) return 1;
        int half = f(n/2, !left);
        if (n % 2 == 0) return left ? half : 2*half;
        return left ? half : 2*half + 1;
    }
public:
    int lastRemaining(int n) { return f(n, true); }
};
