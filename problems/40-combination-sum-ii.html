Combination Sum II

O(2^N) O(N) 59.5% AC
Mô tả
Mỗi candidate dùng tối đa một lần; candidates có thể trùng — không trùng bộ kết quả.
Ví dụ
Ví dụ 1
Incandidates = [10,1,2,7,6,1,5], target = 8
Out[[1,1,6],[1,2,5],[1,7],[2,6]]
Ví dụ 2
Incandidates = [2,5,2,1,2], target = 5
Out[[1,2,2],[5]]
Ý tưởng
Sort. DFS(i, path, remain): skip duplicate candidates[i]==candidates[i-1] khi i>start. Chọn i → dfs(i+1, …).

static void cs40(int* c, int n, int i, int* path, int plen, int rem,
                   int*** out, int** cols, int* rc, int* cap) {
    if (rem == 0) {
        if (*rc >= *cap) { *cap *= 2; *out = (int**)realloc(*out, *cap * sizeof(int*)); *cols = (int*)realloc(*cols, *cap * sizeof(int)); }
        (*out)[*rc] = (int*)malloc(plen * sizeof(int));
        memcpy((*out)[*rc], path, plen * sizeof(int));
        (*cols)[*rc] = plen;
        (*rc)++;
        return;
    }
    if (rem < 0) return;
    for (int j = i; j < n; j++) {
        if (j > i && c[j] == c[j - 1]) continue;
        path[plen] = c[j];
        cs40(c, n, j + 1, path, plen + 1, rem - c[j], out, cols, rc, cap);
    }
}
int cmp40(const void* a, const void* b) { return *(int*)a - *(int*)b; }
int** combinationSum2(int* candidates, int candidatesSize, int target,
                      int* returnSize, int** returnColumnSizes) {
    qsort(candidates, candidatesSize, sizeof(int), cmp40);
    int cap = 32, rc = 0, path[32];
    int** out = (int**)malloc(cap * sizeof(int*));
    *returnColumnSizes = (int*)malloc(cap * sizeof(int));
    cs40(candidates, candidatesSize, 0, path, 0, target, &out, returnColumnSizes, &rc, &cap);
    *returnSize = rc;
    return out;
}
class Solution:
    def combinationSum2(self, c: List[int], t: int) -> List[List[int]]:
        c.sort(); res=[]
        def dfs(i, path, r):
            if r==0: res.append(path[:]); return
            if r<0: return
            for j in range(i, len(c)):
                if j>i and c[j]==c[j-1]: continue
                path.append(c[j]); dfs(j+1, path, r-c[j]); path.pop()
        dfs(0,[],t); return res
class Solution {
    vector<vector<int>> res;
    void dfs(vector<int>& c, int i, vector<int>& p, int r) {
        if (!r) { res.push_back(p); return; }
        if (r < 0) return;
        for (int j = i; j < (int)c.size(); j++) {
            if (j > i && c[j] == c[j-1]) continue;
            p.push_back(c[j]); dfs(c, j+1, p, r - c[j]); p.pop_back();
        }
    }
public:
    vector<vector<int>> combinationSum2(vector<int>& c, int t) {
        sort(c.begin(), c.end()); vector<int> p; dfs(c, 0, p, t); return res;
    }
};
