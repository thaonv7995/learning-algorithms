class Solution:
    def findShortestSubArray(self, nums: List[int]) -> int:
        from collections import defaultdict
        cnt=defaultdict(int); first={}; last={}
        for i,x in enumerate(nums):
            cnt[x]+=1; first.setdefault(x,i); last[x]=i
        deg=max(cnt.values())
        ans=len(nums)
        for x in cnt:
            if cnt[x]==deg: ans=min(ans,last[x]-first[x]+1)
        return ans
class Solution {
public:
    int findShortestSubArray(vector<int>& a) {
        unordered_map<int,int> cnt, fi, la;
        for(int i=0;i<(int)a.size();i++){ cnt[a[i]]++; fi.emplace(a[i],i); la[a[i]]=i; }
        int deg=0; for(auto& p:cnt) deg=max(deg,p.second);
        int ans=a.size();
        for(auto& p:cnt) if(p.second==deg) ans=min(ans, la[p.first]-fi[p.first]+1);
        return ans;
    }
};
