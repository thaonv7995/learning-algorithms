Replace Words

O(n*m) O(n) 68.7% AC
Mô tả
Cho dict và câu. Thay mỗi từ bằng root prefix ngắn nhất trong dict (nếu có).
Ví dụ
Ví dụ 1
Indict=["cat","bat","rat"], sentence="the cattle was rattled by the battery"
Out"the cat was rat by the bat"
Ví dụ 2
Indict=["a","b"], sentence="a a"
Out"a a"
Ý tưởng
Build trie dict; split sentence; mỗi từ walk trie lấy prefix ngắn nhất.

class Solution:
    def replaceWords(self, dictionary: List[str], sentence: str) -> str:
        trie={}
        for w in dictionary:
            node=trie
            for ch in w: node=node.setdefault(ch,{})
            node["#"]=w
        def prefix(w):
            node=trie
            for ch in w:
                if ch not in node: return w
                node=node[ch]
                if "#" in node: return node["#"]
            return w
        return " ".join(prefix(w) for w in sentence.split())
class Solution {
public:
    string replaceWords(vector<string>& dict, string s) {
        struct Node{ unordered_map<char,Node*> nxt; string root; };
        Node* rt=new Node();
        for(auto& w:dict){ Node* cur=rt; for(char c:w){ if(!cur->nxt[c]) cur->nxt[c]=new Node(); cur=cur->nxt[c]; } cur->root=w; }
        auto pref=[&](string w){ Node* cur=rt; string ans;
            for(char c:w){ if(!cur->nxt.count(c)) break; cur=cur->nxt[c]; if(!cur->root.empty()){ ans=cur->root; break; } }
            return ans.empty()?w:ans; };
        stringstream ss(s); string w,out; while(ss>>w){ if(!out.empty()) out+=" "; out+=pref(w);} return out;
    }
};
