Implement Trie (Prefix Tree)

O(m) O(N) 69.5% AC
Mô tả
Cài đặt Trie (prefix tree) hỗ trợ insert(word), search(word) trả bool, startsWith(prefix) trả bool có từ nào có prefix đó không.
Ví dụ
Ví dụ 1
Ininsert "apple"; search "apple" → true; search "app" → false; startsWith "app" → true
Outtrue, false, true
Ví dụ 2
Ininsert "app"; search "app"
Outtrue
Ý tưởng
Mảng con 26 node con + cờ isEnd. insert/search/startsWith duyệt từng ký tự; search cần isEnd, startsWith chỉ cần đi hết prefix.
Mã nguồn
solution.c C99 
typedef struct Node { struct Node* ch[26]; int end; } Node;

static Node* trieWalk(Node* root, const char* s, int needEnd) {
    Node* cur = root;
    for (; *s; s++) {
        int i = *s - 'a';
        if (!cur->ch[i]) return NULL;
        cur = cur->ch[i];
    }
    return (!needEnd || cur->end) ? cur : NULL;
}

void trieInsert(Node* root, const char* word) {
    Node* cur = root;
    for (; *word; word++) {
        int i = *word - 'a';
        if (!cur->ch[i]) cur->ch[i] = calloc(1, sizeof(Node));
        cur = cur->ch[i];
    }
    cur->end = 1;
}

int trieSearch(Node* root, const char* word) { return trieWalk(root, word, 1) != NULL; }
int trieStartsWith(Node* root, const char* p) { return trieWalk(root, p, 0) != NULL; }
class TrieNode:
    __slots__ = ('ch', 'end')
    def __init__(self):
        self.ch = {}
        self.end = False

class Trie:
    def __init__(self):
        self.root = TrieNode()
    def insert(self, word: str) -> None:
        node = self.root
        for c in word:
            node = node.ch.setdefault(c, TrieNode())
        node.end = True
    def search(self, word: str) -> bool:
        node = self.root
        for c in word:
            if c not in node.ch: return False
            node = node.ch[c]
        return node.end
    def startsWith(self, prefix: str) -> bool:
        node = self.root
        for c in prefix:
            if c not in node.ch: return False
            node = node.ch[c]
        return True
class Trie {
    struct Node { Node* ch[26]{}; bool end=false; }* root;
    Node* walk(const string& s, bool needEnd) {
        Node* cur = root;
        for (char c : s) {
            int i = c - 'a';
            if (!cur->ch[i]) return nullptr;
            cur = cur->ch[i];
        }
        return (!needEnd || cur->end) ? cur : nullptr;
    }
public:
    Trie() { root = new Node(); }
    void insert(string word) {
        Node* cur = root;
        for (char c : word) {
            int i = c - 'a';
            if (!cur->ch[i]) cur->ch[i] = new Node();
            cur = cur->ch[i];
        }
        cur->end = true;
    }
    bool search(string word) { return walk(word, true) != nullptr; }
    bool startsWith(string prefix) { return walk(prefix, false) != nullptr; }
};
