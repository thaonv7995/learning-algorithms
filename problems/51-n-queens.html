N-Queens

O(N!) O(N) 75.6% AC
Mô tả
Xếp N quân hậu trên bàn cờ N×N sao cho không hai quân cùng hàng, cột hay đường chéo.
Ví dụ
Ví dụ 1
Inn = 4
Out[[".Q..","...Q","Q...","..Q."], ["..Q.","Q...","...Q",".Q.."]]
Ví dụ 2
Inn = 1
Out[["Q"]]
Ý tưởng
Backtracking từng hàng — thử từng cột, kiểm tra cột và 2 đường chéo bằng set/hash.

/* Backtracking + board build — xem C++ */
class Solution:
    def solveNQueens(self, n: int) -> List[List[str]]:
        res, cols, d1, d2, board = [], set(), set(), set(), []
        def bt(r):
            if r == n:
                res.append([''.join(row) for row in board]); return
            for c in range(n):
                if c in cols or (r-c) in d1 or (r+c) in d2: continue
                cols.add(c); d1.add(r-c); d2.add(r+c)
                board.append(['.']*n); board[-1][c] = 'Q'
                bt(r+1); board.pop()
                cols.remove(c); d1.remove(r-c); d2.remove(r+c)
        bt(0); return res
class Solution {
public:
    vector<vector<string>> solveNQueens(int n) {
        vector<vector<string>> res;
        vector<string> board(n, string(n, '.'));
        vector<bool> col(n), d1(2*n), d2(2*n);
        function<void(int)> bt = [&](int r) {
            if (r == n) { res.push_back(board); return; }
            for (int c = 0; c < n; c++) {
                if (col[c] || d1[r-c+n] || d2[r+c]) continue;
                col[c]=d1[r-c+n]=d2[r+c]=true; board[r][c]='Q';
                bt(r+1); board[r][c]='.';
                col[c]=d1[r-c+n]=d2[r+c]=false;
            }
        };
        bt(0); return res;
    }
};
