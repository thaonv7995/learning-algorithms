N-Queens

O(N!) O(N) (Mảng lưu cấu hình quân hậu) 75.6% AC
Mô tả
Bài toán xếp quân hậu: Đặt n quân hậu trên một bàn cờ kích thước n x n sao cho không có hai quân hậu nào tấn công nhau (không cùng hàng, cột hoặc đường chéo). Hãy trả về tất cả các cấu hình bàn cờ hợp lệ.
Ví dụ
Ví dụ 1
Inn = 4
Out2 cấu hình bàn cờ hợp lệ dạng [[".Q..","...Q","Q...","..Q."], ...]
Ý tưởng
Áp dụng kỹ thuật Đệ quy Quay lui (Backtracking). Ta xếp quân hậu theo từng hàng `row` từ 0 đến N-1. Tại mỗi hàng, ta duyệt qua các cột `col` từ 0 đến N-1, kiểm tra xem việc đặt quân hậu tại `(row, col)` có an toàn không bằng cách sử dụng 3 bảng Hash Set / Mảng trạng thái lưu trữ các cột, đường chéo chính (r - c) và đường chéo phụ (r + c) đã có quân hậu. Nếu an toàn, ta ghi nhận và gọi đệ quy xếp tiếp hàng dưới.

// Solution in C (N-Queens)
void solveNQueensHelper(int row, int n, int* queens, char*** res, int* returnSize) {
    if (row == n) {
        res[*returnSize] = (char**)malloc(n * sizeof(char*));
        for (int i = 0; i < n; i++) {
            res[*returnSize][i] = (char*)malloc((n + 1) * sizeof(char));
            for (int j = 0; j < n; j++) {
                res[*returnSize][i][j] = (queens[i] == j) ? 'Q' : '.';
            }
            res[*returnSize][i][n] = '\0';
        }
        (*returnSize)++;
        return;
    }
    for (int col = 0; col < n; col++) {
        bool isValid = true;
        for (int r = 0; r < row; r++) {
            if (queens[r] == col || abs(queens[r] - col) == abs(r - row)) {
                isValid = false; break;
            }
        }
        if (isValid) {
            queens[row] = col;
            solveNQueensHelper(row + 1, n, queens, res, returnSize);
            queens[row] = -1;
        }
    }
}
# Solution in Python (N-Queens)
class Solution:
    def solveNQueens(self, n: int) -> List[List[str]]:
        res = []
        board = [["."] * n for _ in range(n)]
        cols = set()
        posDiag = set() # (r + c)
        negDiag = set() # (r - c)
        
        def backtrack(r):
            if r == n:
                copy = ["".join(row) for row in board]
                res.append(copy)
                return
            for c in range(n):
                if c in cols or (r + c) in posDiag or (r - c) in negDiag:
                    continue
                cols.add(c)
                posDiag.add(r + c)
                negDiag.add(r - c)
                board[r][c] = "Q"
                
                backtrack(r + 1)
                
                cols.remove(c)
                posDiag.remove(r + c)
                negDiag.remove(r - c)
                board[r][c] = "."
        backtrack(0)
        return res
class Solution {
    vector<vector<string>> res;
    vector<string> board;
    unordered_set<int> cols, diag1, diag2;
    
    void backtrack(int r, int n) {
        if (r == n) { res.push_back(board); return; }
        for (int c = 0; c < n; c++) {
            if (cols.count(c) || diag1.count(r - c) || diag2.count(r + c)) continue;
            
            board[r][c] = 'Q';
            cols.insert(c); diag1.insert(r - c); diag2.insert(r + c);
            
            backtrack(r + 1, n); // De quy xep hang tiep theo
            
            board[r][c] = '.'; // Quay lui (Backtrack)
            cols.erase(c); diag1.erase(r - c); diag2.erase(r + c);
        }
    }
public:
    vector<vector<string>> solveNQueens(int n) {
        board = vector<string>(n, string(n, '.'));
        backtrack(0, n);
        return res;
    }
};
