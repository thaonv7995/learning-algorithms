Longest Valid Parentheses

O(N) O(N) 38.8% AC
Mô tả
Tìm độ dài chuỗi ngoặc hợp lệ dài nhất trong s chỉ gồm '(' và ')'.
Ví dụ
Ví dụ 1
Ins = "(()"
Out2
Ví dụ 2
Ins = ")()())"
Out4
Ví dụ 3
Ins = ""
Out0
Ý tưởng
Stack lưu index. Push -1 làm base. '(' push i; ')' pop — nếu stack rỗng push i làm base mới, else cập nhật max = i - stack.top().

int longestValidParentheses(char* s) {
    int n = (int)strlen(s);
    int* st = (int*)malloc((n + 2) * sizeof(int));
    int top = 0;
    st[top++] = -1;
    int best = 0;
    for (int i = 0; s[i]; i++) {
        if (s[i] == '(') st[top++] = i;
        else {
            top--;
            if (top == 0) st[top++] = i;
            else {
                int len = i - st[top - 1];
                if (len > best) best = len;
            }
        }
    }
    free(st);
    return best;
}
class Solution:
    def longestValidParentheses(self, s: str) -> int:
        st, best = [-1], 0
        for i, c in enumerate(s):
            if c == '(': st.append(i)
            else:
                st.pop()
                if not st: st.append(i)
                else: best = max(best, i - st[-1])
        return best
class Solution {
public:
    int longestValidParentheses(string s) {
        stack<int> st; st.push(-1); int best = 0;
        for (int i = 0; i < (int)s.size(); i++) {
            if (s[i] == '(') st.push(i);
            else {
                st.pop();
                if (st.empty()) st.push(i);
                else best = max(best, i - st.top());
            }
        }
        return best;
    }
};
