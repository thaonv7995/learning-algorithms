Beautiful Arrangement

O(n!) O(n) 64.8% AC
Mô tả
Hoán vị 1..n sao ở vị trí i (1-indexed), i chia hết cho nums[i] hoặc nums[i] chia hết cho i. Đếm số hoán vị "beautiful".
Ví dụ
Ví dụ 1

Out2
Ví dụ 2

Out1
Ý tưởng
Backtracking: thử gán số chưa dùng vào pos i nếu thỏa điều kiện chia hết.

/* Backtracking divisibility — xem C++ */
class Solution:
    def countArrangement(self, n: int) -> int:
        used = [False] * (n + 1)
        def dfs(i):
            if i > n:
                return 1
            ans = 0
            for v in range(1, n + 1):
                if not used[v] and (i % v == 0 or v % i == 0):
                    used[v] = True
                    ans += dfs(i + 1)
                    used[v] = False
            return ans
        return dfs(1)
class Solution {
    int n;
    vector<bool> used;
    int dfs(int i) {
        if (i > n) return 1;
        int ans = 0;
        for (int v = 1; v <= n; v++) {
            if (!used[v] && (i % v == 0 || v % i == 0)) {
                used[v] = true;
                ans += dfs(i + 1);
                used[v] = false;
            }
        }
        return ans;
    }
public:
    int countArrangement(int n) {
        this->n = n;
        used.assign(n + 1, false);
        return dfs(1);
    }
};
