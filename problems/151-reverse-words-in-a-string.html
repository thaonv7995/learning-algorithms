Reverse Words in a String

O(n) O(n) 56.4% AC
Mô tả
Cho chuỗi s chứa từ và khoảng trắng. Đảo thứ tự các từ trong chuỗi, loại bỏ khoảng trắng thừa ở đầu/cuối và giữa các từ chỉ còn một dấu cách. Trả về chuỗi kết quả.
Ví dụ
Ví dụ 1
Ins = "the sky is blue"
Out"blue is sky the"
Ví dụ 2
Ins = " hello world "
Out"world hello"
Ví dụ 3
Ins = "a good example"
Out"example good a"
Ý tưởng
Trim, tách theo whitespace thành mảng từ, đảo mảng rồi join bằng một space. Hoặc reverse toàn chuỗi rồi reverse từng từ in-place O(1) extra nếu mutable char[].

static void reverseRange(char* a, int lo, int hi) {
    while (lo < hi) { char t = a[lo]; a[lo] = a[hi]; a[hi] = t; lo++; hi--; }
}
static void reverseWordsInPlace(char* s) {
    int n = (int)strlen(s), i = 0;
    reverseRange(s, 0, n - 1);
    while (i < n) {
        while (i < n && s[i] == ' ') i++;
        int j = i;
        while (j < n && s[j] != ' ') j++;
        if (i < j) reverseRange(s, i, j - 1);
        i = j;
    }
}
char* reverseWords(char* s) { reverseWordsInPlace(s); return s; }
class Solution:
    def reverseWords(self, s: str) -> str:
        return ' '.join(s.split()[::-1])
class Solution {
public:
    string reverseWords(string s) {
        string word, res;
        int i = 0, n = s.size();
        while (i < n) {
            while (i < n && s[i] == ' ') i++;
            if (i >= n) break;
            int j = i;
            while (j < n && s[j] != ' ') j++;
            word = s.substr(i, j - i);
            res = res.empty() ? word : word + " " + res;
            i = j;
        }
        return res;
    }
};
