Non-decreasing Subsequences

O(n * 2^n) O(n) 62.7% AC
Mô tả
Tìm mọi subsequence length ≥2 của nums sao elements non-decreasing (không nhất liên tiếp). Trả list không trùng.
Ví dụ
Ví dụ 1
Innums = [4,6,7,7]
Out[[4,6],[4,6,7],[4,6,7,7],[4,7],[4,7,7],[6,7],[6,7,7],[7,7]]
Ví dụ 2
Innums = [4,4,3,2,1]
Out[]
Ý tưởng
Backtrack: tại index i chọn start subsequence; dùng set per level tránh duplicate picks cùng value.

class Solution:
    def findSubsequences(self, nums: List[int]) -> List[List[int]]:
        ans = []
        path = []
        def dfs(i):
            if len(path) >= 2: ans.append(path[:])
            used = set()
            for j in range(i, len(nums)):
                if nums[j] in used: continue
                if path and nums[j] < path[-1]: continue
                used.add(nums[j])
                path.append(nums[j])
                dfs(j+1)
                path.pop()
        dfs(0)
        return ans
class Solution {
public:
    vector<vector<int>> findSubsequences(vector<int>& nums) {
        vector<vector<int>> ans; vector<int> path;
        function<void(int)> dfs=[&](int i){
            if(path.size()>=2) ans.push_back(path);
            unordered_set<int> used;
            for(int j=i;j<(int)nums.size();j++){
                if(used.count(nums[j])) continue;
                if(!path.empty()&&nums[j]<path.back()) continue;
                used.insert(nums[j]);
                path.push_back(nums[j]); dfs(j+1); path.pop_back();
            }
        };
        dfs(0); return ans;
    }
};
