Number of Digit One

O(log n) O(1) 38.7% AC
Mô tả
Đếm số lần chữ số '1' xuất hiện khi viết mọi số nguyên từ 1 đến n (bao gồm n).
Ví dụ
Ví dụ 1
Inn = 13
Out6
Ví dụ 2
Inn = 0
Out0
Ý tưởng
Digit DP / math từng vị trí: factor=1,10,100...; higher, cur, lower từ n; cộng count tại mỗi digit.
Mã nguồn
solution.c C99 
int countDigitOne(int n) {
    if (n <= 0) return 0;
    long long count = 0, factor = 1;
    while (factor <= n) {
        long long lower = n - n / factor * factor;
        int cur = n / factor % 10;
        long long higher = n / (factor * 10);
        if (cur == 0) count += higher * factor;
        else if (cur == 1) count += higher * factor + lower + 1;
        else count += (higher + 1) * factor;
        factor *= 10;
    }
    return (int)count;
}
class Solution:
    def countDigitOne(self, n: int) -> int:
        if n <= 0:
            return 0
        count, factor = 0, 1
        while factor <= n:
            lower = n - n // factor * factor
            cur = n // factor % 10

                count += higher * factor
            elif cur == 1:
                count += higher * factor + lower + 1

                count += (higher + 1) * factor
            factor *= 10
        return count
