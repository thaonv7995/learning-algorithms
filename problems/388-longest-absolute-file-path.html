Longest Absolute File Path

O(n) O(d) 49.4% AC
Mô tả
Input mô tả cấu trúc file ảo: mỗi dòng có depth (số tab) + tên. Trả độ dài đường dẫn tuyệt đối dài nhất tới file (không thư mục).
Ví dụ
Ví dụ 1
Ininput = "dir\n\tsubdir1\n\tsubdir2\n\t\tfile.ext"
Out20
Ví dụ 2
Ininput = "dir\n\ta\n\t\tb\n\t\t\tc\n\t\t\tx.txt"
Out32
Ý tưởng
Stack lưu độ dài path tại mỗi depth: đếm tab depth; stack[depth]=stack[depth-1]+len(name)+1; nếu có '.' cập nhật max.

int lengthLongestPath(char* input) {
    int stack[128] = {0}, top = 1, ans = 0;
    char* line = strtok(input, "\n");
    while (line) {
        int depth = 0;
        while (line[depth] == '\t') depth++;
        char* name = line + depth;
        int len = stack[depth] + strlen(name) + (depth ? 1 : 0);
        stack[depth + 1] = len;
        if (strchr(name, '.')) if (len > ans) ans = len;
        line = strtok(NULL, "\n");
    }
    return ans;
}
class Solution:
    def lengthLongestPath(self, input: str) -> int:
        stack, ans = [0], 0
        for line in input.split('\n'):
            depth = line.count('\t')
            name = line[depth:]
            length = stack[depth] + len(name) + (1 if depth else 0)
            if depth + 1 == len(stack): stack.append(length)
            else: stack[depth + 1] = length
            if '.' in name:
                ans = max(ans, length)
        return ans
class Solution {
public:
    int lengthLongestPath(string input) {
        vector<int> stack = {0}; int ans = 0;
        string line;
        for (int i = 0; i <= (int)input.size(); i++) {
            if (i == input.size() || input[i] == '\n') {
                int depth = 0;
                while (depth < (int)line.size() && line[depth] == '\t') depth++;
                string name = line.substr(depth);
                int len = stack[depth] + (int)name.size() + (depth ? 1 : 0);
                if (depth + 1 >= (int)stack.size()) stack.push_back(len);
                else stack[depth + 1] = len;
                if (name.find('.') != string::npos) ans = max(ans, len);
                line.clear();
            } else line.push_back(input[i]);
        }
        return ans;
    }
};
