Implement Magic Dictionary

O(m) O(n) 57.9% AC
Mô tả
Magic Dictionary: buildDict(words); search(word) true nếu đổi đúng 1 ký tự có từ trong dict.
Ví dụ
Ví dụ 1
InbuildDict(["hello","leetcode"]); search("hello")
Outfalse
Ví dụ 2
Insearch("hhllo")
Outtrue
Ý tưởng
Trie hoặc hash; search thử đổi từng vị trí 1 char.
Mã nguồn
solution.c C99 
typedef struct{ char** w; int n; } MagicDictionary;
MagicDictionary* magicDictionaryCreate(){ return calloc(1,sizeof(MagicDictionary)); }
void magicDictionaryBuildDict(MagicDictionary* d,char** w,int n){ d->w=w; d->n=n; }
bool magicDictionarySearch(MagicDictionary* d,char* word){ int i,j,k,wl=strlen(word); for(i=0;i<wl;i++) for(c='a';c<='z';c++){ if(c==word[i])continue; char t[256]; strcpy(t,word); t[i]=c; for(k=0;k<d->n;k++) if(strcmp(t,d->w[k])==0) return 1; } return 0; }
class MagicDictionary:
    def __init__(self):
        self.words=set()
    def buildDict(self, words):
        self.words=set(words)
    def search(self, word):
        for i in range(len(word)):
            for c in "abcdefghijklmnopqrstuvwxyz":
                if c==word[i]: continue
                if word[:i]+c+word[i+1:] in self.words: return True
        return False
class MagicDictionary {
    unordered_set<string> st;
public:
    void buildDict(vector<string> w){ st={w.begin(),w.end()}; }
    bool search(string word){
        for(int i=0;i<(int)word.size();i++)
            for(char c='a';c<='z';c++){
                if(c==word[i]) continue;
                string t=word; t[i]=c;
                if(st.count(t)) return true;
            }
        return false;
    }
};
