class Solution:
    def flipLights(self, n: int, presses: int) -> int:
        if presses==0: return 1
        presses=min(presses,3); n=min(n,3)
        all={tuple([0]*n)}
        for _ in range(presses):
            nxt=set()
            for st in all:
                for op in range(4):
                    a=list(st)
                    if op==0: a=[1-x for x in a]
                    elif op==1 and n>1: a[1]=1-a[1]; a[2]=1-a[2] if n>2 else a[1]
                    elif op==2: a[0]=1-a[0]; a[2]=1-a[2] if n>2 else a[0]
                    elif op==3 and n>2: a[0]=1-a[0]; a[1]=1-a[1]
                    nxt.add(tuple(a))
            all=nxt
        return len(all)
class Solution {
public:
    int flipLights(int n, int p) {
        if(!p) return 1; p=min(p,3); n=min(n,3);
        set<int> st={0};
        for(int t=0;t<p;t++){
            set<int> nx;
            for(int s:st) for(int op=0;op<4;op++){
                int a=s;
                if(op==0) a^=7;
                else if(op==1) a^=6;
                else if(op==2) a^=5;
                else a^=3;
                nx.insert(a&( (1<<n)-1 ));
            }
            st=nx;
        }
        return st.size();
    }
};
