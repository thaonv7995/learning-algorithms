Different Ways to Add Parentheses

O(4^n) O(4^n) 73.3% AC
Mô tả
Chuỗi expression gồm số nguyên và toán tử +, -, * (không có ngoặc sẵn). Trả mọi giá trị có thể khi thêm ngoặc hợp lệ (không đổi thứ tự số và op).
Ví dụ
Ví dụ 1
Inexpression = "2-1-1"
Out[0,2]
Ví dụ 2
Inexpression = "2*3-4*5"
Out[-34,-14,-10,-10,10]
Ý tưởng
Divide & conquer + memo: với mỗi op ở i, tách trái/phải, kết hợp mọi cặp kết quả con the op[i].

static int eval241(int a, int b, char op) {
    if (op == '+') return a + b;
    if (op == '-') return a - b;
    return a * b;
}
static int* dfs241(char* s, int* sz) {
    int allDigit = 1;
    for (int i = 0; s[i]; i++) if (s[i] < '0' || s[i] > '9') { allDigit = 0; break; }
    if (allDigit) { int* r = malloc(sizeof(int)); *r = atoi(s); *sz = 1; return r; }
    int cap = 8, n = 0; int* res = malloc(cap * sizeof(int));
    for (int i = 0; s[i]; i++) {
        if (s[i] == '+' || s[i] == '-' || s[i] == '*') {
            char tmp = s[i]; s[i] = 0;
            int ls, rs;
            int* L = dfs241(s, &ls);
            int* R = dfs241(s + i + 1, &rs);
            s[i] = tmp;
            for (int a = 0; a < ls; a++) for (int b = 0; b < rs; b++) {
                if (n >= cap) { cap *= 2; res = realloc(res, cap * sizeof(int)); }
                res[n++] = eval241(L[a], R[b], tmp);
            }
            free(L); free(R);
        }
    }
    *sz = n; return res;
}
int* diffWaysToCompute(char* expression, int* rs) { return dfs241(expression, rs); }
class Solution:
    def diffWaysToCompute(self, expression: str) -> List[int]:
        memo = {}
        def dfs(s):
            if s in memo:
                return memo[s]
            if s.isdigit():
                return [int(s)]
            res = []
            for i, c in enumerate(s):
                if c in '+-*':
                    for a in dfs(s[:i]):
                        for b in dfs(s[i+1:]):
                            if c == '+': res.append(a + b)
                            elif c == '-': res.append(a - b)
                            else: res.append(a * b)
            memo[s] = res
            return res
        return dfs(expression)
class Solution {
    unordered_map<string, vector<int>> memo;
    vector<int> dfs(string s) {
        if (memo.count(s)) return memo[s];
        vector<int> res;
        bool allDigit = true;
        for (char c : s) if (!isdigit(c)) { allDigit = false; break; }
        if (allDigit) return {stoi(s)};
        for (int i = 0; i < (int)s.size(); i++) {
            if (s[i] == '+' || s[i] == '-' || s[i] == '*') {
                auto L = dfs(s.substr(0, i)), R = dfs(s.substr(i + 1));
                for (int a : L) for (int b : R) {
                    if (s[i] == '+') res.push_back(a + b);
                    else if (s[i] == '-') res.push_back(a - b);
                    else res.push_back(a * b);
                }
            }
        }
        return memo[s] = res;
    }
public:
    vector<int> diffWaysToCompute(string expression) { return dfs(expression); }
};
