Path Sum II

O(n) O(h) 62.2% AC
Mô tả
Cho root và targetSum. Trả về tất cả đường đi từ gốc đến lá có tổng bằng targetSum (mỗi đường là mảng giá trị nút).
Ví dụ
Ví dụ 1
Inroot = [5,4,8,11,null,13,4,7,2,null,null,5,1], targetSum = 22
Out[[5,4,11,2],[5,8,4,5]]
Ví dụ 2
Inroot = [1,2,3], targetSum = 5
Out[]
Ý tưởng
Backtracking DFS: push val vào path; tại lá nếu rem == val thì copy path vào kết quả; pop khi backtrack.
Mã nguồn
solution.c C99 
/* Backtracking DFS thu path — xem C++ */
class Solution:
    def pathSum(self, root: Optional[TreeNode], targetSum: int) -> List[List[int]]:
        res = []
        def dfs(node, rem, path):
            if not node: return
            path.append(node.val)
            if not node.left and not node.right and rem == node.val:
                res.append(path[:])
            else:
                dfs(node.left, rem - node.val, path)
                dfs(node.right, rem - node.val, path)
            path.pop()
        dfs(root, targetSum, [])
        return res
class Solution {
    vector<vector<int>> res;
    void dfs(TreeNode* nd, int rem, vector<int>& path) {
        if (!nd) return;
        path.push_back(nd->val);
        if (!nd->left && !nd->right && rem == nd->val)
            res.push_back(path);
        else {
            dfs(nd->left, rem - nd->val, path);
            dfs(nd->right, rem - nd->val, path);
        }
        path.pop_back();
    }
public:
    vector<vector<int>> pathSum(TreeNode* root, int targetSum) {
        vector<int> path;
        dfs(root, targetSum, path);
        return res;
    }
};
