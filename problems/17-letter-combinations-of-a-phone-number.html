Letter Combinations of a Phone Number

O(4^N) O(N) 66.1% AC
Mô tả
Cho chuỗi digits (2–9), trả về mọi chuỗi chữ cái tương ứng bàn phím điện thoại.
Ví dụ
Ví dụ 1

Out["ad","ae","af","bd","be","bf","cd","ce","cf"]
Ví dụ 2

Out[]
Ý tưởng
Backtracking: map digit → letters, dfs(idx, path). Tại idx chọn từng chữ cái của digits[idx], đệ quy idx+1. Khi idx == len → push path.

void dfs(char* digits, int idx, char* path, int plen, char** res, int* rs, char map[10][5]) {
    if (!digits[idx]) { res[(*rs)++] = strdup(path); return; }
    int d = digits[idx]-'0';
    for (int i = 0; map[d][i]; i++) {
        path[plen] = map[d][i]; path[plen+1] = 0;
        dfs(digits, idx+1, path, plen+1, res, rs, map);
    }
}
class Solution:
    def letterCombinations(self, digits: str) -> List[str]:
        if not digits: return []
        m = {'2':'abc','3':'def','4':'ghi','5':'jkl','6':'mno','7':'pqrs','8':'tuv','9':'wxyz'}
        res = []
        def dfs(i, path):
            if i == len(digits):
                res.append(''.join(path)); return
            for c in m[digits[i]]:
                path.append(c); dfs(i+1, path); path.pop()
        dfs(0, [])
        return res
class Solution {
    map<char,string> m = {{'2',"abc"},{'3',"def"},{'4',"ghi"},{'5',"jkl"},{'6',"mno"},{'7',"pqrs"},{'8',"tuv"},{'9',"wxyz"}};
    vector<string> res;
    string path;
    void dfs(string& d, int i) {
        if (i == (int)d.size()) { res.push_back(path); return; }
        for (char c : m[d[i]]) { path += c; dfs(d, i+1); path.pop_back(); }
    }
public:
    vector<string> letterCombinations(string digits) {
        if (digits.empty()) return {};
        dfs(digits, 0); return res;
    }
};
