Merge k Sorted Lists

O(N log k) O(k) 59.6% AC
Mô tả
Cho mảng k danh sách liên kết đã sắp xếp. Trộn thành một list đã sort. Mỗi phần tử nằm trong đúng một list.
Ví dụ
Ví dụ 1
Inlists = [[1,4,5],[1,3,4],[2,6]]
Out[1,1,2,3,4,4,5,6]
Ví dụ 2
Inlists = []
Out[]
Ví dụ 3
Inlists = [[]]
Out[]
Ý tưởng
Min-heap chứa (val, listIndex, node). Pop nhỏ nhất → nối vào kết quả, push node.next nếu có. Hoặc divide-and-conquer merge từng cặp.
Mã nguồn
solution.c C99 
/* mergeKLists — min-heap hoặc merge từng cặp; xem C++ */
import heapq
class Solution:
    def mergeKLists(self, lists):
        h = []
        for i, l in enumerate(lists):
            if l: heapq.heappush(h, (l.val, i, l))
        dummy = ListNode(); cur = dummy
        while h:
            _, i, node = heapq.heappop(h)
            cur.next = node; cur = cur.next
            if node.next: heapq.heappush(h, (node.next.val, i, node.next))
        return dummy.next
class Solution {
public:
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        auto cmp = [](ListNode* a, ListNode* b){ return a->val > b->val; };
        priority_queue<ListNode*, vector<ListNode*>, decltype(cmp)> pq(cmp);
        for (auto* l : lists) if (l) pq.push(l);
        ListNode dummy(0); ListNode* t = &dummy;
        while (!pq.empty()) {
            auto* n = pq.top(); pq.pop();
            t->next = n; t = t->next;
            if (n->next) pq.push(n->next);
        }
        return dummy.next;
    }
};
