Contains Duplicate

O(n) O(n) 64.4% AC
Mô tả
Cho mảng nums. Trả true nếu có ít nhất một giá trị xuất hiện ít nhất hai lần, ngược lại false.
Ví dụ
Ví dụ 1
Innums = [1,2,3,1]
Outtrue
Ví dụ 2
Innums = [1,2,3,4]
Outfalse
Ví dụ 3
Innums = [1,1,1,3,3,4,3,2,4,2]
Outtrue
Ý tưởng
Duyệt nums, insert vào set; nếu phần tử đã có trong set → true. Hết vòng → false.
Mã nguồn
solution.c C99 
static int cmpInt(const void* a, const void* b) { return *(int*)a - *(int*)b; }
bool containsDuplicate(int* nums, int n) {
    int* a = malloc(n * sizeof(int));
    memcpy(a, nums, n * sizeof(int));
    qsort(a, n, sizeof(int), cmpInt);
    for (int i = 1; i < n; i++)
        if (a[i] == a[i - 1]) { free(a); return true; }
    free(a);
    return false;
}
class Solution:
    def containsDuplicate(self, nums: List[int]) -> bool:
        seen = set()
        for x in nums:
            if x in seen:
                return True
            seen.add(x)
        return False
class Solution {
public:
    bool containsDuplicate(vector<int>& nums) {
        unordered_set<int> seen;
        for (int x : nums) {
            if (seen.count(x)) return true;
            seen.insert(x);
        }
        return false;
    }
};
