int getImportance690(int** emp,int n,int id) {
    int idx=-1; for(int i=0;i<n;i++) if(emp[i][0]==id){idx=i;break;}
    if(idx<0) return 0;
    int sum=emp[idx][1];
    for(int j=2;j<emp[idx][2]+2;j++) sum+=getImportance690(emp,n,emp[idx][j]);
    return sum;
}
class Solution:
    def getImportance(self, employees: List['Employee'], id: int) -> int:
        mp={e.id:e for e in employees}
        def dfs(i):
            e=mp[i]; s=e.importance
            for sub in e.subordinates: s+=dfs(sub)
            return s
        return dfs(id)
class Solution {
public:
    int getImportance(vector<Employee*> employees, int id) {
        unordered_map<int,Employee*> mp;
        for(auto e:employees) mp[e->id]=e;
        function<int(int)> dfs=[&](int i){
            int s=mp[i]->importance;
            for(int sub:mp[i]->subordinates) s+=dfs(sub);
            return s;
        };
        return dfs(id);
    }
};
