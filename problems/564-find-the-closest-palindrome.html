Find the Closest Palindrome

O(log n) O(log n) 32% AC
Mô tả
Cho số nguyên n. Trả palindrome gần n nhất (khác n). Nếu hoà tie chọn nhỏ hơn.
Ví dụ
Ví dụ 1
Inn = 4
Out3
Ví dụ 2
Inn = 123
Out121
Ví dụ 3
Inn = 1234
Out1221
Ý tưởng
Sinh ứng viên từ n: palindrome cùng độ dài (mirror nửa trái ±1), và độ dài n±1 (99, 1001...). Lọc khác n, min |diff|.

/* Generate palindrome candidates — xem C++ */
class Solution:
    def nearestPalindromic(self, n: str) -> str:
        def mirror(s, odd):
            k = len(s)//2 if odd else len(s)//2
            return s[:k] + s[:k][::-1] if odd else s[:k] + s[:k-1][::-1]
        def candidates(s):
            L = len(s)
            res = {str(10**L - 1), str(10**(L-1) + 1)}
            k = L // 2
            left = int(s[:k] if L%2 else s[:k])
            for d in (-1, 0, 1):
                nl = str(left + d)
                if L % 2:
                    res.add(nl + nl[-2::-1])
                else:
                    res.add(nl + nl[::-1])
            return res
        cands = [c for c in candidates(n) if c != n]
        cands.sort(key=lambda x: (abs(int(x)-int(n)), int(x)))
        return cands[0]
class Solution {
    long long build(const string& left, bool odd, const string& s) {
        string p = left;
        if (odd) { for (int i = (int)s.size()/2-1; i >= 0; i--) p += s[i]; }
        else { for (int i = (int)left.size()-1; i >= 0; i--) p += left[i]; }
        return stoll(p);
    }
public:
    string nearestPalindromic(string n) {
        long long num = stoll(n);
        set<long long> cand;
        int L = n.size();
        cand.insert((long long)pow(10,L)-1);
        cand.insert((long long)pow(10,L-1)+1);
        string half = n.substr(0, (L+1)/2);
        long long h = stoll(half);
        for (long long d : {h-1,h,h+1}) {
            string ds = to_string(d);
            cand.insert(build(ds, L%2, n));
        }
        cand.erase(num);
        long long best = *cand.begin(), bestDiff = LLONG_MAX;
        for (long long c : cand) {
            long long diff = llabs(c-num);
            if (diff < bestDiff || (diff==bestDiff && c < best)) { bestDiff=diff; best=c; }
        }
        return to_string(best);
    }
};
