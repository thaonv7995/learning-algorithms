Monotone Increasing Digits

O(n) O(n) 49.7% AC
Mô tả
Cho số nguyên dương N. Trả số lớn nhất ≤ N mà các chữ số không giảm từ trái sang phải (monotone increasing).
Ví dụ
Ví dụ 1
InN = 10
Out9
Ví dụ 2
InN = 1234
Out1234
Ví dụ 3
InN = 332
Out299
Ý tưởng
Quét từ phải: nếu digits[i]<digits[i-1] thì giảm digits[i-1]-- và set digits[i..] = 9.
Mã nguồn
solution.c C99 
int monotoneIncreasingDigits(int N) {
    char s[16]; sprintf(s, "%d", N);
    int n = strlen(s);
    for (int i = n - 1; i > 0; i--)
        if (s[i] < s[i-1]) {
            s[i-1]--;
            for (int j = i; j < n; j++) s[j] = '9';
        }
    return atoi(s);
}
class Solution:
    def monotoneIncreasingDigits(self, N: int) -> int:
        s = list(str(N))
        for i in range(len(s)-1, 0, -1):
            if s[i] < s[i-1]:
                s[i-1] = str(int(s[i-1]) - 1)
                for j in range(i, len(s)): s[j] = '9'
        return int("".join(s))
class Solution {
public:
    int monotoneIncreasingDigits(int N) {
        string s = to_string(N);
        for (int i = s.size()-1; i > 0; i--)
            if (s[i] < s[i-1]) {
                s[i-1]--;
                for (int j = i; j < (int)s.size(); j++) s[j] = '9';
            }
        return stoi(s);
    }
};
