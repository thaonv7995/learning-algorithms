Heaters

O(n log n + m log m) O(1) 41.9% AC
Mô tả
houses và heaters trên trục số. Minimize max distance từ mỗi house tới heater gần nhất (radius tối thiểu).
Ví dụ
Ví dụ 1
Inhouses = [1,2,3], heaters = [2]
Out1
Ví dụ 2
Inhouses = [1,2,3,4], heaters = [1,4]
Out1
Ý tưởng
Sort both; với mỗi house binary search lower_bound heater; dist = min(|h-heater|).

int findRadius(int* houses, int hn, int* heaters, int hen) {
    qsort(houses,hn,sizeof(int),cmpi);
    qsort(heaters,hen,sizeof(int),cmpi);
    int ans=0;
    for(int hi=0;hi<hn;hi++){
        int h=houses[hi], d=INT_MAX;
        int l=0,r=hen;
        while(l<r){ int m=(l+r)/2; if(heaters[m]<h) l=m+1; else r=m; }
        if(l<hen){ int t=heaters[l]-h; if(t<d) d=t; }
        if(l>0){ int t=h-heaters[l-1]; if(t<d) d=t; }
        if(d>ans) ans=d;
    }
    return ans;
}
class Solution:
    def findRadius(self, houses: List[int], heaters: List[int]) -> int:
        houses.sort(); heaters.sort()
        ans = 0
        for h in houses:
            i = bisect.bisect_left(heaters, h)
            d = float("inf")
            if i < len(heaters): d = min(d, abs(heaters[i]-h))
            if i: d = min(d, abs(heaters[i-1]-h))
            ans = max(ans, d)
        return ans
class Solution {
public:
    int findRadius(vector<int>& houses, vector<int>& heaters) {
        sort(houses.begin(), houses.end()); sort(heaters.begin(), heaters.end());
        int ans=0;
        for(int h:houses){
            auto it=lower_bound(heaters.begin(), heaters.end(), h);
            int d=INT_MAX;
            if(it!=heaters.end()) d=min(d, abs(*it-h));
            if(it!=heaters.begin()) d=min(d, abs(*prev(it)-h));
            ans=max(ans,d);
        }
        return ans;
    }
};
