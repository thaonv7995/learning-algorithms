Implement Stack using Queues

O(n) push O(n) 69.9% AC
Mô tả
Cài đặt stack LIFO chỉ dùng hai queue (hoặc một queue). Hỗ trợ push, pop, top, empty.
Ví dụ
Ví dụ 1
Inpush(1); push(2); top() → 2; pop() → 2; empty() → false
Out2, 2, false
Ví dụ 2
Inpush(1); pop(); empty()
Outtrue
Ý tưởng
Một queue: push x thì enqueue x rồi rotate size-1 lần (dequeue→enqueue) để x ở đầu — top/pop O(1).
Mã nguồn
solution.c C99 
typedef struct { int* a; int h, t, cap; } Q;
static void qPush(Q* q, int x) { if (q->h==q->t) { q->cap=q->cap? q->cap*2:4; q->a=realloc(q->a,q->cap*sizeof(int));} q->a[q->t++]=x; if(q->t==q->cap) q->t=0; }
static int qPop(Q* q) { int v=q->a[q->h++]; if(q->h==q->cap) q->h=0; return v; }
static int qFront(Q* q) { return q->a[q->h]; }
static int qSize(Q* q) { return q->t>=q->h? q->t-q->h : q->cap-q->h+q->t; }
void stackPush(Q* s, int x) { qPush(s,x); for(int i=qSize(s)-1;i>0;i--){ qPush(s,qPop(s)); } }
int stackPop(Q* s) { return qPop(s); }
int stackTop(Q* s) { return qFront(s); }
int stackEmpty(Q* s) { return qSize(s)==0; }
class MyStack:
    def __init__(self):
        self.q = deque()
    def push(self, x: int) -> None:
        self.q.append(x)
        for _ in range(len(self.q) - 1):
            self.q.append(self.q.popleft())
    def pop(self) -> int:
        return self.q.popleft()
    def top(self) -> int:
        return self.q[0]
    def empty(self) -> bool:
        return not self.q
class MyStack {
    queue<int> q;
public:
    void push(int x) {
        q.push(x);
        for (int i = q.size() - 1; i > 0; i--) {
            q.push(q.front()); q.pop();
        }
    }
    int pop() { int t = q.front(); q.pop(); return t; }
    int top() { return q.front(); }
    bool empty() { return q.empty(); }
};
