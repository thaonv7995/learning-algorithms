Rotate Image

O(N²) O(1) 80.1% AC
Mô tả
Xoay ma trận vuông n×n 90° theo chiều kim đồng hồ, in-place.
Ví dụ
Ví dụ 1
Inmatrix = [[1,2,3],[4,5,6],[7,8,9]]
Out[[7,4,1],[8,5,2],[9,6,3]]
Ví dụ 2
Inmatrix = [[5,1,9,11],[2,4,8,10],[13,3,6,7],[15,14,12,16]]
Out[[15,13,2,5],[14,3,4,1],[12,6,8,9],[16,7,10,11]]
Ý tưởng
(1) Transpose: swap matrix[i][j] với matrix[j][i]. (2) Reverse từng hàng.
Mã nguồn

void rotate(int** matrix, int n, int* cs) {
    for (int i = 0; i < n; i++)
        for (int j = i + 1; j < n; j++) {
            int t = matrix[i][j]; matrix[i][j] = matrix[j][i]; matrix[j][i] = t;
        }
    for (int i = 0; i < n; i++)
        for (int l = 0, r = n - 1; l < r; l++, r--) {
            int t = matrix[i][l]; matrix[i][l] = matrix[i][r]; matrix[i][r] = t;
        }
}
class Solution:
    def rotate(self, matrix: List[List[int]]) -> None:
        n = len(matrix)
        for i in range(n):
            for j in range(i+1, n):
                matrix[i][j], matrix[j][i] = matrix[j][i], matrix[i][j]
        for row in matrix:
            row.reverse()
class Solution {
public:
    void rotate(vector<vector<int>>& m) {
        int n=m.size();
        for(int i=0;i<n;i++) for(int j=i+1;j<n;j++) swap(m[i][j],m[j][i]);
        for(auto& r:m) reverse(r.begin(),r.end());
    }
};
