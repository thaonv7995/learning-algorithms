Additive Number

O(n^3) O(n) 33.9% AC
Mô tả
Chuỗi s chỉ gồm chữ số. Kiểm tra có thể tách thành dãy additive không: mỗi số (trừ hai số đầu) bằng tổng hai số liền trước, và không có số 0 đứng đầu trừ khi số đó là "0".
Ví dụ
Ví dụ 1
Ins = "112358"
Outtrue
Ví dụ 2
Ins = "199100199"
Outtrue
Ví dụ 3
Ins = "101"
Outfalse
Ý tưởng
Thử mọi cặp số đầu (i,j cắt chuỗi). Dùng big-int string add: nếu s[k..] bắt đầu bằng a+b thì tiếp tục với (b, a+b). Thành công khi ăn hết chuỗi.

static int startsWith(const char* s, int k, const char* pref) {
    return strncmp(s + k, pref, strlen(pref)) == 0;
}
bool isAdditiveNumber(char* s) {
    int n = strlen(s);
    for (int i = 1; i < n; i++)
        for (int j = i + 1; j < n; j++) {
            if ((i > 1 && s[0] == '0') || (j - i > 1 && s[i] == '0')) continue;
            int k = j; char a[32], b[32], c[32];
            strncpy(a, s, i); a[i] = 0;
            strncpy(b, s + i, j - i); b[j - i] = 0;
            while (k < n) {
                long long x = atoll(a) + atoll(b);
                snprintf(c, sizeof(c), "%lld", x);
                if (!startsWith(s, k, c)) break;
                strcpy(a, b); strcpy(b, c); k += strlen(c);
            }
            if (k == n) return true;
        }
    return false;
}
class Solution:
    def isAdditiveNumber(self, s: str) -> bool:
        def add(a, b):
            res, carry, i, j = [], 0, len(a)-1, len(b)-1
            while i >= 0 or j >= 0 or carry:
                x = int(a[i]) if i >= 0 else 0
                y = int(b[j]) if j >= 0 else 0
                t = x + y + carry
                res.append(str(t % 10)); carry = t // 10
                i -= 1; j -= 1
            return ''.join(reversed(res))
        def ok(i, j):
            a, b = s[:i], s[i:j]
            if (len(a) > 1 and a[0] == '0') or (len(b) > 1 and b[0] == '0'):
                return False
            k = j
            while k < len(s):
                c = add(a, b)
                if not s.startswith(c, k):
                    return False
                a, b, k = b, c, k + len(c)
            return True
        n = len(s)
        for i in range(1, n):
            for j in range(i + 1, n):
                if ok(i, j):
                    return True
        return False
class Solution {
    string add(string a, string b) {
        string res; int i = a.size()-1, j = b.size()-1, carry = 0;
        while (i >= 0 || j >= 0 || carry) {
            int x = i >= 0 ? a[i--]-'0' : 0;
            int y = j >= 0 ? b[j--]-'0' : 0;
            int t = x + y + carry;
            res.push_back('0' + t % 10); carry = t / 10;
        }
        reverse(res.begin(), res.end());
        return res;
    }
    bool ok(const string& s, int i, int j) {
        string a = s.substr(0, i), b = s.substr(i, j - i);
        if ((a.size() > 1 && a[0] == '0') || (b.size() > 1 && b[0] == '0')) return false;
        int k = j;
        while (k < (int)s.size()) {
            string c = add(a, b);
            if (s.compare(k, c.size(), c) != 0) return false;
            a = b; b = c; k += c.size();
        }
        return true;
    }
public:
    bool isAdditiveNumber(string s) {
        int n = s.size();
        for (int i = 1; i < n; i++)
            for (int j = i + 1; j < n; j++)
                if (ok(s, i, j)) return true;
        return false;
    }
};
