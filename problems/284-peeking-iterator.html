Peeking Iterator

O(1) O(1) 61.5% AC
Mô tả
Thiết kế Iterator hỗ trợ peek(): trả phần tử kế tiếp mà không advance; next() trả và tiến; hasNext() kiểm tra còn phần tử.
Ví dụ
Ví dụ 1
InIterator [1,2,3]; peek→1; next→1; next→2; hasNext→true
Out1, 1, 2, true
Ví dụ 2
Inpeek called multiple times before next
Outsame value each peek
Ý tưởng
Cache nextVal và hasNextFlag từ underlying iterator; peek trả cache; next consume cache và prefetch lại.
Mã nguồn
solution.c C99 
typedef struct { int* a; int i, n; int buf; int hasBuf; } PeekIter;
void peekInit(PeekIter* p, int* a, int n) {
    p->a=a; p->i=0; p->n=n; p->hasBuf=0;
    if (n) { p->buf=a[0]; p->i=1; p->hasBuf=1; }
}
int peek(PeekIter* p) { return p->buf; }
int peekNext(PeekIter* p) {
    int v = p->buf;
    if (p->i < p->n) { p->buf = p->a[p->i++]; p->hasBuf = 1; }
    else p->hasBuf = 0;
    return v;
}
int peekHasNext(PeekIter* p) { return p->hasBuf; }
class PeekingIterator:
    def __init__(self, iterator):
        self.it = iterator
        self.buf = None
        self.has_buf = False
        self._fill()
    def _fill(self):
        if not self.has_buf and self.it.hasNext():
            self.buf = self.it.next()
            self.has_buf = True
    def peek(self):
        self._fill()
        return self.buf
    def next(self):
        self._fill()
        val = self.buf
        self.has_buf = False
        self._fill()
        return val
    def hasNext(self):
        return self.has_buf
class PeekingIterator : public Iterator {
    int buf;
    bool hasBuf = false;
    void fill() {
        if (!hasBuf && Iterator::hasNext()) {
            buf = Iterator::next();
            hasBuf = true;
        }
    }
public:
    PeekingIterator(Iterator* it) : Iterator(*it) { fill(); }
    int peek() { fill(); return buf; }
    int next() { fill(); hasBuf = false; int v = buf; fill(); return v; }
    bool hasNext() const { return hasBuf; }
};
