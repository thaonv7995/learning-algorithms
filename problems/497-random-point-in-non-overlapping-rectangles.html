Random Point in Non-overlapping Rectangles

O(log n) pick O(n) 39.8% AC
Mô tả
Rectangles không overlap cho pick point uniform trong union các rectangle.
Ví dụ
Ví dụ 1
Inrects = [[1,1,5,5]], pick returns point inside
Outuniform in union
Ví dụ 2
Inmultiple rects by area weight
Outpick rect then point

class Solution:
    def __init__(self, rects: List[List[int]]):
        self.rects = rects
        self.areas = []
        s = 0
        for x1,y1,x2,y2 in rects:
            s += (x2-x1+1)*(y2-y1+1)
            self.areas.append(s)
        self.total = s
    def pick(self) -> List[int]:
        r = random.randint(1, self.total)
        i = bisect.bisect_left(self.areas, r)
        x1,y1,x2,y2 = self.rects[i]
        return [random.randint(x1,x2), random.randint(y1,y2)]
class Solution {
    vector<vector<int>> rects; vector<long long> areas; long long total;
public:
    Solution(vector<vector<int>>& r):rects(r){
        for(auto& v:rects){ total+=(long long)(v[2]-v[0]+1)*(v[3]-v[1]+1); areas.push_back(total); }
    }
    vector<int> pick() {
        long long r=1+rand()%(total);
        int i=lower_bound(areas.begin(),areas.end(),r)-areas.begin();
        auto& v=rects[i];
        return {v[0]+rand()%(v[2]-v[0]+1), v[1]+rand()%(v[3]-v[1]+1)};
    }
};
