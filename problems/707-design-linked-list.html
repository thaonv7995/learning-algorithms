Design Linked List

O(n) get/addAt/remove O(n) 30.3% AC
Mô tả
Thiết kế linked list hỗ trợ get(index), addAtHead(val), addAtTail(val), addAtIndex(index,val), deleteAtIndex(index). Index hợp lệ 0..size-1 trừ addAtIndex có thể = size.
Ví dụ
Ví dụ 1
In["MyLinkedList","addAtHead","addAtTail","addAtIndex","get","deleteAtIndex","get"], [[],[1],[3],[1,2],[1],[1],[1]]
Out[null,null,null,null,2,null,3]
Ví dụ 2
In["MyLinkedList","addAtHead","addAtHead","get","addAtIndex","get"], [[],[4],[5],[0],[1,2],[1]]
Out[null,null,null,5,null,2]
Ý tưởng
Doubly linked list + dummy head/tail hoặc singly với size. Duyệt tới index cho get/add/delete.
Mã nguồn
solution.c C99 
typedef struct Node { int val; struct Node* next; } Node;
typedef struct { Node dummy; int sz; } MyLinkedList;
static Node* nodeAt(MyLinkedList* t, int i){ Node* c=&t->dummy; for(int j=0;j<=i;j++) c=c->next; return c; }
MyLinkedList* myLinkedListCreate(void){ MyLinkedList* o=calloc(1,sizeof(MyLinkedList)); return o; }
int myLinkedListGet(MyLinkedList* o,int idx){ if(idx<0||idx>=o->sz) return -1; return nodeAt(o,idx)->val; }
void myLinkedListAddAtIndex(MyLinkedList* o,int idx,int val){ if(idx<0||idx>o->sz) return; Node* p=nodeAt(o,idx-1); Node* n=malloc(sizeof(Node)); n->val=val; n->next=p->next; p->next=n; o->sz++; }
void myLinkedListDeleteAtIndex(MyLinkedList* o,int idx){ if(idx<0||idx>=o->sz) return; Node* p=nodeAt(o,idx-1); Node* d=p->next; p->next=d->next; free(d); o->sz--; }
class Node:
    def __init__(self, val=0, nxt=None):
        self.val, self.next = val, nxt
class MyLinkedList:
    def __init__(self):
        self.dummy = Node()
        self.size = 0
    def _node(self, index):
        cur = self.dummy
        for _ in range(index + 1):
            cur = cur.next
        return cur
    def get(self, index: int) -> int:
        if index < 0 or index >= self.size: return -1
        return self._node(index).next.val
    def addAtHead(self, val: int) -> None:
        self.addAtIndex(0, val)
    def addAtTail(self, val: int) -> None:
        self.addAtIndex(self.size, val)
    def addAtIndex(self, index: int, val: int) -> None:
        if index < 0 or index > self.size: return
        prev = self._node(index - 1)
        prev.next = Node(val, prev.next)
        self.size += 1
    def deleteAtIndex(self, index: int) -> None:
        if index < 0 or index >= self.size: return
        prev = self._node(index - 1)
        prev.next = prev.next.next
        self.size -= 1
struct Node { int val; Node* next; Node(int v=0, Node* n=nullptr): val(v), next(n) {} };
class MyLinkedList {
    Node dummy; int sz = 0;
    Node* at(int i) { Node* cur = &dummy; for (int j=0;j<=i;j++) cur = cur->next; return cur; }
public:
    MyLinkedList() {}
    int get(int index) { if (index<0||index>=sz) return -1; return at(index)->val; }
    void addAtHead(int val) { addAtIndex(0, val); }
    void addAtTail(int val) { addAtIndex(sz, val); }
    void addAtIndex(int index, int val) {
        if (index<0||index>sz) return;
        Node* prev = at(index-1);
        prev->next = new Node(val, prev->next);
        sz++;
    }
    void deleteAtIndex(int index) {
        if (index<0||index>=sz) return;
        Node* prev = at(index-1);
        Node* del = prev->next;
        prev->next = del->next;
        delete del; sz--;
    }
};
