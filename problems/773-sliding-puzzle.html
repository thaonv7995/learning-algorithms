Sliding Puzzle

O((mn)! ) worst O(mn) 74.4% AC
Mô tả
Puzzle 2×3: mảng 6 số 0-5, 0 là ô trống. Di chuyển ô kề trống. Trả số bước tối thiểu về [1,2,3,4,5,0]; -1 nếu không.
Ví dụ
Ví dụ 1
Inboard = [[1,2,3],[4,0,5]]
Out1
Ví dụ 2
Inboard = [[1,2,3],[5,4,0]]
Out-1
Ví dụ 3
Inboard = [[4,1,2],[5,0,3]]
Out5
Ý tưởng
BFS trên state string 6 digits; neighbors swap blank; target "123450".

class Solution:
    def slidingPuzzle(self, board: List[List[int]]) -> int:
        start = "".join(str(x) for row in board for x in row)
        if start == "123450": return 0
        q = deque([(start, 0)])
        seen = {start}
        moves = {0:[1,3],1:[0,2,4],2:[1,5],3:[0,4],4:[1,3,5],5:[2,4]}
        while q:
            s, d = q.popleft()
            z = s.index("0")
            for j in moves[z]:
                ns = list(s); ns[z], ns[j] = ns[j], ns[z]
                ns = "".join(ns)
                if ns == "123450": return d + 1
                if ns not in seen:
                    seen.add(ns); q.append((ns, d+1))
        return -1
class Solution {
public:
    int slidingPuzzle(vector<vector<int>>& board) {
        string start;
        for(auto& r:board) for(int x:r) start+=char('0'+x);
        if(start=="123450") return 0;
        vector<int> mv[6]={{1,3},{0,2,4},{1,5},{0,4},{1,3,5},{2,4}};
        queue<pair<string,int>> q; q.push({start,0});
        unordered_set<string> seen={start};
        while(!q.empty()){
            auto [s,d]=q.front(); q.pop();
            int z=s.find('0');
            for(int j:mv[z]){
                string ns=s; swap(ns[z],ns[j]);
                if(ns=="123450") return d+1;
                if(!seen.count(ns)){ seen.insert(ns); q.push({ns,d+1}); }
            }
        }
        return -1;
    }
};
