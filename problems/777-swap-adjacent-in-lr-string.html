Swap Adjacent in LR String

O(n) O(1) 38.1% AC
Mô tả
Chuỗi start và result cùng length, chỉ gồm L,R. Một thao tác: chọn X in {L,R}, đổi mọi ký tự đầu thành X rồi xóa. Có thể biến start thành result không?
Ví dụ
Ví dụ 1
Instart = "RLLLRLLRLRRRLLL", result = "LLLLRRLLRRRRLLLL"
Outtrue
Ví dụ 2
Instart = "LLR", result = "RRL"
Outfalse
Ý tưởng
Loại bỏ cặp L/R đối xứng hai đầu (vì có thể xóa dần). So sánh phần còn lại start và result.
Mã nguồn
solution.c C99 
bool canTransform(char* start, char* result) {
    int n=strlen(start); if(n!= (int)strlen(result)) return false;
    for(int i=0,j=0;i<n;i++){ if(start[i]=='X') continue; while(result[j]=='X') j++;
        if(start[i]!=result[j]) return false;
        if(start[i]=='L' && i<j) return false; if(start[i]=='R' && i>j) return false; j++; } return true;
}
class Solution:
    def canTransform(self, start: str, result: str) -> bool:
        if start.replace("X","") != result.replace("X",""): return False
        def strip(s):
            i, j = 0, len(s)-1
            while i <= j:
                if s[i] in "LR": break
                i += 1
            while j >= i:
                if s[j] in "LR": break
                j -= 1
            return s[i:j+1]
        return strip(start) == strip(result) and start.replace("X","")==result.replace("X","")
class Solution {
public:
    bool canTransform(string start, string result) {
        string a,b;
        for(char c:start) if(c=='L'||c=='R') a.push_back(c);
        for(char c:result) if(c=='L'||c=='R') b.push_back(c);
        if(a!=b) return false;
        int n=start.size();
        for(int i=0,j=0;i<n;i++){
            if(start[i]=='X') continue;
            while(result[j]=='X') j++;
            if(start[i]!=result[j]) return false;
            if(start[i]=='L' && i<j) return false;
            if(start[i]=='R' && i>j) return false;
            j++;
        }
        return true;
    }
};
