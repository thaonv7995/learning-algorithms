Kth Smallest Element in a BST

O(h+k) O(h) 76.8% AC
Mô tả
Cho BST root và k. Trả giá trị node nhỏ thứ k (1-indexed) trong thứ tự inorder tăng dần.
Ví dụ
Ví dụ 1
Inroot = [3,1,4,null,2], k = 1
Out1
Ví dụ 2
Inroot = [5,3,6,2,4,null,null,1], k = 3
Out3
Ý tưởng
Inorder iterative: stack push left chain; pop, k--; k==0 return val; sang right.
Mã nguồn
solution.c C99 
int kthSmallest(struct TreeNode* root, int k) {
    struct TreeNode* stack[1000];
    int top = 0;
    struct TreeNode* cur = root;
    while (cur || top) {
        while (cur) { stack[top++] = cur; cur = cur->left; }
        cur = stack[--top];
        if (--k == 0) return cur->val;
        cur = cur->right;
    }
    return -1;
}
class Solution:
    def kthSmallest(self, root: Optional[TreeNode], k: int) -> int:
        stack = []
        cur = root
        while cur or stack:
            while cur:
                stack.append(cur)
                cur = cur.left
            cur = stack.pop()
            k -= 1
            if k == 0:
                return cur.val
            cur = cur.right
class Solution {
public:
    int kthSmallest(TreeNode* root, int k) {
        stack<TreeNode*> st;
        TreeNode* cur = root;
        while (cur || !st.empty()) {
            while (cur) { st.push(cur); cur = cur->left; }
            cur = st.top(); st.pop();
            if (--k == 0) return cur->val;
            cur = cur->right;
        }
        return -1;
    }
};
