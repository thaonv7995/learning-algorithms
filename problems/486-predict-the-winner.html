Predict the Winner

Mô tả
Hai người lấy từ đầu hoặc cuối nums, cộng score. Người 1 đi trước — có thắng (score ≥ đối thủ) không?

bool predictTheWinner(int* nums, int n) {
    int dp[20][20]={0};
    for(int i=0;i<n;i++) dp[i][i]=nums[i];
    for(int len=2;len<=n;len++)
        for(int l=0;l+len-1<n;l++){
            int r=l+len-1;
            int a=nums[l]-dp[l+1][r], b=nums[r]-dp[l][r-1];
            dp[l][r]=a>b?a:b;
        }
    return dp[0][n-1]>=0;
}
class Solution:
    def PredictTheWinner(self, nums: List[int]) -> bool:
        n = len(nums)
        dp = [[0]*n for _ in range(n)]
        for i in range(n): dp[i][i] = nums[i]
        for ln in range(2, n+1):
            for l in range(n-ln+1):
                r = l+ln-1
                dp[l][r] = max(nums[l]-dp[l+1][r], nums[r]-dp[l][r-1])
        return dp[0][n-1] >= 0
