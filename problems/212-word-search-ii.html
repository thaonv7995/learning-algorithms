Word Search II

O(mn·4^L) O(N) 38.5% AC
Mô tả
Lưới ký tự board và danh sách words. Trả mọi từ trong words xuất hiện trên board (nối liền kề 4 hướng, không tái dùng ô trong cùng một đường).
Ví dụ
Ví dụ 1
Inboard = [["o","a","a","n"],["e","t","a","e"],["i","h","k","r"],["i","f","l","v"]], words = ["oath","pea","eat","rain"]
Out["eat","oath"]
Ví dụ 2
Inboard = [["a","b"],["c","d"]], words = ["abcb"]
Out[]
Ý tưởng
Build Trie từ words; DFS từ mỗi ô board theo Trie, đánh dấu visited, khi node.isEnd thì thu từ và cắt nhánh (xóa end) để tránh trùng.

typedef struct WNode { struct WNode* ch[26]; char* word; } WNode;
static void wFree(WNode* n) { if (!n) return; for (int i = 0; i < 26; i++) wFree(n->ch[i]); free(n->word); free(n); }
static void wInsert(WNode* r, char* w) {
    WNode* cur = r;
    for (; *w; w++) { int k = *w - 'a'; if (!cur->ch[k]) cur->ch[k] = calloc(1, sizeof(WNode)); cur = cur->ch[k]; }
    cur->word = strdup(w);
}
static void dfs212(char** b, int m, int n, int i, int j, WNode* node, char** ans, int* asz, int* acap) {
    if (i < 0 || j < 0 || i >= m || j >= n) return;
    char c = b[i][j];
    if (c == '#') return;
    int k = c - 'a';
    if (!node->ch[k]) return;
    WNode* nxt = node->ch[k];
    if (nxt->word) { if (*asz >= *acap) { *acap *= 2; ans[0] = realloc(ans[0], *acap * sizeof(char*)); } ((char**)ans)[(*asz)++] = strdup(nxt->word); free(nxt->word); nxt->word = NULL; }
    b[i][j] = '#';
    int d[4][2] = {{1,0},{-1,0},{0,1},{0,-1}};
    for (int t = 0; t < 4; t++) dfs212(b, m, n, i + d[t][0], j + d[t][1], nxt, ans, asz, acap);
    b[i][j] = c;
}
char** findWords(char** board, int m, char** words, int wn, int* rs) {
    WNode root = {0};
    for (int i = 0; i < wn; i++) wInsert(&root, words[i]);
    char** ans = malloc(16 * sizeof(char*)); int asz = 0, acap = 16;
    for (int i = 0; i < m; i++) for (int j = 0; j < (int)strlen(board[i]); j++) dfs212(board, m, (int)strlen(board[0]), i, j, &root, &ans, &asz, &acap);
    wFree(&root); *rs = asz; return ans;
}
class Solution:
    def findWords(self, board: List[List[str]], words: List[str]) -> List[str]:
        root = {}
        for w in words:
            node = root
            for c in w:
                node = node.setdefault(c, {})
            node['#'] = w
        m, n = len(board), len(board[0])
        ans = []
        def dfs(i, j, node):
            c = board[i][j]
            if c not in node:
                return
            nxt = node[c]
            if '#' in nxt:
                ans.append(nxt['#'])
                del nxt['#']
            board[i][j] = '#'
            for di, dj in ((1,0),(-1,0),(0,1),(0,-1)):
                ni, nj = i + di, j + dj
                if 0 <= ni < m and 0 <= nj < n and board[ni][nj] != '#':
                    dfs(ni, nj, nxt)
            board[i][j] = c
            if not nxt:
                del node[c]
        for i in range(m):
            for j in range(n):
                dfs(i, j, root)
        return ans
class Solution {
    struct Node { unordered_map<char, Node*> ch; string* word=nullptr; };
    vector<string> ans;
    void dfs(vector<vector<char>>& b, int i, int j, Node* node) {
        char c = b[i][j];
        if (!node->ch.count(c)) return;
        Node* nxt = node->ch[c];
        if (nxt->word) { ans.push_back(*nxt->word); delete nxt->word; nxt->word = nullptr; }
        b[i][j] = '#';
        int d[4][2] = {{1,0},{-1,0},{0,1},{0,-1}};
        for (auto& x : d) {
            int ni = i + x[0], nj = j + x[1];
            if (ni>=0 && nj>=0 && ni<(int)b.size() && nj<(int)b[0].size() && b[ni][nj]!='#')
                dfs(b, ni, nj, nxt);
        }
        b[i][j] = c;
        if (nxt->ch.empty() && !nxt->word) { delete nxt; node->ch.erase(c); }
    }
public:
    vector<string> findWords(vector<vector<char>>& board, vector<string>& words) {
        Node root;
        for (string& w : words) {
            Node* cur = &root;
            for (char c : w) {
                if (!cur->ch.count(c)) cur->ch[c] = new Node();
                cur = cur->ch[c];
            }
            cur->word = new string(w);
        }
        for (int i = 0; i < (int)board.size(); i++)
            for (int j = 0; j < (int)board[0].size(); j++)
                dfs(board, i, j, &root);
        return ans;
    }
};
