Find Mode in Binary Search Tree

O(n) O(n) 58.9% AC
Mô tả
Cho cây BST. Trả về tất cả giá trị mode (xuất hiện nhiều nhất). Nếu nhiều mode thì trả tất cả, thứ tự tăng dần. Duyệt inorder vì BST cho thứ tự đã sort.
Ví dụ
Ví dụ 1
Inroot = [1,null,2,2]
Out[2]
Ví dụ 2
Inroot = [0]
Out[0]
Ví dụ 3
Inroot = [1,1,2,2,2]
Out[2]
Ý tưởng
Inorder duyệt tăng dần: đếm tần suất liên tiếp, cập nhật maxCount và danh sách modes khi gặp tần suất mới hoặc bằng max.
Mã nguồn
solution.c C99 
/* Inorder đếm run — xem C++ */
class Solution:
    def findMode(self, root: Optional[TreeNode]) -> List[int]:
        self.prev = None
        self.maxCnt = self.cnt = 0
        self.modes = []
        def dfs(n):
            if not n: return
            dfs(n.left)
            if self.prev is None:
                self.cnt = 1
            elif n.val == self.prev:
                self.cnt += 1
            else:
                self.cnt = 1
            if self.cnt > self.maxCnt:
                self.maxCnt = self.cnt
                self.modes = [n.val]
            elif self.cnt == self.maxCnt:
                self.modes.append(n.val)
            self.prev = n.val
            dfs(n.right)
        dfs(root)
        return self.modes
class Solution {
    TreeNode* prev = nullptr;
    int maxCnt = 0, cnt = 0;
    vector<int> modes;
    void inorder(TreeNode* n) {
        if (!n) return;
        inorder(n->left);
        cnt = (!prev || n->val == prev->val) ? cnt + 1 : 1;
        if (cnt > maxCnt) { maxCnt = cnt; modes = {n->val}; }
        else if (cnt == maxCnt) modes.push_back(n->val);
        prev = n;
        inorder(n->right);
    }
public:
    vector<int> findMode(TreeNode* root) {
        inorder(root);
        return modes;
    }
};
