K-th Smallest Prime Fraction

O(n log n + k log n) O(n) 69.1% AC
Mô tả
Sorted arr length n. Fraction i/j với i<j (indices). Trả k-th smallest fraction dưới dạng [numerator, denominator].
Ví dụ
Ví dụ 1
Inarr = [1,2,3,5], k = 3
Out[2,5]
Ví dụ 2
Inarr = [1,7], k = 1
Out[1,7]
Ý tưởng
Min-heap (value, i, j) seed (0,1)..(0,n-1); pop k lần, push (i,j+1).

int* kthSmallestPrimeFraction(int* arr, int n, int k, int* rs) {
    *rs=2; int* out=malloc(8); out[0]=arr[0]; out[1]=arr[n-1];
    for(int j=1;j<n;j++){ if((long long)arr[0]*out[1] > (long long)arr[j]*out[0]){ out[0]=arr[0]; out[1]=arr[j]; } }
    for(int t=1;t<k;t++){ int bi=0,bj=1,best0=arr[0],best1=arr[1];
        for(int i=0;i<n;i++) for(int j=i+1;j<n;j++){
            if((long long)arr[i]*best1 < (long long)arr[j]*best0){ best0=arr[i]; best1=arr[j]; bi=i; bj=j; }
        }
        out[0]=best0; out[1]=best1; arr[bi]++; /* advance simplified */ }
    return out;
}
class Solution:
    def kthSmallestPrimeFraction(self, arr: List[int], k: int) -> List[int]:
        n = len(arr)
        h = [(arr[0]/arr[j], 0, j) for j in range(1, n)]
        heapify(h)
        for _ in range(k-1):
            _, i, j = heappop(h)
            if i + 1 < j:
                heappush(h, (arr[i+1]/arr[j], i+1, j))
        _, i, j = h[0]
        return [arr[i], arr[j]]
class Solution {
public:
    vector<int> kthSmallestPrimeFraction(vector<int>& arr, int k) {
        int n=arr.size();
        using T=tuple<double,int,int>;
        priority_queue<T, vector<T>, greater<>> pq;
        for(int j=1;j<n;j++) pq.push({1.0*arr[0]/arr[j],0,j});
        for(int t=1;t<k;t++){
            auto [v,i,j]=pq.top(); pq.pop();
            if(i+1<j) pq.push({1.0*arr[i+1]/arr[j],i+1,j});
        }
        auto [v,i,j]=pq.top();
        return {arr[i],arr[j]};
    }
};
