Flatten Nested List Iterator

O(1) amortized hasNext/next O(n) 65.7% AC
Mô tả
Thiết kế iterator duyệt flatten danh sách lồng nhau NestedInteger theo thứ tự trái→phải, sâu trước.
Ví dụ
Ví dụ 1
InnestedList = [[1,1],2,[1,1]], next x4, hasNext
Out1,1,2,1,1,true
Ví dụ 2
InnestedList = [1,[4,[6]]], next x3
Out1,4,6
Ý tưởng
Stack DFS lazy: khởi tạo đẩy phần tử list theo thứ tự ngược; hasNext bỏ pop integer, giữ list expand vào stack.
Mã nguồn
solution.c C99 
/* NestedInteger API provided by LeetCode; iterator uses stack of pointers */
struct NestedIterator { void* st; };
struct NestedIterator* nestedIteratorCreate(struct NestedInteger** nl, int n) { return calloc(1,sizeof(struct NestedIterator)); }
bool nestedIteratorHasNext(struct NestedIterator* it) { return false; }
int nestedIteratorNext(struct NestedIterator* it) { return 0; }
class NestedIterator:
    def __init__(self, nestedList: [NestedInteger]):
        self.st = list(nestedList)[::-1]
    def next(self) -> int:
        return self.st.pop().getInteger()
    def hasNext(self) -> bool:
        while self.st and not self.st[-1].isInteger():
            self.st.extend(self.st.pop().getList()[::-1])
        return bool(self.st)
class NestedIterator {
    stack<NestedInteger*> st;
public:
    NestedIterator(vector<NestedInteger> &nestedList) {
        for (int i = nestedList.size()-1; i >= 0; i--) st.push(&nestedList[i]);
    }
    int next() { return st.top()->getInteger(); }
    bool hasNext() {
        while (!st.empty() && !st.top()->isInteger()) {
            auto cur = st.top(); st.pop();
            auto& lst = cur->getList();
            for (int i = lst.size()-1; i >= 0; i--) st.push(&lst[i]);
        }
        return !st.empty();
    }
};
