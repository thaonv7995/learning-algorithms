Palindrome Partitioning

O(n · 2^n) O(n) 74.1% AC
Mô tả
Cho chuỗi s. Liệt kê mọi cách phân tách s thành các substring sao cho mỗi phần là palindrome.
Ví dụ
Ví dụ 1
Ins = "aab"
Out[["a","a","b"],["aa","b"]]
Ví dụ 2
Ins = "a"
Out[["a"]]
Ý tưởng
Backtracking từ index start: thử mọi end ≥ start; nếu s[start:end+1] palindrome thì thêm vào path, đệ quy start=end+1. start==n → lưu path.
Mã nguồn
solution.c C99 
/* Backtracking + check palindrome — xem C++ */
class Solution:
    def partition(self, s: str) -> List[List[str]]:
        res = []
        def pal(l, r):
            while l < r:
                if s[l] != s[r]: return False
                l += 1; r -= 1
            return True
        def bt(start, path):
            if start == len(s):
                res.append(path[:])
                return
            for end in range(start, len(s)):
                if pal(start, end):
                    path.append(s[start:end+1])
                    bt(end + 1, path)
                    path.pop()
        bt(0, [])
        return res
class Solution {
    bool pal(const string& s, int l, int r) {
        while (l < r) if (s[l++] != s[r--]) return false;
        return true;
    }
    vector<vector<string>> res;
    string s;
    void bt(int start, vector<string>& path) {
        if (start == (int)s.size()) { res.push_back(path); return; }
        for (int end = start; end < (int)s.size(); end++) {
            if (!pal(s, start, end)) continue;
            path.push_back(s.substr(start, end - start + 1));
            bt(end + 1, path);
            path.pop_back();
        }
    }
public:
    vector<vector<string>> partition(string str) {
        s = str;
        vector<string> path;
        bt(0, path);
        return res;
    }
};
