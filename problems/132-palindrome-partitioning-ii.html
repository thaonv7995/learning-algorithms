Palindrome Partitioning II

Mô tả
Cho chuỗi s. Tìm số lần cắt tối thiểu để mọi phần sau khi cắt là palindrome (tương đương số đoạn palindrome ít nhất trừ 1).

Ý tưởng
DP pal[i][j]: s[i..j] palindrome. dp[j] = min cắt cho prefix s[0..j]. Nếu pal[0][j] thì dp[j]=0; else dp[j]=min(dp[i-1]+1) với pal[i][j].

/* DP pal table + min cuts — xem C++ */
class Solution:
    def minCut(self, s: str) -> int:
        n = len(s)
        pal = [[False]*n for _ in range(n)]
        for i in range(n-1, -1, -1):
            for j in range(i, n):
                pal[i][j] = s[i] == s[j] and (j - i < 2 or pal[i+1][j-1])
        dp = [0]*n
        for j in range(n):
            if pal[0][j]:
                dp[j] = 0
            else:
                dp[j] = j
                for i in range(1, j+1):
                    if pal[i][j]:
                        dp[j] = min(dp[j], dp[i-1] + 1)
        return dp[n-1]
class Solution {
public:
    int minCut(string s) {
        int n = s.size();
        vector<vector<bool>> pal(n, vector<bool>(n));
        for (int i = n - 1; i >= 0; i--)
            for (int j = i; j < n; j++)
                pal[i][j] = s[i] == s[j] && (j - i < 2 || pal[i+1][j-1]);
        vector<int> dp(n);
        for (int j = 0; j < n; j++) {
            if (pal[0][j]) dp[j] = 0;
            else {
                dp[j] = j;
                for (int i = 1; i <= j; i++)
                    if (pal[i][j]) dp[j] = min(dp[j], dp[i-1] + 1);
            }
        }
        return dp[n-1];
    }
};
