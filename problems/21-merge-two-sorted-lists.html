Merge Two Sorted Lists

O(N + M) O(1) 68.3% AC
Mô tả
Trộn hai danh sách liên kết đơn đã sắp xếp tăng dần thành một danh sách mới cũng được sắp xếp. Không cần cấp phát nút mới — chỉ nối lại con trỏ.
Ví dụ
Ví dụ 1
Inlist1 = [1,2,4], list2 = [1,3,4]
Out[1,1,2,3,4,4]
Ví dụ 2
Inlist1 = [], list2 = []
Out[]
Ví dụ 3
Inlist1 = [], list2 = [0]
Out[0]
Ý tưởng
Dummy node + con trỏ tail. So sánh head list1/list2, nối nút nhỏ hơn vào tail, tịnh tiến con trỏ tương ứng. Khi một list hết, nối phần còn lại.
Mã nguồn
solution.c C99 
struct ListNode* mergeTwoLists(struct ListNode* l1, struct ListNode* l2) {
    struct ListNode dummy = {0, NULL}, *tail = &dummy;
    while (l1 && l2) {
        if (l1->val <= l2->val) { tail->next = l1; l1 = l1->next; }
        else { tail->next = l2; l2 = l2->next; }
        tail = tail->next;
    }
    tail->next = l1 ? l1 : l2;
    return dummy.next;
}
class Solution:
    def mergeTwoLists(self, list1, list2):
        dummy = ListNode(); tail = dummy
        while list1 and list2:
            if list1.val <= list2.val:
                tail.next = list1; list1 = list1.next
            else:
                tail.next = list2; list2 = list2.next
            tail = tail.next
        tail.next = list1 or list2
        return dummy.next
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {
        ListNode dummy(0); ListNode* tail = &dummy;
        while (l1 && l2) {
            if (l1->val <= l2->val) { tail->next = l1; l1 = l1->next; }
            else { tail->next = l2; l2 = l2->next; }
            tail = tail->next;
        }
        tail->next = l1 ? l1 : l2;
        return dummy.next;
    }
};
