/* Binary digit DP — xem C++ */
class Solution:
    def findIntegers(self, num: int) -> int:
        bits = bin(num)[2:]
        from functools import lru_cache
        @lru_cache(None)
        def dp(i, prev1, tight):
            if i == len(bits):
                return 1
            limit = int(bits[i]) if tight else 1
            ans = 0
            for d in range(0, limit+1):
                if prev1 and d == 1:
                    continue
                ans += dp(i+1, d==1, tight and d==limit)
            return ans
        return dp(0, False, True)
class Solution {
    string bits;
    int dp[35][2][2];
    int dfs(int i, int prev1, int tight) {
        if (i == (int)bits.size()) return 1;
        if (dp[i][prev1][tight] != -1) return dp[i][prev1][tight];
        int limit = tight ? bits[i]-'0' : 1;
        int ans = 0;
        for (int d=0; d<=limit; d++) {
            if (prev1 && d==1) continue;
            ans += dfs(i+1, d==1, tight && d==limit);
        }
        return dp[i][prev1][tight]=ans;
    }
public:
    int findIntegers(int num) {
        bits = bitset<32>(num).to_string();
        bits = bits.substr(bits.find('1'));
        memset(dp, -1, sizeof dp);
        return dfs(0, 0, 1);
    }
};
