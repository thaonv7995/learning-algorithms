Next Greater Element III

O(log n) O(1) 35.3% AC
Mô tả
Cho số nguyên n. Trả next greater với cùng các chữ số (next permutation); không có trả -1.
Ví dụ
Ví dụ 1
Inn = 12
Out21
Ví dụ 2
Inn = 21
Out-1
Ví dụ 3
Inn = 534976
Out536479
Ý tưởng
Chuyển sang mảng chữ số; next_permutation: tìm i giảm dần, swap với số lớn hơn nhỏ nhất bên phải, reverse suffix.
Mã nguồn
solution.c C99 
/* Next permutation digits — xem C++ */
class Solution:
    def nextGreaterElement(self, n: int) -> int:
        a = list(str(n))
        i = len(a) - 2
        while i >= 0 and a[i] >= a[i+1]:
            i -= 1
        if i < 0:
            return -1
        j = len(a) - 1
        while a[j] <= a[i]:
            j -= 1
        a[i], a[j] = a[j], a[i]
        a[i+1:] = reversed(a[i+1:])
        ans = int("".join(a))
        return ans if ans < 2**31 else -1
class Solution {
public:
    int nextGreaterElement(int n) {
        string a = to_string(n);
        int i = (int)a.size()-2;
        while (i>=0 && a[i]>=a[i+1]) i--;
        if (i<0) return -1;
        int j = (int)a.size()-1;
        while (a[j]<=a[i]) j--;
        swap(a[i], a[j]);
        reverse(a.begin()+i+1, a.end());
        long long ans = stoll(a);
        return ans >= (1LL<<31) ? -1 : (int)ans;
    }
};
