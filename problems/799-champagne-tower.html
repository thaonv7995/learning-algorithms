Champagne Tower

O(pour^2) O(pour^2) 64.1% AC
Mô tả
Tháp champagne pour cốc glasses rows. Pour p cups. Mỗi cốc capacity 1; tràn sang hai cốc dưới. Trả tỉ lệ cốc row query_row cột query_glass chứa champagne.
Ví dụ
Ví dụ 1
Inpoured = 1, query_row = 1, query_glass = 1
Out0.0
Ví dụ 2
Inpoured = 2, query_row = 1, query_glass = 1
Out0.5
Ví dụ 3
Inpoured = 100000009, query_row = 10, query_glass = 7
Out0.9375
Ý tưởng
DP[row][col] = lượng champagne tại cốc; flow xuống max(0, amt-1)/2 mỗi nhánh.
Mã nguồn

double champagneTower(int poured, int query_row, int query_glass) {
    double t[101][101]={0}; t[0][0]=poured;
    for(int r=0;r<=query_row;r++) for(int c=0;c<=r;c++){
        double flow=t[r][c]>1?(t[r][c]-1)/2:0;
        t[r+1][c]+=flow; t[r+1][c+1]+=flow;
    }
    double v=t[query_row][query_glass]; return v>1?1:v;
}
class Solution:
    def champagneTower(self, poured: int, query_row: int, query_glass: int) -> float:
        tower = [[0.0]*(query_row+2) for _ in range(query_row+2)]
        tower[0][0] = poured
        for r in range(query_row+1):
            for c in range(r+1):
                flow = (tower[r][c]-1)/2 if tower[r][c]>1 else 0
                if flow>0:
                    tower[r+1][c]+=flow; tower[r+1][c+1]+=flow
        return min(1.0, tower[query_row][query_glass])
