UTF-8 Validation

O(n) O(1) 46.3% AC
Mô tả
Cho mảng byte data. Kiểm tra có phải chuỗi UTF-8 hợp lệ không theo quy tắc encoding 1–4 byte.
Ví dụ
Ví dụ 1
Indata = [197,130,1]
Outtrue
Ví dụ 2
Indata = [235,140,4]
Outfalse
Ý tưởng
Duyệt i: đếm số byte follow từ prefix (1110xxxx → 3 byte total); mỗi follow phải 10xxxxxx; validate range 1–4 byte.

bool validUtf8(int* data, int n) {
    int i = 0;
    while (i < n) {
        int need = 0;
        if ((data[i] >> 7) == 0) i++;
        else if ((data[i] >> 5) == 0b110) { need = 1; i++; }
        else if ((data[i] >> 4) == 0b1110) { need = 2; i++; }
        else if ((data[i] >> 3) == 0b11110) { need = 3; i++; }
        else return false;
        for (int k = 0; k < need; k++, i++)
            if (i >= n || (data[i] >> 6) != 0b10) return false;
    }
    return true;
}
class Solution:
    def validUtf8(self, data: List[int]) -> bool:
        i, n = 0, len(data)
        while i < n:
            b = data[i]
            if b >> 7 == 0: i += 1
            elif b >> 5 == 0b110: need, i = 1, i+1
            elif b >> 4 == 0b1110: need, i = 2, i+1
            elif b >> 3 == 0b11110: need, i = 3, i+1
            else: return False
            for _ in range(need):
                if i >= n or data[i] >> 6 != 0b10: return False
                i += 1
        return True
