Fraction Addition and Subtraction

O(n) O(1) 66.4% AC
Mô tả
Cho biểu thức chuỗi gồm phân số dạng ±a/b cộng trừ nhau. Trả kết quả phân số tối giản dạng "p/q".
Ví dụ
Ví dụ 1
Inexpression = "-1/2+1/2"
Out"0/1"
Ví dụ 2
Inexpression = "-1/2+1/2+1/3"
Out"1/3"
Ví dụ 3
Inexpression = "1/3-1/2"
Out"-1/6"
Ý tưởng
Parse từng phân số; cộng dồn num/den lcm; cuối gcd rút gọn, chuẩn hóa dấu mẫu dương.
Mã nguồn
solution.c C99 
/* Parse fractions lcm gcd — xem C++ */
class Solution:
    def fractionAddition(self, expression: str) -> str:
        import re
        from math import gcd
        nums = list(map(int, re.findall(r'[+-]?\d+', expression)))
        num = den = 0
        for i in range(0, len(nums), 2):
            a, b = nums[i], nums[i+1]
            num = num * b + a * den
            den = den * b if den else b
            g = gcd(abs(num), abs(den))
            num //= g
            den //= g
        if den < 0:
            num, den = -num, -den
        return f"{num}/{den}"
class Solution {
    long long gcdll(long long a, long long b){ return b?gcdll(b,a%b):abs(a); }
public:
    string fractionAddition(string expression) {
        long long num=0, den=1;
        int i=0, n=expression.size();
        while (i<n) {
            long long sign=1; if (expression[i]=='+'||expression[i]=='-'){ sign=expression[i]=='-'?-1:1; i++; }
            long long a=0; while (i<n && isdigit(expression[i])) a=a*10+expression[i++]-'0'; a*=sign;
            i++; long long b=0; while (i<n && isdigit(expression[i])) b=b*10+expression[i++]-'0';
            num = num*b + a*den; den = den*b;
            long long g = gcdll(num, den); num/=g; den/=g;
        }
        if (den<0){ num=-num; den=-den; }
        return to_string(num)+"/"+to_string(den);
    }
};
