Reverse Pairs

O(n log n) O(n) 34.3% AC
Mô tả
Cho mảng nums. Đếm reverse pairs: cặp chỉ số (i,j) với i<j và nums[i] > 2*nums[j]. Trả tổng số cặp thỏa điều kiện.
Ví dụ
Ví dụ 1
Innums = [1,3,2,3,1]
Out2
Ví dụ 2
Innums = [2,4,3,5,1]
Out3
Ý tưởng
Merge sort: khi merge, với mỗi left[i] đếm right[j] sao nums[i]>2*nums[j].

class Solution:
    def reversePairs(self, nums: List[int]) -> int:
        ans = 0
        def sort(arr):
            nonlocal ans
            if len(arr) <= 1: return arr
            mid = len(arr)//2
            L, R = sort(arr[:mid]), sort(arr[mid:])
            i = 0
            for x in L:
                while i < len(R) and x > 2*R[i]: i += 1
                ans += i
            return merge(L, R)
        def merge(a,b):
            out=[]; i=j=0
            while i<len(a) and j<len(b):
                if a[i]<=b[j]: out.append(a[i]); i+=1
                else: out.append(b[j]); j+=1
            return out+a[i:]+b[j:]
        sort(nums); return ans
class Solution {
public:
    int reversePairs(vector<int>& nums) {
        return mergeSort(nums, 0, nums.size()-1);
    }
    long long mergeSort(vector<int>& a, int l, int r){
        if(l>=r) return 0;
        int m=l+(r-l)/2; long long ans=mergeSort(a,l,m)+mergeSort(a,m+1,r);
        int j=m+1;
        for(int i=l;i<=m;i++){
            while(j<=r && (long long)a[i]>2LL*a[j]) j++;
            ans += j-(m+1);
        }
        inplace_merge(a.begin()+l, a.begin()+m+1, a.begin()+r+1);
        return ans;
    }
};
