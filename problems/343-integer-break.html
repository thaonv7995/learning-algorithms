Integer Break

O(n) O(n) 62.3% AC
Mô tả
Chia số nguyên n≥2 thành tổng ít nhất hai số nguyên dương. Trả tích các số hạng lớn nhất.
Ví dụ
Ví dụ 1
Inn = 2
Out1
Ví dụ 2
Inn = 10
Out36
Ý tưởng
DP: dp[i]=max product khi tách i; dp[i]=max(dp[j]*(i-j), j*(i-j)) với j=1..i-1. Hoặc greedy: ưu tiên nhân 3.
Mã nguồn
solution.c C99 
int integerBreak(int n) {
    int* dp = calloc(n + 1, sizeof(int)); dp[1] = 1;
    for (int i = 2; i <= n; i++)
        for (int j = 1; j < i; j++) {
            int v1 = dp[j] * (i - j), v2 = j * (i - j);
            int v = v1 > v2 ? v1 : v2;
            if (v > dp[i]) dp[i] = v;
        }
    int ans = dp[n]; free(dp); return ans;
}
class Solution:
    def integerBreak(self, n: int) -> int:
        dp = [0] * (n + 1)
        dp[1] = 1
        for i in range(2, n + 1):
            for j in range(1, i):
                dp[i] = max(dp[i], dp[j] * (i - j), j * (i - j))
        return dp[n]
class Solution {
public:
    int integerBreak(int n) {
        vector<int> dp(n + 1, 0);
        dp[1] = 1;
        for (int i = 2; i <= n; i++)
            for (int j = 1; j < i; j++)
                dp[i] = max(dp[i], max(dp[j] * (i - j), j * (i - j)));
        return dp[n];
    }
};
