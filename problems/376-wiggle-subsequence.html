Wiggle Subsequence

O(n) O(1) 49.4% AC
Mô tả
Subsequence wiggle: nums[0]<nums[1]>nums[2]<nums[3]... hoặc ngược lại. Tìm độ dài wiggle subsequence dài nhất.
Ví dụ
Ví dụ 1
Innums = [1,7,4,9,2,5]
Out6
Ví dụ 2
Innums = [1,17,5,10,13,15,10,5,16,8]
Out7
Ý tưởng
Greedy: up/down length; nếu nums[i]>nums[i-1] up++; elif nums[i]<nums[i-1] down++; return max(up,down).
Mã nguồn
solution.c C99 
int wiggleMaxLength(int* nums, int n) {
    if (n < 2) return n;
    int up = 1, down = 1;
    for (int i = 1; i < n; i++) {
        if (nums[i] > nums[i-1]) up = down + 1;
        else if (nums[i] < nums[i-1]) down = up + 1;
    }
    return up > down ? up : down;
}
class Solution:
    def wiggleMaxLength(self, nums: List[int]) -> int:
        if len(nums) < 2: return len(nums)
        up = down = 1
        for i in range(1, len(nums)):
            if nums[i] > nums[i-1]: up = down + 1
            elif nums[i] < nums[i-1]: down = up + 1
        return max(up, down)
class Solution {
public:
    int wiggleMaxLength(vector<int>& nums) {
        if (nums.size() < 2) return nums.size();
        int up = 1, down = 1;
        for (int i = 1; i < (int)nums.size(); i++) {
            if (nums[i] > nums[i-1]) up = down + 1;
            else if (nums[i] < nums[i-1]) down = up + 1;
        }
        return max(up, down);
    }
};
