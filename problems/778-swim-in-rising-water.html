Swim in Rising Water

O(n^2 log n) O(n^2) 67.8% AC
Mô tả
Lưới n×n elevation. Thời gian t tại ô (r,c) khi mực nước t. Từ (0,0) đi 4 hướng tới (n-1,n-1), path max elevation <= t. Trả t tối thiểu.
Ví dụ
Ví dụ 1
Ingrid = [[0,2],[1,3]]
Out3
Ví dụ 2
Ingrid = [[0,1,2,3,4],[24,23,22,21,5],[12,13,14,15,16],[11,17,18,19,20],[10,9,8,7,6]]
Out16
Ý tưởng
Binary search t: DFS/BFS kiểm tra có path từ (0,0) tới (n-1,n-1) chỉ qua ô có giá trị <= t.

static int ok778(int** g,int n,int t){
    if(g[0][0]>t) return 0;
    int vis[50][50]={0}; int q[2500][2], h=0,tail=1; q[0][0]=0; q[0][1]=0; vis[0][0]=1;
    int d[4][2]={{1,0},{-1,0},{0,1},{0,-1}};
    while(h<tail){ int r=q[h][0], c=q[h][1]; h++; if(r==n-1&&c==n-1) return 1;
        for(int i=0;i<4;i++){ int nr=r+d[i][0], nc=c+d[i][1];
            if(nr>=0&&nc>=0&&nr<n&&nc<n&&!vis[nr][nc]&&g[nr][nc]<=t){ vis[nr][nc]=1; q[tail][0]=nr; q[tail++][1]=nc; } } }
    return 0;
}
int swimInWater(int** grid, int n, int* cs) {
    int lo=grid[0][0], hi=n*n-1;
    while(lo<hi){ int mid=(lo+hi)/2; if(ok778(grid,n,mid)) hi=mid; else lo=mid+1; }
    return lo;
}
class Solution:
    def swimInWater(self, grid: List[List[int]]) -> int:
        n = len(grid)
        def ok(t):
            if grid[0][0] > t: return False
            vis = [[False]*n for _ in range(n)]
            st = [(0,0)]; vis[0][0]=True
            while st:
                r,c = st.pop()
                if r==n-1 and c==n-1: return True
                for dr,dc in ((1,0),(-1,0),(0,1),(0,-1)):
                    nr, nc = r+dr, c+dc
                    if 0<=nr<n and 0<=nc<n and not vis[nr][nc] and grid[nr][nc]<=t:
                        vis[nr][nc]=True; st.append((nr,nc))
            return False
        lo, hi = grid[0][0], n*n-1
        while lo < hi:
            mid = (lo+hi)//2
            if ok(mid): hi = mid
            else: lo = mid+1
        return lo
class Solution {
public:
    int swimInWater(vector<vector<int>>& g) {
        int n=g.size();
        auto ok=[&](int t){
            if(g[0][0]>t) return false;
            vector<vector<char>> vis(n, vector<char>(n));
            queue<pair<int,int>> q; q.push({0,0}); vis[0][0]=1;
            int d[4][2]={{1,0},{-1,0},{0,1},{0,-1}};
            while(!q.empty()){
                auto [r,c]=q.front(); q.pop();
                if(r==n-1&&c==n-1) return true;
                for(auto& dd:d){ int nr=r+dd[0], nc=c+dd[1];
                    if(nr>=0&&nr<n&&nc>=0&&nc<n&&!vis[nr][nc]&&g[nr][nc]<=t){ vis[nr][nc]=1; q.push({nr,nc}); }
                }
            }
            return false;
        };
        int lo=g[0][0], hi=n*n-1;
        while(lo<hi){ int mid=lo+(hi-lo)/2; if(ok(mid)) hi=mid; else lo=mid+1; }
        return lo;
    }
};
