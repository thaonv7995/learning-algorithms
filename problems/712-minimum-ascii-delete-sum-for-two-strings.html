Minimum ASCII Delete Sum for Two Strings

O(mn) O(mn) 71% AC
Mô tả
Cho hai chuỗi s1, s2. Xóa một số ký tự (không đảo thứ tự) sao hai chuỗi bằng nhau. Chi phí xóa ký tự = mã ASCII của ký tự đó. Trả tổng chi phí xóa tối thiểu.
Ví dụ
Ví dụ 1
Ins1 = "sea", s2 = "eat"
Out231
Ví dụ 2
Ins1 = "delete", s2 = "leet"
Out403
Ví dụ 3
Ins1 = "a", s2 = "b"
Out195
Ý tưởng
LCS DP: dp[i][j] = min cost để khớp s1[:i] và s2[:j]. Nếu s1[i-1]==s2[j-1] giữ; else min(delete s1[i-1], delete s2[j-1]).

int minimumDeleteSum(char* s1, char* s2) {
    int m=strlen(s1), n=strlen(s2);
    int dp[1001][1001]={0};
    for(int i=1;i<=m;i++) dp[i][0]=dp[i-1][0]+s1[i-1];
    for(int j=1;j<=n;j++) dp[0][j]=dp[0][j-1]+s2[j-1];
    for(int i=1;i<=m;i++) for(int j=1;j<=n;j++){
        if(s1[i-1]==s2[j-1]) dp[i][j]=dp[i-1][j-1];
        else dp[i][j]=dp[i-1][j]+s1[i-1] < dp[i][j-1]+s2[j-1] ? dp[i-1][j]+s1[i-1] : dp[i][j-1]+s2[j-1];
    }
    return dp[m][n];
}
class Solution:
    def minimumDeleteSum(self, s1: str, s2: str) -> int:
        m, n = len(s1), len(s2)
        dp = [[0]*(n+1) for _ in range(m+1)]
        for i in range(1, m+1): dp[i][0] = dp[i-1][0] + ord(s1[i-1])
        for j in range(1, n+1): dp[0][j] = dp[0][j-1] + ord(s2[j-1])
        for i in range(1, m+1):
            for j in range(1, n+1):
                if s1[i-1] == s2[j-1]: dp[i][j] = dp[i-1][j-1]
                else: dp[i][j] = min(dp[i-1][j]+ord(s1[i-1]), dp[i][j-1]+ord(s2[j-1]))
        return dp[m][n]
class Solution {
public:
    int minimumDeleteSum(string s1, string s2) {
        int m=s1.size(), n=s2.size();
        vector<vector<int>> dp(m+1, vector<int>(n+1));
        for(int i=1;i<=m;i++) dp[i][0]=dp[i-1][0]+s1[i-1];
        for(int j=1;j<=n;j++) dp[0][j]=dp[0][j-1]+s2[j-1];
        for(int i=1;i<=m;i++) for(int j=1;j<=n;j++){
            if(s1[i-1]==s2[j-1]) dp[i][j]=dp[i-1][j-1];
            else dp[i][j]=min(dp[i-1][j]+s1[i-1], dp[i][j-1]+s2[j-1]);
        }
        return dp[m][n];
    }
};
