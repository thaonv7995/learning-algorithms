Minimum Window Substring

O(|s|+|t|) O(|Σ|) 47.6% AC
Mô tả
Cho chuỗi s và t. Tìm substring nhỏ nhất của s chứa đủ mọi ký tự của t (kể cả trùng lặp). Trả về "" nếu không có.
Ví dụ
Ví dụ 1
Ins = "ADOBECODEBANC", t = "ABC"
Out"BANC"
Ví dụ 2
Ins = "a", t = "a"
Out"a"
Ý tưởng
Sliding window: mở r bằng cách thêm ký tự, đếm need/have; khi have đủ thì thu l từ trái, cập nhật best, giảm have nếu mất ký tự cần.
Mã nguồn
solution.c C99 
/* minWindow — sliding window + count; xem C++ */
class Solution:
    def minWindow(self, s: str, t: str) -> str:
        from collections import Counter
        need = Counter(t); have = 0; req = len(need)
        l = 0; best = (10**9, 0, 0)
        window = {}
        for r, c in enumerate(s):
            window[c] = window.get(c, 0) + 1
            if c in need and window[c] == need[c]: have += 1
            while have == req:
                if r - l + 1 < best[0]: best = (r - l + 1, l, r + 1)
                lc = s[l]
                window[lc] -= 1
                if lc in need and window[lc] < need[lc]: have -= 1
                l += 1
        return s[best[1]:best[2]] if best[0] < 10**9 else ""
class Solution {
public:
    string minWindow(string s, string t) {
        vector<int> need(128,0), win(128,0);
        int req=0;
        for(char c:t) if(!need[c]++) req++;
        int have=0, l=0, bestLen=1e9, bestL=0;
        for(int r=0;r<(int)s.size();r++){
            if(!win[s[r]]++ && need[s[r]]) have++;
            while(have==req){
                if(r-l+1<bestLen){bestLen=r-l+1;bestL=l;}
                if(!--win[s[l]] && need[s[l]]) have--;
                l++;
            }
        }
        return bestLen==1e9?"":s.substr(bestL,bestLen);
    }
};
