Couples Holding Hands

O(n) O(n) 59.4% AC
Mô tả
2n người, couple i ở ghế 2i và 2i+1. Hoán vị row[i]=người tại ghế i. Tối thiểu số lần swap adjacent để mọi cặp ngồi cạnh nhau?
Ví dụ
Ví dụ 1
Inrow = [0,2,1,3]
Out1
Ví dụ 2
Inrow = [3,2,0,1]
Out0
Ý tưởng
Map partner của mỗi người. Duyệt cặp ghế (0,1),(2,3)... nếu không phải couple thì swap count++ và union/find cycle của hoán vị.

class Solution:
    def minSwapsCouples(self, row: List[int]) -> int:
        n=len(row)//2
        pos=[0]*len(row)
        for i,x in enumerate(row): pos[x]=i
        partner=lambda x: x+1 if x%2==0 else x-1
        seen=[False]*n; ans=0
        for i in range(n):
            if seen[i]: continue
            cur=i; cyc=0
            while not seen[cur]:
                seen[cur]=True; cyc+=1
                p=partner(row[2*cur])//2
                cur=p
            ans+=cyc-1
        return ans
class Solution {
public:
    int minSwapsCouples(vector<int>& row) {
        int n=row.size()/2;
        vector<int> pos(row.size());
        for(int i=0;i<(int)row.size();i++) pos[row[i]]=i;
        auto partner=[&](int x){ return x^1; };
        vector<char> seen(n); int ans=0;
        for(int i=0;i<n;i++){
            if(seen[i]) continue;
            int cur=i, cyc=0;
            while(!seen[cur]){
                seen[cur]=1; cyc++;
                int p=partner(row[2*cur])/2;
                cur=p;
            }
            ans+=cyc-1;
        }
        return ans;
    }
};
