Smallest Range Covering Elements from K Lists

O(n log k) O(n) 70.1% AC
Mô tả
Cho k list đã sort. Tìm đoạn [min,max] ngắn nhất chứa ít nhất một phần tử từ mỗi list.
Ví dụ
Ví dụ 1
Innums = [[4,10,15,24,26],[0,9,12,20],[5,18,22,30]]
Out[20,24]
Ví dụ 2
Innums = [[1,2,3],[1,2,3],[1,2,3]]
Out[1,1]
Ý tưởng
Min-heap (val,listIdx,elemIdx); push first of each list; pop min, push next; track min range.

int* smallestRange(int** nums,int k,int* cs,int* rs) {
    int* heapv=malloc(k*4),*heapi=malloc(k*4),*heapj=malloc(k*4),hs=0;
    int mx=-1000000000, bestL=0,bestR=1000000000;
    #define HPUSH(v,i,j){heapv[hs]=v;heapi[hs]=i;heapj[hs]=j;hs++;for(int t=hs-1;t>0;t--)if(heapv[t]<heapv[t-1]){int tv=heapv[t];heapv[t]=heapv[t-1];heapv[t-1]=tv;int ti=heapi[t];heapi[t]=heapi[t-1];heapi[t-1]=ti;ti=heapj[t];heapj[t]=heapj[t-1];heapj[t-1]=ti;}else break;}
    #define HPOP() (hs--)
    for(int i=0;i<k;i++){ HPUSH(nums[i][0],i,0); if(nums[i][0]>mx) mx=nums[i][0]; }
    while(hs){
        int v=heapv[0],i=heapi[0],j=heapj[0];
        if(mx-v<bestR-bestL){ bestL=v; bestR=mx; }
        if(j+1>=cs[i]) break;
        int nv=nums[i][j+1];
        heapv[0]=nv; heapi[0]=i; heapj[0]=j+1;
        if(nv>mx) mx=nv;
        for(int t=0;t<hs-1;t++) if(heapv[t+1]<heapv[t]){int tv=heapv[t];heapv[t]=heapv[t+1];heapv[t+1]=tv;int ti=heapi[t];heapi[t]=heapi[t+1];heapi[t+1]=ti;ti=heapj[t];heapj[t]=heapj[t+1];heapj[t+1]=ti;}
    }
    int* ans=malloc(2*4); ans[0]=bestL; ans[1]=bestR; *rs=2; free(heapv); free(heapi); free(heapj); return ans;
}
class Solution:
    def smallestRange(self, nums: List[List[int]]) -> List[int]:
        import heapq
        h=[(nums[i][0],i,0) for i in range(len(nums))]; heapq.heapify(h)
        curmax=max(x[0] for x in h); best=[-10**9,10**9]
        while h:
            v,i,j=heapq.heappop(h); curmax=max(curmax,v)
            if curmax-v<best[1]-best[0]: best=[v,curmax]
            if j+1==len(nums[i]): break
            nv=nums[i][j+1]; heapq.heappush(h,(nv,i,j+1)); curmax=max(curmax,nv)
        return best
class Solution {
public:
    vector<int> smallestRange(vector<vector<int>>& nums) {
        using T=array<int,3>;
        priority_queue<T,vector<T>,greater<T>> pq;
        int mx=INT_MIN;
        for(int i=0;i<(int)nums.size();i++){ pq.push({nums[i][0],i,0}); mx=max(mx,nums[i][0]); }
        vector<int> ans={-1000000,1000000};
        while(!pq.empty()){
            auto [v,i,j]=pq.top(); pq.pop();
            if(mx-v<ans[1]-ans[0]) ans={v,mx};
            if(j+1==(int)nums[i].size()) break;
            int nv=nums[i][j+1]; pq.push({nv,i,j+1}); mx=max(mx,nv);
        }
        return ans;
    }
};
