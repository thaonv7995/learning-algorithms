Letter Case Permutation

O(2^n) O(n) 75.9% AC
Mô tả
Cho chuỗi S gồm chữ và số. Trả mọi hoán vị từ S bằng cách chọn hoa/thường cho chữ cái (số giữ nguyên).
Ví dụ
Ví dụ 1
InS = "a1b2"
Out["a1b2","a1B2","A1b2","A1B2"]
Ví dụ 2
InS = "3z4"
Out["3z4","3Z4"]
Ví dụ 3
InS = "12345"
Out["12345"]
Ý tưởng
Backtrack index i: nếu chữ cái thử lower và upper; số append cố định.

class Solution:
    def letterCasePermutation(self, s: str) -> List[str]:
        ans = []
        def dfs(i, path):
            if i == len(s):
                ans.append("".join(path)); return
            if s[i].isalpha():
                path.append(s[i].lower()); dfs(i+1, path); path.pop()
                path.append(s[i].upper()); dfs(i+1, path); path.pop()
            else:
                path.append(s[i]); dfs(i+1, path); path.pop()
        dfs(0, [])
        return ans
class Solution {
public:
    vector<string> letterCasePermutation(string s) {
        vector<string> ans;
        string cur;
        function<void(int)> dfs = [&](int i){
            if(i==(int)s.size()){ ans.push_back(cur); return; }
            if(isalpha(s[i])){
                cur.push_back(tolower(s[i])); dfs(i+1); cur.pop_back();
                cur.push_back(toupper(s[i])); dfs(i+1); cur.pop_back();
            } else { cur.push_back(s[i]); dfs(i+1); cur.pop_back(); }
        };
        dfs(0); return ans;
    }
};
