Arithmetic Slices II - Subsequence

O(n^2) O(n^2) 55.1% AC
Mô tả
Đếm số arithmetic subsequence (≥3 phần tử, không nhất thiệp liên tiếp) trong nums.
Ví dụ
Ví dụ 1
Innums = [2,4,6,8,10]
Out7
Ví dụ 2
Innums = [7,7,7,7,7]
Out6
Ý tưởng
dp[i][diff] = số seq kết thúc tại i với diff; với j<i, diff=nums[i]-nums[j], cộng dp[j][diff]+1 nếu đã có seq length≥2.

class Solution:
    def numberOfArithmeticSlices(self, nums: List[int]) -> int:
        n = len(nums)
        dp = [defaultdict(int) for _ in range(n)]
        ans = 0
        for i in range(n):
            for j in range(i):
                d = nums[i] - nums[j]
                cnt = dp[j][d]
                if cnt: ans += cnt
                dp[i][d] += cnt + 1
        return ans
class Solution {
public:
    int numberOfArithmeticSlices(vector<int>& nums) {
        int n=nums.size(), ans=0;
        vector<unordered_map<long long,int>> dp(n);
        for(int i=0;i<n;i++)
            for(int j=0;j<i;j++){
                long long d=(long long)nums[i]-nums[j];
                int cnt=dp[j][d];
                if(cnt) ans+=cnt;
                dp[i][d]+=cnt+1;
            }
        return ans;
    }
};
