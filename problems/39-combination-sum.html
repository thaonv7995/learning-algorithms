Combination Sum

O(2^T) O(T) 76.5% AC
Mô tả
Tìm mọi bộ candidates (có thể lặp) có tổng = target.
Ví dụ
Ví dụ 1
Incandidates = [2,3,6,7], target = 7
Out[[2,2,3],[7]]
Ý tưởng
Sort candidates. DFS(start, path, remain): nếu remain=0 push path; nếu remain<0 return; từ start chọn candidates[i] lặp lại (i, không i+1).

static void cs39(int* c, int n, int i, int* path, int plen, int rem,
                   int*** out, int** cols, int* rc, int* cap) {
    if (rem == 0) {
        if (*rc >= *cap) { *cap *= 2; *out = (int**)realloc(*out, *cap * sizeof(int*)); *cols = (int*)realloc(*cols, *cap * sizeof(int)); }
        (*out)[*rc] = (int*)malloc(plen * sizeof(int));
        memcpy((*out)[*rc], path, plen * sizeof(int));
        (*cols)[*rc] = plen;
        (*rc)++;
        return;
    }
    if (rem < 0) return;
    for (int j = i; j < n; j++) {
        path[plen] = c[j];
        cs39(c, n, j, path, plen + 1, rem - c[j], out, cols, rc, cap);
    }
}
int cmp39(const void* a, const void* b) { return *(int*)a - *(int*)b; }
int** combinationSum(int* candidates, int candidatesSize, int target,
                     int* returnSize, int** returnColumnSizes) {
    qsort(candidates, candidatesSize, sizeof(int), cmp39);
    int cap = 32, rc = 0, path[32];
    int** out = (int**)malloc(cap * sizeof(int*));
    *returnColumnSizes = (int*)malloc(cap * sizeof(int));
    cs39(candidates, candidatesSize, 0, path, 0, target, &out, returnColumnSizes, &rc, &cap);
    *returnSize = rc;
    return out;
}
class Solution:
    def combinationSum(self, c: List[int], t: int) -> List[List[int]]:
        c.sort(); res=[]
        def dfs(i, path, r):
            if r==0: res.append(path[:]); return
            if r<0: return
            for j in range(i, len(c)):
                path.append(c[j]); dfs(j, path, r-c[j]); path.pop()
        dfs(0,[],t); return res
class Solution {
    vector<vector<int>> res;
    void dfs(vector<int>& c, int i, vector<int>& p, int r) {
        if (!r) { res.push_back(p); return; }
        if (r < 0) return;
        for (int j = i; j < (int)c.size(); j++) {
            p.push_back(c[j]); dfs(c, j, p, r - c[j]); p.pop_back();
        }
    }
public:
    vector<vector<int>> combinationSum(vector<int>& c, int t) {
        sort(c.begin(), c.end()); vector<int> p; dfs(c, 0, p, t); return res;
    }
};
