Remove Invalid Parentheses

O(2^n) O(n) 49.9% AC
Mô tả
Cho chuỗi s chứa '(' , ')' và chữ cái. Xóa số ký tự ngoặc tối thiểu để chuỗi còn lại hợp lệ. Trả về mọi chuỗi hợp lệ có thể, không trùng, thứ tự bất kỳ.
Ví dụ
Ví dụ 1
Ins = "()())()"
Out["(())()","()()()"]
Ví dụ 2
Ins = "(a)())()"
Out["(a())()","(a)()()"]
Ví dụ 3
Ins = ")("
Out[]
Ý tưởng
BFS theo số ngoặc đã xóa: từ s, sinh mọi chuỗi xóa một '('. Nếu chuỗi hợp lệ thì thu; nếu chưa gặp hợp lệ thì tiếp tục tầng tiếp theo. Dùng set tránh trùng.

static int valid(const char* t) {
    int bal = 0;
    for (int i = 0; t[i]; i++) {
        if (t[i] == '(') bal++;
        else if (t[i] == ')') { if (--bal < 0) return 0; }
    }
    return bal == 0;
}

class Solution:
    def removeInvalidParentheses(self, s: str) -> List[str]:
        def valid(t):
            bal = 0
            for c in t:
                if c == '(':
                    bal += 1
                elif c == ')':
                    bal -= 1
                    if bal < 0:
                        return False
            return bal == 0
        level, seen = {s}, {s}
        while level:
            good = [x for x in level if valid(x)]
            if good:
                return good
            nxt = set()
            for cur in level:
                for i, c in enumerate(cur):
                    if c not in '()':
                        continue
                    t = cur[:i] + cur[i + 1:]
                    if t not in seen:
                        seen.add(t)
                        nxt.add(t)
            level = nxt
        return []
class Solution {
    bool valid(const string& t) {
        int bal = 0;
        for (char c : t) {
            if (c == '(') bal++;
            else if (c == ')') { if (--bal < 0) return false; }
        }
        return bal == 0;
    }
public:
    vector<string> removeInvalidParentheses(string s) {
        unordered_set<string> level{s}, seen{s};
        while (!level.empty()) {
            vector<string> good;
            for (auto& x : level) if (valid(x)) good.push_back(x);
            if (!good.empty()) return good;
            unordered_set<string> nxt;
            for (auto& cur : level)
                for (int i = 0; i < (int)cur.size(); i++)
                    if (cur[i] == '(' || cur[i] == ')') {
                        string t = cur.substr(0, i) + cur.substr(i + 1);
                        if (!seen.count(t)) { seen.insert(t); nxt.insert(t); }
                    }
            level = move(nxt);
        }
        return {};
    }
};
