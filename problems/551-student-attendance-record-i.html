Student Attendance Record I

O(n) O(1) 50.2% AC
Mô tả
Kiểm tra chuỗi attendance hợp lệ: tối đa một 'A' (absent) và không có 3 'L' liên tiếp.
Ví dụ
Ví dụ 1
Ins = "PPALLP"
Outtrue
Ví dụ 2
Ins = "PPALLL"
Outfalse
Ví dụ 3
Ins = "P"
Outtrue
Ý tưởng
Duyệt một lần: đếm A, streak L reset khi gặp P/A.
Mã nguồn
solution.c C99 
/* Scan A count L streak — xem C++ */
class Solution:
    def checkRecord(self, s: str) -> bool:
        a = l = 0
        for c in s:
            if c == "A":
                a += 1
                if a > 1: return False
                l = 0
            elif c == "L":
                l += 1
                if l >= 3: return False
            else:
                l = 0
        return True
class Solution {
public:
    bool checkRecord(string s) {
        int a = 0, l = 0;
        for (char c : s) {
            if (c == 'A') { if (++a > 1) return false; l = 0; }
            else if (c == 'L') { if (++l >= 3) return false; }
            else l = 0;
        }
        return true;
    }
};
