Flatten a Multilevel Doubly Linked List

O(n) O(1) 62.9% AC
Mô tả
Flatten danh sách kép đa cấp: mỗi node có next, prev, child (head list con). Trả list phẳng theo thứ tự pre-order, dùng next/prev nối liền.
Ví dụ
Ví dụ 1
Inhead = [1,2,3,4,5,6,null,null,null,7,8,9,10,null,null,11,12]
Out[1,2,3,7,8,11,12,9,10,4,5,6]
Ví dụ 2
Inhead = [1,2,null,3]
Out[1,3,2]
Ý tưởng
DFS: flatten(child) trước; nối child list vào giữa node và node.next; prev pointer cập nhật.
Mã nguồn
solution.c C99 
struct Node* flatten(struct Node* head) { return head; /* doubly linked — see C++ */ }
class Solution:
    def flatten(self, head: Node) -> Node:
        if not head: return head
        cur = head
        while cur:
            if cur.child:
                nxt = cur.next
                child = self.flatten(cur.child)
                cur.next = child
                child.prev = cur
                tail = child
                while tail.next: tail = tail.next
                tail.next = nxt
                if nxt: nxt.prev = tail
                cur.child = None
            cur = cur.next
        return head
class Solution {
public:
    Node* flatten(Node* head) {
        if (!head) return head;
        Node* cur = head;
        while (cur) {
            if (cur->child) {
                Node* nxt = cur->next;
                Node* ch = flatten(cur->child);
                cur->next = ch; ch->prev = cur;
                Node* tail = ch;
                while (tail->next) tail = tail->next;
                tail->next = nxt; if (nxt) nxt->prev = tail;
                cur->child = nullptr;
            }
            cur = cur->next;
        }
        return head;
    }
};
