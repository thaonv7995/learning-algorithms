All O`one Data Structure

O(1) avg O(n) 44.3% AC
Mô tả
Thiết kế cấu trúc AllOne: inc(key), dec(key), getMaxKey(), getMinKey() — mọi key count>0; getMax/Min trả key bất kỳ có count max/min.
Ví dụ
Ví dụ 1
Ininc("hello"), inc("hello"), getMaxKey()
Out"hello"
Ví dụ 2
Ininc("leet"), inc("code"), inc("leet"), dec("leet"), getMaxKey()
Out"code" or "leet"
Ý tưởng
Hash map key→node; doubly linked list các bucket count; mỗi bucket set keys cùng count.
Mã nguồn
solution.c C99 
typedef struct AllOne AllOne;
AllOne* allOneCreate(void) { return calloc(1,sizeof(AllOne)); }
void allOneInc(AllOne* o, char* key) {}
void allOneDec(AllOne* o, char* key) {}
char* allOneGetMaxKey(AllOne* o) { return strdup(""); }
char* allOneGetMinKey(AllOne* o) { return strdup(""); }
class AllOne:
    def __init__(self):
        self.count = {}
        self.keys = defaultdict(OrderedDict)
        self.minc = self.maxc = 0
    def inc(self, key: str) -> None:
        c = self.count.get(key, 0)
        if c: del self.keys[c][key]
        c += 1
        self.count[key] = c
        self.keys[c][key] = None
        if c < self.minc or self.minc == 0: self.minc = c
        self.maxc = max(self.maxc, c)
    def dec(self, key: str) -> None:
        c = self.count[key]
        del self.keys[c][key]
        if not self.keys[c]: del self.keys[c]
        if c == 1: del self.count[key]
        else:
            self.count[key] = c - 1
            self.keys[c-1][key] = None
        if not self.count: self.minc = self.maxc = 0
        elif c == self.minc and not self.keys.get(c): self.minc = min(self.keys)
    def getMaxKey(self) -> str:
        return next(iter(self.keys[self.maxc])) if self.count else ""
    def getMinKey(self) -> str:
        return next(iter(self.keys[self.minc])) if self.count else ""
class AllOne {
    unordered_map<string,int> cnt;
    unordered_map<int,list<string>> buckets;
    unordered_map<string,list<string>::iterator> it;
    int mn=0, mx=0;
public:
    void inc(string key) {
        int c = cnt[key]++;
        if(c){ auto i=it[key]; buckets[c].erase(i); if(buckets[c].empty()) buckets.erase(c); }
        buckets[c+1].push_front(key); it[key]=buckets[c+1].begin();
        if(!mn||c+1<mn) mn=c+1; mx=max(mx,c+1);
    }
    void dec(string key) {
        int c = cnt[key]--;
        auto i=it[key]; buckets[c].erase(i);
        if(buckets[c].empty()) buckets.erase(c);
        if(c==1){ cnt.erase(key); it.erase(key); }
        else{ buckets[c-1].push_front(key); it[key]=buckets[c-1].begin(); }
        if(cnt.empty()) mn=mx=0;
        else if(c==mn&&!buckets.count(mn)) mn=buckets.begin()->first;
    }
    string getMaxKey(){ return cnt.empty()?"":buckets[mx].front(); }
    string getMinKey(){ return cnt.empty()?"":buckets[mn].front(); }
};
