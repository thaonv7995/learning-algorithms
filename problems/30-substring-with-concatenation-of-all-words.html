Substring with Concatenation of All Words

O(N·M·L) O(M·L) 34.4% AC
Mô tả
Tìm mọi index bắt đầu trong s sao cho substring gồm đủ tất cả words (cùng độ dài) nối liên tiếp.
Ví dụ
Ví dụ 1
Ins = "barfoothefoobarman", words = ["foo","bar"]
Out[0,9]
Ví dụ 2
Ins = "wordgoodgoodgoodbestword", words = ["word","good","best","word"]
Out[]
Ý tưởng
wordLen = len(words[0]). Với offset 0..wordLen-1, sliding window trên s đếm tần suất từ khớp map words. Khi window có đủ words.length từ đúng freq → ghi index.

/* Sliding window theo offset mod wordLen — freq map cho từng word */
int* findSubstring(char* s, char** words, int wordsSize, int* returnSize) {
    *returnSize = 0;
    if (!wordsSize || !words[0][0]) return NULL;
    int wl = (int)strlen(words[0]), sl = (int)strlen(s);
    int need = wordsSize * wl;
    if (sl < need) return NULL;
    /* Brute: thử mọi start i sao đủ len */
    int cap = 64, cnt = 0;
    int* res = (int*)malloc(cap * sizeof(int));
    for (int i = 0; i + need <= sl; i++) {
        int ok = 1;
        for (int w = 0; w < wordsSize && ok; w++) {
            if (strncmp(s + i + w * wl, words[w], wl) != 0) ok = 0;
        }
        if (ok) {
            if (cnt == cap) { cap *= 2; res = (int*)realloc(res, cap * sizeof(int)); }
            res[cnt++] = i;
        }
    }
    *returnSize = cnt;
    return res;
}
class Solution:
    def findSubstring(self, s: str, words: List[str]) -> List[int]:
        if not words: return []
        wl, n, m = len(words[0]), len(s), len(words)
        from collections import Counter
        need = Counter(words)
        res = []
        for off in range(wl):
            left = off; cnt = Counter(); matched = 0
            for right in range(off, n, wl):
                w = s[right:right+wl]
                if w not in need:
                    cnt.clear(); matched = 0; left = right + wl; continue
                cnt[w] += 1
                if cnt[w] == need[w]: matched += 1
                elif cnt[w] == need[w] + 1: matched -= 1
                while matched == m:
                    if left - off == 0 or (left-off) % wl == 0:
                        if (right-left)//wl + 1 == m: res.append(left)
                    cnt[s[left:left+wl]] -= 1
                    if cnt[s[left:left+wl]] == need[s[left:left+wl]]: matched -= 1
                    left += wl
        return sorted(set(res)) if res else []
class Solution {
public:
    vector<int> findSubstring(string s, vector<string>& words) {
        vector<int> res;
        if (words.empty()) return res;
        int wl = words[0].size(), m = words.size(), n = s.size();
        unordered_map<string,int> need;
        for (auto& w : words) need[w]++;
        for (int off = 0; off < wl; off++) {
            unordered_map<string,int> cnt;
            int left = off, matched = 0;
            for (int right = off; right + wl <= n; right += wl) {
                string w = s.substr(right, wl);
                if (!need.count(w)) { cnt.clear(); matched = 0; left = right + wl; continue; }
                cnt[w]++;
                if (cnt[w] == need[w]) matched++;
                else if (cnt[w] == need[w] + 1) matched--;
                while (matched == m) {
                    if ((right - left) / wl + 1 == m) res.push_back(left);
                    string lw = s.substr(left, wl);
                    cnt[lw]--;
                    if (cnt[lw] == need[lw]) matched--;
                    left += wl;
                }
            }
        }
        return res;
    }
};
