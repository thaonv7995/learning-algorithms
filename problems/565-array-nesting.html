Array Nesting

O(n) O(1) 56.6% AC
Mô tả
Mảng là permutation 0..n-1. nums[i] trỏ tới nums[nums[i]]. Tìm độ dài nesting dài nhất (chu trình).
Ví dụ
Ví dụ 1
Innums = [5,4,0,3,1,6,2]
Out4
Ví dụ 2
Innums = [0,1,2]
Out1
Ý tưởng
Duyệt i chưa visit; đi theo nums[i] đếm bước trong cycle; đánh dấu visited.
Mã nguồn
solution.c C99 
/* Cycle detection walk — xem C++ */
class Solution:
    def arrayNesting(self, nums: List[int]) -> int:
        seen = [False]*len(nums)
        ans = 0
        for i in range(len(nums)):
            if seen[i]: continue
            cnt = 0
            j = i
            while not seen[j]:
                seen[j] = True
                j = nums[j]
                cnt += 1
            ans = max(ans, cnt)
        return ans
class Solution {
public:
    int arrayNesting(vector<int>& nums) {
        vector<bool> seen(nums.size());
        int ans = 0;
        for (int i = 0; i < (int)nums.size(); i++) {
            if (seen[i]) continue;
            int cnt = 0, j = i;
            while (!seen[j]) { seen[j]=true; j=nums[j]; cnt++; }
            ans = max(ans, cnt);
        }
        return ans;
    }
};
