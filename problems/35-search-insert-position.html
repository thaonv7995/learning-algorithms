Search Insert Position

O(log N) O(1) 51.3% AC
Mô tả
Tìm index chèn target vào mảng đã sort để vẫn sort (hoặc index nếu đã có).

Ý tưởng
Binary search: nếu nums[mid] < target thì l = mid+1, ngược lại r = mid-1. Trả l.
Mã nguồn

int searchInsert(int* nums, int n, int target) {
    int l=0,r=n-1;
    while(l<=r){int m=l+(r-l)/2;
        if(nums[m]<target)l=m+1; else r=m-1;}
    return l;
}
class Solution:
    def searchInsert(self, nums: List[int], target: int) -> int:
        l, r = 0, len(nums)-1
        while l <= r:
            m = (l+r)//2
            if nums[m] < target: l = m+1
            else: r = m-1
        return l
