Matchsticks to Square

O(n * sum^3) O(sum) 41.9% AC
Mô tả
matchsticks[i] là độ dài que. Có ghép thành hình vuông (4 cạnh bằng nhau, dùng hết que) không?
Ví dụ
Ví dụ 1
Inmatchsticks = [1,1,2,2,2]
Outtrue
Ví dụ 2
Inmatchsticks = [3,3,3,3,4]
Outfalse
Ý tưởng
Tổng phải chia 4; side=sum/4. Backtracking + DP bitmask phân que vào 4 nhóm sum side.

class Solution:
    def makesquare(self, matchsticks: List[int]) -> bool:
        s = sum(matchsticks)
        if s % 4: return False
        side = s // 4
        matchsticks.sort(reverse=True)
        sides = [0]*4
        def dfs(i):
            if i == len(matchsticks): return True
            seen = set()
            for j in range(4):
                if sides[j] in seen: continue
                if sides[j] + matchsticks[i] > side: continue
                seen.add(sides[j])
                sides[j] += matchsticks[i]
                if dfs(i+1): return True
                sides[j] -= matchsticks[i]
            return False
        return dfs(0)
class Solution {
public:
    bool makesquare(vector<int>& m) {
        int s=accumulate(m.begin(),m.end(),0); if(s%4) return false;
        int side=s/4; sort(m.begin(),m.end(), greater<int>());
        array<int,4> sides{};
        function<bool(int)> dfs=[&](int i)->bool{
            if(i==(int)m.size()) return true;
            unordered_set<int> seen;
            for(int j=0;j<4;j++){
                if(seen.count(sides[j])) continue;
                if(sides[j]+m[i]>side) continue;
                seen.insert(sides[j]);
                sides[j]+=m[i];
                if(dfs(i+1)) return true;
                sides[j]-=m[i];
            }
            return false;
        };
        return dfs(0);
    }
};
