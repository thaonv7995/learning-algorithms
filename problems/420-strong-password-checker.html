Strong Password Checker

O(n) O(1) 15.8% AC
Mô tả
Chuỗi password cần: ≥6 ký tự, có chữ thường, hoa, digit. Trả số thao tác tối thiểu (insert/delete/replace) để hợp lệ. Delete chỉ khi len>20 và chưa đủ 3 loại ký tự.
Ví dụ
Ví dụ 1
Inpassword = "aA1"
Out2
Ví dụ 2
Inpassword = "1337"
Out1
Ví dụ 3
Inpassword = "a"
Out5
Ý tưởng
Đếm missing types; đếm runs ≥3 cùng ký tự. Nếu len<6: insert max(6-len, missing). len 6-20: replace = max(missing, fix runs). len>20: delete dư trước, rồi replace.

int strongPasswordChecker(char* s) {
    int n = (int)strlen(s), lower = 0, upper = 0, digit = 0;
    for (int i = 0; s[i]; i++) {
        if (s[i] >= 'a' && s[i] <= 'z') lower = 1;
        else if (s[i] >= 'A' && s[i] <= 'Z') upper = 1;
        else if (s[i] >= '0' && s[i] <= '9') digit = 1;
    }
    int missing = 3 - lower - upper - digit, rep = 0, one = 0, two = 0;
    for (int i = 0; i < n; ) {
        int j = i;
        while (j < n && s[j] == s[i]) j++;
        int L = j - i;
        if (L >= 3) {
            rep += L / 3;
            if (L % 3 == 0) one++;
            else if (L % 3 == 1) two++;
        }
        i = j;
    }
    if (n < 6) return (6 - n > missing) ? 6 - n : missing;
    if (n <= 20) return (rep > missing) ? rep : missing;
    int del = n - 20;
    rep -= (del < one) ? del : one;
    del -= one; if (del < 0) del = 0;
    int t2 = two * 2;
    rep -= (del < t2) ? del / 2 : two;
    del -= t2; if (del < 0) del = 0;
    rep -= del / 3;
    if (rep < missing) rep = missing;
    return (n - 20) + rep;
}
class Solution:
    def strongPasswordChecker(self, password: str) -> int:
        n = len(password)
        lower = upper = digit = 0
        for c in password:
            if c.islower(): lower = 1
            elif c.isupper(): upper = 1
            elif c.isdigit(): digit = 1
        missing = 3 - lower - upper - digit
        i = 0
        replace = one = two = 0
        while i < n:
            j = i
            while j < n and password[j] == password[i]: j += 1
            L = j - i
            if L >= 3:
                replace += L // 3
                if L % 3 == 0: one += 1
                elif L % 3 == 1: two += 1
            i = j
        if n < 6: return max(6 - n, missing)
        if n <= 20: return max(missing, replace)
        del_cnt = n - 20
        replace -= min(del_cnt, one)
        replace -= min(max(del_cnt - one, 0), two * 2) // 2
        replace -= max(del_cnt - one - two * 2, 0) // 3
        return del_cnt + max(missing, replace)

        int n=s.size(), lower=0, upper=0, digit=0;

            int L=j-i; if(L>=3){ rep+=L/3; if(L%3==0) one++; else if(L%3==1) two++; }
            i=j;
        }
        if(n<6) return max(6-n, missing);
        if(n<=20) return max(missing, rep);
        int del=n-20; rep-=min(del,one); del=max(del-one,0);
        rep-=min(del,two*2)/2; del=max(del-two*2,0); rep-=del/3;
