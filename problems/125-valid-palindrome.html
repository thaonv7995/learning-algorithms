Valid Palindrome

O(n) O(1) 53.3% AC
Mô tả
Cho chuỗi s. Kiểm tra sau khi bỏ ký tự không phải chữ/số và không phân biệt hoa thường, chuỗi có phải palindrome hay không.
Ví dụ
Ví dụ 1
Ins = "A man, a plan, a canal: Panama"
Outtrue
Ví dụ 2
Ins = "race a car"
Outfalse
Ý tưởng
Two pointers l, r: bỏ qua ký tự không alphanumeric, so sánh lower(s[l]) và lower(s[r]). Lệch → false; gặp nhau → true.
Mã nguồn
solution.c C99 
/* Two pointers skip non-alnum — xem C++ */
class Solution:
    def isPalindrome(self, s: str) -> bool:
        l, r = 0, len(s) - 1
        while l < r:
            while l < r and not s[l].isalnum(): l += 1
            while l < r and not s[r].isalnum(): r -= 1
            if s[l].lower() != s[r].lower(): return False
            l += 1; r -= 1
        return True
class Solution {
    bool alnum(char c) {
        return isalnum(static_cast<unsigned char>(c));
    }
    char lower(char c) {
        return tolower(static_cast<unsigned char>(c));
    }
public:
    bool isPalindrome(string s) {
        int l = 0, r = (int)s.size() - 1;
        while (l < r) {
            while (l < r && !alnum(s[l])) l++;
            while (l < r && !alnum(s[r])) r--;
            if (lower(s[l]) != lower(s[r])) return false;
            l++; r--;
        }
        return true;
    }
};
