Product of Array Except Self

O(n) O(1) 68.9% AC
Mô tả
Cho mảng nums. Trả mảng answer cùng độ dài: answer[i] = tích mọi phần tử nums trừ nums[i]. Không dùng phép chia; output array không tính extra space.
Ví dụ
Ví dụ 1
Innums = [1,2,3,4]
Out[24,12,8,6]
Ví dụ 2
Innums = [-1,1,0,-3,3]
Out[0,0,9,0,0]
Ý tưởng
Pass trái: answer[i]=prefix product; pass phải nhân suffix vào answer. Hoặc dùng answer làm prefix rồi biến suffix.
Mã nguồn
solution.c C99 
int* productExceptSelf(int* nums, int n, int* rs) {
    int* ans = malloc(n * sizeof(int));
    int prefix = 1;
    for (int i = 0; i < n; i++) { ans[i] = prefix; prefix *= nums[i]; }
    int suffix = 1;
    for (int i = n - 1; i >= 0; i--) { ans[i] *= suffix; suffix *= nums[i]; }
    *rs = n;
    return ans;
}
class Solution:
    def productExceptSelf(self, nums: List[int]) -> List[int]:
        n = len(nums)
        ans = [1] * n
        prefix = 1
        for i in range(n):
            ans[i] = prefix
            prefix *= nums[i]
        suffix = 1
        for i in range(n - 1, -1, -1):
            ans[i] *= suffix
            suffix *= nums[i]
        return ans
class Solution {
public:
    vector<int> productExceptSelf(vector<int>& nums) {
        int n = nums.size();
        vector<int> ans(n, 1);
        int prefix = 1;
        for (int i = 0; i < n; i++) { ans[i] = prefix; prefix *= nums[i]; }
        int suffix = 1;
        for (int i = n - 1; i >= 0; i--) { ans[i] *= suffix; suffix *= nums[i]; }
        return ans;
    }
};
