Design HashSet

O(1) avg O(n) 68% AC
Mô tả
Thiết kế HashSet: add(key), remove(key), contains(key). Không dùng built-in hash set của ngôn ngữ.
Ví dụ
Ví dụ 1
In["MyHashSet","add","add","contains","contains","add","remove","contains"], [[],[1],[2],[1],[3],[2],[2],[2]]
Out[null,null,null,true,false,null,null,false]
Ví dụ 2
In["MyHashSet","add","remove","contains"], [[],[1],[1],[1]]
Out[null,null,null,false]
Ý tưởng
Chaining: mảng bucket size BASE, hash(key)%BASE, mỗi bucket linked list int. add/remove/contains duyệt bucket.
Mã nguồn
solution.c C99 
#define HSZ 769
static int buckets[HSZ][128], cnt[HSZ];
static int hkey(int k){ return ((k%HSZ)+HSZ)%HSZ; }
void myHashSetAdd(int key){ int b=hkey(key); for(int i=0;i<cnt[b];i++) if(buckets[b][i]==key) return; buckets[b][cnt[b]++]=key; }
void myHashSetRemove(int key){ int b=hkey(key); for(int i=0;i<cnt[b];i++) if(buckets[b][i]==key){ buckets[b][i]=buckets[b][--cnt[b]]; return; } }
bool myHashSetContains(int key){ int b=hkey(key); for(int i=0;i<cnt[b];i++) if(buckets[b][i]==key) return true; return false; }
class MyHashSet:
    BASE = 769
    def __init__(self):
        self.data = [[] for _ in range(self.BASE)]
    def _hash(self, key): return key % self.BASE
    def add(self, key: int) -> None:
        h = self._hash(key)
        if key not in self.data[h]: self.data[h].append(key)
    def remove(self, key: int) -> None:
        h = self._hash(key)
        if key in self.data[h]: self.data[h].remove(key)
    def contains(self, key: int) -> bool:
        return key in self.data[self._hash(key)]
class MyHashSet {
    static const int BASE = 769;
    vector<list<int>> data;
    int hash(int key) { return key % BASE; }
public:
    MyHashSet() : data(BASE) {}
    void add(int key) {
        int h = hash(key);
        for (int x : data[h]) if (x == key) return;
        data[h].push_back(key);
    }
    void remove(int key) {
        int h = hash(key);
        data[h].remove(key);
    }
    bool contains(int key) {
        int h = hash(key);
        for (int x : data[h]) if (x == key) return true;
        return false;
    }
};
Design O(1) O(n)

  2. 2Duyệt chính — add/remove/contains duyệt bucket.

Hash định vị bucket; list xử lý collision — add/remove/contains đúng membership.
