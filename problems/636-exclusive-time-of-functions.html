Exclusive Time of Functions

O(n) O(n) 66.2% AC
Mô tả
Cho logs Exclusive Time: function_id start/end và timestamp. Tính thời gian exclusive (ms) mỗi function.
Ví dụ
Ví dụ 1
Inn=2, logs=["0:start:0","1:start:2","1:end:5","0:end:6"]
Out[3,3]
Ví dụ 2
InMột function start/end
Outend-start+1
Ý tưởng
Stack function ids; on start push; on end pop, cộng delta trừ thời gian nested cho parent.
Mã nguồn

int* exclusiveTime(int n, char** logs, int ln, int* rs){ int* ans=calloc(n,sizeof(int)); int st[256],top=0,prev=0; for(int i=0;i<ln;i++){ int fid,ts; char typ[8]; sscanf(logs[i],"%d:%7s:%d",&fid,typ,&ts); if(typ[0]=='s'){ if(top) ans[st[top-1]]+=ts-prev; st[top++]=fid; prev=ts;} else { ans[st[--top]]+=ts-prev+1; prev=ts+1;} } *rs=n; return ans; }
class Solution:
    def exclusiveTime(self, n: int, logs: List[str]) -> List[int]:
        ans=[0]*n; st=[]; prev=0
        for log in logs:
            fid,typ,ts= log.split(":"); fid=int(fid); ts=int(ts)
            if typ=="start":
                if st: ans[st[-1]]+=ts-prev
                st.append(fid); prev=ts
            else:
                ans[st.pop()]+=ts-prev+1; prev=ts+1
        return ans
class Solution {
public:
    vector<int> exclusiveTime(int n, vector<string>& logs) {
        vector<int> ans(n); stack<int> st; int prev=0;
        for(auto& log:logs){
            int i=log.find(':'); int j=log.rfind(':');
            string typ=log.substr(i+1,j-i-1); int fid=stoi(log.substr(0,i)), ts=stoi(log.substr(j+1));
            if(typ=="start"){ if(!st.empty()) ans[st.top()]+=ts-prev; st.push(fid); prev=ts; }
            else { ans[st.top()]+=ts-prev+1; st.pop(); prev=ts+1; }
        }
        return ans;
    }
};
