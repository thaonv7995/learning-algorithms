Bulls and Cows

O(n) O(1) 52.5% AC
Mô tả
Trò Mastermind: secret và guess là chuỗi 4 chữ số. Bulls = đúng vị trí; Cows = đúng digit sai vị trí (không đếm bulls). Trả "xAyB".
Ví dụ
Ví dụ 1
Insecret = "1807", guess = "7810"
Out"1A3B"
Ví dụ 2
Insecret = "1123", guess = "0111"
Out"1A1B"
Ý tưởng
Một pass: bulls khi secret[i]==guess[i]; else tăng count secret digit và guess digit; cows += min(counts) sau pass.
Mã nguồn
solution.c C99 
char* getHint(char* secret, char* guess) {
    int bulls = 0, cntS[10] = {0}, cntG[10] = {0};
    for (int i = 0; secret[i]; i++) {
        if (secret[i] == guess[i]) bulls++;
        else { cntS[secret[i]-'0']++; cntG[guess[i]-'0']++; }
    }
    int cows = 0;
    for (int i = 0; i < 10; i++) cows += cntS[i] < cntG[i] ? cntS[i] : cntG[i];
    char* out = malloc(16);
    snprintf(out, 16, "%dA%dB", bulls, cows);
    return out;
}
class Solution:
    def getHint(self, secret: str, guess: str) -> str:
        bulls = 0
        cnt_s = [0] * 10
        cnt_g = [0] * 10
        for a, b in zip(secret, guess):
            if a == b:
                bulls += 1
            else:
                cnt_s[int(a)] += 1
                cnt_g[int(b)] += 1
        cows = sum(min(cnt_s[i], cnt_g[i]) for i in range(10))
        return f"{bulls}A{cows}B"
class Solution {
public:
    string getHint(string secret, string guess) {
        int bulls = 0, cntS[10] = {}, cntG[10] = {};
        for (int i = 0; i < (int)secret.size(); i++) {
            if (secret[i] == guess[i]) bulls++;
            else { cntS[secret[i]-'0']++; cntG[guess[i]-'0']++; }
        }
        int cows = 0;
        for (int i = 0; i < 10; i++) cows += min(cntS[i], cntG[i]);
        return to_string(bulls) + "A" + to_string(cows) + "B";
    }
};
