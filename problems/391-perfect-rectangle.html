Perfect Rectangle

O(n) O(n) 38% AC
Mô tả
Cho các hình chữ nhật axis-aligned [x1,y1,x2,y2]. Kiểm tra chúng ghép thành một hình chữ nhật lớn không overlap, không gap.
Ví dụ
Ví dụ 1
Inrectangles = [[1,1,3,3],[3,1,4,2],[3,2,4,4],[1,3,2,4],[2,3,3,4]]
Outtrue
Ví dụ 2
Inrectangles = [[1,1,2,3],[1,3,2,4],[3,1,4,2],[3,2,4,4]]
Outfalse
Ý tưởng
XOR corner points: góc trong xuất hiện 4 lần cancel; góc ngoài 1 lần; cạnh trong 2 lần. Kiểm tra area tổng = (maxX-minX)*(maxY-minY) và set size=4.

bool isRectangleCover(int** rectangles, int n, int* cs) {
    long long area=0; int minx=INT_MAX,miny=INT_MAX,maxx=INT_MIN,maxy=INT_MIN;
    for(int i=0;i<n;i++){
        int x1=rectangles[i][0],y1=rectangles[i][1],x2=rectangles[i][2],y2=rectangles[i][3];
        area+=(long long)(x2-x1)*(y2-y1);
        if(x1<minx)minx=x1; if(y1<miny)miny=y1; if(x2>maxx)maxx=x2; if(y2>maxy)maxy=y2;
    }
    return area==(long long)(maxx-minx)*(maxy-miny);
}
class Solution:
    def isRectangleCover(self, rectangles: List[List[int]]) -> bool:
        corners = set()
        area, minx, miny, maxx, maxy = 0, inf, inf, -inf, -inf
        for x1,y1,x2,y2 in rectangles:
            area += (x2-x1)*(y2-y1)
            minx, miny = min(minx,x1), min(miny,y1)
            maxx, maxy = max(maxx,x2), max(maxy,y2)
            for p in ((x1,y1),(x1,y2),(x2,y1),(x2,y2)):
                corners.symmetric_difference_update({p})
        if area != (maxx-minx)*(maxy-miny): return False
        return len(corners) == 4
class Solution {
public:
    bool isRectangleCover(vector<vector<int>>& r) {
        set<pair<int,int>> c;
        long long area = 0;
        int minx=INT_MAX,miny=INT_MAX,maxx=INT_MIN,maxy=INT_MIN;
        for (auto& x : r) {
            area += 1LL*(x[2]-x[0])*(x[3]-x[1]);
            minx=min(minx,x[0]); miny=min(miny,x[1]);
            maxx=max(maxx,x[2]); maxy=max(maxy,x[3]);
            auto toggle = [&](int a,int b){ if(c.count({a,b})) c.erase({a,b}); else c.insert({a,b}); };
            toggle(x[0],x[1]); toggle(x[0],x[3]); toggle(x[2],x[1]); toggle(x[2],x[3]);
        }
        if (area != 1LL*(maxx-minx)*(maxy-miny)) return false;
        return c.size() == 4;
    }
};
