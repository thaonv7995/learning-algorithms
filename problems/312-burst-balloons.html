Burst Balloons

O(n^3) O(n^2) 63.5% AC
Mô tả
Cho mảng nums gồm n bóng bay có số. Nếu nổ bóng i cuối cùng trong đoạn, điểm = nums[left]*nums[i]*nums[right]. Trả điểm tối đa khi nổ hết bóng.
Ví dụ
Ví dụ 1
Innums = [3,1,5,8]
Out167
Ví dụ 2
Innums = [1,5]
Out10
Ý tưởng
Interval DP: thêm nums[0]=nums[n+1]=1. dp[l][r] = max điểm trong (l,r) exclusive. Thử bóng k nổ cuối: dp[l][r]=max(dp[l][k]+dp[k][r]+a[l]*a[k]*a[r]).

int maxCoins(int* nums, int numsSize) {
    int n = numsSize + 2;
    int* a = malloc(n * sizeof(int));
    a[0] = a[n-1] = 1;
    for (int i = 0; i < numsSize; i++) a[i+1] = nums[i];
    int** dp = malloc(n * sizeof(int*));
    for (int i = 0; i < n; i++) { dp[i] = calloc(n, sizeof(int)); }
    for (int len = 3; len <= n; len++)
        for (int l = 0; l + len - 1 < n; l++) {
            int r = l + len - 1;
            for (int k = l + 1; k < r; k++) {
                int v = dp[l][k] + dp[k][r] + a[l]*a[k]*a[r];
                if (v > dp[l][r]) dp[l][r] = v;
            }
        }
    int ans = dp[0][n-1];
    for (int i = 0; i < n; i++) free(dp[i]);
    free(dp); free(a);
    return ans;
}
class Solution:
    def maxCoins(self, nums: List[int]) -> int:
        a = [1] + nums + [1]
        n = len(a)
        dp = [[0]*n for _ in range(n)]
        for length in range(3, n+1):
            for l in range(0, n-length+1):
                r = l + length - 1
                for k in range(l+1, r):
                    dp[l][r] = max(dp[l][r], dp[l][k] + dp[k][r] + a[l]*a[k]*a[r])
        return dp[0][n-1]
class Solution {
public:
    int maxCoins(vector<int>& nums) {
        vector<int> a = {1};
        a.insert(a.end(), nums.begin(), nums.end());
        a.push_back(1);
        int n = a.size();
        vector<vector<int>> dp(n, vector<int>(n, 0));
        for (int len = 3; len <= n; len++)
            for (int l = 0; l + len - 1 < n; l++) {
                int r = l + len - 1;
                for (int k = l + 1; k < r; k++)
                    dp[l][r] = max(dp[l][r], dp[l][k] + dp[k][r] + a[l]*a[k]*a[r]);
            }
        return dp[0][n-1];
    }
};
