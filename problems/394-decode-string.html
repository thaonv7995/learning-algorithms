Decode String

O(n) O(n) 62.6% AC
Mô tả
Chuỗi mã hóa dạng k[encoded_string] — lặp encoded_string k lần, có thể lồng nhau. Trả chuỗi decoded.
Ví dụ
Ví dụ 1
Ins = "3[a]2[bc]"
Out"aaabcbc"
Ví dụ 2
Ins = "3[a2[c]]"
Out"accaccacc"
Ý tưởng
Stack: gặp digit tích lũy k; '[' push (curString, k); gặp ']' pop và cur = prev + k*cur; chữ cái append cur.

        else if (s[i]=='['){ stackStr[top]=strdup(cur); stackK[top++]=k; cur[0]=0; k=0; }
        else if (s[i]==']'){

        } else { int len=strlen(cur); cur[len]=s[i]; cur[len+1]=0; }
    }
    return cur;

class Solution:
    def decodeString(self, s: str) -> str:
        st, cur, k = [], '', 0
        for c in s:
            if c.isdigit():
                k = k * 10 + int(c)
            elif c == '[':
                st.append((cur, k)); cur, k = '', 0
            elif c == ']':
                prev, num = st.pop()
                cur = prev + num * cur
            else:
                cur += c
        return cur
class Solution {
public:
    string decodeString(string s) {
        stack<pair<string,int>> st;
        string cur; int k = 0;
        for (char c : s) {
            if (isdigit(c)) k = k*10 + (c-'0');
            else if (c == '[') { st.push({cur,k}); cur.clear(); k=0; }
            else if (c == ']') {
                auto [prev, num] = st.top(); st.pop();
                string rep;
                for (int i=0;i<num;i++) rep += cur;
                cur = prev + rep;
            } else cur.push_back(c);
        }
        return cur;
    }
};
