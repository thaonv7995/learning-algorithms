/* permuteUnique — sort + skip dup; xem C++ */
class Solution:
    def permuteUnique(self, nums: List[int]) -> List[List[int]]:
        nums.sort(); res = []
        def dfs(path, used):
            if len(path) == len(nums):
                res.append(path[:]); return
            for i in range(len(nums)):
                if used[i]: continue
                if i and nums[i]==nums[i-1] and not used[i-1]: continue
                used[i]=True; path.append(nums[i])
                dfs(path, used)
                path.pop(); used[i]=False
        dfs([], [False]*len(nums))
        return res
class Solution {
    vector<vector<int>> res;
    void dfs(vector<int>& nums, vector<int>& p, vector<bool>& u) {
        if (p.size()==nums.size()){ res.push_back(p); return; }
        for(int i=0;i<(int)nums.size();i++){
            if(u[i]) continue;
            if(i&&nums[i]==nums[i-1]&&!u[i-1]) continue;
            u[i]=true; p.push_back(nums[i]); dfs(nums,p,u); p.pop_back(); u[i]=false;
        }
    }
public:
    vector<vector<int>> permuteUnique(vector<int>& nums) {
        sort(nums.begin(),nums.end());
        vector<int> p; vector<bool> u(nums.size());
        dfs(nums,p,u); return res;
    }
};
