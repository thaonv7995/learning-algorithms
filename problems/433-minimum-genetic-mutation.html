Minimum Genetic Mutation

O(n * 8^L) O(n) 56.6% AC
Mô tả
Gene start, end, bank (mutations hợp lệ). Mỗi bước đổi đúng 1 ký tự A/C/G/T. Trả số bước mutation tối thiểu start→end, -1 nếu không được.
Ví dụ
Ví dụ 1
Instart = "AACCGGTT", end = "AACCGGTA", bank = ["AACCGGTA"]
Out1
Ví dụ 2
Instart = "AACCGGTT", end = "AAACGGTA", bank = ["AACCGGTA","AACCGCTA","AAACGGTA"]
Out3
Ý tưởng
BFS từ start; với mỗi gene thử đổi từng vị trí 4 ký tự; nếu trong bank và chưa thăm thì enqueue.

class Solution:
    def minMutation(self, startGene: str, endGene: str, bank: List[str]) -> int:
        bank = set(bank)
        if endGene not in bank: return -1
        q = deque([(startGene, 0)])
        seen = {startGene}
        genes = "ACGT"
        while q:
            g, d = q.popleft()
            if g == endGene: return d
            for i in range(len(g)):
                for c in genes:
                    if c == g[i]: continue
                    ng = g[:i] + c + g[i+1:]
                    if ng in bank and ng not in seen:
                        seen.add(ng); q.append((ng, d+1))
        return -1
class Solution {
public:
    int minMutation(string start, string end, vector<string>& bank) {
        unordered_set<string> B(bank.begin(), bank.end());
        if (!B.count(end)) return -1;
        queue<pair<string,int>> q; q.push({start,0});
        unordered_set<string> seen={start};
        string genes="ACGT";
        while(!q.empty()){
            auto [g,d]=q.front(); q.pop();
            if(g==end) return d;
            for(int i=0;i<(int)g.size();i++)
                for(char c:genes){
                    if(c==g[i]) continue;
                    string ng=g; ng[i]=c;
                    if(B.count(ng)&&!seen.count(ng)){ seen.insert(ng); q.push({ng,d+1}); }
                }
        }
        return -1;
    }
};
