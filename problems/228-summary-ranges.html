Summary Ranges

O(n) O(1) 54.2% AC
Mô tả
Cho mảng nums đã sort tăng dần (có thể trùng?). Gom các dãy liên tiếp thành chuỗi "a->b" hoặc "a" nếu một phần tử.
Ví dụ
Ví dụ 1
Innums = [0,1,2,4,5,7]
Out["0->2","4->5","7"]
Ví dụ 2
Innums = [0,2,3,4,6,8,9]
Out["0","2->4","6","8->9"]
Ví dụ 3
Innums = []
Out[]
Ý tưởng
Two pointers: start=i; while i+1 < n và nums[i+1]==nums[i]+1 thì i++; format range từ start đến i.

char** summaryRanges(int* nums, int n, int* retSize) {
    *retSize = 0;
    if (!n) return NULL;
    char** out = malloc(n * sizeof(char*));
    for (int i = 0; i < n; ) {
        int start = i;
        while (i + 1 < n && nums[i + 1] == nums[i] + 1) i++;
        char buf[64];
        if (start == i) snprintf(buf, sizeof(buf), "%d", nums[start]);
        else snprintf(buf, sizeof(buf), "%d->%d", nums[start], nums[i]);
        out[(*retSize)++] = strdup(buf);
        i++;
    }
    return out;
}
class Solution:
    def summaryRanges(self, nums: List[int]) -> List[str]:
        res, i, n = [], 0, len(nums)
        while i < n:
            start = i
            while i + 1 < n and nums[i + 1] == nums[i] + 1:
                i += 1
            if start == i:
                res.append(str(nums[start]))
            else:
                res.append(f"{nums[start]}->{nums[i]}")
            i += 1
        return res
class Solution {
public:
    vector<string> summaryRanges(vector<int>& nums) {
        vector<string> res;
        int n = nums.size(), i = 0;
        while (i < n) {
            int start = i;
            while (i + 1 < n && nums[i + 1] == nums[i] + 1) i++;
            if (start == i) res.push_back(to_string(nums[start]));
            else res.push_back(to_string(nums[start]) + "->" + to_string(nums[i]));
            i++;
        }
        return res;
    }
};
