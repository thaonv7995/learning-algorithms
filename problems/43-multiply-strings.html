Multiply Strings

O(M·N) O(M+N) 44.2% AC
Mô tả
Nhân hai số không âm dạng chuỗi, trả kết quả dạng chuỗi. Không dùng BigInteger hay ép kiểu int trực tiếp.
Ví dụ
Ví dụ 1
Innum1 = "2", num2 = "3"
Out"6"
Ví dụ 2
Innum1 = "123", num2 = "456"
Out"56088"
Ý tưởng
Nhân tay: mảng pos size m+n. Duyệt từng cặp chữ số i,j: mul = d1*d2 + pos[i+j+1]; pos[i+j+1]=mul%10; pos[i+j]+=mul/10. Bỏ leading zero.

char* multiply(char* num1, char* num2) {
    int m = (int)strlen(num1), n = (int)strlen(num2);
    if (num1[0]=='0'||num2[0]=='0') return strdup("0");
    int* pos = (int*)calloc(m + n, sizeof(int));
    for (int i = m - 1; i >= 0; i--)
        for (int j = n - 1; j >= 0; j--) {
            int mul = (num1[i]-'0')*(num2[j]-'0');
            int s = mul + pos[i+j+1];
            pos[i+j+1] = s % 10;
            pos[i+j] += s / 10;
        }
    int k = 0;
    while (k < m + n - 1 && pos[k] == 0) k++;
    char* res = (char*)malloc(m + n - k + 1);
    int p = 0;
    for (; k < m + n; k++) res[p++] = (char)('0' + pos[k]);
    res[p] = '\0';
    free(pos);
    return res;
}
class Solution:
    def multiply(self, num1: str, num2: str) -> str:
        if num1 == "0" or num2 == "0": return "0"
        m, n = len(num1), len(num2)
        pos = [0] * (m + n)
        for i in range(m - 1, -1, -1):
            for j in range(n - 1, -1, -1):
                mul = int(num1[i]) * int(num2[j])
                s = mul + pos[i + j + 1]
                pos[i + j + 1] = s % 10
                pos[i + j] += s // 10
        k = 0
        while k < len(pos) - 1 and pos[k] == 0: k += 1
        return "".join(map(str, pos[k:]))
class Solution {
public:
    string multiply(string a, string b) {
        if (a=="0"||b=="0") return "0";
        int m=a.size(), n=b.size();
        vector<int> pos(m+n,0);
        for(int i=m-1;i>=0;i--)
            for(int j=n-1;j>=0;j--){
                int mul=(a[i]-'0')*(b[j]-'0');
                int s=mul+pos[i+j+1];
                pos[i+j+1]=s%10; pos[i+j]+=s/10;
            }
        int k=0; while(k<(int)pos.size()-1&&pos[k]==0) k++;
        string res; for(;k<(int)pos.size();k++) res+=char('0'+pos[k]);
        return res;
    }
};
