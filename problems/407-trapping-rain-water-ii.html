Trapping Rain Water II

O(mn log(mn)) O(mn) 64.1% AC
Mô tả
Ma trận heightMap m×n cho độ cao. Tính lượng nước mưa giữ lại sau khi nước chảy ra biên (không qua điểm cao hơn mực nước local).
Ví dụ
Ví dụ 1
InheightMap = [[1,4,3,1,3,2],[3,2,1,3,2,4],[2,3,3,2,3,1]]
Out4
Ví dụ 2
InheightMap = [[3,3,3,3,3],[3,2,2,2,3],[3,2,1,2,3],[3,2,2,2,3],[3,3,3,3,3]]
Out10
Ý tưởng
Min-heap từ mọi ô biên; pop ô thấp nhất, lan sang láng giềng chưa thăm: nước += max(0, wallHeight - neighborHeight), push neighbor với height max(neighbor, wall).

typedef struct { int h, r, c; } Cell;
static int cmpCell(const void* a, const void* b) {
    return ((Cell*)a)->h - ((Cell*)b)->h;
}
int trapRainWater(int** hm, int m, int* cs, int n) {
    if (!m || !n) return 0;
    char** vis = (char**)malloc(m * sizeof(char*));
    Cell* heap = (Cell*)malloc(m * n * sizeof(Cell));
    int hlen = 0, ans = 0;
    for (int i = 0; i < m; i++) {
        vis[i] = (char*)calloc(n, 1);
        vis[i][0] = vis[i][n-1] = 1;
        heap[hlen++] = (Cell){hm[i][0], i, 0};
        heap[hlen++] = (Cell){hm[i][n-1], i, n-1};
    }
    for (int j = 1; j < n - 1; j++) {
        vis[0][j] = vis[m-1][j] = 1;
        heap[hlen++] = (Cell){hm[0][j], 0, j};
        heap[hlen++] = (Cell){hm[m-1][j], m-1, j};
    }
    int dr[4] = {1,-1,0,0}, dc[4] = {0,0,1,-1};
    while (hlen) {
        qsort(heap, hlen, sizeof(Cell), cmpCell);
        Cell cur = heap[0];
        memmove(heap, heap + 1, (--hlen) * sizeof(Cell));
        for (int k = 0; k < 4; k++) {
            int nr = cur.r + dr[k], nc = cur.c + dc[k];
            if (nr < 0 || nr >= m || nc < 0 || nc >= n || vis[nr][nc]) continue;
            vis[nr][nc] = 1;
            if (hm[nr][nc] < cur.h) ans += cur.h - hm[nr][nc];
            int nh = hm[nr][nc] > cur.h ? hm[nr][nc] : cur.h;
            heap[hlen++] = (Cell){nh, nr, nc};
        }
    }
    for (int i = 0; i < m; i++) free(vis[i]);
    free(vis); free(heap);
    return ans;
}
class Solution:
    def trapRainWater(self, heightMap: List[List[int]]) -> int:
        if not heightMap or not heightMap[0]: return 0
        m, n = len(heightMap), len(heightMap[0])
        vis = [[False]*n for _ in range(m)]
        heap = []
        for i in range(m):
            for j in (0, n-1):
                heapq.heappush(heap, (heightMap[i][j], i, j)); vis[i][j]=True
        for j in range(1, n-1):
            for i in (0, m-1):
                heapq.heappush(heap, (heightMap[i][j], i, j)); vis[i][j]=True
        ans = 0
        while heap:
            h, r, c = heapq.heappop(heap)
            for dr, dc in ((1,0),(-1,0),(0,1),(0,-1)):
                nr, nc = r+dr, c+dc
                if 0<=nr<m and 0<=nc<n and not vis[nr][nc]:
                    vis[nr][nc]=True
                    ans += max(0, h - heightMap[nr][nc])
                    heapq.heappush(heap, (max(h, heightMap[nr][nc]), nr, nc))
        return ans
class Solution {
public:
    int trapRainWater(vector<vector<int>>& hm) {
        int m=hm.size(); if(!m) return 0; int n=hm[0].size();
        vector<vector<char>> vis(m, vector<char>(n));
        priority_queue<array<int,3>, vector<array<int,3>>, greater<>> pq;
        auto push=[&](int r,int c){ vis[r][c]=1; pq.push({hm[r][c],r,c}); };
        for(int i=0;i<m;i++){ push(i,0); push(i,n-1); }
        for(int j=1;j<n-1;j++){ push(0,j); push(m-1,j); }
        int ans=0, d[4][2]={{1,0},{-1,0},{0,1},{0,-1}};
        while(!pq.empty()){
            auto [h,r,c]=pq.top(); pq.pop();
            for(auto& dd:d){ int nr=r+dd[0], nc=c+dd[1];
                if(nr>=0&&nr<m&&nc>=0&&nc<n&&!vis[nr][nc]){
                    vis[nr][nc]=1; ans+=max(0,h-hm[nr][nc]);
                    pq.push({max(h,hm[nr][nc]),nr,nc});
                }
            }
        }
        return ans;
    }
};
