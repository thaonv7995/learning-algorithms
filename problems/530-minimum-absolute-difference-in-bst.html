Minimum Absolute Difference in BST

O(n) O(h) 59.4% AC
Mô tả
Cho BST. Trả minimum absolute difference giữa hai giá trị bất kỳ trong cây.
Ví dụ
Ví dụ 1
Inroot = [4,2,6,1,3]
Out1
Ví dụ 2
Inroot = [1,0,48,null,null,12,49]
Out1
Ý tưởng
Inorder BST → dãy tăng dần; min diff = min(nums[i]-nums[i-1]).
Mã nguồn
solution.c C99 
/* Inorder min adjacent — xem C++ */
class Solution:
    def getMinimumDifference(self, root: Optional[TreeNode]) -> int:
        self.prev = None
        self.ans = 10**18
        def dfs(n):
            if not n: return
            dfs(n.left)
            if self.prev is not None:
                self.ans = min(self.ans, n.val - self.prev)
            self.prev = n.val
            dfs(n.right)
        dfs(root)
        return self.ans
class Solution {
    TreeNode* prev = nullptr;
    int ans = INT_MAX;
    void inorder(TreeNode* n) {
        if (!n) return;
        inorder(n->left);
        if (prev) ans = min(ans, n->val - prev->val);
        prev = n;
        inorder(n->right);
    }
public:
    int getMinimumDifference(TreeNode* root) {
        inorder(root);
        return ans;
    }
};
