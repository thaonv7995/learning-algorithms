double knightProbability688(int n,int k,int r,int c) {
    double dp[25][25]={0}, nd[25][25]={0};
    dp[r][c]=1.0;
    int mv[8][2]={{-2,-1},{-2,1},{-1,-2},{-1,2},{2,-1},{2,1},{1,-2},{1,2}};
    for(int t=0;t<k;t++){
        for(int i=0;i<n;i++) for(int j=0;j<n;j++) nd[i][j]=0;
        for(int i=0;i<n;i++) for(int j=0;j<n;j++) if(dp[i][j])
            for(int m=0;m<8;m++){
                int ni=i+mv[m][0], nj=j+mv[m][1];
                if(ni>=0&&ni<n&&nj>=0&&nj<n) nd[ni][nj]+=dp[i][j]/8.0;
            }
        for(int i=0;i<n;i++) for(int j=0;j<n;j++) dp[i][j]=nd[i][j];
    }
    double s=0; for(int i=0;i<n;i++) for(int j=0;j<n;j++) s+=dp[i][j]; return s;
}
class Solution:
    def knightProbability(self, n: int, k: int, row: int, column: int) -> float:
        dp=[[0.0]*n for _ in range(n)]; dp[row][column]=1.0
        moves=[(-2,-1),(-2,1),(-1,-2),(-1,2),(2,-1),(2,1),(1,-2),(1,2)]
        for _ in range(k):
            nd=[[0.0]*n for _ in range(n)]
            for r in range(n):
                for c in range(n):
                    if dp[r][c]:
                        for dr,dc in moves:
                            nr,nc=r+dr,c+dc
                            if 0<=nr<n and 0<=nc<n: nd[nr][nc]+=dp[r][c]/8.0
            dp=nd
        return sum(sum(row) for row in dp)
