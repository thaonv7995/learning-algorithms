Count Primes

O(n log log n) O(n) 36.1% AC
Mô tả
Cho số nguyên n. Đếm số lượng số nguyên tố nhỏ hơn n (0 và 1 không được tính là số nguyên tố).
Ví dụ
Ví dụ 1
Inn = 10
Out4
Ví dụ 2
Inn = 0
Out0
Ví dụ 3
Inn = 1
Out0
Ý tưởng
Sieve of Eratosthenes: mảng bool isPrime; với i từ 2, nếu isPrime[i] thì đánh dấu mọi bội j=i*i, i+i, ... < n là composite.
Mã nguồn
solution.c C99 
int countPrimes(int n) {
    if (n <= 2) return 0;
    char* sieve = calloc(n, 1);
    sieve[0] = sieve[1] = 1;
    int cnt = 0;
    for (int i = 2; i * i < n; i++)
        if (!sieve[i])
            for (int j = i * i; j < n; j += i)
                sieve[j] = 1;
    for (int i = 2; i < n; i++) if (!sieve[i]) cnt++;
    free(sieve);
    return cnt;
}
class Solution:
    def countPrimes(self, n: int) -> int:
        if n <= 2:
            return 0
        sieve = [True] * n
        sieve[0] = sieve[1] = False
        for i in range(2, int(n ** 0.5) + 1):
            if sieve[i]:
                for j in range(i * i, n, i):
                    sieve[j] = False
        return sum(sieve)
class Solution {
public:
    int countPrimes(int n) {
        if (n <= 2) return 0;
        vector<bool> sieve(n, true);
        sieve[0] = sieve[1] = false;
        for (int i = 2; i * i < n; i++)
            if (sieve[i])
                for (int j = i * i; j < n; j += i)
                    sieve[j] = false;
        return count(sieve.begin(), sieve.end(), true);
    }
};
