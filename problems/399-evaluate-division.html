Evaluate Division

O(q×(V+E)) O(V+E) 64.2% AC
Mô tả
Cho equations [a,b] và values a/b. Trả lời queries c/b; nếu không suy ra trả -1.0.
Ví dụ
Ví dụ 1
Inequations = [["a","b"],["b","c"]], values = [2.0,3.0], queries = [["a","c"],["b","a"],["a","e"],["a","a"],["x","x"]]
Out[6.0,0.5,-1.0,1.0,-1.0]
Ví dụ 2
Inequations = [["a","b"],["c","d"]], values = [1.0,1.0], queries = [["a","c"],["c","c"]]
Out[-1.0,1.0]
Ý tưởng
Weighted graph: a→b weight val, b→a 1/val. DFS/BFS từ query num tìm den nhân cạnh; a/a=1; biến lạ -1.

class Solution:
    def calcEquation(self, equations: List[List[str]], values: List[float], queries: List[List[str]]) -> List[float]:
        graph = defaultdict(dict)
        for (a,b), v in zip(equations, values):
            graph[a][b] = v; graph[b][a] = 1/v
        def dfs(src, dst, seen):
            if src not in graph or dst not in graph: return -1.0
            if src == dst: return 1.0
            seen.add(src)
            for nei, w in graph[src].items():
                if nei in seen: continue
                sub = dfs(nei, dst, seen)
                if sub != -1.0: return w * sub
            return -1.0
        return [dfs(a, b, set()) for a, b in queries]
class Solution {
    unordered_map<string, unordered_map<string,double>> g;
    double dfs(string& src, string& dst, unordered_set<string>& vis) {
        if (!g.count(src) || !g.count(dst)) return -1.0;
        if (src == dst) return 1.0;
        vis.insert(src);
        for (auto& [nei, w] : g[src]) {
            if (vis.count(nei)) continue;
            double sub = dfs(nei, dst, vis);
            if (sub >= 0) return w * sub;
        }
        return -1.0;
    }
public:
    vector<double> calcEquation(vector<vector<string>>& eq, vector<double>& val, vector<vector<string>>& q) {
        for (int i = 0; i < (int)eq.size(); i++) {
            g[eq[i][0]][eq[i][1]] = val[i];
            g[eq[i][1]][eq[i][0]] = 1.0 / val[i];
        }
        vector<double> ans;
        for (auto& query : q) {
            unordered_set<string> vis;
            ans.push_back(dfs(query[0], query[1], vis));
        }
        return ans;
    }
};
