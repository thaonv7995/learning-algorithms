Scramble String

O(n⁴) O(n³) 44.6% AC
Mô tả
Hai chuỗi s1, s2 cùng độ dài. Kiểm tra s2 có phải scrambled string của s1 không: có thể chia s1 thành hai phần không rỗng và hoán đổi hoặc scramble đệ quy hai phần.
Ví dụ
Ví dụ 1
Ins1 = "great", s2 = "rgeat"
Outtrue
Ví dụ 2
Ins1 = "abcde", s2 = "caebd"
Outfalse
Ý tưởng
DP/lệnh ghi nhớ: isScramble(i1,i2,len) — thử mọi split k: scramble trái-trái phải-phải HOẶC trái-phải phải-trái. Base len=1 so ký tự.

/* isScramble — DP 3D memo split; xem C++ */
class Solution:
    def isScramble(self, s1: str, s2: str) -> bool:
        from functools import lru_cache
        n = len(s1)
        if sorted(s1) != sorted(s2): return False
        @lru_cache(None)
        def dfs(i, j, l):
            if l == 1: return s1[i] == s2[j]
            for k in range(1, l):
                if (dfs(i, j, k) and dfs(i+k, j+k, l-k)) or \
                   (dfs(i, j+k, l-k) and dfs(i+k, j, k)):
                    return True
            return False
        return dfs(0, 0, n)
class Solution {
    string a,b;
    vector<vector<vector<int>>> memo;
    bool dp(int i,int j,int l) {
        if (l==1) return a[i]==b[j];
        if (memo[i][j][l]!=-1) return memo[i][j][l];
        for (int k=1;k<l;k++)
            if ((dp(i,j,k)&&dp(i+k,j+k,l-k))||(dp(i,j+k,l-k)&&dp(i+k,j,k)))
                return memo[i][j][l]=1;
        return memo[i][j][l]=0;
    }
public:
    bool isScramble(string s1, string s2) {
        if (s1.size()!=s2.size()) return false;
        a=s1; b=s2; int n=a.size();
        memo.assign(n, vector<vector<int>>(n, vector<int>(n+1,-1)));
        return dp(0,0,n);
    }
};
