My Calendar III

O(n) per book O(n) 71.7% AC
Mô tả
My Calendar III: book(start,end) luôn thành công; trả max số booking overlap tại cùng một thời điểm sau mỗi lần book.
Ví dụ

Ý tưởng
Difference array / timeline: +1 tại start, -1 tại end; sweep sorted events lấy prefix max.
Mã nguồn
solution.c C99 
typedef struct { int t, v; } Ev;
static Ev ev[2000]; static int en;
int myCalendarThreeBook(int start, int end) {
    ev[en].t=start; ev[en++].v=1;
    ev[en].t=end; ev[en++].v=-1;
    qsort(ev, en, sizeof(Ev), [](const void* a,const void* b){ return ((Ev*)a)->t - ((Ev*)b)->t; });
    int cur=0, ans=0;
    for(int i=0;i<en;i++){ cur+=ev[i].v; if(cur>ans) ans=cur; }
    return ans;
}
class MyCalendarThree:
    def __init__(self):
        self.delta = defaultdict(int)
    def book(self, start: int, end: int) -> int:
        self.delta[start] += 1
        self.delta[end] -= 1
        cur = ans = 0
        for t in sorted(self.delta.keys()):
            cur += self.delta[t]
            ans = max(ans, cur)
        return ans
class MyCalendarThree {
    map<int,int> d;
public:
    int book(int start, int end) {
        d[start]++; d[end]--;
        int cur=0, ans=0;
        for (auto& [t,v]: d) { cur+=v; ans=max(ans,cur); }
        return ans;
    }
};
