Magical String

O(n) O(n) 55.3% AC
Mô tả
Magical string S: bắt đầu "122". S[i] cho biết lặp bao nhiêu lần ký tự 1 hoặc 2 kế tiếp. Đếm số '1' trong n ký tự đầu của S.
Ví dụ
Ví dụ 1
Inn = 6
Out3
Ví dụ 2
Inn = 1
Out1
Ví dụ 3
Inn = 4
Out2
Ý tưởng
Mảng s; i=2; lặp s[i] lần thêm (last%2?2:1); đếm 1 trong n phần tử đầu.

int magicalString(int n) {
    if(!n) return 0;
    int* s=malloc((n+100)*sizeof(int)); int sz=3, i=2;
    s[0]=1;s[1]=2;s[2]=2;
    while(sz<n){
        int rep=s[i], val=s[sz-1]==1?2:1;
        while(rep--) s[sz++]=val;
        i++;
    }
    int ans=0; for(int j=0;j<n;j++) if(s[j]==1) ans++;
    free(s); return ans;
}
class Solution:
    def magicalString(self, n: int) -> int:
        if n == 0: return 0
        s = [1, 2, 2]
        i = 2
        while len(s) < n:
            s.extend([s[-1] % 2 + 1] * s[i])
            i += 1
        return sum(s[:n])
class Solution {
public:
    int magicalString(int n) {
        if(!n) return 0;
        vector<int> s={1,2,2}; int i=2, ans=0;
        while((int)s.size()<n){
            int rep=s[i], val=s.back()==1?2:1;
            while(rep--) s.push_back(val);
            i++;
        }
        for(int j=0;j<n;j++) ans+=s[j]==1;
        return ans;
    }
};
