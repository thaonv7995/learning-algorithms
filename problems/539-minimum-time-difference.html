Minimum Time Difference

O(n log n) O(n) 62.6% AC
Mô tả
Cho mảng time dạng "HH:MM" (24h). Trả chênh lệch thời gian nhỏ nhất (phút) giữa hai mốc bất kỳ (có thể qua nửa đêm).

Out120
Ý tưởng
Chuyển mỗi time sang phút 0..1439; sort; min diff giữa kề + wrap (1440 - last + first).
Mã nguồn
solution.c C99 
/* Parse minutes sort wrap — xem C++ */
class Solution:
    def findMinDifference(self, timeList: List[str]) -> int:
        mins = []
        for t in timeList:
            h, m = map(int, t.split(":"))
            mins.append(h * 60 + m)
        mins.sort()
        ans = 24 * 60
        for i in range(1, len(mins)):
            ans = min(ans, mins[i] - mins[i-1])
        ans = min(ans, 24*60 - mins[-1] + mins[0])
        return ans
class Solution {
public:
    int findMinDifference(vector<string>& timeList) {
        vector<int> m;
        for (auto& t : timeList) {
            int h = stoi(t.substr(0,2)), mi = stoi(t.substr(3,2));
            m.push_back(h*60+mi);
        }
        sort(m.begin(), m.end());
        int ans = 1440;
        for (int i = 1; i < (int)m.size(); i++) ans = min(ans, m[i]-m[i-1]);
        ans = min(ans, 1440 - m.back() + m.front());
        return ans;
    }
};
