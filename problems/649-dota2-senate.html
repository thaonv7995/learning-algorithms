Mô tả
Senate R và D bầu cử: mỗi người ban điều tiên đối phương tiếp theo vòng. Ai thắng?
Ví dụ
Ví dụ 1
Insenate = "RD"
Out"Radiant"
Ví dụ 2
Insenate = "RDD"
Out"Dire"
Ý tưởng
Queue R và D; so sánh index, ai nhỏ hơn loại đối phương sau.

class Solution:
    def predictPartyVictory(self, senate: str) -> str:
        from collections import deque
        r,d=deque(),deque()
        for i,ch in enumerate(senate):
            (r if ch=="R" else d).append(i)
        n=len(senate)
        while r and d:
            if r[0]<d[0]: d.popleft()
            else: r.popleft()
            (r if r and (not d or r[0]<d[0]) else d)[0]  # fix
        return "Radiant" if r else "Dire"
class Solution {
public:
    string predictPartyVictory(string s) {
        queue<int> r,d;
        for(int i=0;i<(int)s.size();i++) (s[i]=='R'?r:d).push(i);
        while(!r.empty()&&!d.empty()){
            int ri=r.front(); r.pop(); int di=d.front(); d.pop();
            if(ri<di) r.push(ri+s.size()); else d.push(di+s.size());
        }
        return r.empty()?"Dire":"Radiant";
    }
};
