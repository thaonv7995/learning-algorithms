Shortest Palindrome

O(n) O(n) 42.5% AC
Mô tả
Cho chuỗi s. Thêm ký tự vào đầu s (không đổi thứ tự ký tự gốc) để thu palindrome ngắn nhất có thể. Trả chuỗi kết quả.
Ví dụ
Ví dụ 1
Ins = "aacecaaa"
Out"aaacecaaa"
Ví dụ 2
Ins = "abcd"
Out"dcbabcd"
Ý tưởng
Tìm prefix dài nhất của s là palindrome (KMP trên s + '#' + reverse(s)). Reverse phần suffix còn lại và prepend.

char* shortestPalindrome(const char* s) {
    int n = (int)strlen(s);
    if (!n) return strdup("");
    char* t = malloc(2*n + 2);
    strcpy(t, s); t[n] = '#'; reverseCopy(t+n+1, s);
    int m = 2*n+1;
    int* pi = calloc(m, sizeof(int));
    for (int i=1;i<m;i++) {
        int j=pi[i-1];
        while (j && t[i]!=t[j]) j=pi[j-1];
        if (t[i]==t[j]) j++;
        pi[i]=j;
    }
    int k=pi[m-1];
    char* out = malloc(n + (n-k) + 1);
    for (int i=n-1;i>=k;i--) out[n-1-i]=s[i];
    strcpy(out+(n-k), s);
    free(t); free(pi);
    return out;
}
class Solution:
    def shortestPalindrome(self, s: str) -> str:
        if not s:
            return s
        t = s + '#' + s[::-1]
        pi = [0] * len(t)
        for i in range(1, len(t)):
            j = pi[i - 1]
            while j and t[i] != t[j]:
                j = pi[j - 1]
            if t[i] == t[j]:
                j += 1
            pi[i] = j
        k = pi[-1]
        return s[k:][::-1] + s
class Solution {
public:
    string shortestPalindrome(string s) {
        string rs = s;
        reverse(rs.begin(), rs.end());
        string t = s + '#' + rs;
        vector<int> pi(t.size());
        for (int i = 1; i < (int)t.size(); i++) {
            int j = pi[i - 1];
            while (j && t[i] != t[j]) j = pi[j - 1];
            if (t[i] == t[j]) j++;
            pi[i] = j;
        }
        int k = pi.back();
        return rs.substr(0, s.size() - k) + s;
    }
};
