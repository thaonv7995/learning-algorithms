Basic Calculator II

O(n) O(1) 46.9% AC
Mô tả
Chuỗi s gồm số nguyên không âm và '+', '-', '*', '/' (chia lấy phần nguyên). Không có ngoặc. Tính giá trị biểu thức.
Ví dụ
Ví dụ 1
Ins = "3+2*2"
Out7
Ví dụ 2
Ins = " 3/2 "
Out1
Ví dụ 3
Ins = " 3+5 / 2 "
Out5
Ý tưởng
Duyệt: num hiện tại; gặp */ tính ngay với last; gặp +- push last vào result, last=±num. Cuối cộng result+last.

int calculate2(char* s) {
    int num = 0, last = 0, result = 0;
    char op = '+';
    for (int i = 0; s[i]; i++) {
        if (s[i] >= '0' && s[i] <= '9') num = num * 10 + (s[i] - '0');
        if ((!isdigit((unsigned char)s[i]) && !isspace((unsigned char)s[i])) || s[i+1]==0) {
            if (op == '+') { result += last; last = num; }
            else if (op == '-') { result += last; last = -num; }
            else if (op == '*') last *= num;
            else if (op == '/') last /= num;
            op = s[i]; num = 0;
        }
    }
    return result + last;
}
class Solution:
    def calculate(self, s: str) -> int:
        s = s.replace(' ', '')
        n, num, last, result, op = len(s), 0, 0, 0, '+'
        for i, c in enumerate(s):
            if c.isdigit():
                num = num * 10 + int(c)
            if not c.isdigit() or i == n - 1:
                if op == '+':
                    result += last
                    last = num
                elif op == '-':
                    result += last
                    last = -num
                elif op == '*':
                    last *= num
                elif op == '/':
                    last = int(last / num)
                op = c
                num = 0
        return result + last

            if (!isdigit(s[i]) && !isspace(s[i]) || i == n - 1) {
