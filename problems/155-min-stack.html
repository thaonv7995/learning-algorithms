Min Stack

O(1) per op O(n) 58.2% AC
Mô tả
Thiết kế stack hỗ trợ push, pop, top và getMin — tất cả O(1). getMin trả giá trị nhỏ nhất hiện có trên stack tại mọi thời điểm.
Ví dụ
Ví dụ 1
Inpush(-2), push(0), push(-3), getMin() → -3, pop(), top() → 0, getMin() → -2
Out[null,null,null,-3,null,0,-2]
Ví dụ 2
Inpush(2), push(0), push(3), getMin() → 0, pop(), getMin() → 0
Out[null,null,null,0,null,0]
Ý tưởng
Hai stack: stack chính và minStack lưu min tại mỗi mức. Push: đẩy val và min(val, minStack.top). Pop: pop cả hai. getMin: minStack.top.
Mã nguồn
solution.c C99 
typedef struct { int* data; int* mins; int top; int cap; } MinStack;
static void msGrow(MinStack* s) {
    s->cap *= 2;
    s->data = realloc(s->data, s->cap * sizeof(int));
    s->mins = realloc(s->mins, s->cap * sizeof(int));
}
MinStack* minStackCreate(void) {
    MinStack* s = malloc(sizeof(MinStack));
    s->cap = 8; s->top = -1;
    s->data = malloc(s->cap * sizeof(int));
    s->mins = malloc(s->cap * sizeof(int));
    return s;
}
void minStackPush(MinStack* s, int val) {
    if (++s->top >= s->cap) msGrow(s);
    s->data[s->top] = val;
    s->mins[s->top] = s->top ? (val < s->mins[s->top-1] ? val : s->mins[s->top-1]) : val;
}
void minStackPop(MinStack* s) { s->top--; }
int minStackTop(MinStack* s) { return s->data[s->top]; }
int minStackGetMin(MinStack* s) { return s->mins[s->top]; }
class MinStack:
    def __init__(self):
        self.st, self.mn = [], []
    def push(self, val: int) -> None:
        self.st.append(val)
        self.mn.append(val if not self.mn else min(self.mn[-1], val))
    def pop(self) -> None:
        self.st.pop(); self.mn.pop()
    def top(self) -> int:
        return self.st[-1]
    def getMin(self) -> int:
        return self.mn[-1]
class MinStack {
    stack<int> st, mn;
public:
    MinStack() {}
    void push(int val) {
        st.push(val);
        mn.push(mn.empty() ? val : min(mn.top(), val));
    }
    void pop() { st.pop(); mn.pop(); }
    int top() { return st.top(); }
    int getMin() { return mn.top(); }
};
