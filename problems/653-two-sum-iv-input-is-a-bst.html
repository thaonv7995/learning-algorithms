bool findTarget(struct TreeNode* r,int k){ int v[10000],n=0; void in(struct TreeNode* x){if(!x)return;in(x->left);v[n++]=x->val;in(x->right);} in(r); int i=0,j=n-1; while(i<j){ int s=v[i]+v[j]; if(s==k)return 1; if(s<k)i++; else j--; } return 0; }
class Solution:
    def findTarget(self, root: Optional[TreeNode], k: int) -> bool:
        vals=[]
        def dfs(n):
            if not n: return
            dfs(n.left); vals.append(n.val); dfs(n.right)
        dfs(root); i,j=0,len(vals)-1
        while i<j:
            s=vals[i]+vals[j]
            if s==k: return True
            if s<k: i+=1
            else: j-=1
        return False
class Solution {
    void in(TreeNode* n, vector<int>& v){ if(!n) return; in(n->left,v); v.push_back(n->val); in(n->right,v);}
public:
    bool findTarget(TreeNode* root, int k) {
        vector<int> v; in(root,v); int i=0,j=v.size()-1;
        while(i<j){ int s=v[i]+v[j]; if(s==k) return true; if(s<k) i++; else j--; }
        return false;
    }
};
