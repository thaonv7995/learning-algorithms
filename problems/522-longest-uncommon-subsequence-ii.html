Longest Uncommon Subsequence II

O(n^2 * L) O(n) 44.9% AC
Mô tả
Cho mảng strs. Trả độ dài subsequence không chung dài nhất xuất hiện trong đúng một chuỗi (không phải subsequence của chuỗi khác).
Ví dụ
Ví dụ 1
Instrs = ["aba","cdc","eae"]
Out3
Ví dụ 2
Instrs = ["aaa","aaa","aa"]
Out-1
Ý tưởng
Sort strs theo length giảm dần. Với mỗi s, kiểm tra có phải subsequence của chuỗi khác không; nếu không → trả len(s).
Mã nguồn
solution.c C99 
/* Sort + subsequence check — xem C++ */
class Solution:
    def findLUSlength(self, strs: List[str]) -> int:
        def subseq(a, b):
            i = 0
            for c in b:
                if i < len(a) and a[i] == c:
                    i += 1
            return i == len(a)
        strs.sort(key=len, reverse=True)
        for i, s in enumerate(strs):
            if all(i == j or not subseq(s, strs[j]) for j in range(len(strs))):
                return len(s)
        return -1
class Solution {
    bool isSub(const string& a, const string& b) {
        int i = 0;
        for (char c : b) if (i < (int)a.size() && a[i] == c) i++;
        return i == (int)a.size();
    }
public:
    int findLUSlength(vector<string>& strs) {
        sort(strs.begin(), strs.end(), [](auto& x, auto& y){ return x.size() > y.size(); });
        for (int i = 0; i < (int)strs.size(); i++) {
            bool ok = true;
            for (int j = 0; j < (int)strs.size(); j++)
                if (i != j && isSub(strs[i], strs[j])) { ok = false; break; }
            if (ok) return strs[i].size();
        }
        return -1;
    }
};
