Add Strings

O(max(m,n)) O(max(m,n)) 52.2% AC
Mô tả
Cộng hai số không âm num1 và num2 (chuỗi chữ số). Trả tổng dạng chuỗi, không dùng BigInteger.
Ví dụ
Ví dụ 1
Innum1 = "11", num2 = "123"
Out"134"
Ví dụ 2
Innum1 = "456", num2 = "77"
Out"533"
Ví dụ 3
Innum1 = "0", num2 = "0"
Out"0"
Ý tưởng
Two pointers từ cuối, cộng digit + carry, build kết quả đảo ngược.

char* addStrings(char* num1, char* num2) {
    int i=strlen(num1)-1, j=strlen(num2)-1, carry=0, cap=2000;
    char* ans=malloc(cap); int p=0;
    while(i>=0||j>=0||carry){
        int d=carry;
        if(i>=0) d+=num1[i--]-'0';
        if(j>=0) d+=num2[j--]-'0';
        ans[p++]='0'+d%10; carry=d/10;
    }
    ans[p]=0; for(int a=0,b=p-1;a<b;a++,b--){ char t=ans[a]; ans[a]=ans[b]; ans[b]=t; }
    return ans;
}
class Solution:
    def addStrings(self, num1: str, num2: str) -> str:
        i, j, carry, ans = len(num1)-1, len(num2)-1, 0, []
        while i >= 0 or j >= 0 or carry:
            d = carry
            if i >= 0: d += ord(num1[i]) - 48; i -= 1
            if j >= 0: d += ord(num2[j]) - 48; j -= 1
            ans.append(chr(d % 10 + 48)); carry = d // 10
        return "".join(reversed(ans))
class Solution {
public:
    string addStrings(string num1, string num2) {

        while(i>=0||j>=0||carry){

            if(j>=0) d+=num2[j--]-'0';
            ans.push_back('0'+d%10); carry=d/10;
        }
        reverse(ans.begin(), ans.end());
        return ans;
