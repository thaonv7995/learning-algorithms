Flood Fill

O(mn) O(mn) 68.3% AC
Mô tả
Ảnh image m×n, pixel (sr,sc) và newColor. Flood fill: đổi màu vùng liên thông 4-hướng cùng màu gốc thành newColor. Trả image sau fill.
Ví dụ
Ví dụ 1
Inimage = [[1,1,1],[1,1,0],[1,0,1]], sr = 1, sc = 1, newColor = 2
Out[[2,2,2],[2,2,0],[2,0,1]]
Ví dụ 2
Inimage = [[0,0,0],[0,0,0]], sr = 0, sc = 0, newColor = 0
Out[[0,0,0],[0,0,0]]
Ý tưởng
DFS/BFS từ (sr,sc): nếu image[r][c]==oldColor thì gán newColor và lan 4 hướng.

void dfs733(int** img,int m,int n,int r,int c,int old,int nw){
    if(r<0||c<0||r>=m||c>=n||img[r][c]!=old) return;
    img[r][c]=nw;
    dfs733(img,m,n,r+1,c,old,nw); dfs733(img,m,n,r-1,c,old,nw);
    dfs733(img,m,n,r,c+1,old,nw); dfs733(img,m,n,r,c-1,old,nw);
}

class Solution:
    def floodFill(self, image: List[List[int]], sr: int, sc: int, newColor: int) -> List[List[int]]:
        old = image[sr][sc]
        if old == newColor: return image
        def dfs(r, c):
            if r < 0 or c < 0 or r >= len(image) or c >= len(image[0]) or image[r][c] != old: return
            image[r][c] = newColor
            dfs(r+1,c); dfs(r-1,c); dfs(r,c+1); dfs(r,c-1)
        dfs(sr, sc)
        return image
class Solution {
    int oldC, newC;
    void dfs(vector<vector<int>>& g, int r, int c) {
        if (r<0||c<0||r>=(int)g.size()||c>=(int)g[0].size()||g[r][c]!=oldC) return;
        g[r][c]=newC;
        dfs(g,r+1,c); dfs(g,r-1,c); dfs(g,r,c+1); dfs(g,r,c-1);
    }
public:
    vector<vector<int>> floodFill(vector<vector<int>>& image, int sr, int sc, int newColor) {
        oldC=image[sr][sc]; newC=newColor;
        if (oldC==newC) return image;
        dfs(image,sr,sc);
        return image;
    }
};
