Number of Boomerangs

O(n^2) O(n) 57.6% AC
Mô tả
Đếm boomerang: bộ ba (i,j,k) distinct sao dist(i,j)==dist(i,k). Mỗi boomerang đếm ordered triple.
Ví dụ
Ví dụ 1

Out2
Ví dụ 2

Out2
Ý tưởng
Với mỗi i, map dist² tới j: cộng ans += cnt*(cnt-1) khi thêm dist trùng.

int numberOfBoomerangs(int** points, int n, int* cs) {
    int ans=0;
    for(int i=0;i<n;i++){
        int cnt[200000]={0};
        for(int j=0;j<n;j++){
            int dx=points[i][0]-points[j][0], dy=points[i][1]-points[j][1];
            int d=dx*dx+dy*dy;
            ans += 2*cnt[d+100000]; cnt[d+100000]++;
        }
    }
    return ans;
}
class Solution:
    def numberOfBoomerangs(self, points: List[List[int]]) -> int:
        ans = 0
        for x1, y1 in points:
            cnt = defaultdict(int)
            for x2, y2 in points:
                d = (x1-x2)**2 + (y1-y2)**2
                cnt[d] += 1
            for c in cnt.values():
                ans += c * (c - 1)
        return ans
class Solution {
public:
    int numberOfBoomerangs(vector<vector<int>>& p) {
        int ans=0;
        for(auto& a:p){
            unordered_map<int,int> cnt;
            for(auto& b:p){
                int d=(a[0]-b[0])*(a[0]-b[0])+(a[1]-b[1])*(a[1]-b[1]);
                ans += 2*cnt[d]; cnt[d]++;
            }
        }
        return ans;
    }
};
