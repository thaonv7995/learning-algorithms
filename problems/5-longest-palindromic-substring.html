Longest Palindromic Substring

O(N²) O(1) 37.9% AC
Mô tả
Cho chuỗi s, trả về chuỗi con palindrome dài nhất trong s.
Ví dụ
Ví dụ 1
Ins = "babad"
Out"bab" hoặc "aba"
Ví dụ 2
Ins = "cbbd"
Out"bb"
Ý tưởng
Expand around center: với mỗi vị trí i, mở rộng palindrome lẻ (i,i) và chẵn (i,i+1) bằng two pointers left/right. Cập nhật chuỗi dài nhất. O(N²) thời gian, O(1) bộ nhớ.
Mã nguồn
solution.c C99 
char* longestPalindrome(char* s) {
    int n = strlen(s);
    if (n < 2) return s;
    int bestL = 0, bestLen = 1;
    for (int i = 0; i < n; i++) {
        int l = i, r = i;
        while (l >= 0 && r < n && s[l] == s[r]) { l--; r++; }
        if (r - l - 1 > bestLen) { bestL = l + 1; bestLen = r - l - 1; }
        l = i; r = i + 1;
        while (l >= 0 && r < n && s[l] == s[r]) { l--; r++; }
        if (r - l - 1 > bestLen) { bestL = l + 1; bestLen = r - l - 1; }
    }
    char* res = (char*)malloc(bestLen + 1);
    strncpy(res, s + bestL, bestLen);
    res[bestLen] = '\0';
    return res;
}
class Solution:
    def longestPalindrome(self, s: str) -> str:
        def expand(l, r):
            while l >= 0 and r < len(s) and s[l] == s[r]:
                l -= 1; r += 1
            return s[l+1:r]
        best = ""
        for i in range(len(s)):
            best = max(best, expand(i, i), expand(i, i+1), key=len)
        return best
class Solution {
public:
    string longestPalindrome(string s) {
        auto expand = [&](int l, int r) {
            while (l >= 0 && r < (int)s.size() && s[l] == s[r]) { l--; r++; }
            return s.substr(l + 1, r - l - 1);
        };
        string best;
        for (int i = 0; i < (int)s.size(); i++) {
            string a = expand(i, i), b = expand(i, i + 1);
            if (a.size() > best.size()) best = a;
            if (b.size() > best.size()) best = b;
        }
        return best;
    }
};
