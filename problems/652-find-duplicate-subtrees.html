Find Duplicate Subtrees

O(n) O(n) 60.7% AC
Mô tả
Tìm mọi subtree trùng cấu trúc (serialize giống nhau) trong cây nhị phân.
Ví dụ
Ví dụ 1
Inroot có hai subtree [2,4] giống nhau
Out[[2,4],[4]]
Ví dụ 2
Inroot = [1,2,3]
Out[]
Ý tưởng
Postorder serialize; map serial→list nodes; trả nodes có serial trùng ≥2.

struct TreeNode** findDuplicateSubtrees652(struct TreeNode* r,int* rs) {
    struct TreeNode** ans=malloc(256*sizeof(*ans)); *rs=0;
    char keys[256][512]; int cnt[256]={0}; int nk=0;
    void ser(struct TreeNode* n, char* s, int* p){
        if(!n){ *p+=sprintf(s+*p,"0,"); return; }
        *p+=sprintf(s+*p,"%d,",n->val); ser(n->left,s,p); ser(n->right,s,p);
    }
    void walk(struct TreeNode* n){
        if(!n) return;
        char s[512]={0}; int p=0; ser(n,s,&p);
        int ki=-1; for(int k=0;k<nk;k++) if(!strcmp(keys[k],s)){ki=k;break;}
        if(ki<0){ ki=nk++; strcpy(keys[ki],s); }
        cnt[ki]++;
        if(cnt[ki]==2) ans[(*rs)++]=n;
        walk(n->left); walk(n->right);
    }
    walk(r); return ans;
}
class Solution:
    def findDuplicateSubtrees(self, root: Optional[TreeNode]) -> List[Optional[TreeNode]]:
        mp=defaultdict(int); ans=[]
        def ser(n):
            if not n: return "0"
            s=f"{n.val},{ser(n.left)},{ser(n.right)}"
            mp[s]+=1
            if mp[s]==2: ans.append(n)
            return s
        ser(root); return ans
class Solution {
    unordered_map<string,int> mp; vector<TreeNode*> ans;
    string ser(TreeNode* n){
        if(!n) return "0";
        string s=to_string(n->val)+","+ser(n->left)+","+ser(n->right);
        if(++mp[s]==2) ans.push_back(n); return s;
    }
public:
    vector<TreeNode*> findDuplicateSubtrees(TreeNode* root){ ser(root); return ans; }
};
