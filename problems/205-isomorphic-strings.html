Isomorphic Strings

O(n) O(1) 48.4% AC
Mô tả
Hai chuỗi s và t isomorphic nếu có thể hoán vị ký tự s → t một-một (thứ tự tương đối giữ nguyên). Kiểm tra hai chuỗi có isomorphic không.
Ví dụ
Ví dụ 1
Ins = "egg", t = "add"
Outtrue
Ví dụ 2
Ins = "foo", t = "bar"
Outfalse
Ví dụ 3
Ins = "paper", t = "title"
Outtrue
Ý tưởng
Hai map s→t và t→s; duyệt từng cặp ký tự: nếu map khác nhau hoặc vi phạm injectivity → false.
Mã nguồn
solution.c C99 
bool isIsomorphic(char* s, char* t) {
    char ms[256] = {0}, mt[256] = {0};
    for (int i = 0; s[i] && t[i]; i++) {
        if (ms[(unsigned char)s[i]] == 0) ms[(unsigned char)s[i]] = t[i];
        if (mt[(unsigned char)t[i]] == 0) mt[(unsigned char)t[i]] = s[i];
        if (ms[(unsigned char)s[i]] != t[i] || mt[(unsigned char)t[i]] != s[i])
            return false;
    }
    return s[0] == 0 && t[0] == 0;
}
class Solution:
    def isIsomorphic(self, s: str, t: str) -> bool:
        if len(s) != len(t):
            return False
        ms, mt = {}, {}
        for a, b in zip(s, t):
            if ms.get(a, b) != b or mt.get(b, a) != a:
                return False
            ms[a] = b
            mt[b] = a
        return True
class Solution {
public:
    bool isIsomorphic(string s, string t) {
        if (s.size() != t.size()) return false;
        char ms[256] = {}, mt[256] = {};
        for (int i = 0; i < (int)s.size(); i++) {
            if (ms[(unsigned char)s[i]] == 0) ms[(unsigned char)s[i]] = t[i];
            if (mt[(unsigned char)t[i]] == 0) mt[(unsigned char)t[i]] = s[i];
            if (ms[(unsigned char)s[i]] != t[i] || mt[(unsigned char)t[i]] != s[i])
                return false;
        }
        return true;
    }
};
