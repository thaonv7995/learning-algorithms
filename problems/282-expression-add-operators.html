Expression Add Operators

O(4^n) O(n) 43.2% AC
Mô tả
Cho chuỗi num chỉ chữ số và target. Chèn +, -, hoặc * giữa các chữ số (không đổi thứ tự) để biểu thức = target. Trả mọi biểu thức hợp lệ.
Ví dụ
Ví dụ 1
Innum = "123", target = 6
Out["1+2+3", "1*2*3"]
Ví dụ 2
Innum = "232", target = 8
Out["2*3+2", "2+3*2"]
Ví dụ 3
Innum = "105", target = 5
Out["1*0+5","10-5"]
Ý tưởng
Backtracking: track val (tổng signed) và last (operand cuối cho *); * cập nhật val-last+last*num mới.

char** addOperators(char* num, int target, int* rs) {
    char** out = malloc(8 * sizeof(char*)); int osz = 0, ocap = 8;
    dfs282(num, target, 0, "", 0, 0, 0, &out, &osz, &ocap);
    *rs = osz; return out;
}
class Solution:
    def addOperators(self, num: str, target: int) -> List[str]:
        res = []
        def dfs(i, path, val, last):
            if i == len(num):
                if val == target:
                    res.append(path)
                return
            for j in range(i, len(num)):
                if j > i and num[i] == '0':
                    break
                cur = int(num[i:j+1])
                if i == 0:
                    dfs(j+1, str(cur), cur, cur)
                else:
                    dfs(j+1, path+'+'+str(cur), val+cur, cur)
                    dfs(j+1, path+'-'+str(cur), val-cur, -cur)
                    dfs(j+1, path+'*'+str(cur), val-last+last*cur, last*cur)
        dfs(0, '', 0, 0)
        return res
class Solution {
    vector<string> res;
    void dfs(string& num, int target, int i, string path, long val, long last) {
        if (i == (int)num.size()) {
            if (val == target) res.push_back(path);
            return;
        }
        for (int j = i; j < (int)num.size(); j++) {
            if (j > i && num[i] == '0') break;
            long cur = stol(num.substr(i, j - i + 1));
            if (i == 0) dfs(num, target, j + 1, to_string(cur), cur, cur);
            else {
                dfs(num, target, j + 1, path + '+' + to_string(cur), val + cur, cur);
                dfs(num, target, j + 1, path + '-' + to_string(cur), val - cur, -cur);
                dfs(num, target, j + 1, path + '*' + to_string(cur), val - last + last * cur, last * cur);
            }
        }
    }
public:
    vector<string> addOperators(string num, int target) {
        dfs(num, target, 0, "", 0, 0);
        return res;
    }
};
