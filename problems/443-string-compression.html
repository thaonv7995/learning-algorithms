String Compression

O(n) O(1) 60% AC
Mô tả
Nén chars in-place: nhóm ký tự liên tiếp thành ký tự + số lần (nếu >1). Trả độ dài mới.
Ví dụ
Ví dụ 1
Inchars = ["a","a","b","b","c","c","c"]
Out6, chars = ["a","2","b","2","c","3"]
Ví dụ 2
Inchars = ["a"]
Out1
Ý tưởng
Two pointers write: read group, write char rồi count string nếu len>1.
Mã nguồn
solution.c C99 
int compress(char* chars, int n) {
    int w=0,i=0;
    while(i<n){
        char ch=chars[i]; int j=i;
        while(j<n&&chars[j]==ch) j++;
        chars[w++]=ch; int cnt=j-i;
        if(cnt>1){
            char buf[12]; sprintf(buf,"%d",cnt);
            for(int k=0;buf[k];k++) chars[w++]=buf[k];
        }
        i=j;
    }
    return w;
}
class Solution:
    def compress(self, chars: List[str]) -> int:
        w = i = 0
        while i < len(chars):
            ch = chars[i]; j = i
            while j < len(chars) and chars[j] == ch: j += 1
            chars[w] = ch; w += 1
            cnt = j - i
            if cnt > 1:
                for c in str(cnt): chars[w] = c; w += 1
            i = j
        return w
class Solution {
public:
    int compress(vector<char>& c) {
        int w=0, i=0, n=c.size();
        while(i<n){
            char ch=c[i]; int j=i;
            while(j<n&&c[j]==ch) j++;
            c[w++]=ch; int cnt=j-i;
            if(cnt>1){ string s=to_string(cnt); for(char x:s) c[w++]=x; }
            i=j;
        }
        return w;
    }
};
