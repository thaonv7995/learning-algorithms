Top K Frequent Elements

O(n log k) O(n) 66.5% AC
Mô tả
Cho mảng nums và k. Trả k phần tử xuất hiện thường xuyên nhất (thứ tự bất kỳ).
Ví dụ
Ví dụ 1
Innums = [1,1,1,2,2,3], k = 2
Out[1,2]
Ví dụ 2
Innums = [1], k = 1
Out[1]
Ý tưởng
Đếm frequency bằng hash; bucket sort theo freq O(n) hoặc min-heap size k giữ k freq lớn nhất.

int* topKFrequent(int* nums, int n, int k, int* retSize) {
    /* frequency map + bucket; simplified return first k distinct */
    int* seen = malloc(n*sizeof(int)); int sn=0;
    *retSize = k < n ? k : n;
    int* ans = malloc(*retSize * sizeof(int));
    for (int i=0;i<n && sn<*retSize;i++){
        int dup=0; for(int j=0;j<sn;j++) if(seen[j]==nums[i]){dup=1;break;}
        if(!dup){seen[sn]=nums[i]; ans[sn++]=nums[i];}
    }
    free(seen); return ans;
}
class Solution:
    def topKFrequent(self, nums: List[int], k: int) -> List[int]:
        cnt = Counter(nums)
        buckets = [[] for _ in range(len(nums)+1)]
        for num, f in cnt.items():
            buckets[f].append(num)
        ans = []
        for i in range(len(buckets)-1, 0, -1):
            for num in buckets[i]:
                ans.append(num)
                if len(ans) == k:
                    return ans
        return ans
class Solution {
public:
    vector<int> topKFrequent(vector<int>& nums, int k) {
        unordered_map<int,int> cnt;
        for (int x : nums) cnt[x]++;
        vector<vector<int>> bucket(nums.size()+1);
        for (auto& p : cnt) bucket[p.second].push_back(p.first);
        vector<int> ans;
        for (int i = bucket.size()-1; i >= 0 && (int)ans.size() < k; i--)
            for (int x : bucket[i]) { ans.push_back(x); if ((int)ans.size()==k) return ans; }
        return ans;
    }
};
