static int shopDfs(int* price,int pn,int** sp,int sn,int* need,int w,int* memo,int key){
    int base=0; for(int i=0;i<w;i++) base+=price[i]*need[i];
    int ans=base;
    for(int s=0;s<sn;s++){
        int ok=1,nxt[16]; for(int i=0;i<w;i++){ nxt[i]=need[i]-sp[s][i]; if(nxt[i]<0) ok=0; }
        if(!ok) continue;
        int sub=shopDfs(price,pn,sp,sn,nxt,w,memo,0);
        if(sp[s][w]+sub<ans) ans=sp[s][w]+sub;
    }
    return ans;
}

class Solution:
    def shoppingOffers(self, price: List[int], special: List[List[int]], needs: List[int]) -> int:
        memo={}
        def dfs(need):
            key=tuple(need)
            if key in memo: return memo[key]
            ans=sum(p*n for p,n in zip(price,need))
            for sp in special:
                nxt=[need[i]-sp[i] for i in range(len(need))]
                if min(nxt)>=0: ans=min(ans,sp[-1]+dfs(nxt))
            memo[key]=ans; return ans
        return dfs(needs)
class Solution {
public:
    int shoppingOffers(vector<int>& price, vector<vector<int>>& special, vector<int>& needs) {
        map<vector<int>,int> memo;
        function<int(vector<int>&)> dfs=[&](vector<int>& need)->int{
            if(memo.count(need)) return memo[need];
            int ans=0; for(int i=0;i<(int)price.size();i++) ans+=price[i]*need[i];
            for(auto& sp:special){
                vector<int> nxt=need; bool ok=1;
                for(int i=0;i<(int)need.size();i++){ nxt[i]-=sp[i]; if(nxt[i]<0) ok=0; }
                if(ok) ans=min(ans, sp.back()+dfs(nxt));
            }
            return memo[need]=ans;
        };
        return dfs(needs);
    }
};
