Perfect Squares

O(n^{3/2}) O(n) 56.5% AC
Mô tả
Cho số nguyên n. Trả số lượng số hoàn hảo (square) nhỏ nhất sao cho tổng bằng n (Lagrange: tối đa 4, đề cho DP/BFS).
Ví dụ
Ví dụ 1
Inn = 12
Out3
Ví dụ 2
Inn = 13
Out2
Ý tưởng
DP: dp[i]=min(dp[i-j*j]+1) với j*j<=i; dp[0]=0. Hoặc BFS layer trên i trừ square.
Mã nguồn
solution.c C99 
int numSquares(int n) {
    int* dp = malloc((n + 1) * sizeof(int));
    for (int i = 0; i <= n; i++) dp[i] = 2147483647;
    dp[0] = 0;
    for (int i = 1; i <= n; i++)
        for (int j = 1; j * j <= i; j++) {
            int v = dp[i - j * j] + 1;
            if (v < dp[i]) dp[i] = v;
        }
    int ans = dp[n];
    free(dp);
    return ans;
}
class Solution:
    def numSquares(self, n: int) -> int:
        dp = [0] + [10**9] * n
        for i in range(1, n + 1):
            j = 1
            while j * j <= i:
                dp[i] = min(dp[i], dp[i - j * j] + 1)
                j += 1
        return dp[n]
class Solution {
public:
    int numSquares(int n) {
        vector<int> dp(n + 1, INT_MAX);
        dp[0] = 0;
        for (int i = 1; i <= n; i++)
            for (int j = 1; j * j <= i; j++)
                dp[i] = min(dp[i], dp[i - j * j] + 1);
        return dp[n];
    }
};
