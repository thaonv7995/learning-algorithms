Zigzag Conversion

O(N) O(N) 54.2% AC
Mô tả
Cho chuỗi s và số hàng numRows, sắp xếp s theo zigzag trên numRows hàng rồi đọc từng hàng nối lại.
Ví dụ
Ví dụ 1
Ins = "PAYPALISHIRING", numRows = 3
Out"PAHNAPLSIIGYIR"
Ví dụ 2
Ins = "PAYPALISHIRING", numRows = 4
Out"PINALSIGYAHRPI"
Ý tưởng
Dùng mảng rows[numRows] làm bucket. Duyệt từng ký tự, gán vào row hiện tại, đảo chiều khi chạm hàng 0 hoặc numRows-1. Cuối cùng nối các bucket.
Mã nguồn
solution.c C99 
char* convert(char* s, int numRows) {
    if (numRows == 1) return s;
    int n = strlen(s), r = 0, dir = 1;
    char** rows = calloc(numRows, sizeof(char*));
    for (int i = 0; i < numRows; i++) rows[i] = calloc(n + 1, 1);
    for (int i = 0; s[i]; i++) {
        rows[r][strlen(rows[r])] = s[i];
        if (r == 0) dir = 1; else if (r == numRows - 1) dir = -1;
        r += dir;
    }
    char* out = malloc(n + 1); out[0] = 0;
    for (int i = 0; i < numRows; i++) strcat(out, rows[i]);
    return out;
}
class Solution:
    def convert(self, s: str, numRows: int) -> str:
        if numRows == 1: return s
        rows = [''] * numRows
        r, d = 0, 1
        for c in s:
            rows[r] += c
            if r == 0: d = 1
            elif r == numRows - 1: d = -1
            r += d
        return ''.join(rows)
class Solution {
public:
    string convert(string s, int numRows) {
        if (numRows == 1) return s;
        vector<string> rows(numRows);
        int r = 0, d = 1;
        for (char c : s) {
            rows[r] += c;
            if (r == 0) d = 1; else if (r == numRows - 1) d = -1;
            r += d;
        }
        string out;
        for (auto& row : rows) out += row;
        return out;
    }
};
