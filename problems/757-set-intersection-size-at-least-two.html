Set Intersection Size At Least Two

O(n log n) O(n) 58.1% AC
Mô tả
Cho mảng intervals[i]=[start,end]. Chọn tối thiểu số điểm sao mỗi interval chứa ít nhất 2 điểm đã chọn (có thể trùng interval). Trả số điểm tối thiểu.
Ví dụ
Ví dụ 1
Inintervals = [[1,3],[1,4],[2,5],[3,5]]
Out2
Ví dụ 2
Inintervals = [[1,2],[2,3],[2,3],[3,4],[3,4]]
Out2
Ví dụ 3
Inintervals = [[1,2],[2,3],[2,3],[4,5],[4,5]]
Out2
Ý tưởng
Sort theo end. Greedy: với mỗi interval thiếu điểm, thêm end và end-1 (hoặc end và end nếu overlap).

int intersectionSizeTwo(int** intervals, int n, int* cs) {
    int ans[2000], an=0;
    for(int i=0;i<n;i++){
        int s=intervals[i][0], e=intervals[i][1], cnt=0;
        for(int j=0;j<an;j++) if(s<=ans[j]&&ans[j]<=e) cnt++;
        if(cnt>=2) continue;
        if(cnt==0){ ans[an++]=e-1; ans[an++]=e; }
        else ans[an++]=(ans[an-1]==e?e-1:e);
    }
    return an;
}
class Solution:
    def intersectionSizeTwo(self, intervals: List[List[int]]) -> int:
        intervals.sort(key=lambda x: x[1])
        ans = []
        for s, e in intervals:
            have = [x for x in ans if s <= x <= e]
            if len(have) >= 2: continue
            if not have:
                ans.extend([e-1, e])
            elif len(have) == 1:
                ans.append(e if have[0] != e else e-1)
        return len(set(ans))
class Solution {
public:
    int intersectionSizeTwo(vector<vector<int>>& a) {
        sort(a.begin(), a.end(), [](auto& x, auto& y){ return x[1]<y[1]; });
        vector<int> ans;
        for(auto& iv: a){
            int s=iv[0], e=iv[1], cnt=0;
            for(int x: ans) if(s<=x && x<=e) cnt++;
            if(cnt>=2) continue;
            if(cnt==0){ ans.push_back(e-1); ans.push_back(e); }
            else ans.push_back(ans.back()==e?e-1:e);
        }
        sort(ans.begin(), ans.end());
        ans.erase(unique(ans.begin(), ans.end()), ans.end());
        return ans.size();
    }
};
