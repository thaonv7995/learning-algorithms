Repeated DNA Sequences

O(n) O(n) 53.3% AC
Mô tả
Chuỗi s gồm A,C,G,T. Tìm mọi substring dài 10 ký tự xuất hiện hơn một lần trong s. Trả danh sách (thứ tự bất kỳ).
Ví dụ
Ví dụ 1
Ins = "AAAAACCCCCAAAAACCCCCCAAAAAGGGTTT"
Out["AAAAACCCCC","CCCCCAAAAA"]
Ví dụ 2
Ins = "AAAAAAAAAAAAA"
Out["AAAAAAAAAA"]
Ý tưởng
Sliding window 10: hash set seen và repeated. Duyệt mọi substr 10 ký tự; lần 2 gặp thì thêm vào kết quả (dùng set tránh trùng output).
Mã nguồn
solution.c C99 
char** findRepeatedDnaSequences(char* s, int* retSize) {
    *retSize = 0;
    int n = (int)strlen(s);
    if (n < 11) return NULL;
    char** out = NULL; int cap = 0;
    /* simple O(n^2) check with fixed small alphabet */
    for (int i = 0; i + 10 <= n; i++) {
        int cnt = 0;
        for (int j = 0; j + 10 <= n; j++) if (j != i && strncmp(s+i, s+j, 10) == 0) { cnt = 1; break; }
        if (!cnt) continue;
        int dup = 0;
        for (int k = 0; k < *retSize; k++) if (strncmp(out[k], s+i, 10) == 0) { dup = 1; break; }
        if (dup) continue;
        if (*retSize >= cap) { cap = cap ? cap*2 : 4; out = realloc(out, cap*sizeof(char*)); }
        out[(*retSize)++] = strndup(s+i, 10);
    }
    return out;
}
class Solution:
    def findRepeatedDnaSequences(self, s: str) -> List[str]:
        seen, rep = set(), set()
        for i in range(len(s) - 9):
            sub = s[i:i+10]
            if sub in seen:
                rep.add(sub)
            else:
                seen.add(sub)
        return list(rep)
class Solution {
public:
    vector<string> findRepeatedDnaSequences(string s) {
        unordered_set<string> seen, rep;
        vector<string> res;
        for (int i = 0; i + 10 <= (int)s.size(); i++) {
            string sub = s.substr(i, 10);
            if (seen.count(sub)) rep.insert(sub);
            else seen.insert(sub);
        }
        for (auto& x : rep) res.push_back(x);
        return res;
    }
};
