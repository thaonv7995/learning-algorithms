Validate Binary Search Tree

O(n) O(h) 35.8% AC
Mô tả
Cho root cây nhị phân. Kiểm tra cây có phải BST hợp lệ không: mọi nút trái < root < mọi nút phải, đệ quy trên toàn cây.
Ví dụ
Ví dụ 1
Inroot = [2,1,3]
Outtrue
Ví dụ 2
Inroot = [5,1,4,null,null,3,6]
Outfalse
Ý tưởng
DFS với khoảng (lo, hi): node.val phải trong (lo, hi); kiểm tra left (lo, val) và right (val, hi). Hoặc inorder tăng strict.
Mã nguồn
solution.c C99 
/* isValidBST — DFS bounds; xem C++ */
class Solution:
    def isValidBST(self, root: Optional[TreeNode]) -> bool:
        def dfs(node, lo, hi):
            if not node: return True
            if not (lo < node.val < hi): return False
            return dfs(node.left, lo, node.val) and dfs(node.right, node.val, hi)
        return dfs(root, float('-inf'), float('inf'))
class Solution {
    bool dfs(TreeNode* n, long lo, long hi) {
        if (!n) return true;
        if (n->val <= lo || n->val >= hi) return false;
        return dfs(n->left, lo, n->val) && dfs(n->right, n->val, hi);
    }
public:
    bool isValidBST(TreeNode* root) {
        return dfs(root, LONG_MIN, LONG_MAX);
    }
};
