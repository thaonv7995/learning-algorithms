Remove Comments

O(n) O(n) 40.2% AC
Mô tả
Cho mảng chuỗi source mô tả code C/C++. Trả mảng code sau khi xóa comment: // đến hết dòng, /* ... */ có thể nhiều dòng. Giữ xuống dòng ngoài comment.
Ví dụ
Ví dụ 1
Insource = ["/*Test program */", "int main()", "{ ", " // variable declaration ", "int a, b, c;", "/* This is a test", " multiline ", " comment for ", " testing */", "a = b + c;", "}"]
Out["int main()","{ "," ","int a, b, c;","a = b + c;","}"]
Ví dụ 2
Insource = ["a/*comment", "line", "more_comment*/b"]
Out["ab"]
Ý tưởng
Duyệt từng dòng ký tự: state inCode/blockComment/lineComment. blockComment /* */; lineComment // đến \n.
Mã nguồn
solution.c C99 
char** removeComments(char** source, int n, int* retSize) {
    char** ans = malloc(n * sizeof(char*)); *retSize = 0;
    char buf[10000]; int bi = 0, block = 0;
    for (int t = 0; t < n; t++) {
        char* line = source[t];
        for (int i = 0; line[i]; ) {
            if (block) {
                char* p = strstr(line + i, "*/");
                if (!p) break;
                i = (int)(p - line) + 2; block = 0;
            } else if (line[i]=='/' && line[i+1]=='/') break;
            else if (line[i]=='/' && line[i+1]=='*') { block=1; i+=2; }
            else buf[bi++] = line[i++];
        }
        if (!block) { buf[bi]=0; ans[(*retSize)++]=strdup(buf); bi=0; }
    }
    return ans;
}
class Solution:
    def removeComments(self, source: List[str]) -> List[str]:
        res, buf, block = [], [], False
        for line in source:
            i, n = 0, len(line)
            while i < n:
                if block:
                    j = line.find("*/", i)
                    if j == -1: i = n; break
                    i = j + 2; block = False
                elif i + 1 < n and line[i:i+2] == "//": break
                elif i + 1 < n and line[i:i+2] == "/*": block = True; i += 2
                else: buf.append(line[i]); i += 1
            if not block:
                res.append("".join(buf)); buf = []
        return res
class Solution {
public:
    vector<string> removeComments(vector<string>& src) {
        vector<string> ans;
        string cur;
        bool block = false;
        for (auto& line : src) {
            for (int i = 0; i < (int)line.size(); ) {
                if (block) {
                    auto p = line.find("*/", i);
                    if (p == string::npos) break;
                    i = p + 2; block = false;
                } else if (i+1<(int)line.size() && line[i]=='/' && line[i+1]=='/') break;
                else if (i+1<(int)line.size() && line[i]=='/' && line[i+1]=='*') { block=true; i+=2; }
                else cur.push_back(line[i++]);
            }
            if (!block) { ans.push_back(cur); cur.clear(); }
        }
        return ans;
    }
};
