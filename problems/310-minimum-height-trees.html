Minimum Height Trees

O(n) O(n) 42.6% AC
Mô tả
Cho cây vô hướng n nút (n-1 cạnh). Root tại bất kỳ nút nào tạo cây có chiều cao. Trả về mọi root cho chiều cao cây nhỏ nhất (MHT).
Ví dụ
Ví dụ 1
Inn = 4, edges = [[1,0],[1,2],[1,3]]
Out[1]
Ví dụ 2
Inn = 6, edges = [[3,0],[3,1],[3,2],[3,4],[5,4]]
Out[3,4]
Ý tưởng
Topological peel: bắt đầu từ lá (degree=1), lột dần từng lớp. Nút còn lại cuối (≤2) là tâm cây — root MHT.

int* findMinHeightTrees(int n, int** edges, int edgesSize, int* retSize) {
    if (n <= 2) { int* r = malloc(n*sizeof(int)); for (int i=0;i<n;i++) r[i]=i; *retSize=n; return r; }
    int* deg = calloc(n, sizeof(int));
    int** adj = calloc(n, sizeof(int*)); int* alen = calloc(n, sizeof(int));
    for (int i = 0; i < edgesSize; i++) {
        int u=edges[i][0], v=edges[i][1]; deg[u]++; deg[v]++;
        adj[u] = realloc(adj[u], deg[u]*sizeof(int)); adj[u][alen[u]++]=v;
        adj[v] = realloc(adj[v], deg[v]*sizeof(int)); adj[v][alen[v]++]=u;
    }
    int* q = malloc(n*sizeof(int)), qs=0, rem=n;
    for (int i=0;i<n;i++) if (deg[i]==1) q[qs++]=i;
    while (rem > 2) {
        int sz=qs; rem-=sz; int nqs=0;
        for (int t=0;t<sz;t++) { int u=q[t]; for (int j=0;j<alen[u];j++) { int v=adj[u][j]; if (--deg[v]==1) q[nqs++]=v; } }
        qs=nqs;
    }
    *retSize=qs; return q;
}
class Solution:
    def findMinHeightTrees(self, n: int, edges: List[List[int]]) -> List[int]:
        if n <= 2:
            return list(range(n))
        adj = [[] for _ in range(n)]
        deg = [0]*n
        for u, v in edges:
            adj[u].append(v); adj[v].append(u)
            deg[u] += 1; deg[v] += 1
        q = [i for i in range(n) if deg[i] == 1]
        rem = n
        while rem > 2:
            rem -= len(q)
            nxt = []
            for u in q:
                for v in adj[u]:
                    deg[v] -= 1
                    if deg[v] == 1:
                        nxt.append(v)
            q = nxt
        return q
class Solution {
public:
    vector<int> findMinHeightTrees(int n, vector<vector<int>>& edges) {
        if (n <= 2) { vector<int> r(n); iota(r.begin(), r.end(), 0); return r; }
        vector<vector<int>> adj(n);
        vector<int> deg(n, 0);
        for (auto& e : edges) {
            adj[e[0]].push_back(e[1]);
            adj[e[1]].push_back(e[0]);
            deg[e[0]]++; deg[e[1]]++;
        }
        queue<int> q;
        for (int i = 0; i < n; i++) if (deg[i] == 1) q.push(i);
        int rem = n;
        while (rem > 2) {
            int sz = q.size(); rem -= sz;
            for (int t = 0; t < sz; t++) {
                int u = q.front(); q.pop();
                for (int v : adj[u]) if (--deg[v] == 1) q.push(v);
            }
        }
        vector<int> ans;
        while (!q.empty()) { ans.push_back(q.front()); q.pop(); }
        return ans;
    }
};
