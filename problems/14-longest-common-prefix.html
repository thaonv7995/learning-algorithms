Longest Common Prefix

O(S) O(1) 47.6% AC
Mô tả
Cho mảng chuỗi strs, trả về tiền tố chung dài nhất.
Ví dụ
Ví dụ 1
Instrs = ["flower","flow","flight"]
Out"fl"
Ví dụ 2
Instrs = ["dog","racecar","car"]
Out""
Ý tưởng
Lấy strs[0] làm prefix, lặp i=1..n: rút ngắn prefix cho đến khi strs[i].startsWith(prefix).
Mã nguồn
solution.c C99 
char* longestCommonPrefix(char** strs, int strsSize) {
    if (!strsSize) return "";
    char* pre = strdup(strs[0]);
    for (int i = 1; i < strsSize; i++) {
        int j = 0;
        while (pre[j] && strs[i][j] && pre[j] == strs[i][j]) j++;
        pre[j] = 0;
        if (!pre[0]) break;
    }
    return pre;
}
class Solution:
    def longestCommonPrefix(self, strs: List[str]) -> str:
        if not strs: return ""
        pre = strs[0]
        for s in strs[1:]:
            while not s.startswith(pre):
                pre = pre[:-1]
                if not pre: return ""
        return pre
class Solution {
public:
    string longestCommonPrefix(vector<string>& strs) {
        if (strs.empty()) return "";
        string pre = strs[0];
        for (int i = 1; i < (int)strs.size(); i++) {
            while (strs[i].find(pre) != 0) {
                pre.pop_back();
                if (pre.empty()) return "";
            }
        }
        return pre;
    }
};
