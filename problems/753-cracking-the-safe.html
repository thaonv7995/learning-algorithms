Cracking the Safe

O(k^n) O(k^n) 58.4% AC
Mô tả
Safe n chữ số k-ary (0..k-1). Mỗi lần nhập một chữ số, mọi chuỗi con dài n xuất hiện đúng một lần trong output. Trả chuỗi ngắn nhất (De Bruijn).
Ví dụ
Ví dụ 1
Inn = 1, k = 2
Out"01"
Ví dụ 2
Inn = 2, k = 2
Out"01100"
Ví dụ 3
Inn = 2, k = 3
Out"011122233020010012021"
Ý tưởng
Eulerian path trên De Bruijn graph: edge từ node (n-1)-gram tới node bằng shift+new digit. Hierholzer DFS.

static char seen753[100000][16]; static int sn753;
static int has753(char* e){ for(int i=0;i<sn753;i++) if(strcmp(seen753[i],e)==0) return 1; return 0; }
static void add753(char* e){ if(!has753(e)) strcpy(seen753[sn753++],e); }
static void dfs753(char* node, int n, int k, char* ans, int* ap){
    for(int d=0; d<k; d++){
        char edge[16]; snprintf(edge,16,"%s%d",node,d);
        if(has753(edge)) continue; add753(edge);
        char nxt[16]; strcpy(nxt,edge+1);
        dfs753(nxt,n,k,ans,ap);
        ans[(*ap)++]='0'+d;
    }
}
char* crackSafe(int n, int k, int* retSize) {
    if(n==1){ char* s=malloc(k+1); for(int i=0;i<k;i++) s[i]='0'+i; s[k]=0; *retSize=k; return s; }
    char start[16]; for(int i=0;i<n-1;i++) start[i]='0'; start[n-1]=0;
    char* ans=malloc(100000); int ap=0; sn753=0; dfs753(start,n,k,ans,&ap); ans[ap]=0;
    char* out=malloc(n+ap+1); strcpy(out,start); strcat(out,ans); *retSize=strlen(out); free(ans); return out;
}
class Solution:
    def crackSafe(self, n: int, k: int) -> str:
        if n == 1: return "".join(str(i) for i in range(k))
        seen = set()
        ans = []
        def dfs(node):
            for d in map(str, range(k)):
                edge = node + d
                if edge in seen: continue
                seen.add(edge)
                dfs(edge[1:])
                ans.append(d)
        start = "0" * (n-1)
        dfs(start)
        return start + "".join(ans)
class Solution {
public:
    string crackSafe(int n, int k) {
        if(n==1){ string s; for(int i=0;i<k;i++) s+=char('0'+i); return s; }
        unordered_set<string> seen;
        string ans, start(n-1,'0');
        function<void(string)> dfs = [&](string node){
            for(int d=0;d<k;d++){
                string edge=node+char('0'+d);
                if(seen.count(edge)) continue;
                seen.insert(edge);
                dfs(edge.substr(1));
                ans.push_back(char('0'+d));
            }
        };
        dfs(start);
        return start+ans;
    }
};
