Word Ladder II

O(n · m · 26) O(n · m) 27.7% AC
Mô tả
Cho beginWord, endWord và wordList. Mỗi bước đổi đúng một chữ cái thành từ trong list. Trả về tất cả chuỗi biến đổi ngắn nhất từ beginWord đến endWord (mỗi chuỗi là danh sách từ).
Ví dụ
Ví dụ 1
InbeginWord = "hit", endWord = "cog", wordList = ["hot","dot","dog","lot","log","cog"]
Out[["hit","hot","dot","dog","cog"],["hit","hot","lot","log","cog"]]
Ví dụ 2
InbeginWord = "hit", endWord = "cog", wordList = ["hot","dot","dog","lot","log"]
Out[]
Ý tưởng
BFS tầng để gán dist cho mọi từ đạt được. Xây đồ thị: u→v nếu dist[v]=dist[u]+1 và khác 1 chữ. DFS/backtrack từ beginWord theo đồ thị để thu mọi đường đến endWord.

/* BFS dist + DFS backtrack — xem C++ */
class Solution:
    def findLadders(self, beginWord: str, endWord: str, wordList: List[str]) -> List[List[str]]:
        words = set(wordList)
        if endWord not in words: return []
        from collections import deque, defaultdict
        dist = {beginWord: 0}
        q = deque([beginWord])
        found = False
        while q and not found:
            for _ in range(len(q)):
                w = q.popleft()
                for i in range(len(w)):
                    for c in 'abcdefghijklmnopqrstuvwxyz':
                        nw = w[:i] + c + w[i+1:]
                        if nw not in words or nw in dist: continue
                        dist[nw] = dist[w] + 1
                        q.append(nw)
                        if nw == endWord: found = True
        if endWord not in dist: return []
        adj = defaultdict(list)
        for w, d in dist.items():
            if w == endWord: continue
            for i in range(len(w)):
                for c in 'abcdefghijklmnopqrstuvwxyz':
                    nw = w[:i] + c + w[i+1:]
                    if nw in dist and dist[nw] == d + 1:
                        adj[w].append(nw)
        res = []
        def dfs(w, path):
            if w == endWord:
                res.append(path[:])
                return
            for nxt in adj[w]:
                path.append(nxt)
                dfs(nxt, path)
                path.pop()
        dfs(beginWord, [beginWord])
        return res
class Solution {
    bool diff1(const string& a, const string& b) {
        int c = 0;
        for (int i = 0; i < (int)a.size(); i++) if (a[i] != b[i]) c++;
        return c == 1;
    }
public:
    vector<vector<string>> findLadders(string begin, string end, vector<string>& wl) {
        unordered_set<string> words(wl.begin(), wl.end());
        if (!words.count(end)) return {};
        unordered_map<string, int> dist;
        dist[begin] = 0;
        queue<string> q;
        q.push(begin);
        bool found = false;
        while (q.size() && !found) {
            int sz = q.size();
            while (sz--) {
                string w = q.front(); q.pop();
                for (int i = 0; i < (int)w.size(); i++) {
                    for (char c = 'a'; c <= 'z'; c++) {
                        string nw = w; nw[i] = c;
                        if (!words.count(nw) || dist.count(nw)) continue;
                        dist[nw] = dist[w] + 1;
                        q.push(nw);
                        if (nw == end) found = true;
                    }
                }
            }
        }
        if (!dist.count(end)) return {};
        unordered_map<string, vector<string>> adj;
        for (auto& [w, d] : dist) {
            if (w == end) continue;
            for (int i = 0; i < (int)w.size(); i++) {
                for (char c = 'a'; c <= 'z'; c++) {
                    string nw = w; nw[i] = c;
                    if (dist.count(nw) && dist[nw] == d + 1) adj[w].push_back(nw);
                }
            }
        }
        vector<vector<string>> res;
        vector<string> path = {begin};
        function<void(string&)> dfs = [&](string& w) {
            if (w == end) { res.push_back(path); return; }
            for (auto& nxt : adj[w]) {
                path.push_back(nxt);
                dfs(nxt);
                path.pop_back();
            }
        };
        dfs(begin);
        return res;
    }
};
