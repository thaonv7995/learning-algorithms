Shuffle an Array

O(n) reset/shuffle O(n) 59.7% AC
Mô tả
Thiết kế Solution nhận mảng nums: reset trả bản gốc, shuffle trả hoán vị ngẫu nhiên đều (Fisher-Yates).
Ví dụ
Ví dụ 1
Innums = [1,2,3], reset(), shuffle()
Out[1,2,3], random permutation
Ví dụ 2
Innums = [1], reset(), shuffle()
Out[1], [1]
Ý tưởng
Lưu bản gốc orig; reset copy orig; shuffle Fisher-Yates trên bản copy từ cuối về đầu swap random j≤i.

typedef struct { int* orig; int* arr; int n; } Solution;
Solution* solutionCreate(int* nums, int n) {
    Solution* s = malloc(sizeof(Solution)); s->n = n;
    s->orig = malloc(n*4); s->arr = malloc(n*4);
    memcpy(s->orig, nums, n*4); memcpy(s->arr, nums, n*4);
    return s;
}
int* solutionReset(Solution* s, int* rs) { memcpy(s->arr, s->orig, s->n*4); *rs=s->n; return s->arr; }
int* solutionShuffle(Solution* s, int* rs) {
    for (int i=s->n-1;i>0;i--){ int j=rand()%(i+1); int t=s->arr[i]; s->arr[i]=s->arr[j]; s->arr[j]=t; }
    *rs=s->n; return s->arr;
}
class Solution:
    def __init__(self, nums: List[int]):
        self.orig = nums[:]
        self.arr = nums[:]
    def reset(self) -> List[int]:
        self.arr = self.orig[:]
        return self.arr
    def shuffle(self) -> List[int]:
        a = self.arr[:]
        for i in range(len(a)-1, 0, -1):
            j = random.randint(0, i)
            a[i], a[j] = a[j], a[i]
        self.arr = a
        return a
class Solution {
    vector<int> orig, arr;
public:
    Solution(vector<int>& nums) : orig(nums), arr(nums) {}
    vector<int> reset() { arr = orig; return arr; }
    vector<int> shuffle() {
        for (int i = arr.size()-1; i > 0; i--) {
            int j = rand() % (i+1);
            swap(arr[i], arr[j]);
        }
        return arr;
    }
};
