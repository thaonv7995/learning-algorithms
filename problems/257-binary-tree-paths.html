Binary Tree Paths

O(n) O(h) 68.7% AC
Mô tả
Cho binary tree root. Trả mọi đường đi từ root đến leaf dạng chuỗi "a->b->c" (theo thứ tự bất kỳ).
Ví dụ
Ví dụ 1
Inroot = [1,2,3,null,5]
Out["1->2->5","1->3"]
Ví dụ 2
Inroot = [1]
Out["1"]
Ý tưởng
DFS: path string tích lũy; tại leaf push path; else đệ quy left/right với path + "->" + val.
Mã nguồn
solution.c C99 
void collectPaths(struct TreeNode* node, char* path, int len, char*** out, int* sz, int* cap) {
    if (!node) return;
    len += snprintf(path + len, 32, "%s%d", len ? "->" : "", node->val);
    if (!node->left && !node->right) {
        if (*sz >= *cap) { *cap *= 2; *out = realloc(*out, *cap * sizeof(char*)); }
        (*out)[(*sz)++] = strdup(path);
        return;
    }
    collectPaths(node->left, path, len, out, sz, cap);
    collectPaths(node->right, path, len, out, sz, cap);
}
char** binaryTreePaths(struct TreeNode* root, int* rs) {
    char path[256] = {0}; char** out = NULL; int sz = 0, cap = 8;
    out = malloc(cap * sizeof(char*));
    collectPaths(root, path, 0, &out, &sz, &cap);
    *rs = sz; return out;
}
class Solution:
    def binaryTreePaths(self, root: Optional[TreeNode]) -> List[str]:
        res = []
        def dfs(node, path):
            if not node:
                return
            path += str(node.val)
            if not node.left and not node.right:
                res.append(path)
                return
            dfs(node.left, path + '->')
            dfs(node.right, path + '->')
        dfs(root, '')
        return res
class Solution {
    void dfs(TreeNode* node, string path, vector<string>& res) {
        if (!node) return;
        path += to_string(node->val);
        if (!node->left && !node->right) { res.push_back(path); return; }
        dfs(node->left, path + "->", res);
        dfs(node->right, path + "->", res);
    }
public:
    vector<string> binaryTreePaths(TreeNode* root) {
        vector<string> res;
        dfs(root, "", res);
        return res;
    }
};
