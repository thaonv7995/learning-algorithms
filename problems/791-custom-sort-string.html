Custom Sort String

O(n+m) O(1) 72.3% AC
Mô tả
Cho order (permutation các chữ cái distinct) và s. Sort s sao ký tự trong order đứng trước ký tự không trong order; trong cùng nhóm sort alphabet.
Ví dụ
Ví dụ 1
Inorder = "cba", s = "abcd"
Out"cbad"
Ví dụ 2
Inorder = "bcafg", s = "abcd"
Out"bcad"
Ý tưởng
Đếm freq s; output theo thứ tự order rồi các chữ còn lại sort.
Mã nguồn
solution.c C99 
char* customSortString(char* order, char* s) {
    int cnt[26]={0}; for(;*s;s++) cnt[*s-'a']++;
    char* ans=malloc(strlen(s)+100); int p=0;
    for(char* o=order;*o;o++) while(cnt[*o-'a']--) ans[p++]=*o;
    for(int i=0;i<26;i++) while(cnt[i]--) ans[p++]=i+'a'; ans[p]=0; return ans;
}
class Solution:
    def customSortString(self, order: str, s: str) -> str:
        cnt = Counter(s)
        ans = []
        for c in order:
            ans.append(c * cnt[c]); del cnt[c]
        for c in sorted(cnt):
            ans.append(c * cnt[c])
        return "".join(ans)
class Solution {
public:
    string customSortString(string order, string s) {
        int cnt[26]={}; for(char c:s) cnt[c-'a']++;
        string ans;
        for(char c:order) while(cnt[c-'a']--) ans.push_back(c);
        for(int i=0;i<26;i++) while(cnt[i]--) ans.push_back(i+'a');
        return ans;
    }
};
