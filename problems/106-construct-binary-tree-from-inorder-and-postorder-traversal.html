Construct Binary Tree from Inorder and Postorder Traversal

O(n) O(n) 68.7% AC
Mô tả
Cho hai mảng inorder và postorder (giá trị không trùng). Dựng lại cây nhị phân duy nhất tương ứng.
Ví dụ
Ví dụ 1
Ininorder = [9,3,15,20,7], postorder = [9,15,7,20,3]
Out[3,9,20,null,null,15,7]
Ví dụ 2
Ininorder = [-1], postorder = [-1]
Out[-1]
Ý tưởng
Phần tử cuối postorder là root. Tìm root trong inorder → cắt cây trái/phải. Vì pop từ cuối postorder, đệ quy cây phải trước rồi cây trái. Dùng hash map vị trí inorder.
Mã nguồn
solution.c C99 
/* Map inorder + đệ quy postorder — xem C++ */
class Solution:
    def buildTree(self, inorder: List[int], postorder: List[int]) -> Optional[TreeNode]:
        pos = {v: i for i, v in enumerate(inorder)}
        def build(l, r):
            if l > r: return None
            v = postorder.pop()
            root = TreeNode(v)
            k = pos[v]
            root.right = build(k + 1, r)
            root.left = build(l, k - 1)
            return root
        return build(0, len(inorder) - 1)
class Solution {
    unordered_map<int,int> pos;
    int pi;
    TreeNode* build(int l, int r, vector<int>& post) {
        if (l > r) return nullptr;
        int v = post[pi--];
        auto* nd = new TreeNode(v);
        nd->right = build(pos[v]+1, r, post);
        nd->left = build(l, pos[v]-1, post);
        return nd;
    }
public:
    TreeNode* buildTree(vector<int>& in, vector<int>& post) {
        for (int i = 0; i < (int)in.size(); i++) pos[in[i]] = i;
        pi = post.size() - 1;
        return build(0, in.size()-1, post);
    }
};
