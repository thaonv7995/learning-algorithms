Remove K Digits

O(n) O(n) 36.9% AC
Mô tả
Cho chuỗi số num và số nguyên k. Xóa đúng k chữ số sao cho chuỗi còn lại nhỏ nhất theo thứ tự số (không có số 0 đứng đầu trừ khi chỉ còn "0").
Ví dụ
Ví dụ 1
Innum = "1432219", k = 3
Out"1219"
Ví dụ 2
Innum = "10200", k = 1
Out"200"
Ví dụ 3
Innum = "10", k = 2
Out"0"
Ý tưởng
Monotonic stack tăng: push từng digit; khi top > digit hiện tại và còn k thì pop. Nếu còn k thì cắt đuôi. Bỏ số 0 đầu.
Mã nguồn
solution.c C99 
char* removeKdigits(char* num, int k) {
    int n = strlen(num), top = 0;
    char* st = malloc(n + 2);
    for (int i = 0; i < n; i++) {
        while (k && top && st[top-1] > num[i]) { top--; k--; }
        st[top++] = num[i];
    }
    top -= k; if (top < 0) top = 0;
    int s = 0; while (s < top && st[s] == '0') s++;
    if (s == top) { st[0]='0'; st[1]=0; return st; }
    memmove(st, st+s, top-s+1);
    return st;
}
class Solution:
    def removeKdigits(self, num: str, k: int) -> str:
        st = []
        for ch in num:
            while k and st and st[-1] > ch:
                st.pop(); k -= 1
            st.append(ch)
        st = st[:len(st)-k] if k else st
        return ("".join(st).lstrip("0") or "0")
class Solution {
public:
    string removeKdigits(string num, int k) {
        string st;
        for (char ch : num) {
            while (k && !st.empty() && st.back() > ch) { st.pop_back(); k--; }
            st.push_back(ch);
        }
        if (k) st.resize(st.size() - k);
        size_t i = st.find_first_not_of('0');
        return i == string::npos ? "0" : st.substr(i);
    }
};
