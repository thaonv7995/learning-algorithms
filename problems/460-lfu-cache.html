LFU Cache

O(1) O(capacity) 49.2% AC
Mô tả
LFU Cache: get/put O(1); evict least frequently used; tie-break LRU among same freq.
Ví dụ
Ví dụ 1
InLFUCache(2), put(1,1), put(2,2), get(1), put(3,3), get(2), get(3)
Out1,-1,3
Ví dụ 2
Input evicts min freq key
OutLFU policy
Ý tưởng
key→(val,freq,iter); freq→OrderedDict keys; minFreq pointer; inc freq move bucket.
Mã nguồn

class LFUCache:
    def __init__(self, capacity: int):
        self.cap = capacity
        self.kv = {}
        self.freq = defaultdict(OrderedDict)
        self.minf = 0
    def _touch(self, key):
        val, f = self.kv[key]
        del self.freq[f][key]
        if not self.freq[f] and f == self.minf: self.minf += 1
        f += 1
        self.freq[f][key] = None
        self.kv[key] = (val, f)
    def get(self, key: int) -> int:
        if key not in self.kv: return -1
        self._touch(key)
        return self.kv[key][0]
    def put(self, key: int, value: int) -> None:
        if self.cap == 0: return
        if key in self.kv:
            self.kv[key] = (value, self.kv[key][1])
            self._touch(key); return
        if len(self.kv) == self.cap:
            ev, _ = self.freq[self.minf].popitem(last=False)
            del self.kv[ev]
        self.kv[key] = (value, 1)
        self.freq[1][key] = None
        self.minf = 1
class LFUCache {
    int cap, minf;
    unordered_map<int,pair<int,int>> kv;
    unordered_map<int,list<int>> freq;
    unordered_map<int,list<int>::iterator> it;
public:
    LFUCache(int c):cap(c),minf(0){}
    int get(int key){
        if(!kv.count(key)) return -1;
        int v=kv[key].first, f=kv[key].second;
        freq[f].erase(it[key]); if(freq[f].empty()&&f==minf) minf++;
        f++; freq[f].push_front(key); it[key]=freq[f].begin(); kv[key]={v,f};
        return v;
    }
    void put(int key,int val){
        if(!cap) return;
        if(kv.count(key)){ kv[key].first=val; get(key); return; }
        if((int)kv.size()==cap){ int ev=freq[minf].back(); freq[minf].pop_back(); kv.erase(ev); it.erase(ev); }
        kv[key]={val,1}; freq[1].push_front(key); it[key]=freq[1].begin(); minf=1;
    }
};
