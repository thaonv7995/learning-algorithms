Range Sum Query - Mutable

O(log n) cập nhật/truy vấn O(n) 43.1% AC
Mô tả
Cho mảng nums và hỗ trợ update(index, val) cùng sumRange(left, right). Cần cấu trúc cho phép sửa và truy vấn tổng đoạn hiệu quả.
Ví dụ
Ví dụ 1
Innums = [1,3,5], update(1,2), sumRange(0,2)
Out8
Ví dụ 2
Innums = [0,0,0], update(0,1), update(1,2), sumRange(0,2)
Out3
Ý tưởng
Binary Indexed Tree (Fenwick): update thêm delta tại index; prefix sum tại i cho tổng [0..i]; sumRange = prefix(r)-prefix(l-1).
Mã nguồn
solution.c C99 
typedef struct { int* bit; int* nums; int n; } NumArray;
static void bitAdd(int* bit, int n, int i, int d) { for (; i <= n; i += i & -i) bit[i] += d; }
static int bitSum(int* bit, int i) { int s = 0; for (; i > 0; i -= i & -i) s += bit[i]; return s; }
NumArray* numArrayCreate(int* nums, int n) {
    NumArray* o = malloc(sizeof(NumArray)); o->n = n;
    o->bit = calloc(n+1, sizeof(int)); o->nums = malloc(n*sizeof(int));
    for (int i = 0; i < n; i++) { o->nums[i] = nums[i]; bitAdd(o->bit, n, i+1, nums[i]); }
    return o;
}
void numArrayUpdate(NumArray* o, int index, int val) {
    bitAdd(o->bit, o->n, index+1, val - o->nums[index]); o->nums[index] = val;
}
int numArraySumRange(NumArray* o, int l, int r) { return bitSum(o->bit, r+1) - bitSum(o->bit, l); }
class NumArray:
    def __init__(self, nums: List[int]):
        self.n = len(nums)
        self.bit = [0]*(self.n+1)
        self.nums = nums[:]
        for i, x in enumerate(nums):
            self._add(i+1, x)
    def _add(self, i, delta):
        while i <= self.n:
            self.bit[i] += delta
            i += i & -i
    def _sum(self, i):
        s = 0
        while i > 0:
            s += self.bit[i]
            i -= i & -i
        return s
    def update(self, index: int, val: int) -> None:
        self._add(index+1, val - self.nums[index])
        self.nums[index] = val
    def sumRange(self, left: int, right: int) -> int:
        return self._sum(right+1) - self._sum(left)
class NumArray {
    vector<int> bit, nums;
    int n;
    void add(int i, int d) { for (; i <= n; i += i & -i) bit[i] += d; }
    int sum(int i) { int s = 0; for (; i > 0; i -= i & -i) s += bit[i]; return s; }
public:
    NumArray(vector<int>& a) : nums(a), n(a.size()), bit(n+1, 0) {
        for (int i = 0; i < n; i++) add(i+1, a[i]);
    }
    void update(int index, int val) { add(index+1, val - nums[index]); nums[index] = val; }
    int sumRange(int l, int r) { return sum(r+1) - sum(l); }
};
