Serialize and Deserialize Binary Tree

O(n) O(n) 60.8% AC
Mô tả
Thiết kế thuật toán serialize binary tree thành chuỗi và deserialize ngược lại. Cấu trúc tree không cố định.
Ví dụ
Ví dụ 1
Inroot = [1,2,3,null,null,4,5]
Outserialized then deserialized equals original
Ví dụ 2
Inroot = []
Out[]
Ý tưởng
BFS/preorder với marker null: '1,2,3,null,null,4,5'. Deserialize queue/list build level order hoặc recursive preorder.

char* serializeTree(struct TreeNode* root) {
    if (!root) return strdup("");
    char* out = calloc(4096, 1);
    struct TreeNode** q = malloc(4096 * sizeof(struct TreeNode*));
    int h = 0, t = 0;
    q[t++] = root;
    while (h < t) {
        struct TreeNode* node = q[h++];
        if (node) {
            char buf[32]; snprintf(buf, sizeof(buf), "%d,", node->val);
            strcat(out, buf);
            q[t++] = node->left; q[t++] = node->right;
        } else strcat(out, "null,");
    }
    int len = (int)strlen(out);
    if (len && out[len - 1] == ',') out[len - 1] = 0;
    free(q); return out;
}
struct TreeNode* deserializeTree(char* data) {
    if (!data || !data[0]) return NULL;
    char* copy = strdup(data);
    char* tok = strtok(copy, ",");
    struct TreeNode* root = malloc(sizeof(struct TreeNode));
    root->val = atoi(tok); root->left = root->right = NULL;
    struct TreeNode** q = malloc(4096 * sizeof(struct TreeNode*));
    int h = 0, t = 0; q[t++] = root;
    while (h < t && (tok = strtok(NULL, ","))) {
        struct TreeNode* node = q[h++];
        if (strcmp(tok, "null")) { node->left = malloc(sizeof(struct TreeNode)); node->left->val = atoi(tok); node->left->left = node->left->right = NULL; q[t++] = node->left; }
        if (!(tok = strtok(NULL, ","))) break;
        if (strcmp(tok, "null")) { node->right = malloc(sizeof(struct TreeNode)); node->right->val = atoi(tok); node->right->left = node->right->right = NULL; q[t++] = node->right; }
    }
    free(copy); free(q); return root;
}
class Codec:
    def serialize(self, root: Optional[TreeNode]) -> str:
        if not root:
            return ""
        out = []
        q = deque([root])
        while q:
            node = q.popleft()
            if node:
                out.append(str(node.val))
                q.append(node.left)
                q.append(node.right)
            else:
                out.append('null')
        while out and out[-1] == 'null':
            out.pop()
        return ','.join(out)
    def deserialize(self, data: str) -> Optional[TreeNode]:
        if not data:
            return None
        vals = data.split(',')
        root = TreeNode(int(vals[0]))
        q = deque([root])
        i = 1
        while q and i < len(vals):
            node = q.popleft()
            if vals[i] != 'null':
                node.left = TreeNode(int(vals[i]))
                q.append(node.left)
            i += 1
            if i < len(vals) and vals[i] != 'null':
                node.right = TreeNode(int(vals[i]))
                q.append(node.right)
            i += 1
        return root
class Codec {
public:
    string serialize(TreeNode* root) {
        if (!root) return "";
        string res;
        queue<TreeNode*> q;
        q.push(root);
        while (!q.empty()) {
            TreeNode* node = q.front(); q.pop();
            if (node) {
                res += to_string(node->val) + ",";
                q.push(node->left);
                q.push(node->right);
            } else res += "null,";
        }
        while (!res.empty() && res.back() == ',') res.pop_back();
        if (res.size() >= 5 && res.substr(res.size()-5) == ",null") { /* trim trailing nulls optional */ }
        return res;
    }
    TreeNode* deserialize(string data) {
        if (data.empty()) return nullptr;
        vector<string> vals;
        stringstream ss(data);
        string v;
        while (getline(ss, v, ',')) vals.push_back(v);
        TreeNode* root = new TreeNode(stoi(vals[0]));
        queue<TreeNode*> q;
        q.push(root);
        int i = 1;
        while (!q.empty() && i < (int)vals.size()) {
            TreeNode* node = q.front(); q.pop();
            if (vals[i] != "null") { node->left = new TreeNode(stoi(vals[i])); q.push(node->left); }
            i++;
            if (i < (int)vals.size() && vals[i] != "null") { node->right = new TreeNode(stoi(vals[i])); q.push(node->right); }
            i++;
        }
        return root;
    }
};
