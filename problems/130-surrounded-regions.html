Surrounded Regions

O(m · n) O(m · n) 45.3% AC
Mô tả
Cho lưới ký tự 'X' và 'O'. Đảo mọi 'O' bị bao quanh hoàn toàn bởi 'X' thành 'X'. 'O' nối biên không bị đảo.
Ví dụ
Ví dụ 1
Inboard = [["X","X","X","X"],["X","O","O","X"],["X","X","O","X"],["X","O","X","X"]]
Out[["X","X","X","X"],["X","X","X","X"],["X","X","X","X"],["X","O","X","X"]]
Ví dụ 2
Inboard = [["X"]]
Out[["X"]]
Ý tưởng
DFS/BFS từ mọi 'O' trên biên, đánh dấu 'T' (safe). Sau đó: 'O'→'X', 'T'→'O'.
Mã nguồn
solution.c C99 
/* Flood fill biên O → T, flip còn lại — xem C++ */
class Solution:
    def solve(self, board: List[List[str]]) -> None:
        if not board: return
        m, n = len(board), len(board[0])
        def dfs(i, j):
            if i < 0 or i >= m or j < 0 or j >= n or board[i][j] != 'O': return
            board[i][j] = 'T'
            dfs(i+1,j); dfs(i-1,j); dfs(i,j+1); dfs(i,j-1)
        for i in range(m):
            dfs(i, 0); dfs(i, n-1)
        for j in range(n):
            dfs(0, j); dfs(m-1, j)
        for i in range(m):
            for j in range(n):
                if board[i][j] == 'O': board[i][j] = 'X'
                elif board[i][j] == 'T': board[i][j] = 'O'
class Solution {
    void dfs(vector<vector<char>>& b, int i, int j) {
        int m = b.size(), n = b[0].size();
        if (i < 0 || i >= m || j < 0 || j >= n || b[i][j] != 'O') return;
        b[i][j] = 'T';
        dfs(b, i+1, j); dfs(b, i-1, j); dfs(b, i, j+1); dfs(b, i, j-1);
    }
public:
    void solve(vector<vector<char>>& board) {
        if (board.empty()) return;
        int m = board.size(), n = board[0].size();
        for (int i = 0; i < m; i++) { dfs(board, i, 0); dfs(board, i, n-1); }
        for (int j = 0; j < n; j++) { dfs(board, 0, j); dfs(board, m-1, j); }
        for (int i = 0; i < m; i++)
            for (int j = 0; j < n; j++) {
                if (board[i][j] == 'O') board[i][j] = 'X';
                else if (board[i][j] == 'T') board[i][j] = 'O';
            }
    }
};
