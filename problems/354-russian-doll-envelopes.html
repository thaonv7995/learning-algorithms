Russian Doll Envelopes

O(n log n) O(n) 37.9% AC
Mô tả
Envelope [w,h] chứa được envelope khác nếu cả w và h đều lớn hơn (không bằng). Tìm số envelope tối đa có thể lồng nhau.
Ví dụ
Ví dụ 1
Inenvelopes = [[5,4],[6,4],[6,7],[2,3]]
Out3
Ví dụ 2
Inenvelopes = [[1,1],[1,1],[1,1]]
Out1
Ý tưởng
Sort theo w tăng, h giảm (trùng w); LIS trên h bằng patience sorting O(n log n).

int maxEnvelopes(int** envelopes, int n, int* cs) {
    /* sort + LIS on heights; qsort comparator by w asc, h desc */
    int* tails = malloc(n*sizeof(int)); int t=0;
    for (int i=0;i<n;i++){
        int h=envelopes[i][1], lo=0, hi=t;
        while(lo<hi){ int m=(lo+hi)/2; if(tails[m]<h) lo=m+1; else hi=m; }
        tails[lo]=h; if(lo==t) t++;
    }
    free(tails); return t;
}
class Solution:
    def maxEnvelopes(self, envelopes: List[List[int]]) -> int:
        envelopes.sort(key=lambda x: (x[0], -x[1]))
        tails = []
        for _, h in envelopes:
            i = bisect_left(tails, h)
            if i == len(tails): tails.append(h)
            else: tails[i] = h
        return len(tails)
class Solution {
public:
    int maxEnvelopes(vector<vector<int>>& env) {
        sort(env.begin(), env.end(), [](auto& a, auto& b){
            return a[0] < b[0] || (a[0]==b[0] && a[1] > b[1]);
        });
        vector<int> tails;
        for (auto& e : env) {
            int h = e[1];
            auto it = lower_bound(tails.begin(), tails.end(), h);
            if (it == tails.end()) tails.push_back(h);
            else *it = h;
        }
        return tails.size();
    }
};
