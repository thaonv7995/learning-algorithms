Longest Univalue Path

O(n) O(h) 43.9% AC
Mô tả
Độ dài path dài nhất cùng giá trị (edges). Path không nhất thiết qua root.
Ví dụ
Ví dụ 1
Inroot=[5,4,5,1,1,5]
Out2
Ví dụ 2
Inroot=[1,1,1]
Out3
Ý tưởng
Postorder: extend same val left/right; global max through node.

class Solution:
    def longestUnivaluePath(self, root: Optional[TreeNode]) -> int:
        self.ans=0
        def dfs(n):
            if not n: return 0
            l=dfs(n.left); r=dfs(n.right)
            ll=rr=0
            if n.left and n.left.val==n.val: ll=l+1
            if n.right and n.right.val==n.val: rr=r+1
            self.ans=max(self.ans,ll+rr)
            return max(ll,rr)
        dfs(root); return self.ans
class Solution {
    int ans=0;
    int dfs(TreeNode* n){
        if(!n) return 0;
        int l=dfs(n->left), r=dfs(n->right), ll=0, rr=0;
        if(n->left&&n->left->val==n->val) ll=l+1;
        if(n->right&&n->right->val==n->val) rr=r+1;
        ans=max(ans,ll+rr); return max(ll,rr);
    }
public:
    int longestUnivaluePath(TreeNode* root){ dfs(root); return ans; }
};
