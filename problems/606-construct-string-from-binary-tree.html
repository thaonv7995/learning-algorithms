Construct String from Binary Tree

O(n) O(n) 70.7% AC
Mô tả
Cho cây nhị phân root. Trả chuỗi "N" + preorder (val) + "L" + left + "R" + right + ")" theo quy ước LeetCode construct string.
Ví dụ
Ví dụ 1
Inroot = [1,2,3,4]
Out"1(2(4))(3)"
Ví dụ 2
Inroot = [1,2,null,4]
Out"1(2(4))"
Ví dụ 3
Inroot = [1]
Out"1"
Ý tưởng
DFS preorder: nếu không con trả val; nếu chỉ right bỏ L rỗng; ghép val(L)(R).
Mã nguồn

void tree2strRec(struct TreeNode* r, char* buf, int* p) {
    if(!r) return;
    *p += sprintf(buf+*p, "%d", r->val);
    if(!r->left && !r->right) return;
    if(!r->left){ *p+=sprintf(buf+*p,"()("); tree2strRec(r->right,buf,p); *p+=sprintf(buf+*p,")"); return; }
    *p+=sprintf(buf+*p,"("); tree2strRec(r->left,buf,p); *p+=sprintf(buf+*p,")(");
    if(r->right) tree2strRec(r->right,buf,p); *p+=sprintf(buf+*p,")");
}
class Solution:
    def tree2str(self, root: Optional[TreeNode]) -> str:
        if not root: return ""
        if not root.left and not root.right: return str(root.val)
        if not root.left: return f"{root.val}()({self.tree2str(root.right)})"
        return f"{root.val}({self.tree2str(root.left)})({self.tree2str(root.right) if root.right else ''})"
class Solution {
public:
    string tree2str(TreeNode* r) {
        if(!r) return "";
        string s=to_string(r->val);
        if(!r->left&&!r->right) return s;
        if(!r->left) return s+"()("+tree2str(r->right)+")";
        s+="("+tree2str(r->left)+")";
        if(r->right) s+="("+tree2str(r->right)+")";
        else s+="()";
        return s;
    }
};
