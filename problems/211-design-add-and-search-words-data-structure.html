Design Add and Search Words Data Structure

O(m) O(N) 48.5% AC
Mô tả
Thiết kế cấu trúc WordDictionary: addWord(word) thêm từ; search(word) tìm từ — ký tự '.' khớp bất kỳ chữ cái nào một lần.
Ví dụ
Ví dụ 1
Inadd "bad","dad","mad"; search "pad" → false; search "bad" → true; search ".ad" → true; search "b.." → true
Outfalse, true, true, true
Ví dụ 2
Inadd "a"; search "."
Outtrue
Ý tưởng
Trie lưu từ; search DFS tại node hiện tại: nếu '.' thì thử mọi con 26 nhánh, else đi theo ký tự cụ thể.
Mã nguồn
solution.c C99 
typedef struct WNode { struct WNode* ch[26]; int end; } WNode;

static int wSearch(WNode* node, const char* w, int i) {
    if (!w[i]) return node->end;
    if (w[i] == '.') {
        for (int k = 0; k < 26; k++)
            if (node->ch[k] && wSearch(node->ch[k], w, i + 1)) return 1;
        return 0;
    }
    int k = w[i] - 'a';
    return node->ch[k] && wSearch(node->ch[k], w, i + 1);
}

void wAdd(WNode* root, const char* word) {
    WNode* cur = root;
    for (; *word; word++) {
        int k = *word - 'a';
        if (!cur->ch[k]) cur->ch[k] = calloc(1, sizeof(WNode));
        cur = cur->ch[k];
    }
    cur->end = 1;
}

int wFind(WNode* root, const char* word) { return wSearch(root, word, 0); }
class WordDictionary:
    def __init__(self):
        self.root = {}
    def addWord(self, word: str) -> None:
        node = self.root
        for c in word:
            node = node.setdefault(c, {})
        node['#'] = True
    def search(self, word: str) -> bool:
        def dfs(node, i):
            if i == len(word):
                return node.get('#', False)
            c = word[i]
            if c == '.':
                return any(dfs(child, i + 1) for k, child in node.items() if k != '#')
            if c not in node:
                return False
            return dfs(node[c], i + 1)
        return dfs(self.root, 0)
class WordDictionary {
    struct Node { unordered_map<char, Node*> ch; bool end=false; }* root;
    bool dfs(Node* node, const string& w, int i) {
        if (i == (int)w.size()) return node->end;
        if (w[i] == '.') {
            for (auto& [c, child] : node->ch)
                if (dfs(child, w, i + 1)) return true;
            return false;
        }
        auto it = node->ch.find(w[i]);
        return it != node->ch.end() && dfs(it->second, w, i + 1);
    }
public:
    WordDictionary() { root = new Node(); }
    void addWord(string word) {
        Node* cur = root;
        for (char c : word) {
            if (!cur->ch.count(c)) cur->ch[c] = new Node();
            cur = cur->ch[c];
        }
        cur->end = true;
    }
    bool search(string word) { return dfs(root, word, 0); }
};
