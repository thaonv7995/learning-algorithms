char*** findDuplicate(char** paths, int n, int* rs, int** cs) {
    char*** ans = NULL; *rs = 0;
    char keys[256][256]; int cnt[256] = {0}; int nk = 0;
    char* lists[256][64]; int ls[256] = {0};
    for (int i = 0; i < n; i++) {
        char* p = paths[i]; char* sp = strchr(p, ' ');
        char root[256]; int rl = sp ? (int)(sp - p) : (int)strlen(p);
        memcpy(root, p, rl); root[rl] = 0;
        for (char* q = sp ? sp + 1 : p; *q;) {
            char* lp = strchr(q, '('); char* rp = strchr(lp, ')');
            char content[256]; int cl = (int)(rp - lp - 1); memcpy(content, lp + 1, cl); content[cl] = 0;
            char full[512]; snprintf(full, sizeof(full), "%s/%.*s", root, (int)(lp - q), q);
            int ki = -1; for (int k = 0; k < nk; k++) if (!strcmp(keys[k], content)) { ki = k; break; }
            if (ki < 0) { ki = nk++; strcpy(keys[ki], content); }
            strcpy(lists[ki][ls[ki]++], full);
            q = rp + 1; while (*q == ' ') q++;
        }
    }
    for (int k = 0; k < nk; k++) if (ls[k] > 1) (*rs)++;
    ans = malloc(*rs * sizeof(char**)); cs = malloc(*rs * sizeof(int)); int ai = 0;
    for (int k = 0; k < nk; k++) if (ls[k] > 1) {
        ans[ai] = malloc(ls[k] * sizeof(char*)); cs[ai] = ls[k];
        for (int j = 0; j < ls[k]; j++) ans[ai][j] = lists[k][j];
        ai++;
    }
    return ans;
}
class Solution:
    def findDuplicate(self, paths: List[str]) -> List[List[str]]:
        mp = defaultdict(list)
        for p in paths:
            parts = p.split()
            root = parts[0]
            for f in parts[1:]:
                i = f.index('(')
                mp[f[i+1:-1]].append(root + '/' + f[:i])
        return [[v[0]]+v for v in mp.values() if len(v)>1]
class Solution {
public:
    vector<vector<string>> findDuplicate(vector<string>& paths) {
        unordered_map<string, vector<string>> mp;
        for(auto& p:paths){
            int sp=p.find(' '); string root=p.substr(0,sp);
            for(int i=sp+1;i<(int)p.size();){
                int j=p.find('(',i); int k=p.find(')',j);
                string file=p.substr(i,j-i), content=p.substr(j+1,k-j-1);
                mp[content].push_back(root+"/"+file);
                i=k+2;
            }
        }
        vector<vector<string>> ans;
        for(auto& [c,v]:mp) if(v.size()>1){ vector<string> row={v[0]}; row.insert(row.end(),v.begin(),v.end()); ans.push_back(row); }
        return ans;
    }
};
