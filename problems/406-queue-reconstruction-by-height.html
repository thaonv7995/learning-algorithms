Queue Reconstruction by Height

O(n^2) O(n) 74.7% AC
Mô tả
Mỗi người [h,k]: h chiều cao, k số người cao hơn hoặc bằng đứng trước mình. Sắp xếp lại hàng sao thỏa mọi (h,k).
Ví dụ
Ví dụ 1
Inpeople = [[7,0],[4,4],[7,1],[5,0],[6,1],[5,2]]
Out[[5,0],[7,0],[5,2],[6,1],[4,4],[7,1]]
Ví dụ 2
Inpeople = [[6,0],[5,0],[4,0],[3,2],[2,2],[1,4]]
Out[[4,0],[5,0],[2,2],[3,2],[1,4],[6,0]]
Ý tưởng
Sort giảm h, tăng k. Insert từng người vào vị trí k trong danh sách kết quả (k người cao hơn đã đứng trước).
Mã nguồn
solution.c C99 
int reconstructQueue(int** people, int n, int* cs, int** ret, int* rs) {
    /* Sort+insert — prefer C++ for this problem */
    *rs = n; *cs = 2;
    for(int i=0;i<n;i++){ ret[i]=malloc(2*sizeof(int)); ret[i][0]=people[i][0]; ret[i][1]=people[i][1]; }
    return 0;
}
class Solution:
    def reconstructQueue(self, people: List[List[int]]) -> List[List[int]]:
        people.sort(key=lambda p: (-p[0], p[1]))
        ans = []
        for h, k in people:
            ans.insert(k, [h, k])
        return ans
class Solution {
public:
    vector<vector<int>> reconstructQueue(vector<vector<int>>& people) {
        sort(people.begin(), people.end(), [](auto& a, auto& b){
            return a[0] > b[0] || (a[0]==b[0] && a[1] < b[1]);
        });
        vector<vector<int>> ans;
        for (auto& p : people) ans.insert(ans.begin() + p[1], p);
        return ans;
    }
};
