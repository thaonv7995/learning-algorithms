Dungeon Game

O(m×n) O(m×n) 41.4% AC
Mô tả
Lưới dungeon: mỗi ô có số (máu +/-). Từ góc trên-trái tới góc dưới-phải, HP luôn > 0. Tìm HP ban đầu tối thiểu (làm tròn lên) để sống sót.
Ví dụ
Ví dụ 1
Indungeon = [[-2,-3,3],[-5,-10,1],[10,30,-5]]
Out7
Ví dụ 2
Indungeon = [[0]]
Out1
Ý tưởng
DP ngược từ (m-1,n-1): need[i][j] = min HP cần tại (i,j). need = max(1, min(need[i+1][j], need[i][j+1]) - dungeon[i][j]).

int calculateMinimumHP(int** d, int m, int* cs, int n) {
    int** need = malloc(m*sizeof(int*));
    for (int i=0;i<m;i++) need[i]=malloc(n*sizeof(int));
    need[m-1][n-1] = 1-d[m-1][n-1]>0?1-d[m-1][n-1]:1;
    for (int i=m-2;i>=0;i--) { int v=need[i+1][n-1]-d[i][n-1]; need[i][n-1]=v>0?v:1; }
    for (int j=n-2;j>=0;j--) { int v=need[m-1][j+1]-d[m-1][j]; need[m-1][j]=v>0?v:1; }
    for (int i=m-2;i>=0;i--) for (int j=n-2;j>=0;j--) {
        int v=(need[i+1][j]<need[i][j+1]?need[i+1][j]:need[i][j+1])-d[i][j];
        need[i][j]=v>0?v:1;
    }
    int ans=need[0][0];
    for (int i=0;i<m;i++) free(need[i]); free(need); return ans;
}
class Solution:
    def calculateMinimumHP(self, d: List[List[int]]) -> int:
        m, n = len(d), len(d[0])
        need = [[0]*n for _ in range(m)]
        need[-1][-1] = max(1, 1 - d[-1][-1])
        for i in range(m-2, -1, -1):
            need[i][-1] = max(1, need[i+1][-1] - d[i][-1])
        for j in range(n-2, -1, -1):
            need[-1][j] = max(1, need[-1][j+1] - d[-1][j])
        for i in range(m-2, -1, -1):
            for j in range(n-2, -1, -1):
                need[i][j] = max(1, min(need[i+1][j], need[i][j+1]) - d[i][j])
        return need[0][0]
class Solution {
public:
    int calculateMinimumHP(vector<vector<int>>& d) {
        int m = d.size(), n = d[0].size();
        vector<vector<int>> need(m, vector<int>(n));
        need[m-1][n-1] = max(1, 1 - d[m-1][n-1]);
        for (int i = m - 2; i >= 0; i--) need[i][n-1] = max(1, need[i+1][n-1] - d[i][n-1]);
        for (int j = n - 2; j >= 0; j--) need[m-1][j] = max(1, need[m-1][j+1] - d[m-1][j]);
        for (int i = m - 2; i >= 0; i--)
            for (int j = n - 2; j >= 0; j--)
                need[i][j] = max(1, min(need[i+1][j], need[i][j+1]) - d[i][j]);
        return need[0][0];
    }
};
