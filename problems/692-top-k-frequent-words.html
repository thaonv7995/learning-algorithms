Top K Frequent Words

O(n log n) O(n) 60.2% AC
Mô tả
Cho mảng words và k. Trả k từ xuất hiện nhiều nhất; nếu cùng tần suất thì sắp xếp theo thứ tự từ điển tăng dần.
Ví dụ
Ví dụ 1
Inwords=["i","love","leetcode","i","love","coding"], k=2
Out["i","love"]
Ví dụ 2
Inwords=["the","day","is","sunny","the","the","the","sunny","is","is"], k=4
Out["the","is","sunny","day"]
Ý tưởng
Count freq; heap/custom sort by (-freq, word).
Mã nguồn

char** topKFrequent692(char** words,int n,int k,int* rs) {
    char uniq[256][64]; int cnt[256]={0}, nu=0;
    for(int i=0;i<n;i++){
        int fi=-1; for(int j=0;j<nu;j++) if(!strcmp(uniq[j],words[i])){fi=j;break;}
        if(fi<0){ strcpy(uniq[nu],words[i]); cnt[nu]=1; nu++; } else cnt[fi]++;
    }
    for(int i=0;i<nu;i++) for(int j=i+1;j<nu;j++){
        if(cnt[j]>cnt[i]||(cnt[j]==cnt[i]&&strcmp(uniq[j],uniq[i])<0)){
            char t[64]; strcpy(t,uniq[i]); strcpy(uniq[i],uniq[j]); strcpy(uniq[j],t);
            int tc=cnt[i]; cnt[i]=cnt[j]; cnt[j]=tc;
        }
    }
    char** ans=malloc(k*sizeof(char*)); *rs=k;
    for(int i=0;i<k;i++){ ans[i]=malloc(strlen(uniq[i])+1); strcpy(ans[i],uniq[i]); }
    return ans;
}
class Solution:
    def topKFrequent(self, words: List[str], k: int) -> List[str]:
        cnt=Counter(words)
        return sorted(cnt.keys(), key=lambda w:(-cnt[w],w))[:k]
class Solution {
public:
    vector<string> topKFrequent(vector<string>& words, int k) {
        unordered_map<string,int> cnt;
        for(auto& w:words) cnt[w]++;
        vector<string> keys; for(auto& p:cnt) keys.push_back(p.first);
        sort(keys.begin(),keys.end(),[&](auto& a,auto& b){
            return cnt[a]>cnt[b]||(cnt[a]==cnt[b]&&a<b);
        });
        return vector<string>(keys.begin(),keys.begin()+k);
    }
};
