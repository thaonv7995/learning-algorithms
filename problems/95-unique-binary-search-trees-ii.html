Unique Binary Search Trees II

O(C_n·n) O(C_n·n) 62.5% AC
Mô tả
Cho n ≥ 1. Trả về mọi cây BST khác nhau có n nút nhãn 1..n (cấu trúc khác nhau). Mỗi cây là TreeNode root.
Ví dụ
Ví dụ 1
Inn = 3
Out[[1,null,2,null,3],[1,null,3,2],[2,1,3],[3,1,null,null,2],[3,2,null,1]]
Ví dụ 2
Inn = 1
Out[[1]]
Ý tưởng
Đệ quy generate(start,end): mọi root i ∈ [start,end]; left = gen(start,i-1), right = gen(i+1,end); ghép mọi cặp (L,R) thành TreeNode(i).
Mã nguồn
solution.c C99 
/* generateTrees — đệ quy split root; xem C++ */
class Solution:
    def generateTrees(self, n: int) -> List[Optional[TreeNode]]:
        if n == 0: return []
        def build(lo, hi):
            if lo > hi: return [None]
            res = []
            for r in range(lo, hi + 1):
                for left in build(lo, r - 1):
                    for right in build(r + 1, hi):
                        res.append(TreeNode(r, left, right))
            return res
        return build(1, n)
class Solution {
    vector<TreeNode*> build(int lo, int hi) {
        vector<TreeNode*> res;
        if (lo>hi) { res.push_back(nullptr); return res; }
        for (int r=lo;r<=hi;r++) {
            auto L=build(lo,r-1), R=build(r+1,hi);
            for (auto* a:L) for (auto* b:R)
                res.push_back(new TreeNode(r,a,b));
        }
        return res;
    }
public:
    vector<TreeNode*> generateTrees(int n) {
        if (!n) return {};
        return build(1,n);
    }
};
