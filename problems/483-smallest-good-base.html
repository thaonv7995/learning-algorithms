Smallest Good Base

O(log^2 n) O(1) 46.2% AC
Mô tả
Tìm cơ số k nhỏ nhất (k≥2) sao n viết được toàn chữ số 1 trong hệ k (số repunit).
Ví dụ
Ví dụ 1
Inn = 13
Out"3"
Ví dụ 2
Inn = 4681
Out"8"
Ví dụ 3
Inn = 10001
Out"142857"
Ý tưởng
Thử độ dài len giảm dần; binary search k sao (k^len-1)/(k-1)=n.
Mã nguồn
solution.c C99

class Solution:
    def smallestGoodBase(self, n: int) -> str:
        import math
        for len1 in range(int(math.log(n, 2))+2, 1, -1):
            k = int(n ** (1.0/len1))
            for cand in (k, k+1, k-1):
                if cand >= 2 and (pow(cand, len1) - 1) == (cand - 1) * n:
                    return str(cand)
        return str(n - 1)
class Solution {
public:
    string smallestGoodBase(long long n) {
        for(int len=60;len>=2;len--){
            long long lo=2, hi=n-1;
            while(lo<=hi){
                long long mid=lo+(hi-lo)/2;
                __int128 s=0, p=1;
                for(int i=0;i<len;i++){ s+=p; p*=mid; if(s>n) break; }
                if(s==n) return to_string(mid);
                if(s>n) hi=mid-1; else lo=mid+1;
            }
        }
        return to_string(n-1);
    }
};
