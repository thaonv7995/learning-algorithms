3Sum Closest

O(N²) O(1) 48.6% AC
Mô tả
Cho mảng nums và target, tìm tổng ba số gần target nhất (|sum - target| nhỏ nhất).
Ví dụ
Ví dụ 1
Innums = [-1,2,1,-4], target = 1
Out2
Ví dụ 2
Innums = [0,0,0], target = 1
Out0
Ý tưởng
Sort nums. Với mỗi i, two-pointer left=i+1, right=n-1: cập nhật best nếu |sum-target| nhỏ hơn. Di chuyển left/right theo sum so với target.
Mã nguồn
solution.c C99 
int cmp(const void* a, const void* b) { return *(int*)a - *(int*)b; }
int threeSumClosest(int* nums, int n, int target) {
    qsort(nums, n, sizeof(int), cmp);
    int best = nums[0]+nums[1]+nums[2];
    for (int i = 0; i < n-2; i++) {
        int l = i+1, r = n-1;
        while (l < r) {
            int sum = nums[i]+nums[l]+nums[r];
            if (abs(sum-target) < abs(best-target)) best = sum;
            if (sum < target) l++; else if (sum > target) r--; else return sum;
        }
    }
    return best;
}
class Solution:
    def threeSumClosest(self, nums: List[int], target: int) -> int:
        nums.sort()
        best = sum(nums[:3])
        for i in range(len(nums)-2):
            l, r = i+1, len(nums)-1
            while l < r:
                s = nums[i]+nums[l]+nums[r]
                if abs(s-target) < abs(best-target): best = s
                if s < target: l += 1
                elif s > target: r -= 1
                else: return s
        return best
class Solution {
public:
    int threeSumClosest(vector<int>& nums, int target) {
        sort(nums.begin(), nums.end());
        int best = nums[0]+nums[1]+nums[2];
        for (int i = 0; i+2 < (int)nums.size(); i++) {
            int l = i+1, r = nums.size()-1;
            while (l < r) {
                int s = nums[i]+nums[l]+nums[r];
                if (abs(s-target) < abs(best-target)) best = s;
                if (s < target) l++; else if (s > target) r--; else return s;
            }
        }
        return best;
    }
};
