String to Integer (atoi)

O(N) O(1) 21% AC
Mô tả
Chuyển chuỗi s thành số nguyên 32-bit (atoi). Bỏ whitespace đầu, đọc dấu, đọc digits, clamp overflow.
Ví dụ
Ví dụ 1
Ins = "42"
Out42
Ví dụ 2
Ins = " -42"
Out-42
Ví dụ 3
Ins = "4193 with words"
Out4193
Ý tưởng
State machine: (1) skip space (2) đọc sign (3) đọc digits, mỗi bước check overflow INT_MAX/INT_MIN trước khi res = res*10 + d.
Mã nguồn
solution.c C99 
int myAtoi(char* s) {
    int i = 0, sign = 1;
    long res = 0;
    while (s[i] == ' ') i++;
    if (s[i] == '+' || s[i] == '-') { sign = s[i++] == '-' ? -1 : 1; }
    while (s[i] >= '0' && s[i] <= '9') {
        res = res * 10 + (s[i++] - '0');
        if (sign * res >= 2147483647) return 2147483647;
        if (sign * res <= -2147483648L) return -2147483648;
    }
    return (int)(sign * res);
}
class Solution:
    def myAtoi(self, s: str) -> int:
        i, sign, res = 0, 1, 0
        while i < len(s) and s[i] == ' ': i += 1
        if i < len(s) and s[i] in '+-': sign = -1 if s[i] == '-' else 1; i += 1
        while i < len(s) and s[i].isdigit():
            res = res * 10 + int(s[i]); i += 1
            if sign * res >= 2**31 - 1: return 2**31 - 1
            if sign * res <= -2**31: return -2**31
        return sign * res
class Solution {
public:
    int myAtoi(string s) {
        int i = 0, sign = 1;
        long res = 0;
        while (i < (int)s.size() && s[i] == ' ') i++;
        if (i < (int)s.size() && (s[i] == '+' || s[i] == '-')) sign = s[i++] == '-' ? -1 : 1;
        while (i < (int)s.size() && isdigit(s[i])) {
            res = res * 10 + (s[i++] - '0');
            if (sign * res >= INT_MAX) return INT_MAX;
            if (sign * res <= INT_MIN) return INT_MIN;
        }
        return (int)(sign * res);
    }
};
