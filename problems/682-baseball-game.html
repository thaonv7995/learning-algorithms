int calPoints682(char** ops,int n){ int st[1000],top=0,i,s=0; for(i=0;i<n;i++){ if(ops[i][0]=='+') st[top]=st[top-1]+st[top-2],top++; else if(ops[i][0]=='D') st[top]=st[top-1]*2,top++; else if(ops[i][0]=='C') top--; else st[top++]=atoi(ops[i]); } for(i=0;i<top;i++) s+=st[i]; return s; }
class Solution:
    def calPoints(self, ops: List[str]) -> int:
        st=[]
        for op in ops:
            if op=="+": st.append(st[-1]+st[-2])
            elif op=="D": st.append(st[-1]*2)
            elif op=="C": st.pop()
            else: st.append(int(op))
        return sum(st)
class Solution {
public:
    int calPoints(vector<string>& ops) {
        vector<int> st;
        for(auto& op:ops){
            if(op=="+") st.push_back(st[st.size()-1]+st[st.size()-2]);
            else if(op=="D") st.push_back(st.back()*2);
            else if(op=="C") st.pop_back();
            else st.push_back(stoi(op));
        }
        return accumulate(st.begin(),st.end(),0);
    }
};
