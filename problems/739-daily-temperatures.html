Daily Temperatures

O(N) O(N) 68.7% AC
Mô tả
Cho một mảng số nguyên temperatures đại diện cho nhiệt độ hàng ngày. Hãy trả về một mảng ans sao cho ans[i] là số ngày bạn phải đợi cho đến khi có nhiệt độ ấm hơn. Nếu không có ngày nào ấm hơn, gán ans[i] = 0.
Ví dụ
Ví dụ 1
Intemperatures = [73,74,75,71,69,72,76,73]
Out[1,1,4,2,1,1,0,0]
Ý tưởng
Sử dụng cấu trúc dữ liệu Ngăn xếp Đơn điệu Giảm dần (Monotonic Decreasing Stack). Ta duyệt qua mảng nhiệt độ từ trái sang phải, lưu trữ các chỉ số (index) vào Stack. Khi gặp một ngày có nhiệt độ lớn hơn nhiệt độ của ngày ở index đỉnh Stack, ta liên tục rút (pop) đỉnh Stack ra, tính khoảng cách ngày `i - prevIndex` điền vào mảng kết quả, cho đến khi Stack trống hoặc nhiệt độ đỉnh Stack lớn hơn ngày hiện tại.
Mã nguồn
solution.c C99 
// Solution in C (Daily Temperatures)
int* dailyTemperatures(int* T, int TSize, int* returnSize) {
    *returnSize = TSize;
    int* ans = (int*)calloc(TSize, sizeof(int));
    int* stack = (int*)malloc(TSize * sizeof(int));
    int top = -1;
    for (int i = 0; i < TSize; i++) {
        while (top != -1 && T[i] > T[stack[top]]) {
            int idx = stack[top--];
            ans[idx] = i - idx;
        }
        stack[++top] = i;
    }
    free(stack);
    return ans;
}
# Solution in Python (Daily Temperatures)
class Solution:
    def dailyTemperatures(self, T: List[int]) -> List[int]:
        ans = [0] * len(T)
        stack = [] # indices stores
        for i, t in enumerate(T):
            while stack and t > T[stack[-1]]:
                idx = stack.pop()
                ans[idx] = i - idx
            stack.append(i)
        return ans
class Solution {
public:
    vector<int> dailyTemperatures(vector<int>& T) {
        int n = T.size();
        vector<int> ans(n, 0);
        stack<int> st; // Luu tru chi so index
        for (int i = 0; i < n; i++) {
            while (!st.empty() && T[i] > T[st.top()]) {
                int idx = st.top(); st.pop();
                ans[idx] = i - idx; // Khoang cach ngay cho doi
            }
            st.push(i);
        }
        return ans;
    }
};
