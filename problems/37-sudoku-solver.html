static bool sudokuOk(char** b, int r, int c, char d) {
    for (int i = 0; i < 9; i++)
        if (b[r][i] == d || b[i][c] == d) return false;
    int br = (r / 3) * 3, bc = (c / 3) * 3;
    for (int i = 0; i < 3; i++)
        for (int j = 0; j < 3; j++)
            if (b[br + i][bc + j] == d) return false;
    return true;
}
static bool sudokuDfs(char** b) {
    for (int r = 0; r < 9; r++)
        for (int c = 0; c < 9; c++) {
            if (b[r][c] != '.') continue;
            for (char d = '1'; d <= '9'; d++) {
                if (sudokuOk(b, r, c, d)) {
                    b[r][c] = d;
                    if (sudokuDfs(b)) return true;
                    b[r][c] = '.';
                }
            }
            return false;
        }
    return true;
}
void solveSudoku(char** board, int boardSize, int* boardColSize) {
    (void)boardSize; (void)boardColSize;
    sudokuDfs(board);
}
class Solution:
    def solveSudoku(self, board: List[List[str]]) -> None:
        def ok(r,c,d):
            for i in range(9):
                if board[r][i]==d or board[i][c]==d: return False
            br,bc=r//3*3,c//3*3
            for i in range(3):
                for j in range(3):
                    if board[br+i][bc+j]==d: return False
            return True
        def dfs():
            for r in range(9):
                for c in range(9):
                    if board[r][c] != '.': continue
                    for d in '123456789':
                        if ok(r,c,d):
                            board[r][c]=d
                            if dfs(): return True
                            board[r][c]='.'
                    return False
            return True
        dfs()
class Solution {
    bool ok(vector<vector<char>>& b,int r,int c,char d){
        for(int i=0;i<9;i++) if(b[r][i]==d||b[i][c]==d) return false;
        int br=r/3*3,bc=c/3*3;
        for(int i=0;i<3;i++) for(int j=0;j<3;j++) if(b[br+i][bc+j]==d) return false;
        return true;
    }
    bool dfs(vector<vector<char>>& b){
        for(int r=0;r<9;r++) for(int c=0;c<9;c++) if(b[r][c]=='.'){
            for(char d='1';d<='9';d++) if(ok(b,r,c,d)){
                b[r][c]=d; if(dfs(b)) return true; b[r][c]='.';
            } return false;
        } return true;
    }
public:
    void solveSudoku(vector<vector<char>>& b){ dfs(b); }
};
