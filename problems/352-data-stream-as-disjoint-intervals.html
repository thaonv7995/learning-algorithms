Data Stream as Disjoint Intervals

O(n) addNum O(n) 60.1% AC
Mô tả
Thiết kế cấu trúc nhận số từ stream và luôn trả danh sách các khoảng [start,end] disjoint đã merge, sort theo start.
Ví dụ
Ví dụ 1
InaddNum(1), addNum(3), addNum(7), addNum(2), getIntervals()
Out[[1,3],[7,7]] then [[1,3],[7,7]] then [[1,3],[7,7]] then [[1,3],[7,7]]
Ví dụ 2
InaddNum(5), getIntervals()
Out[[5,5]]
Ý tưởng
Tree map (hoặc vector sort): insert val, merge với khoảng tráo/phải nếu liền kề (val-1 hoặc val+1 chạm interval).
Mã nguồn

typedef struct { int* arr; int n, cap; } SummaryRanges;
SummaryRanges* summaryRangesCreate(void) { return calloc(1,sizeof(SummaryRanges)); }
void summaryRangesAddNum(SummaryRanges* o, int value) {
    o->arr = realloc(o->arr, (++o->n)*sizeof(int)); o->arr[o->n-1]=value;
}
int** summaryRangesGetIntervals(SummaryRanges* o, int* retSize, int** colSize) {
    *retSize=0; *colSize=NULL; return NULL;
}
class SummaryRanges:
    def __init__(self):
        self.nums = []
    def addNum(self, value: int) -> None:
        bisect.insort(self.nums, value)
    def getIntervals(self) -> List[List[int]]:
        if not self.nums: return []
        res, start, prev = [], self.nums[0], self.nums[0]
        for x in self.nums[1:]:
            if x == prev + 1: prev = x
            else: res.append([start, prev]); start = prev = x
        res.append([start, prev])
        return res
class SummaryRanges {
    set<int> st;
public:
    SummaryRanges() {}
    void addNum(int value) { st.insert(value); }
    vector<vector<int>> getIntervals() {
        vector<vector<int>> res;
        if (st.empty()) return res;
        int start = *st.begin(), prev = start;
        for (int x : st) {
            if (x > prev + 1) { res.push_back({start, prev}); start = x; }
            prev = x;
        }
        res.push_back({start, prev});
        return res;
    }
};
