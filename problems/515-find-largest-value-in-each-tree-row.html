Find Largest Value in Each Tree Row

O(n) O(w) 66.3% AC
Mô tả
Cho cây nhị phân. Trả mảng giá trị lớn nhất ở mỗi hàng (level) từ trên xuống.
Ví dụ
Ví dụ 1
Inroot = [1,3,2,5,3,null,9]
Out[1,3,9]
Ví dụ 2
Inroot = [1,2,3]
Out[1,3]
Ý tưởng
BFS theo level: duyệt từng hàng, cập nhật max val trong hàng, push vào answer.
Mã nguồn
solution.c C99 
/* BFS max per level — xem C++ */
class Solution:
    def largestValues(self, root: Optional[TreeNode]) -> List[int]:
        if not root: return []
        from collections import deque
        q = deque([root])
        ans = []
        while q:
            mx = float("-inf")
            for _ in range(len(q)):
                n = q.popleft()
                mx = max(mx, n.val)
                if n.left: q.append(n.left)
                if n.right: q.append(n.right)
            ans.append(mx)
        return ans
class Solution {
public:
    vector<int> largestValues(TreeNode* root) {
        if (!root) return {};
        queue<TreeNode*> q;
        q.push(root);
        vector<int> ans;
        while (!q.empty()) {
            int sz = q.size(), mx = INT_MIN;
            while (sz--) {
                auto* n = q.front(); q.pop();
                mx = max(mx, n->val);
                if (n->left) q.push(n->left);
                if (n->right) q.push(n->right);
            }
            ans.push_back(mx);
        }
        return ans;
    }
};
