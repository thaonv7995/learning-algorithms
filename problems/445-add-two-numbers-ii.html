Add Two Numbers II

O(m+n) O(m+n) 62.6% AC
Mô tả
Hai linked list không rỗng biểu diễn số không âm (digit reverse). Trả sum dạng list (không reverse input).
Ví dụ
Ví dụ 1
Inl1 = [7,2,4,3], l2 = [5,6,4]
Out[7,8,0,7]
Ví dụ 2
Inl1 = [0], l2 = [0]
Out[0]
Ý tưởng
Push digits vào stack cả hai list; pop + carry build result list đầu.
Mã nguồn
solution.c C99 
struct ListNode* addTwoNumbers(struct ListNode* l1, struct ListNode* l2) { return NULL; }
class Solution:
    def addTwoNumbers(self, l1: ListNode, l2: ListNode) -> ListNode:
        s1, s2 = [], []
        while l1: s1.append(l1.val); l1 = l1.next
        while l2: s2.append(l2.val); l2 = l2.next
        carry, head = 0, None
        while s1 or s2 or carry:
            carry += (s1.pop() if s1 else 0) + (s2.pop() if s2 else 0)
            node = ListNode(carry % 10)
            node.next = head; head = node
            carry //= 10
        return head
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        stack<int> s1,s2;
        while(l1){ s1.push(l1->val); l1=l1->next; }
        while(l2){ s2.push(l2->val); l2=l2->next; }
        ListNode* head=nullptr; int carry=0;
        while(!s1.empty()||!s2.empty()||carry){
            if(!s1.empty()){ carry+=s1.top(); s1.pop(); }
            if(!s2.empty()){ carry+=s2.top(); s2.pop(); }
            auto* n=new ListNode(carry%10); n->next=head; head=n; carry/=10;
        }
        return head;
    }
};
