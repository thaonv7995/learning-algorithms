Count and Say

O(N·L) O(L) 62.9% AC
Mô tả
Dãy count-and-say: base "1", mỗi bước mô tả bước trước (vd "1211" → "111221"). Trả về term thứ n.
Ví dụ
Ví dụ 1
Inn = 4
Out"1211"
Ví dụ 2
Inn = 1
Out"1"
Ý tưởng
cur = "1". Lặp n-1 lần: duyệt cur, đếm run cùng ký tự, append count+digit vào next.
Mã nguồn
solution.c C99 
char* countAndSay(int n) {
    char* cur = (char*)malloc(2);
    strcpy(cur, "1");
    for (int k = 1; k < n; k++) {
        int len = (int)strlen(cur), cap = len * 2 + 16, pos = 0;
        char* nxt = (char*)malloc(cap);
        nxt[0] = '\0';
        for (int i = 0; cur[i]; ) {
            int j = i;
            while (cur[j] && cur[j] == cur[i]) j++;
            pos += sprintf(nxt + pos, "%d%c", j - i, cur[i]);
            i = j;
        }
        free(cur);
        cur = nxt;
    }
    return cur;
}
class Solution:
    def countAndSay(self, n: int) -> str:
        cur = '1'
        for _ in range(n-1):
            nxt, i = [], 0
            while i < len(cur):
                j = i
                while j < len(cur) and cur[j] == cur[i]: j += 1
                nxt.append(str(j-i)+cur[i]); i = j
            cur = ''.join(nxt)
        return cur
