Third Maximum Number

O(n) O(1) 39.4% AC
Mô tả
Trả giá trị lớn thứ ba khác nhau trong mảng nums. Nếu không đủ 3 giá trị khác nhau thì trả max.
Ví dụ
Ví dụ 1
Innums = [3,2,1]
Out1
Ví dụ 2
Innums = [1,2]
Out2
Ví dụ 3
Innums = [2,2,3,1]
Out1
Ý tưởng
Giữ ba biến first, second, third (long min). Duyệt x: cập nhật nếu x lớn hơn từng mức, bỏ qua trùng.
Mã nguồn

int thirdMax(int* nums, int n) {
    long a=LLONG_MIN,b=LLONG_MIN,c=LLONG_MIN;
    for(int i=0;i<n;i++){
        long x=nums[i];
        if(x==a||x==b||x==c) continue;
        if(x>a){ c=b;b=a;a=x; }
        else if(x>b){ c=b;b=x; }
        else if(x>c) c=x;
    }
    return c>LLONG_MIN/2 ? (int)c : (int)a;
}
class Solution:
    def thirdMax(self, nums: List[int]) -> int:
        a = b = c = -10**18
        for x in nums:
            if x == a or x == b or x == c: continue
            if x > a: a, b, c = x, a, b
            elif x > b: b, c = x, b
            elif x > c: c = x
        return c if c > -10**17 else a
