Valid Parentheses

O(n) O(n) 44.2% AC
Mô tả
Cho chuỗi s chỉ gồm '(', ')', '{', '}', '[', ']'. Kiểm tra chuỗi hợp lệ: mở đúng loại và đóng đúng thứ tự LIFO.
Ví dụ
Ví dụ 1
Ins = "()[]{}"
Outtrue
Ví dụ 2
Ins = "(]"
Outfalse
Ý tưởng
Stack: mở → push; đóng → pop và khớp cặp. Cuối stack phải rỗng.
Mã nguồn
solution.c C99 
/* Stack char — xem C++ */
class Solution:
    def isValid(self, s: str) -> bool:
        st = []
        pair = {')':'(', '}':'{', ']':'['}
        for c in s:
            if c in '({[': st.append(c)
            elif not st or st.pop() != pair[c]: return False
        return not st
class Solution {
public:
    bool isValid(string s) {
        stack<char> st;
        for (char c : s) {
            if (c=='('||c=='{'||c=='[') st.push(c);
            else {
                if (st.empty()) return false;
                char t = st.top(); st.pop();
                if ((c==')'&&t!='(')||(c=='}'&&t!='{')||(c==']'&&t!='[')) return false;
            }
        }
        return st.empty();
    }
};
