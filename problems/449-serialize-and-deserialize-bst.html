Serialize and Deserialize BST

O(n) O(n) 59.6% AC
Mô tả
Serialize và deserialize BST thành chuỗi. Codec phải khôi phục đúng cây BST gốc.
Ví dụ
Ví dụ 1
Inroot = [2,1,3]
Outserialize then deserialize → [2,1,3]
Ví dụ 2
Inroot = []
Out[]
Ý tưởng
Preorder với marker null: "val,left,right". Deserialize đọc tuần tự, BST property chọn nhánh.

class Codec:
    def serialize(self, root):
        def dfs(n):
            if not n: return "null,"
            return str(n.val)+","+dfs(n.left)+dfs(n.right)
        return dfs(root)
    def deserialize(self, data):
        it = iter(data.split(","))
        def build():
            v = next(it)
            if v == "null": return None
            node = TreeNode(int(v))
            node.left = build(); node.right = build()
            return node
        return build()
class Codec {
public:
    string serialize(TreeNode* root) {
        if(!root) return "null,";
        return to_string(root->val)+","+serialize(root->left)+serialize(root->right);
    }
    TreeNode* deserialize(string data) {
        queue<string> q;
        string s; for(char c:data){ if(c==','){ q.push(s); s=""; } else s+=c; }
        function<TreeNode*()> build=[&]()->TreeNode*{
            string v=q.front(); q.pop();
            if(v=="null") return nullptr;
            TreeNode* n=new TreeNode(stoi(v));
            n->left=build(); n->right=build();
            return n;
        };
        return build();
    }
};
