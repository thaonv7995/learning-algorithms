Diameter of Binary Tree

Mô tả
Trả đường kính cây nhị phân (số cạnh dài nhất giữa hai nút bất kỳ).
Ví dụ

Ý tưởng
DFS: tại mỗi node, diameter candidate = depth(left)+depth(right); trả max depth lên parent.
Mã nguồn
solution.c C99 
/* DFS diameter depth — xem C++ */
class Solution:
    def diameterOfBinaryTree(self, root: Optional[TreeNode]) -> int:
        self.ans = 0
        def depth(n):
            if not n: return 0
            l = depth(n.left)
            r = depth(n.right)
            self.ans = max(self.ans, l + r)
            return 1 + max(l, r)
        depth(root)
        return self.ans
class Solution {
    int ans = 0;
    int depth(TreeNode* n) {
        if (!n) return 0;
        int l = depth(n->left), r = depth(n->right);
        ans = max(ans, l + r);
        return 1 + max(l, r);
    }
public:
    int diameterOfBinaryTree(TreeNode* root) {
        depth(root);
        return ans;
    }
};
