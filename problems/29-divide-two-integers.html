Divide Two Integers

O(log N) O(1) 19.8% AC
Mô tả
Chia dividend cho divisor trả thương nguyên 32-bit (không dùng *, /, %). Làm tròn về 0.
Ví dụ
Ví dụ 1
Individend = 10, divisor = 3
Out3
Ví dụ 2
Individend = 7, divisor = -3
Out-2
Ý tưởng
Dùng long cho dấu. Lặp: tìm lớn nhất power of 2 sao cho divisor<<k <= dividend, trừ và cộng quotient, đến dividend=0.
Mã nguồn
solution.c C99 
int divide(int dividend, int divisor) {
    if (dividend == INT_MIN && divisor == -1) return INT_MAX;
    long sign = ((dividend < 0) ^ (divisor < 0)) ? -1L : 1L;
    long a = labs((long)dividend), b = labs((long)divisor), q = 0;
    while (a >= b) {
        long k = 1;
        while (a >= (b << (k + 1))) k <<= 1;
        a -= b << k;
        q += 1L << k;
    }
    q *= sign;
    if (q > INT_MAX) return INT_MAX;
    if (q < INT_MIN) return INT_MIN;
    return (int)q;
}
class Solution:
    def divide(self, dividend: int, divisor: int) -> int:
        if dividend == -2**31 and divisor == -1: return 2**31 - 1
        sign = -1 if (dividend < 0) ^ (divisor < 0) else 1
        a, b, q = abs(dividend), abs(divisor), 0
        while a >= b:
            k = 1
            while a >= (b << (k+1)): k += 1
            a -= b << k; q += 1 << k
        q *= sign
        return max(-2**31, min(2**31-1, q))
class Solution {
public:
    int divide(int dividend, int divisor) {
        if (dividend == INT_MIN && divisor == -1) return INT_MAX;
        long sign = ((dividend < 0) ^ (divisor < 0)) ? -1 : 1;
        long a = labs(dividend), b = labs(divisor), q = 0;
        while (a >= b) {
            long k = 1;
            while (a >= (b << (k+1))) k <<= 1;
            a -= b << k; q += 1 << k;
        }
        q *= sign;
        return (int)max((long)INT_MIN, min((long)INT_MAX, q));
    }
};
