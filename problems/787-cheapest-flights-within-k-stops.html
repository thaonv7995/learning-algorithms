Cheapest Flights Within K Stops

O(K * (E log V)) O(VK) 41.8% AC
Mô tả
n cities, flights [u,v,w] cost, src, dst, k stops max. Trả chi phí vé rẻ nhất từ src tới dst với at most k stops; -1 nếu không.
Ví dụ
Ví dụ 1
Inn = 4, flights = [[0,1,100],[1,2,100],[2,0,100],[1,3,600],[2,3,200]], src = 0, dst = 3, k = 1
Out700
Ví dụ 2
Inn = 3, flights = [[0,1,100],[1,2,100],[0,2,500]], src = 0, dst = 2, k = 1
Out200
Ý tưởng
Bellman-Ford k+1 lần hoặc BFS level Dijkstra: dist[stops][node] min cost.

int findCheapestPrice(int n, int** flights, int fn, int* cs, int src, int dst, int k) {
    const int INF=1000000000; int dist[100], nd[100];
    for(int i=0;i<n;i++) dist[i]=INF; dist[src]=0;
    for(int s=0;s<=k;s++){ for(int i=0;i<n;i++) nd[i]=dist[i];
        for(int i=0;i<fn;i++){ int u=flights[i][0],v=flights[i][1],w=flights[i][2];
            if(dist[u]!=INF && dist[u]+w<nd[v]) nd[v]=dist[u]+w; }
        for(int i=0;i<n;i++) dist[i]=nd[i]; }
    return dist[dst]==INF?-1:dist[dst];
}
class Solution:
    def findCheapestPrice(self, n: int, flights: List[List[int]], src: int, dst: int, k: int) -> int:
        INF = 10**9
        dist = [INF] * n
        dist[src] = 0
        for _ in range(k + 1):
            nd = dist[:]
            for u, v, w in flights:
                if dist[u] + w < nd[v]:
                    nd[v] = dist[u] + w
            dist = nd
        return dist[dst] if dist[dst] < INF else -1
class Solution {
public:
    int findCheapestPrice(int n, vector<vector<int>>& f, int src, int dst, int k) {
        const int INF=1e9;
        vector<int> dist(n,INF); dist[src]=0;
        for(int s=0;s<=k;s++){
            vector<int> nd=dist;
            for(auto& e:f){
                int u=e[0],v=e[1],w=e[2];
                if(dist[u]!=INF) nd[v]=min(nd[v],dist[u]+w);
            }
            dist=move(nd);
        }
        return dist[dst]==INF?-1:dist[dst];
    }
};
