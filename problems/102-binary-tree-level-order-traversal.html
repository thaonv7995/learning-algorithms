Binary Tree Level Order Traversal

O(n) O(n) 72.7% AC
Mô tả
Trả về giá trị các nút theo từng tầng (trái → phải), mỗi tầng là một mảng con.
Ví dụ
Ví dụ 1
Inroot = [3,9,20,null,null,15,7]
Out[[3],[9,20],[15,7]]
Ý tưởng
BFS: queue, mỗi vòng lặp xử lý đúng size hiện tại của queue = một tầng.
Mã nguồn
solution.c C99 
/* BFS tầng — xem C++ */
class Solution:
    def levelOrder(self, root: Optional[TreeNode]) -> List[List[int]]:
        if not root: return []
        res, q = [], deque([root])
        while q:
            row, n = [], len(q)
            for _ in range(n):
                node = q.popleft()
                row.append(node.val)
                if node.left: q.append(node.left)
                if node.right: q.append(node.right)
            res.append(row)
        return res
class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>> res;
        if (!root) return res;
        queue<TreeNode*> q; q.push(root);
        while (!q.empty()) {
            int n = q.size(); vector<int> row;
            for (int i = 0; i < n; i++) {
                auto* nd = q.front(); q.pop();
                row.push_back(nd->val);
                if (nd->left) q.push(nd->left);
                if (nd->right) q.push(nd->right);
            }
            res.push_back(row);
        }
        return res;
    }
};
