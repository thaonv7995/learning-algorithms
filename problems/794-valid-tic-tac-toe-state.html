Valid Tic-Tac-Toe State

O(1) O(1) 34.8% AC
Mô tả
Bàn cờ tic-tac-toe 3×3 với X,O. Kiểm tra board có valid state không (đúng lượt, không ai thắng hai, số X = O hoặc O+1).
Ví dụ
Ví dụ 1
Inboard = ["O "," "," "]
Outtrue
Ví dụ 2
Inboard = ["XOX"," X "," "]
Outfalse
Ví dụ 3
Inboard = ["XOX","O O","XOX"]
Outfalse
Ý tưởng
Đếm X,O; check win lines; valid nếu count ok và không conflict winners và winner matches count.

bool validTicTacToe(char** board, int n) {
    int x=0,o=0; for(int i=0;i<3;i++) for(int j=0;j<3;j++){ if(board[i][j]=='X') x++; if(board[i][j]=='O') o++; }
    if(o!=x && o!=x-1) return false; return true;
}
class Solution:
    def validTicTacToe(self, board: List[str]) -> bool:
        b = board
        def win(c):
            return any(all(b[r][c2]==c for c2 in range(3)) for r in range(3)) or \
                   any(all(b[r2][c]==c for r2 in range(3)) for c in range(3)) or \
                   all(b[i][i]==c for i in range(3)) or all(b[i][2-i]==c for i in range(3))
        x = sum(row.count("X") for row in b)
        o = sum(row.count("O") for row in b)
        if o not in (x, x-1): return False
        if win("X") and o == x: return False
        if win("O") and x != o: return False
        if win("X") and win("O"): return False
        return True
class Solution {
public:
    bool validTicTacToe(vector<string>& b) {
        auto win=[&](char c){
            for(int i=0;i<3;i++){ if(b[i][0]==c&&b[i][1]==c&&b[i][2]==c) return true; if(b[0][i]==c&&b[1][i]==c&&b[2][i]==c) return true; }
            return (b[0][0]==c&&b[1][1]==c&&b[2][2]==c)||(b[0][2]==c&&b[1][1]==c&&b[2][0]==c);
        };
        int x=0,o=0; for(auto& r:b){ for(char c:r){ if(c=='X') x++; if(c=='O') o++; } }
        if(o!=x && o!=x-1) return false;
        if(win('X') && o==x) return false;
        if(win('O') && x!=o) return false;
        if(win('X') && win('O')) return false;
        return true;
    }
};
