Insert Delete GetRandom O(1)

O(1) amortized mỗi thao tác O(n) 55.4% AC
Mô tả
Thiết kế RandomizedSet: insert, remove, getRandom đều trung bình O(1), không trùng phần tử.
Ví dụ
Ví dụ 1
Ininsert(1), remove(2), insert(2), getRandom()
Outtrue,false,true,2
Ví dụ 2
Ininsert(1), remove(1), insert(2), getRandom()
Outtrue,true,true,2
Ý tưởng
Vector lưu giá trị + hash map val→index; remove: swap phần tử cần xóa với phần tử cuối, pop O(1).

class RandomizedSet:
    def __init__(self):
        self.arr = []
        self.pos = {}
    def insert(self, val: int) -> bool:
        if val in self.pos: return False
        self.pos[val] = len(self.arr)
        self.arr.append(val)
        return True
    def remove(self, val: int) -> bool:
        if val not in self.pos: return False
        i = self.pos[val]
        last = self.arr[-1]
        self.arr[i] = last
        self.pos[last] = i
        self.arr.pop()
        del self.pos[val]
        return True
    def getRandom(self) -> int:
        return random.choice(self.arr)
class RandomizedSet {
    vector<int> arr;
    unordered_map<int,int> pos;
public:
    RandomizedSet() {}
    bool insert(int val) {
        if (pos.count(val)) return false;
        pos[val] = arr.size(); arr.push_back(val); return true;
    }
    bool remove(int val) {
        if (!pos.count(val)) return false;
        int i = pos[val], last = arr.back();
        arr[i] = last; pos[last] = i;
        arr.pop_back(); pos.erase(val); return true;
    }
    int getRandom() { return arr[rand() % arr.size()]; }
};
