Add One Row to Tree

O(n) O(n) 64.1% AC
Mô tả
Thêm một hàng val vào cây nhị phân tại depth d (1-indexed). Nếu depth vượt chiều cao, thêm làm leaf.
Ví dụ
Ví dụ 1
Inroot=[4,2,6,3,1,5], val=1, depth=2
Out[4,1,1,2,null,6,null,3,null,5]
Ví dụ 2
Inroot=[1], val=2, depth=1
Out[2,1]
Ý tưởng
DFS đến depth-1: nếu node null tạo mới; nếu depth==1 chèn val làm parent wrap node cũ.

class Solution:
    def addOneRow(self, root: Optional[TreeNode], val: int, depth: int) -> Optional[TreeNode]:
        if depth == 1: return TreeNode(val, root, None)
        def dfs(node, d):
            if not node: return
            if d == depth - 1:
                node.left = TreeNode(val, node.left, None)
                node.right = TreeNode(val, None, node.right)
            else:
                dfs(node.left, d+1); dfs(node.right, d+1)
        dfs(root, 1); return root
class Solution {
public:
    TreeNode* addOneRow(TreeNode* root, int val, int depth) {
        if(depth==1){ auto* n=new TreeNode(val); n->left=root; return n; }
        function<void(TreeNode*,int)> dfs=[&](TreeNode* node,int d){
            if(!node) return;
            if(d==depth-1){
                auto* l=new TreeNode(val); l->left=node->left; node->left=l;
                auto* r=new TreeNode(val); r->right=node->right; node->right=r;
            } else { dfs(node->left,d+1); dfs(node->right,d+1); }
        };
        dfs(root,1); return root;
    }
};
