Reconstruct Original Digits from English

O(n) O(1) 52.9% AC
Mô tả
Cho chuỗi s gồm chữ cái tiếng Anh viết thường của các số zero–nine đảo thứ tự. Trả các chữ số gốc theo thứ tự tăng dần.
Ví dụ
Ví dụ 1
Ins = "owoztneoer"
Out"012"
Ví dụ 2
Ins = "fviefuro"
Out"45"
Ví dụ 3
Ins = "nnei"
Out"9"
Ý tưởng
Đếm chữ cái; zero chỉ có z, two chỉ có w, four chỉ có u, six chỉ có x, eight chỉ có g. Trừ dần rồi suy three(seven), five, nine, one.

class Solution:
    def originalDigits(self, s: str) -> str:
        cnt = Counter(s)
        order = "zowtufxsgn"
        digits = "0123456789"
        uniq = "zowtufxg"
        ans = []
        for d, u in zip("02468", uniq):
            k = cnt[u]
            ans += [d] * k
            for c in ["zero","two","four","six","eight"][int(d)//2]:
                cnt[c] -= k
        for d, word in zip("3579", ["three","five","seven","nine"]):
            k = cnt[word[0]]
            ans += [d] * k
            for c in word: cnt[c] -= k
        ans += ["1"] * cnt["o"]
        return "".join(sorted(ans))
class Solution {
public:
    string originalDigits(string s) {
        int c[26]={};
        for(char ch:s) c[ch-'a']++;
        int cnt[10]={};
        cnt[0]=c['z'-'a']; cnt[2]=c['w'-'a']; cnt[4]=c['u'-'a'];
        cnt[6]=c['x'-'a']; cnt[8]=c['g'-'a'];
        c['o'-'a']-=cnt[0]+cnt[2]+cnt[4];
        c['t'-'a']-=cnt[2]+cnt[3]+cnt[8];
        c['f'-'a']-=cnt[4]+cnt[5];
        c['s'-'a']-=cnt[6]+cnt[7];
        c['g'-'a']-=cnt[8];
        cnt[3]=c['t'-'a']; cnt[5]=c['f'-'a']; cnt[7]=c['s'-'a']; cnt[9]=c['n'-'a'];
        cnt[1]=c['o'-'a'];
        string ans;
        for(int d=0;d<10;d++) ans += string(cnt[d], '0'+d);
        return ans;
    }
};
