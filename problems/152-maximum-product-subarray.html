Maximum Product Subarray

O(n) O(1) 36.4% AC
Mô tả
Cho mảng nums số nguyên. Tìm tích lớn nhất của một đoạn con liên tiếp (subarray) không rỗng. Mảng có thể chứa số âm và số 0.
Ví dụ
Ví dụ 1
Innums = [2,3,-2,4]
Out6
Ví dụ 2
Innums = [-2,0,-1]
Out0
Ví dụ 3
Innums = [-2]
Out-2
Ý tưởng
Kadane biến thể: duy trì curMax và curMin vì số âm đảo dấu tích. Với mỗi x: curMax = max(x, curMax*x, curMin*x); curMin tương tự; cập nhật best.
Mã nguồn
solution.c C99 
int maxProduct(int* nums, int n) {
    int best = nums[0], curMax = nums[0], curMin = nums[0];
    for (int i = 1; i < n; i++) {
        int x = nums[i];
        if (x < 0) { int t = curMax; curMax = curMin; curMin = t; }
        int a = x * curMax, b = x * curMin;
        curMax = x > a ? x : (a > b ? a : b);
        curMin = x < a ? x : (a < b ? a : b);
        if (curMax > best) best = curMax;
    }
    return best;
}
class Solution:
    def maxProduct(self, nums: List[int]) -> int:
        best = cur_max = cur_min = nums[0]
        for x in nums[1:]:
            if x < 0:
                cur_max, cur_min = cur_min, cur_max
            cur_max = max(x, cur_max * x)
            cur_min = min(x, cur_min * x)
            best = max(best, cur_max)
        return best
class Solution {
public:
    int maxProduct(vector<int>& nums) {
        int best = nums[0], curMax = nums[0], curMin = nums[0];
        for (int i = 1; i < (int)nums.size(); i++) {
            int x = nums[i];
            if (x < 0) swap(curMax, curMin);
            curMax = max(x, curMax * x);
            curMin = min(x, curMin * x);
            best = max(best, curMax);
        }
        return best;
    }
};
