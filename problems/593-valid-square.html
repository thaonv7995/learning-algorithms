Valid Square

O(1) O(1) 45% AC
Mô tả
Cho 4 điểm 2D. Kiểm tra có tạo thành hình vuông hợp lệ (4 cạnh bằng, 2 đường chéo bằng, không trùng điểm).
Ví dụ
Ví dụ 1
Inpoints = [[0,0],[1,1],[1,0],[0,1]]
Outtrue
Ví dụ 2
Inpoints = [[0,0],[1,1],[1,0],[2,1]]
Outfalse
Ví dụ 3
Inpoints = [[1,0],[0,1],[1,1],[0,0]]
Outtrue
Ý tưởng
Tính 6 khoảng cách bình; multiset phải có 4 cạnh bằng + 2 diagonal bằng (diagonal = 2*cạnh).
Mã nguồn

/* Six distances pattern — xem C++ */
class Solution:
    def validSquare(self, p1: List[int], p2: List[int], p3: List[int], p4: List[int]) -> bool:
        pts = [p1,p2,p3,p4]
        dists = []
        for i in range(4):
            for j in range(i+1,4):
                dx=pts[i][0]-pts[j][0]; dy=pts[i][1]-pts[j][1]
                dists.append(dx*dx+dy*dy)
        dists.sort()
        return dists[0]>0 and dists[0]==dists[1]==dists[2]==dists[3] and dists[4]==dists[5]==2*dists[0]
class Solution {
    int dist(vector<int>& a, vector<int>& b) {
        return (a[0]-b[0])*(a[0]-b[0])+(a[1]-b[1])*(a[1]-b[1]);
    }
public:
    bool validSquare(vector<int>& p1, vector<int>& p2, vector<int>& p3, vector<int>& p4) {
        vector<vector<int>*> pts = {&p1,&p2,&p3,&p4};
        vector<int> d;
        for (int i=0;i<4;i++) for (int j=i+1;j<4;j++) d.push_back(dist(*pts[i],*pts[j]));
        sort(d.begin(), d.end());
        return d[0]>0 && d[0]==d[1]&&d[1]==d[2]&&d[2]==d[3] && d[4]==d[5]&&d[5]==2*d[0];
    }
};
