Find All Duplicates in an Array

O(n) O(1) 77% AC
Mô tả
Mảng nums độ dài n, mỗi phần tử trong [1,n]. Một số giá trị xuất hiện hai lần, còn lại một lần. Trả danh sách các số xuất hiện hai lần. O(n) thời gian, O(1) bộ nhớ phụ.
Ví dụ
Ví dụ 1
Innums = [4,3,2,7,8,2,3,1]
Out[5,6]
Ví dụ 2
Innums = [1,1]
Out[2]
Ý tưởng
Đánh dấu: với x đặt nums[abs(x)-1] âm; cuối cùng index i dương → thiếu i+1.
Mã nguồn
solution.c C99 
int* findDuplicates(int* nums, int n, int* rs) {
    *rs=0; int cap=n; int* ans=malloc(cap*sizeof(int));
    for(int i=0;i<n;i++){
        int j=abs(nums[i])-1;
        if(nums[j]<0) ans[(*rs)++]=j+1;
        else nums[j]=-nums[j];
    }
    return ans;
}
class Solution:
    def findDuplicates(self, nums: List[int]) -> List[int]:
        ans = []
        for x in nums:
            i = abs(x) - 1
            if nums[i] < 0:
                ans.append(i + 1)
            else:
                nums[i] = -nums[i]
        return ans
class Solution {
public:
    vector<int> findDuplicates(vector<int>& nums) {
        vector<int> ans;
        for(int x:nums){
            int i=abs(x)-1;
            if(nums[i]<0) ans.push_back(i+1);
            else nums[i]=-nums[i];
        }
        return ans;
    }
};
