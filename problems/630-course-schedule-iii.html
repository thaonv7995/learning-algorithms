Course Schedule III

O(n log n) O(n) 41.7% AC
Mô tả
Cho courses[i]=[duration, deadline]. Làm tối đa bao nhiêu course sao tổng thời gian mỗi course ≤ deadline.
Ví dụ
Ví dụ 1
Incourses = [[100,200],[200,1300],[1000,1250],[2000,3200]]
Out3
Ví dụ 2
Incourses = [[5,5],[8,8],[7,7]]
Out3
Ý tưởng
Sort theo deadline; max-heap thời gian; nếu tổng>deadline pop course dài nhất.

typedef struct { int d, dl; } Course;
int cmpCourse(const void* a, const void* b){ return ((Course*)a)->dl - ((Course*)b)->dl; }
int scheduleCourse(int** c, int n, int* cs) {
    Course* a=malloc(n*sizeof(Course));
    for(int i=0;i<n;i++){ a[i].d=c[i][0]; a[i].dl=c[i][1]; }
    qsort(a,n,sizeof(Course),cmpCourse);
    int* heap=malloc(n*sizeof(int)); int hs=0; long t=0;
    for(int i=0;i<n;i++){
        t+=a[i].d; heap[hs++]=a[i].d;
        for(int j=hs-1;j>0;j--) if(heap[j]>heap[j-1]){ int tmp=heap[j]; heap[j]=heap[j-1]; heap[j-1]=tmp; } else break;
        if(t>a[i].dl){ t-=heap[0]; heap[0]=heap[--hs-1]; hs--; if(hs>1){ int mx=0; for(int j=1;j<hs;j++) if(heap[j]>heap[mx]) mx=j; int tmp=heap[0]; heap[0]=heap[mx]; heap[mx]=tmp; } }
    }
    int ans=hs; free(a); free(heap); return ans;
}
class Solution:
    def scheduleCourse(self, courses: List[List[int]]) -> int:
        courses.sort(key=lambda x: x[1])
        heap=[]; t=0
        for d,dl in courses:
            heapq.heappush(heap,-d); t+=d
            if t>dl: t+=heapq.heappop(heap)
        return len(heap)
class Solution {
public:
    int scheduleCourse(vector<vector<int>>& c) {
        sort(c.begin(),c.end(),[](auto&a,auto&b){return a[1]<b[1];});
        priority_queue<int> pq; long t=0;
        for(auto& v:c){
            t+=v[0]; pq.push(v[0]);
            if(t>v[1]){ t-=pq.top(); pq.pop(); }
        }
        return pq.size();
    }
};
