Partition to K Equal Sum Subsets

O(n*2^n) O(2^n) 38.6% AC
Mô tả
Chia nums thành k nhóm cùng tổng (partition equal subset k ways).
Ví dụ
Ví dụ 1
Innums=[4,3,2,3,5,2,1], k=4
Outtrue
Ví dụ 2
Innums=[1,2,3,4], k=3
Outfalse
Ý tưởng
If sum%k!=0 false; else backtrack k buckets with pruning.

class Solution:
    def canPartitionKSubsets(self, nums: List[int], k: int) -> bool:
        s=sum(nums)
        if s%k: return False
        target=s//k; nums.sort(reverse=True)
        used=[False]*len(nums)
        def dfs(start,k,cur):
            if k==1: return True
            if cur==target: return dfs(0,k-1,0)
            for i in range(start,len(nums)):
                if used[i] or cur+nums[i]>target: continue
                used[i]=True
                if dfs(i+1,k,cur+nums[i]): return True
                used[i]=False
            return False
        return dfs(0,k,0)
class Solution {

        }
        return false;
    }
