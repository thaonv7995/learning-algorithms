Solve the Equation

O(n) O(1) 46.2% AC
Mô tả
Giải phương trình ax+b=cx+d dạng chuỗi. Trả "x=val", "Infinite solutions" hoặc "No solution".
Ví dụ
Ví dụ 1
Inequation = "2x=x+2"
Out"x=2"
Ví dụ 2
Inequation = "x=x"
Out"Infinite solutions"
Ý tưởng
Parse hai vế; gom hệ số x và hằng; coef=0 → infinite/no; else x=-const/coef.

class Solution:
    def solveEquation(self, equation: str) -> str:
        def parse(s):
            coef=const=0; i=0; sign=1
            while i<len(s):
                if s[i]=='+': sign=1; i+=1
                elif s[i]=='-': sign=-1; i+=1
                j=i
                while j<len(s) and (s[j].isdigit() or s[j]=='x'): j+=1
                term=s[i:j]; i=j
                if 'x' in term:
                    c=term[:-1]; c=int(c) if c not in ('','+','-') else (sign if c in ('+','-') else 1)*sign
                    if c==0: c=sign
                    coef+=c
                else:
                    const+=sign*int(term)
            return coef,const
        left,right=equation.split('=')
        lc,lk=parse(left); rc,rk=parse(right)
        coef=lc-rc; const=rk-lk
        if coef==0: return "Infinite solutions" if const==0 else "No solution"
        if const%coef: return f"x={const/coef}"
        return f"x={const//coef}"
class Solution {
public:
    string solveEquation(string eq) {
        auto parse=[](string s){ long coef=0,const_=0,sign=1; for(int i=0;i<(int)s.size();){
            if(s[i]=='+'){sign=1;i++;} else if(s[i]=='-'){sign=-1;i++;}
            int j=i; while(j<(int)s.size()&&(isdigit(s[j])||s[j]=='x')) j++;
            string t=s.substr(i,j-i); i=j;
            if(t.find('x')!=string::npos){ long c=1; if(t.size()>1) c=stol(t.substr(0,t.size()-1)); else if(t[0]=='-') c=-1; coef+=sign*c;}
            else const_+=sign*stol(t);
        } return pair<long,long>{coef,const_};};
        int p=eq.find('='); auto [lc,lk]=parse(eq.substr(0,p)); auto [rc,rk]=parse(eq.substr(p+1));
        long coef=lc-rc, cst=rk-lk;
        if(!coef) return !cst?"Infinite solutions":"No solution";
        return "x="+to_string(cst/coef);
    }
};
