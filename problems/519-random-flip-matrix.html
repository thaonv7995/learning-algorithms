Random Flip Matrix

O(1) amortized per flip O(k) 45.9% AC
Mô tả
Ma trận m×n, flip ngẫu nhiên một ô chưa flip (uniform). reset() đưa về chưa flip. Implement class với flip() và reset().
Ví dụ
Ví dụ 1
InRandomFlipMatrix(1,2); flip(); flip(); reset()
OutHai ô khác nhau rồi reset
Ví dụ 2
Inm=2,n=2 nhiều flip
OutMỗi ô chưa flip có xác suất bằng nhau
Ý tưởng
Map index đã flip sang slot mới; pool = m*n - flipped. Chọn r ngẫu nhiên, hoán đổi với phần tử cuối pool (Fisher-Yates ảo).
Mã nguồn
solution.c C99 
/* Random flip hash map — xem C++ */
class Solution:
    def __init__(self, m: int, n: int):
        self.m, self.n = m, n
        self.total = m * n
        self.avail = self.total
        self.idx_map = {}
        self.reset()
    def flip(self) -> List[int]:
        import random
        r = random.randint(0, self.avail - 1)
        idx = self.idx_map.get(r, r)
        self.idx_map[r] = self.idx_map.get(self.avail - 1, self.avail - 1)
        self.avail -= 1
        return [idx // self.n, idx % self.n]
    def reset(self) -> None:
        self.avail = self.total
        self.idx_map.clear()
class Solution {
    int m, n, total, avail;
    unordered_map<int,int> mp;
public:
    Solution(int m, int n): m(m), n(n), total(m*n), avail(total) {}
    vector<int> flip() {
        int r = rand() % avail;
        int idx = mp.count(r) ? mp[r] : r;
        mp[r] = mp.count(avail-1) ? mp[avail-1] : avail-1;
        avail--;
        return {idx / n, idx % n};
    }
    void reset() { avail = total; mp.clear(); }
};
