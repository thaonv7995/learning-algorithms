Find Median from Data Stream

O(log n) O(n) 54.5% AC
Mô tả
Thiết kế cấu trúc nhận số từ stream: addNum(num) thêm số; findMedian() trả median (trung vị) hiện tại.
Ví dụ
Ví dụ 1
InaddNum(1); addNum(2); findMedian()→1.5; addNum(3); findMedian()→2
Out1.5, 2
Ví dụ 2
InaddNum(5); findMedian()→5
Out5
Ý tưởng
Hai heap: max-heap nửa thấp (lo), min-heap nửa cao (hi); cân |lo|≥|hi| và |lo|-|hi|≤1; median từ top.
Mã nguồn
solution.c C99 
typedef struct { int* lo; int* hi; int loN, hiN, loCap, hiCap; } MedianFinder;
static void pushLo(MedianFinder* m, int v) {
    if (m->loN == m->loCap) { m->loCap = m->loCap ? m->loCap * 2 : 4; m->lo = realloc(m->lo, m->loCap * 4); }
    m->lo[m->loN++] = v;
    for (int i = m->loN - 1; i > 0 && m->lo[i] > m->lo[i - 1]; i--) { int t = m->lo[i]; m->lo[i] = m->lo[i - 1]; m->lo[i - 1] = t; }
}
static int popLo(MedianFinder* m) { return m->lo[--m->loN]; }
static int topLo(MedianFinder* m) { return m->lo[m->loN - 1]; }
static void pushHi(MedianFinder* m, int v) {
    if (m->hiN == m->hiCap) { m->hiCap = m->hiCap ? m->hiCap * 2 : 4; m->hi = realloc(m->hi, m->hiCap * 4); }
    m->hi[m->hiN++] = v;
    for (int i = m->hiN - 1; i > 0 && m->hi[i] < m->hi[i - 1]; i--) { int t = m->hi[i]; m->hi[i] = m->hi[i - 1]; m->hi[i - 1] = t; }
}
static int popHi(MedianFinder* m) { return m->hi[--m->hiN]; }
static int topHi(MedianFinder* m) { return m->hi[m->hiN - 1]; }
void mfAdd(MedianFinder* m, int num) {
    pushLo(m, num);
    pushHi(m, popLo(m));
    if (m->loN < m->hiN) pushLo(m, popHi(m));
}
double mfMedian(MedianFinder* m) {
    if (m->loN > m->hiN) return topLo(m);
    return (topLo(m) + topHi(m)) / 2.0;
}
class MedianFinder:
    def __init__(self):
        self.lo = []  # max-heap via negate
        self.hi = []  # min-heap
    def addNum(self, num: int) -> None:
        heapq.heappush(self.lo, -num)
        heapq.heappush(self.hi, -heapq.heappop(self.lo))
        if len(self.lo) < len(self.hi):
            heapq.heappush(self.lo, -heapq.heappop(self.hi))
    def findMedian(self) -> float:
        if len(self.lo) > len(self.hi):
            return -self.lo[0]
        return (-self.lo[0] + self.hi[0]) / 2
class MedianFinder {
    priority_queue<int> lo;
    priority_queue<int, vector<int>, greater<int>> hi;
public:
    void addNum(int num) {
        lo.push(num);
        hi.push(lo.top()); lo.pop();
        if (lo.size() < hi.size()) { lo.push(hi.top()); hi.pop(); }
    }
    double findMedian() {
        if (lo.size() > hi.size()) return lo.top();
        return (lo.top() + hi.top()) / 2.0;
    }
};
