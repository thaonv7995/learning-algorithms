License Key Formatting

O(n) O(n) 46% AC
Mô tả
Format license key: bỏ dash, uppercase, chia nhóm k ký tự từ trái (nhóm đầu có thể ngắn hơn).
Ví dụ
Ví dụ 1
InS = "5F3Z-2e-9", k = 4
Out"5F3Z-2E9"
Ví dụ 2
InS = "2-5g-3-J", k = 2
Out"2-5G-3J"
Ví dụ 3
InS = "1", k = 1
Out"1"
Ý tưởng
Lọc alnum uppercase; rem=len%k; nhóm rem rồi các block k nối dash.

    char t[2000]; int p=0;
    for(int i=0;s[i];i++) if(s[i]!='-') t[p++]=toupper((unsigned char)s[i]);
    t[p]=0; if(!p){ char* r=malloc(1); r[0]=0; return r; }
    int rem=p%k; char* ans=malloc(p+p/k+2); int q=0, i=0;
    if(rem){ memcpy(ans,t,rem); q=rem; i=rem; if(i<p) ans[q++]='-'; }
    for(;i<p;i+=k){ memcpy(ans+q,t+i,k); q+=k; if(i+k<p) ans[q++]='-'; }
    ans[q]=0; return ans;

class Solution:
    def licenseKeyFormatting(self, s: str, k: int) -> str:
        t = "".join(ch.upper() for ch in s if ch != "-")
        if not t: return ""
        rem = len(t) % k
        parts = []
        if rem: parts.append(t[:rem])
        for i in range(rem, len(t), k):
            parts.append(t[i:i+k])
        return "-".join(parts)
class Solution {
public:
    string licenseKeyFormatting(string s, int k) {
        string t;
        for(char c:s) if(c!='-') t.push_back(toupper(c));
        if(t.empty()) return "";
        int rem=t.size()%k; string ans;
        int i=0;
        if(rem){ ans=t.substr(0,rem); i=rem; if(i<(int)t.size()) ans+='-'; }
        for(;i<(int)t.size();i+=k){ ans+=t.substr(i,k); if(i+k<(int)t.size()) ans+='-'; }
        return ans;
    }
};
