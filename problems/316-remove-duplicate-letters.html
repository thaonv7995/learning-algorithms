Remove Duplicate Letters

O(n) O(1) 53.3% AC
Mô tả
Cho chuỗi s chỉ chữ thường. Xóa ký tự trùng sao mỗi chữ còn đúng một lần và chuỗi kết quả nhỏ nhất theo thứ tự từ điển.
Ví dụ
Ví dụ 1
Ins = "bcabc"
Out"abc"
Ví dụ 2
Ins = "cbacdcbc"
Out"acdb"
Ý tưởng
Monotonic stack + count còn lại: duyệt s, giảm count[c]; nếu c chưa trong stack và stack.top()>c còn xuất hiện sau thì pop; push c.

char* removeDuplicateLetters(char* s) {
    int n = strlen(s), last[26], cnt[26]={0}, used[26]={0};
    for (int i=0;i<26;i++) last[i]=-1;
    for (int i=0;i<n;i++){last[s[i]-'a']=i;cnt[s[i]-'a']++;}
    char st[256]; int top=0;
    for (int i=0;i<n;i++){
        int c=s[i]-'a'; if (--cnt[c]<0) continue;
        if (used[c]) continue;
        while (top>0 && st[top-1]>s[i] && cnt[st[top-1]-'a']>0){used[st[--top]-'a']=0;}
        st[top++]=s[i]; used[c]=1;
    }
    st[top]=0; return strdup(st);
}
class Solution:
    def removeDuplicateLetters(self, s: str) -> str:
        last = {c: i for i, c in enumerate(s)}
        stack, in_stack = [], set()
        for i, c in enumerate(s):
            if c in in_stack:
                continue
            while stack and stack[-1] > c and last[stack[-1]] > i:
                in_stack.remove(stack.pop())
            stack.append(c); in_stack.add(c)
        return ''.join(stack)
class Solution {
public:
    string removeDuplicateLetters(string s) {
        vector<int> last(26, -1), cnt(26, 0);
        for (int i = 0; i < (int)s.size(); i++) { last[s[i]-'a'] = i; cnt[s[i]-'a']++; }
        string st; vector<bool> used(26, false);
        for (int i = 0; i < (int)s.size(); i++) {
            int c = s[i]-'a'; if (--cnt[c] < 0) continue;
            if (used[c]) continue;
            while (!st.empty() && st.back() > s[i] && cnt[st.back()-'a'] > 0) {
                used[st.back()-'a'] = false; st.pop_back();
            }
            st.push_back(s[i]); used[c] = true;
        }
        return st;
    }
};
