Ý tưởng

int calculate(char* s) {
    int stack[1000], top = 0, result = 0, num = 0, sign = 1;
    for (int i = 0; s[i]; i++) {
        char c = s[i];
        if (c >= '0' && c <= '9') num = num * 10 + (c - '0');
        else if (c == '+' || c == '-') {
            result += sign * num; num = 0; sign = (c == '+') ? 1 : -1;
        } else if (c == '(') {
            stack[top++] = result; stack[top++] = sign;
            result = 0; sign = 1; num = 0;
        } else if (c == ')') {
            result += sign * num; num = 0;
            result *= stack[--top];
            result += stack[--top];
        }
    }
    return result + sign * num;
}
class Solution:
    def calculate(self, s: str) -> int:
        stack, result, num, sign = [], 0, 0, 1
        for c in s:
            if c.isdigit():
                num = num * 10 + int(c)
            elif c in '+-':
                result += sign * num
                num = 0
                sign = 1 if c == '+' else -1
            elif c == '(':
                stack.append(result)
                stack.append(sign)
                result, sign, num = 0, 1, 0
            elif c == ')':
                result += sign * num
                num = 0
                result *= stack.pop()
                result += stack.pop()
        return result + sign * num

public:
    int calculate(string s) {

            } else if (c == '(') {
                st.push(result); st.push(sign);
                result = 0; sign = 1; num = 0;
            } else if (c == ')') {
                result += sign * num; num = 0;
                result *= st.top(); st.pop();
                result += st.top(); st.pop();
            }
        }

    }
