Accounts Merge

O(n log n) O(n) 61.3% AC
Mô tả
Cho danh sách accounts, mỗi phần tử [name, email1, email2, ...]. Gộp các account có email chung (transitive). Trả danh sách đã gộp: mỗi account sort email, danh sách sort theo name.
Ví dụ
Ví dụ 1
Inaccounts = [["John","johnsmith@mail.com","john_newyork@mail.com"],["John","johnsmith@mail.com","john00@mail.com"],["Mary","mary@mail.com"],["John","johnnybravo@mail.com"]]
Out[["John","john00@mail.com","john_newyork@mail.com","johnsmith@mail.com"],["Mary","mary@mail.com"],["John","johnnybravo@mail.com"]]
Ví dụ 2
Inaccounts = [["Gabe","Gabe0@gmail.com","Gabe1@gmail.com","Gabe3@gmail.com"],["Kevin","Kevin3@gmail.com","Kevin5@gmail.com"],["Ethan","Ethan1@gmail.com","Ethan2@gmail.com","Ethan0@gmail.com"]]
Out[["Ethan","Ethan0@gmail.com","Ethan1@gmail.com","Ethan2@gmail.com"],["Gabe","Gabe0@gmail.com","Gabe1@gmail.com","Gabe3@gmail.com"],["Kevin","Kevin3@gmail.com","Kevin5@gmail.com"]]
Ý tưởng
Union-Find trên index account: union khi email trùng (map email→first index). Mỗi component gom email unique, sort, output [name, ...emails].

static int par[1000];
static int uf_find(int x){ while(par[x]!=x){ par[x]=par[par[x]]; x=par[x]; } return x; }
static void uf_union(int a,int b){ a=uf_find(a); b=uf_find(b); if(a!=b) par[b]=a; }

class Solution:
    def accountsMerge(self, accounts: List[List[str]]) -> List[List[str]]:
        n = len(accounts)
        parent = list(range(n))
        def find(x):
            while parent[x] != x:
                parent[x] = parent[parent[x]]
                x = parent[x]
            return x
        def union(a, b):
            ra, rb = find(a), find(b)
            if ra != rb: parent[rb] = ra
        email_owner = {}
        for i, acc in enumerate(accounts):
            for e in acc[1:]:
                if e in email_owner: union(i, email_owner[e])
                else: email_owner[e] = i
        groups = defaultdict(set)
        names = {}
        for i, acc in enumerate(accounts):
            r = find(i)
            names[r] = acc[0]
            groups[r].update(acc[1:])
        out = []
        for r, emails in groups.items():
            row = [names[r]] + sorted(emails)
            out.append(row)
        out.sort(key=lambda x: x[0])
        return out
class Solution {
    vector<int> p;
    int find(int x){ return p[x]==x?x:p[x]=find(p[x]); }
    void uni(int a,int b){ a=find(a); b=find(b); if(a!=b) p[b]=a; }
public:
    vector<vector<string>> accountsMerge(vector<vector<string>>& acc) {
        int n=acc.size(); p.resize(n); iota(p.begin(),p.end(),0);
        unordered_map<string,int> own;
        for(int i=0;i<n;i++) for(int j=1;j<(int)acc[i].size();j++){
            string& e=acc[i][j];
            if(own.count(e)) uni(i, own[e]); else own[e]=i;
        }
        unordered_map<int, set<string>> g;
        unordered_map<int,string> name;
        for(int i=0;i<n;i++){
            int r=find(i); name[r]=acc[i][0]; g[r].insert(acc[i].begin()+1, acc[i].end());
        }
        vector<vector<string>> ans;
        for(auto& kv: g){
            vector<string> row={name[kv.first]};
            for(auto& e: kv.second) row.push_back(e);
            ans.push_back(move(row));
        }
        sort(ans.begin(), ans.end(), [](auto& a, auto& b){ return a[0]<b[0]; });
        return ans;
    }
};
