Minimum Size Subarray Sum

O(n) O(1) 51.7% AC
Mô tả
Cho mảng số nguyên dương nums và target. Tìm độ dài subarray liên tiếp nhỏ nhất có tổng ≥ target. Trả 0 nếu không tồn tại.
Ví dụ
Ví dụ 1
Intarget = 7, nums = [2,3,1,2,4,3]
Out2
Ví dụ 2
Intarget = 4, nums = [1,4,4]
Out1
Ví dụ 3
Intarget = 11, nums = [1,1,1,1,1,1,1,1]
Out0
Ý tưởng
Sliding window: r mở rộng tổng, khi sum≥target thu l left++ cập nhật minLen, lặp đến hết.
Mã nguồn
solution.c C99 
int minSubArrayLen(int target, int* nums, int n) {
    int ans = 2147483647, s = 0, l = 0;
    for (int r = 0; r < n; r++) {
        s += nums[r];
        while (s >= target) {
            if (r - l + 1 < ans) ans = r - l + 1;
            s -= nums[l++];
        }
    }
    return ans == 2147483647 ? 0 : ans;
}
class Solution:
    def minSubArrayLen(self, target: int, nums: List[int]) -> int:
        n, ans, s, l = len(nums), float('inf'), 0, 0
        for r in range(n):
            s += nums[r]
            while s >= target:
                ans = min(ans, r - l + 1)
                s -= nums[l]
                l += 1
        return 0 if ans == float('inf') else ans
class Solution {
public:
    int minSubArrayLen(int target, vector<int>& nums) {
        int n = nums.size(), ans = INT_MAX, s = 0, l = 0;
        for (int r = 0; r < n; r++) {
            s += nums[r];
            while (s >= target) {
                ans = min(ans, r - l + 1);
                s -= nums[l++];
            }
        }
        return ans == INT_MAX ? 0 : ans;
    }
};
