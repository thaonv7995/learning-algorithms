Find K-th Smallest Pair Distance

O(n log n + n log W) O(1) 46.5% AC
Mô tả
Cho mảng nums và k. Với mọi cặp (i,j), distance = |nums[i]-nums[j]|. Trả k-th smallest trong tất cả n(n-1)/2 khoảng cách (1-indexed).
Ví dụ
Ví dụ 1
Innums = [1,3,1,4,1], k = 3
Out0
Ví dụ 2
Innums = [1,6,1], k = 3
Out5
Ví dụ 3
Innums = [1,2,3,4], k = 6
Out2
Ý tưởng
Sort nums. Binary search answer d: đếm số cặp có distance <= d bằng two pointers trên mảng sort. Tìm min d sao count>=k.

static int cmp(const void* a,const void* b){ return *(int*)a-*(int*)b; }
int smallestDistancePair(int* nums, int n, int k) {
    qsort(nums,n,4,cmp);
    int lo=0, hi=nums[n-1]-nums[0];
    while(lo<hi){
        int mid=(lo+hi)/2, c=0, j=0;
        for(int i=0;i<n;i++){ while(j<n&&nums[j]-nums[i]<=mid) j++; c+=j-i-1; }
        if(c>=k) hi=mid; else lo=mid+1;
    }
    return lo;
}
class Solution:
    def smallestDistancePair(self, nums: List[int], k: int) -> int:
        nums.sort()
        n = len(nums)
        lo, hi = 0, nums[-1] - nums[0]
        def count(mid):
            c = j = 0
            for i in range(n):
                while j < n and nums[j] - nums[i] <= mid:
                    j += 1
                c += j - i - 1
            return c >= k
        while lo < hi:
            mid = (lo + hi) // 2
            if count(mid): hi = mid
            else: lo = mid + 1
        return lo
class Solution {
public:
    int smallestDistancePair(vector<int>& nums, int k) {
        sort(nums.begin(), nums.end());
        int n = nums.size(), lo = 0, hi = nums.back() - nums[0];
        auto enough = [&](int mid){
            int c = 0, j = 0;
            for (int i = 0; i < n; i++) {
                while (j < n && nums[j] - nums[i] <= mid) j++;
                c += j - i - 1;
                if (c >= k) return true;
            }
            return false;
        };
        while (lo < hi) {
            int mid = lo + (hi - lo) / 2;
            if (enough(mid)) hi = mid;
            else lo = mid + 1;
        }
        return lo;
    }
};
