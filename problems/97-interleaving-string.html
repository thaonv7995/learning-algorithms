Interleaving String

O(m·n) O(m·n) 44% AC
Mô tả
Cho s1, s2, s3 cùng độ dài |s1|+|s2|=|s3|. Kiểm tra s3 có phải interleaving của s1 và s2 không (giữ thứ tự ký tự trong từng chuỗi gốc).
Ví dụ
Ví dụ 1
Ins1 = "aabcc", s2 = "dbbca", s3 = "aadbbcbcac"
Outtrue
Ví dụ 2
Ins1 = "aabcc", s2 = "dbbca", s3 = "aadbbbaccc"
Outfalse
Ý tưởng
DP[i][j] = s3[0..i+j) interleave được từ s1[:i] và s2[:j]. Chuyển từ (i-1,j) nếu s1[i-1]==s3[i+j-1] hoặc (i,j-1) nếu s2[j-1]==s3[i+j-1].
Mã nguồn
solution.c C99 
/* isInterleave — DP 2D; xem C++ */
class Solution:
    def isInterleave(self, s1: str, s2: str, s3: str) -> bool:
        m, n = len(s1), len(s2)
        if m + n != len(s3): return False
        dp = [[False]*(n+1) for _ in range(m+1)]
        dp[0][0] = True
        for i in range(m+1):
            for j in range(n+1):
                k = i + j
                if i and s1[i-1]==s3[k-1]: dp[i][j] |= dp[i-1][j]
                if j and s2[j-1]==s3[k-1]: dp[i][j] |= dp[i][j-1]
        return dp[m][n]
class Solution {
public:
    bool isInterleave(string a, string b, string c) {
        int m=a.size(), n=b.size();
        if (m+n!=(int)c.size()) return false;
        vector<vector<bool>> dp(m+1, vector<bool>(n+1));
        dp[0][0]=true;
        for (int i=0;i<=m;i++) for (int j=0;j<=n;j++) {
            int k=i+j;
            if (i&&a[i-1]==c[k-1]) dp[i][j]|=dp[i-1][j];
            if (j&&b[j-1]==c[k-1]) dp[i][j]|=dp[i][j-1];
        }
        return dp[m][n];
    }
};
