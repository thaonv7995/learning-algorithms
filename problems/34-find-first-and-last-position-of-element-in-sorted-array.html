Find First and Last Position of Element in Sorted Array

O(log N) O(1) 48.9% AC
Mô tả
Cho mảng đã sort và target, trả về [first, last] index của target hoặc [-1,-1].
Ví dụ
Ví dụ 1
Innums = [5,7,7,8,8,10], target = 8
Out[3,4]
Ví dụ 2
Innums = [5,7,7,8,8,10], target = 6
Out[-1,-1]
Ý tưởng
Hai lần binary search: tìm leftmost (bias trái khi nums[mid]==target) và rightmost (bias phải).
Mã nguồn
solution.c C99 
int* searchRange(int* nums, int n, int target, int* rs) { *rs=2; /* two binary searches */ return rs; }
class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        def bs(left_bias):
            l, r, ans = 0, len(nums)-1, -1
            while l <= r:
                m = (l+r)//2
                if nums[m] == target:
                    ans = m
                    if left_bias: r = m-1
                    else: l = m+1
                elif nums[m] < target: l = m+1
                else: r = m-1
            return ans
        return [bs(True), bs(False)]
class Solution {
public:
    vector<int> searchRange(vector<int>& nums, int target) {
        auto bs = [&](bool left) {
            int l=0,r=nums.size()-1,ans=-1;
            while(l<=r){int m=l+(r-l)/2;
                if(nums[m]==target){ans=m; if(left)r=m-1; else l=m+1;}
                else if(nums[m]<target)l=m+1; else r=m-1;}
            return ans;
        };
        return {bs(true), bs(false)};
    }
};
