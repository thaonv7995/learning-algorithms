K Inverse Pairs Array

O(n*k) O(k) 49% AC
Mô tả
Đếm số permutation của 1..n có đúng k inverse pairs (i<j nhưng arr[i]>arr[j]).
Ví dụ
Ví dụ 1
Inn = 3, k = 0
Out1
Ví dụ 2
Inn = 3, k = 1
Out2
Ý tưởng
DP[i][j] = số perm 1..i với j inversions; thêm i: dp[i][j]+=dp[i-1][j-k] for k=0..min(j,i-1).
Mã nguồn
solution.c C99 
int kInversePairs(int n, int k) {
    const int MOD=1000000007;
    int* dp=calloc(k+1,sizeof(int)); dp[0]=1;
    for(int i=2;i<=n;i++){
        int* nd=calloc(k+1,sizeof(int)); long s=0;
        for(int j=0;j<=k;j++){
            s=(s+dp[j])%MOD; if(j>=i) s=(s-dp[j-i]+MOD)%MOD;
            nd[j]=(int)s;
        }
        free(dp); dp=nd;
    }
    int ans=dp[k]; free(dp); return ans;
}
class Solution:
    def kInversePairs(self, n: int, k: int) -> int:
        MOD=10**9+7
        dp=[1]+[0]*k
        for i in range(2,n+1):
            nd=[0]*(k+1); s=0
            for j in range(k+1):
                s=(s+dp[j])%(MOD)
                if j>=i: s=(s-dp[j-i])%MOD
                nd[j]=s
            dp=nd
        return dp[k]
