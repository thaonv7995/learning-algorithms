Ones and Zeroes

O(n*m*maxM) O(m*maxM) 53.4% AC
Mô tả
Mỗi strs[i] có m[i] số 0 và n[i] số 1. Chọn tối đa bao nhiêu chuỗi với tổng 0≤m và 1≤n.
Ví dụ
Ví dụ 1
Instrs = ["10","0001","111001","1","0"], m = 5, n = 3
Out4
Ví dụ 2
Instrs = ["10","0","1"], m = 1, n = 1
Out2
Ý tưởng
2D knapsack dp[i][j] = max strings with i zeros j ones; for each string update dp backward.
Mã nguồn
solution.c C99 
int findMaxForm(char** strs, int sn, int m, int n) {
    int dp[101][101]={0};
    for(int k=0;k<sn;k++){
        int a=0; for(char* p=strs[k];*p;p++) if(*p=='0') a++;
        int b=strlen(strs[k])-a;
        for(int i=m;i>=a;i--) for(int j=n;j>=b;j--)
            if(dp[i-a][j-b]+1>dp[i][j]) dp[i][j]=dp[i-a][j-b]+1;
    }
    return dp[m][n];
}
class Solution:
    def findMaxForm(self, strs: List[str], m: int, n: int) -> int:
        dp = [[0]*(n+1) for _ in range(m+1)]
        for s in strs:
            a = s.count("0"); b = len(s)-a
            for i in range(m, a-1, -1):
                for j in range(n, b-1, -1):
                    dp[i][j] = max(dp[i][j], dp[i-a][j-b]+1)
        return dp[m][n]
class Solution {
public:
    int findMaxForm(vector<string>& strs, int m, int n) {
        vector<vector<int>> dp(m+1, vector<int>(n+1));
        for(auto& s:strs){
            int a=count(s.begin(),s.end(),'0'), b=s.size()-a;
            for(int i=m;i>=a;i--)

        }
