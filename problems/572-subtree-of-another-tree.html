Subtree of Another Tree

O(m * n) O(h) 51.6% AC
Mô tả
Kiểm tra subRoot có phải subtree của root không (cùng cấu trúc và giá trị).
Ví dụ
Ví dụ 1
Inroot=[3,4,5,1,2], subRoot=[4,1,2]
Outtrue
Ví dụ 2
Inroot=[3,4,5,1,2,null,null,null,0], subRoot=[4,1,2]
Outfalse
Ý tưởng
DFS root: tại mỗi node gọi isSameTree(node, subRoot); nếu khớp return true.
Mã nguồn
solution.c C99 
/* DFS isSameTree — xem C++ */
class Solution:
    def isSubtree(self, root: Optional[TreeNode], subRoot: Optional[TreeNode]) -> bool:
        def same(a, b):
            if not a and not b: return True
            if not a or not b or a.val != b.val: return False
            return same(a.left,b.left) and same(a.right,b.right)
        if not root: return False
        if same(root, subRoot): return True
        return self.isSubtree(root.left, subRoot) or self.isSubtree(root.right, subRoot)
class Solution {
    bool same(TreeNode* a, TreeNode* b) {
        if (!a && !b) return true;
        if (!a || !b || a->val != b->val) return false;
        return same(a->left,b->left) && same(a->right,b->right);
    }
public:
    bool isSubtree(TreeNode* root, TreeNode* subRoot) {
        if (!root) return false;
        if (same(root, subRoot)) return true;
        return isSubtree(root->left, subRoot) || isSubtree(root->right, subRoot);
    }
};
