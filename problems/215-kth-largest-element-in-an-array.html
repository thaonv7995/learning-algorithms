Kth Largest Element in an Array

O(n) avg O(1) 68.9% AC
Mô tả
Cho mảng nums không rỗng và số nguyên k. Trả phần tử lớn thứ k trong mảng (1-indexed, không phải thứ tự distinct).
Ví dụ
Ví dụ 1
Innums = [3,2,1,5,6,4], k = 2
Out5
Ví dụ 2
Innums = [3,2,3,1,2,4,5,5,6], k = 4
Out4
Ý tưởng
Quickselect: partition quanh pivot ngẫu nhiên/cuối; nếu pivot index == n-k thì trả; else recurse trái/phải.

static int partition(int* a, int l, int r) {
    int pivot = a[r], i = l, tmp;
    for (int j = l; j < r; j++)
        if (a[j] <= pivot) { tmp=a[i]; a[i]=a[j]; a[j]=tmp; i++; }
    tmp=a[i]; a[i]=a[r]; a[r]=tmp;
    return i;
}
static int selectK(int* a, int l, int r, int t) {
    int p = partition(a, l, r);
    if (p == t) return a[p];
    return p > t ? selectK(a, l, p-1, t) : selectK(a, p+1, r, t);
}
int findKthLargest(int* nums, int n, int k) {
    return selectK(nums, 0, n-1, n-k);
}
class Solution:
    def findKthLargest(self, nums: List[int], k: int) -> int:
        target = len(nums) - k
        def select(l, r):
            pivot = nums[r]
            i = l
            for j in range(l, r):
                if nums[j] <= pivot:
                    nums[i], nums[j] = nums[j], nums[i]
                    i += 1
            nums[i], nums[r] = nums[r], nums[i]
            if i == target:
                return nums[i]
            return select(l, i - 1) if i > target else select(i + 1, r)
        return select(0, len(nums) - 1)
class Solution {
    int partition(vector<int>& a, int l, int r) {
        int pivot = a[r], i = l;
        for (int j = l; j < r; j++)
            if (a[j] <= pivot) swap(a[i++], a[j]);
        swap(a[i], a[r]);
        return i;
    }
    int select(vector<int>& a, int l, int r, int t) {
        int p = partition(a, l, r);
        if (p == t) return a[p];
        return p > t ? select(a, l, p - 1, t) : select(a, p + 1, r, t);
    }
public:
    int findKthLargest(vector<int>& nums, int k) {
        return select(nums, 0, nums.size() - 1, (int)nums.size() - k);
    }
};
