Integer to English Words

O(log n) O(log n) 35% AC
Mô tả
Chuyển số nguyên không âm num (0 ≤ num ≤ 2^31-1) thành chữ tiếng Anh (units, tens, hundreds, thousand, million, billion).
Ví dụ
Ví dụ 1
Innum = 123
Out"One Hundred Twenty Three"
Ví dụ 2
Innum = 12345
Out"Twelve Thousand Three Hundred Forty Five"
Ví dụ 3
Innum = 0
Out"Zero"
Ý tưởng
Chia khối 1000 (billion/million/thousand); helper convert 0-999 bằng bảng dưới 20, tens, hundreds.

static const char* below20[] = {"One","Two","Three","Four","Five","Six","Seven","Eight","Nine","Ten","Eleven","Twelve","Thirteen","Fourteen","Fifteen","Sixteen","Seventeen","Eighteen","Nineteen"};
static const char* tens[] = {"Ten","Twenty","Thirty","Forty","Fifty","Sixty","Seventy","Eighty","Ninety"};
static void under1000(int n, char* out) {
    out[0] = 0;
    if (!n) return;
    if (n < 20) { strcpy(out, below20[n - 1]); return; }
    if (n < 100) {
        strcpy(out, tens[n / 10 - 1]);
        if (n % 10) { strcat(out, " "); strcat(out, below20[n % 10 - 1]); }
        return;
    }
    strcpy(out, below20[n / 100 - 1]); strcat(out, " Hundred");
    if (n % 100) { char tail[64]; under1000(n % 100, tail); strcat(out, " "); strcat(out, tail); }
}
char* numberToWords(int num) {
    if (num == 0) return strdup("Zero");
    char* out = calloc(256, 1);
    struct { int u; const char* lab; } units[] = {{1000000000,"Billion"},{1000000,"Million"},{1000,"Thousand"}};
    for (int i = 0; i < 3; i++) if (num >= units[i].u) {
        char part[64]; under1000(num / units[i].u, part);
        if (out[0]) strcat(out, " ");
        strcat(out, part); strcat(out, " "); strcat(out, units[i].lab);
        num %= units[i].u;
    }
    if (num) { char part[64]; under1000(num, part); if (out[0]) strcat(out, " "); strcat(out, part); }
    return out;
}
class Solution:
    def numberToWords(self, num: int) -> str:
        if num == 0:
            return "Zero"
        below20 = "One Two Three Four Five Six Seven Eight Nine Ten Eleven Twelve Thirteen Fourteen Fifteen Sixteen Seventeen Eighteen Nineteen".split()
        tens = "Ten Twenty Thirty Forty Fifty Sixty Seventy Eighty Ninety".split()
        def under1000(n):
            if n == 0:
                return ""
            if n < 20:
                return below20[n - 1]
            if n < 100:
                return tens[n // 10 - 1] + (" " + below20[n % 10 - 1] if n % 10 else "")
            return below20[n // 100 - 1] + " Hundred" + (" " + under1000(n % 100) if n % 100 else "")
        parts = []
        for unit, label in ((1_000_000_000, "Billion"), (1_000_000, "Million"), (1000, "Thousand")):
            if num >= unit:
                parts.append(under1000(num // unit) + " " + label)
                num %= unit
        if num:
            parts.append(under1000(num))
        return " ".join(parts)
class Solution {
    vector<string> below20 = {"One","Two","Three","Four","Five","Six","Seven","Eight","Nine","Ten","Eleven","Twelve","Thirteen","Fourteen","Fifteen","Sixteen","Seventeen","Eighteen","Nineteen"};
    vector<string> tens = {"Ten","Twenty","Thirty","Forty","Fifty","Sixty","Seventy","Eighty","Ninety"};
    string under1000(int n) {
        if (!n) return "";
        if (n < 20) return below20[n-1];
        if (n < 100) return tens[n/10-1] + (n%10 ? " " + below20[n%10-1] : "");
        return below20[n/100-1] + string(" Hundred") + (n%100 ? " " + under1000(n%100) : "");
    }
public:
    string numberToWords(int num) {
        if (!num) return "Zero";
        string res;
        vector<pair<int,string>> units = {{1000000000,"Billion"},{1000000,"Million"},{1000,"Thousand"}};
        for (auto& [u, lab] : units)
            if (num >= u) { res += under1000(num/u) + " " + lab + " "; num %= u; }
        if (num) res += under1000(num);
        while (!res.empty() && res.back() == ' ') res.pop_back();
        return res;
    }
};
