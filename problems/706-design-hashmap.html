Design HashMap

O(1) avg O(n) 66.6% AC
Mô tả
Thiết kế HashMap: put(key,value), get(key), remove(key). Key và value trong [0, 10^6].
Ví dụ
Ví dụ 1
In["MyHashMap","put","put","get","get","put","get","remove","get"], [[],[1,1],[2,2],[1],[3],[2,1],[2],[2],[2]]
Out[null,null,null,1,-1,null,1,null,-1]
Ví dụ 2
In["MyHashMap","put","get","remove","get"], [[],[0,1],[0],[0],[0]]
Out[null,null,1,null,-1]
Ý tưởng
Chaining như HashSet nhưng mỗi bucket lưu pair (key,val). put upsert; get trả -1 nếu thiếu.
Mã nguồn
solution.c C99 
#define MSZ 769
static int mk[MSZ][128], mv[MSZ][128], mc[MSZ];
static int mhash(int k){ return ((k%MSZ)+MSZ)%MSZ; }
void myHashMapPut(int key,int val){ int b=mhash(key); for(int i=0;i<mc[b];i++) if(mk[b][i]==key){ mv[b][i]=val; return; } mk[b][mc[b]]=key; mv[b][mc[b]++]=val; }
int myHashMapGet(int key){ int b=mhash(key); for(int i=0;i<mc[b];i++) if(mk[b][i]==key) return mv[b][i]; return -1; }
void myHashMapRemove(int key){ int b=mhash(key); for(int i=0;i<mc[b];i++) if(mk[b][i]==key){ mk[b][i]=mk[b][mc[b]-1]; mv[b][i]=mv[b][mc[b]-1]; mc[b]--; return; } }
class MyHashMap:
    BASE = 769
    def __init__(self):
        self.data = [[] for _ in range(self.BASE)]
    def _hash(self, key): return key % self.BASE
    def put(self, key: int, value: int) -> None:
        h = self._hash(key)
        for i,(k,v) in enumerate(self.data[h]):
            if k == key:
                self.data[h][i] = (key, value); return
        self.data[h].append((key, value))
    def get(self, key: int) -> int:
        for k,v in self.data[self._hash(key)]:
            if k == key: return v
        return -1
    def remove(self, key: int) -> None:
        h = self._hash(key)
        self.data[h] = [(k,v) for k,v in self.data[h] if k != key]
class MyHashMap {
    static const int BASE = 769;
    vector<vector<pair<int,int>>> data;
    int hash(int k){ return k % BASE; }
public:
    MyHashMap() : data(BASE) {}
    void put(int key, int value) {
        auto& b = data[hash(key)];
        for (auto& p : b) if (p.first == key) { p.second = value; return; }
        b.emplace_back(key, value);
    }
    int get(int key) {
        for (auto& p : data[hash(key)]) if (p.first == key) return p.second;
        return -1;
    }
    void remove(int key) {
        auto& b = data[hash(key)];
        b.erase(remove_if(b.begin(), b.end(), [&](auto& p){ return p.first == key; }), b.end());
    }
};
