Concatenated Words

O(n * L^2) O(n * L) 49.8% AC
Mô tả
words và danh sách từ. Trả mọi từ có thể ghép từ ≥2 từ nhỏ hơn trong words (mỗi từ nhỏ dùng một lần).
Ví dụ
Ví dụ 1
Inwords = ["cat","cats","catsdogcats","dog","catdogcatdog"]
Out["catdogcatdog","catsdogcats"]
Ví dụ 2
Inwords = ["cat","dog","catdog"]
Out["catdog"]
Ý tưởng
Trie insert words; DFS mỗi word kiểm tra có thể segment thành ≥2 dict words (DP on prefix).

class Solution:
    def findAllConcatenatedWordsInADict(self, words: List[str]) -> List[str]:
        words.sort(key=len)
        trie = {}
        def add(w):
            node = trie
            for c in w: node = node.setdefault(c, {})
            node["#"] = True
        def ok(w):
            if not w: return True
            node = trie
            for i, c in enumerate(w):
                if c not in node: return False
                node = node[c]
                if "#" in node and ok(w[i+1:]): return True
            return False
        ans = []
        for w in words:
            if not w: continue
            if ok(w): ans.append(w)
            add(w)
        return ans
class Solution {
public:
    vector<string> findAllConcatenatedWordsInADict(vector<string>& words) {
        sort(words.begin(), words.end(), [](auto& a, auto& b){ return a.size()<b.size(); });
        unordered_set<string> dict; vector<string> ans;
        function<bool(const string&)> can=[&](const string& w){
            if(w.empty()) return true;
            for(int i=1;i<=(int)w.size();i++)
                if(dict.count(w.substr(0,i)) && can(w.substr(i))) return true;
            return false;
        };
        for(auto& w:words){
            if(!w.empty() && can(w)) ans.push_back(w);
            dict.insert(w);
        }
        return ans;
    }
};
