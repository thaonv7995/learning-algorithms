class Solution:
    def minStickers(self, stickers: List[str], target: str) -> int:
        from collections import Counter
        def match(t, s):
            cnt=Counter(s); rem=[]
            for c in t:
                if cnt[c]: cnt[c]-=1
                else: rem.append(c)
            return "".join(rem)
        dp={ "": 0 }
        for _ in range(len(target)+1):
            nxt=dict(dp)
            for state, steps in dp.items():
                for st in stickers:
                    ns=match(state if state else target, st)
                    if ns not in nxt or steps+1<nxt[ns]: nxt[ns]=steps+1
            dp=nxt
            if "" in dp: return dp[""]
        return -1
class Solution {
public:
    int minStickers(vector<string>& st, string target) {
        queue<pair<string,int>> q; q.push({target,0});
        unordered_set<string> vis={target};
        while(!q.empty()){
            auto [t,step]=q.front(); q.pop();
            if(t.empty()) return step;
            for(auto& s:st){
                string rem; array<int,26> cnt{};
                for(char c:s) cnt[c-'a']++;
                for(char c:t){ if(cnt[c-'a']) cnt[c-'a']--; else rem.push_back(c); }
                if(!vis.count(rem)){ vis.insert(rem); q.push({rem,step+1}); }
            }
        }
        return -1;
    }
};
