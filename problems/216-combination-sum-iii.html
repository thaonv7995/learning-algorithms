Combination Sum III

O(C(9,k)) O(k) 73.2% AC
Mô tả
Tìm mọi bộ k số khác nhau từ 1..9 có tổng bằng n. Mỗi số dùng tối đa một lần. Trả danh sách các bộ (thứ tự tăng trong mỗi bộ).
Ví dụ
Ví dụ 1
Ink = 3, n = 7
Out[[1,2,4]]
Ví dụ 2
Ink = 3, n = 9
Out[[1,2,6],[1,3,5],[2,3,4]]
Ví dụ 3
Ink = 4, n = 1
Out[]
Ý tưởng
Backtracking: start từ 1, thử thêm i nếu còn slot và n-i≥0; path đủ k và sum=n thì push.

static void dfs216(int* path, int depth, int k, int start, int rem, int** out, int* outSz, int* cap) {
    if (k == 0) {
        if (rem == 0) {
            int* row = malloc(depth * sizeof(int));
            memcpy(row, path, depth * sizeof(int));
            if (*outSz >= *cap) { *cap *= 2; out[0] = realloc(out[0], *cap * sizeof(int*)); }
            ((int**)out)[(*outSz)++] = row;
        }
        return;
    }
    if (rem <= 0) return;
    for (int i = start; i <= 9; i++) {
        if (i > rem) break;
        path[depth] = i;
        dfs216(path, depth+1, k-1, i+1, rem-i, out, outSz, cap);
    }
}
int** combinationSum3(int k, int n, int* retCol, int* retSize) {
    int cap=16, sz=0, path[9];
    int** out = malloc(cap*sizeof(int*));
    dfs216(path,0,k,1,n,&out,&sz,&cap);
    *retSize=sz; *retCol=k; return out;
}
class Solution:
    def combinationSum3(self, k: int, n: int) -> List[List[int]]:
        ans, path = [], []
        def dfs(start, rem, left):
            if left == 0:
                if rem == 0:
                    ans.append(path[:])
                return
            if rem <= 0:
                return
            for i in range(start, 10):
                if i > rem:
                    break
                path.append(i)
                dfs(i + 1, rem - i, left - 1)
                path.pop()
        dfs(1, n, k)
        return ans
class Solution {
    vector<vector<int>> ans;
    vector<int> path;
    void dfs(int start, int rem, int k) {
        if (k == 0) { if (rem == 0) ans.push_back(path); return; }
        if (rem <= 0) return;
        for (int i = start; i <= 9; i++) {
            if (i > rem) break;
            path.push_back(i);
            dfs(i + 1, rem - i, k - 1);
            path.pop_back();
        }
    }
public:
    vector<vector<int>> combinationSum3(int k, int n) {
        dfs(1, n, k);
        return ans;
    }
};
