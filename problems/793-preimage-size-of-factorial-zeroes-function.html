Preimage Size of Factorial Zeroes Function

O(log K) O(1) 47% AC
Mô tả
f(x) = số x sao cho x! có đúng K số 0 ở cuối (preimage size). Cho K, trả |{x : f(x)=K}|.
Ví dụ
Ví dụ 1
InK = 0
Out0
Ví dụ 2
InK = 5
Out0
Ví dụ 3
InK = 3
Out5
Ý tưởng
f(x) staircase non-decreasing. Tìm range [lo,hi] where f(x)=K bằng BS on x counting trailing zeros in x!.

static int f793(long long x){ int s=0; while(x){ x/=5; s+=x; } return s; }
int preimageSizeFZF(int k) {
    if(k==0) return 0;
    long long lo=0, hi=5LL*(k+1);
    while(lo<hi){ long long mid=(lo+hi)/2; if(f793(mid)<k) lo=mid+1; else hi=mid; }
    return f793(lo)==k?5:0;
}
class Solution:
    def preimageSizeFZF(self, k: int) -> int:
        def f(x):
            s = 0
            while x:
                x //= 5; s += x
            return s
        if k == 0: return 0
        lo, hi = 0, 5 * (k + 1)
        while lo < hi:
            mid = (lo + hi) // 2
            if f(mid) < k: lo = mid + 1
            else: hi = mid
        if f(lo) != k: return 0
        return 5
class Solution {
public:
    int preimageSizeFZF(int k) {
        auto f=[](long long x){ int s=0; while(x){ x/=5; s+=x; } return s; };
        if(k==0) return 0;
        long long lo=0, hi=5LL*(k+1);
        while(lo<hi){ long long mid=(lo+hi)/2; if(f(mid)<k) lo=mid+1; else hi=mid; }
        if(f(lo)!=k) return 0;
        return 5;
    }
};
