Maximum Distance in Arrays

O(n) O(1) 45.6% AC
Mô tả
Cho m arrays đã sort. Max distance |a[i]-b[j]| với a,b từ hai array khác nhau (một phần tử mỗi array).
Ví dụ
Ví dụ 1
Inarrays = [[1,2,3],[4,5],[1,2,3]]
Out4
Ví dụ 2
Inarrays = [[1],[1]]
Out0
Ý tưởng
Track global min/max first element seen; với array mới max dist = max(max-minArr, maxArr-min).
Mã nguồn
solution.c C99 
int maxDistance(int** arrays, int m, int* sizes) {
    int mn=arrays[0][0], mx=arrays[0][sizes[0]-1], ans=0;
    for(int i=1;i<m;i++){
        int lo=arrays[i][0], hi=arrays[i][sizes[i]-1];
        int d1=hi-mn; if(d1<0)d1=-d1; int d2=mx-lo; if(d2<0)d2=-d2;
        if(d1>ans) ans=d1; if(d2>ans) ans=d2;
        if(lo<mn) mn=lo; if(hi>mx) mx=hi;
    }
    return ans;
}
class Solution:
    def maxDistance(self, arrays: List[List[int]]) -> int:
        mn, mx, ans = arrays[0][0], arrays[0][-1], 0
        for arr in arrays[1:]:
            ans = max(ans, abs(arr[-1]-mn), abs(mx-arr[0]))
            mn = min(mn, arr[0]); mx = max(mx, arr[-1])
        return ans
class Solution {
public:
    int maxDistance(vector<vector<int>>& a) {
        int mn=a[0][0], mx=a[0].back(), ans=0;
        for(int i=1;i<(int)a.size();i++){
            ans=max({ans, abs(a[i].back()-mn), abs(mx-a[i][0])});
            mn=min(mn,a[i][0]); mx=max(mx,a[i].back());
        }
        return ans;
    }
};
