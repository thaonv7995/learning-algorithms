K-th Smallest in Lexicographical Order

O(log n) O(1) 46.3% AC
Mô tả
Trả số thứ k (1-indexed) trong thứ tự lexicographical của số 1..n.
Ví dụ
Ví dụ 1
Inn = 13, k = 2
Out10
Ví dụ 2
Inn = 1, k = 1
Out1
Ví dụ 3
Inn = 20, k = 5
Out14
Ý tưởng
Trie walk: tại prefix cur đếm số trong [cur, cur+1) ≤ n; nếu count<k thì cur++, else k-=count và cur*=10.
Mã nguồn
solution.c C99 
long long countSteps(long long cur, long long n){
    long long s=0,nxt=cur+1;
    while(cur<=n){ s+=(n+1<nxt?n+1:nxt)-cur; cur*=10; nxt*=10; }
    return s;
}
int findKthNumber(int n, int k){
    long long cur=1; k--;
    while(k){
        long long cnt=countSteps(cur,n);
        if(cnt<=k){ k-=cnt; cur++; } else { cur*=10; k--; }
    }
    return (int)cur;
}
class Solution:
    def findKthNumber(self, n: int, k: int) -> int:
        def count(cur, n):
            s, nxt = 0, cur + 1
            while cur <= n:
                s += min(n + 1, nxt) - cur
                cur *= 10; nxt *= 10
            return s
        cur = 1; k -= 1
        while k:
            cnt = count(cur, n)
            if cnt <= k: k -= cnt; cur += 1
            else: cur *= 10; k -= 1
        return cur
