Integer Replacement

O(log n) O(log n) 37.5% AC
Mô tả
Bắt đầu từ n, mỗi bước: n chẵn thì /2; n lẻ thì +1 hoặc -1. Tìm số bước tối thiểu về 1.
Ví dụ
Ví dụ 1
Inn = 8
Out3
Ví dụ 2
Inn = 7
Out4
Ý tưởng
Memo DFS: n==1 return 0; n chẵn 1+dfs(n/2); n lẻ 1+min(dfs(n+1), dfs(n-1)). Greedy: n lẻ chọn +1 trừ n==3 hoặc n%4==1 và n!=3.
Mã nguồn
solution.c C99 
static int dfs(long long n) {
    if (n == 1) return 0;
    if (n % 2 == 0) return 1 + dfs(n / 2);
    int a = dfs(n + 1), b = dfs(n - 1);
    return 1 + (a < b ? a : b);
}
int integerReplacement(int n) { return dfs(n); }
class Solution:
    def integerReplacement(self, n: int) -> int:
        memo = {}
        def dfs(x):
            if x == 1: return 0
            if x in memo: return memo[x]
            if x % 2 == 0:
                memo[x] = 1 + dfs(x // 2)
            else:
                memo[x] = 1 + min(dfs(x + 1), dfs(x - 1))
            return memo[x]
        return dfs(n)
class Solution {
    unordered_map<long long,int> memo;
    int dfs(long long n) {
        if (n == 1) return 0;
        if (memo.count(n)) return memo[n];
        if (n % 2 == 0) return memo[n] = 1 + dfs(n/2);
        return memo[n] = 1 + min(dfs(n+1), dfs(n-1));
    }
public:
    int integerReplacement(int n) { return dfs(n); }
};
