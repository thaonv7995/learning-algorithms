Lexicographical Numbers

O(n) O(1) 76.2% AC
Mô tả
Trả mảng số từ 1..n theo thứ tự từ điển (coi số như chuỗi: 1,10,11,...,2,20,...).
Ví dụ
Ví dụ 1
Inn = 13
Out[1,10,11,12,13,2,3,4,5,6,7,8,9]
Ví dụ 2
Inn = 2
Out[1,2]
Ý tưởng
DFS trên cây 10-ary ảo: từ cur, thử cur*10 và cur+1 nếu ≤n; không cần stack — vòng while next number.
Mã nguồn

int* lexicalOrder(int n, int* retSize) {
    int* ans = malloc(n * sizeof(int)); *retSize = n;
    int cur = 1;
    for (int i = 0; i < n; i++) {
        ans[i] = cur;
        if (cur * 10 <= n) cur *= 10;
        else if (cur % 10 != 9 && cur + 1 <= n) cur++;
        else { cur /= 10; while (cur % 10 == 9) cur /= 10; cur++; }
    }
    return ans;
}
class Solution:
    def lexicalOrder(self, n: int) -> List[int]:
        ans, cur = [], 1
        for _ in range(n):
            ans.append(cur)
            if cur * 10 <= n:
                cur *= 10
            elif cur % 10 != 9 and cur + 1 <= n:
                cur += 1
            else:
                cur //= 10
                while cur % 10 == 9:
                    cur //= 10
                cur += 1
        return ans

public:
    vector<int> lexicalOrder(int n) {
        vector<int> ans; int cur = 1;
        for (int i = 0; i < n; i++) {
            ans.push_back(cur);
            if (cur * 10 <= n) cur *= 10;
