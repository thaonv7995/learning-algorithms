Super Pow

O(n) O(1) 37% AC
Mô tả
Tính a^b mod 1337 với b là mảng chữ số (số rất lớn).
Ví dụ
Ví dụ 1
Ina = 2, b = [3]
Out8
Ví dụ 2
Ina = 2, b = [1,0]
Out1024
Ý tưởng
Modular exponentiation kết hợp xử lý b từng digit: result = pow(result, 10, mod) * pow(a, digit, mod) % mod.
Mã nguồn

static long long modPow(long long a, long long e, int mod) {
    long long r = 1; a %= mod;
    while (e) { if (e & 1) r = r * a % mod; a = a * a % mod; e >>= 1; }
    return r;
}
int superPow(int a, int* b, int bSize) {
    const int mod = 1337;
    long long ans = 1; a %= mod;
    for (int i = 0; i < bSize; i++) ans = modPow(ans, 10, mod) * modPow(a, b[i], mod) % mod;
    return (int)ans;
}
class Solution:
    def superPow(self, a: int, b: List[int]) -> int:
        mod = 1337
        a %= mod
        ans = 1
        for d in b:
            ans = pow(ans, 10, mod) * pow(a, d, mod) % mod
        return ans

public:
