Add Binary

O(N) O(N) 58.1% AC
Mô tả
Cộng hai chuỗi nhị phân a và b, trả kết quả dạng chuỗi nhị phân.
Ví dụ
Ví dụ 1
Ina = "11", b = "1"
Out"100"
Ví dụ 2
Ina = "1010", b = "1011"
Out"10101"
Ý tưởng
Two pointers từ cuối, cộng bit + carry. Push vào result, reverse.
Mã nguồn
solution.c C99 
char* addBinary(char* a, char* b) {
    int la=strlen(a), lb=strlen(b), mx=la+lb+2;
    char* r=(char*)malloc(mx); int k=0, i=la-1, j=lb-1, c=0;
    while(i>=0||j>=0||c){ int s=c; if(i>=0)s+=a[i--]-'0'; if(j>=0)s+=b[j--]-'0'; r[k++]=('0'+s%2); c=s/2; }
    r[k]=0; for(int l=0,h=k-1;l<h;l++,h--){char t=r[l];r[l]=r[h];r[h]=t;} return r;
}
class Solution:
    def addBinary(self, a: str, b: str) -> str:
        i, j, carry, res = len(a)-1, len(b)-1, 0, []
        while i >= 0 or j >= 0 or carry:
            s = carry
            if i >= 0: s += int(a[i]); i -= 1
            if j >= 0: s += int(b[j]); j -= 1
            res.append(str(s % 2)); carry = s // 2
        return ''.join(reversed(res))
class Solution {
public:
    string addBinary(string a, string b) {
        string res; int i=a.size()-1, j=b.size()-1, c=0;
        while(i>=0||j>=0||c){
            int s=c;
            if(i>=0) s+=a[i--]-'0';
            if(j>=0) s+=b[j--]-'0';
            res+=char('0'+s%2); c=s/2;
        }
        reverse(res.begin(),res.end());
        return res;
    }
};
