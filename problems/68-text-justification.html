Text Justification

O(N) O(N) 51.2% AC
Mô tả
Căn đều từng dòng trong words sao cho mỗi dòng (trừ dòng cuối) có đúng maxWidth ký tự, từ cách đều bằng space.
Ví dụ
Ví dụ 1
Inwords = ["This","is","an","example"], maxWidth = 16
Out["This is an","example "]
Ý tưởng
Gom từ vào dòng cho đến khi vượt maxWidth. Phân bổ space: base = totalSpaces/slots, extra = totalSpaces%slots cho slot đầu.
Mã nguồn

/* fullJustify — simulation phân bổ space; xem C++ */
class Solution:
    def fullJustify(self, words: List[str], w: int) -> List[str]:
        res, i, n = [], 0, len(words)
        while i < n:
            j = i; ln = 0
            while j < n and ln + len(words[j]) + (j - i) <= w:
                ln += len(words[j]); j += 1
            cnt = j - i
            if j == n or cnt == 1:
                line = ' '.join(words[i:j])
                res.append(line + ' ' * (w - len(line)))
            else:
                spaces = w - ln; gap, extra = divmod(spaces, cnt - 1)
                s = ''
                for k in range(cnt - 1):
                    s += words[i+k] + ' ' * (gap + (1 if k < extra else 0))
                s += words[j-1]; res.append(s)
            i = j
        return res
class Solution {
public:
    vector<string> fullJustify(vector<string>& words, int w) {
        vector<string> res; int i=0,n=words.size();
        while(i<n){
            int j=i,len=0;
            while(j<n&&len+(int)words[j].size()+(j-i)<=w){len+=words[j].size();j++;}
            int cnt=j-i; string line;
            if(j==n||cnt==1){
                for(int k=i;k<j;k++){line+=words[k];if(k<j-1)line+=' ';}
                line+=string(w-line.size(),' ');
            }else{
                int sp=w-len, gap=sp/(cnt-1), ex=sp%(cnt-1);
                for(int k=0;k<cnt-1;k++){line+=words[i+k]+string(gap+(k<ex), ' ');}
                line+=words[j-1];
            }
            res.push_back(line); i=j;
        }
        return res;
    }
};
