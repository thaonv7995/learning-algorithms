Minimum Moves to Equal Array Elements II

O(n log n) O(1) 61.9% AC
Mô tả
Mỗi move chọn i và tăng hoặc giảm nums[i] đúng 1. Min moves để mọi phần tử bằng nhau.
Ví dụ
Ví dụ 1
Innums = [1,2,3]
Out2
Ví dụ 2
Innums = [1,0,0,8,6]
Out14
Ý tưởng
Target optimal là median. Answer = sum |nums[i] - median|.
Mã nguồn
solution.c C99 
int cmpi462(const void* a,const void* b){ return *(int*)a-*(int*)b; }
int minMoves2(int* nums, int n){
    qsort(nums,n,sizeof(int),cmpi462);
    int m=nums[n/2], ans=0;
    for(int i=0;i<n;i++){ int d=nums[i]-m; ans+=d<0?-d:d; }
    return ans;
}
class Solution:
    def minMoves2(self, nums: List[int]) -> int:
        nums.sort()
        m = nums[len(nums)//2]
        return sum(abs(x - m) for x in nums)
class Solution {
public:
    int minMoves2(vector<int>& nums) {
        nth_element(nums.begin(), nums.begin()+nums.size()/2, nums.end());
        int m=nums[nums.size()/2], ans=0;
        for(int x:nums) ans+=abs(x-m);
        return ans;
    }
};
