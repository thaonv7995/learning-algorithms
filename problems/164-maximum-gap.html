Maximum Gap

O(n) O(n) 52.1% AC
Mô tả
Cho mảng nums số nguyên không âm, sort và tìm khoảng cách lớn nhất giữa hai phần tử liên tiếp trong mảng đã sort. Yêu cầu O(n) nếu có thể.
Ví dụ
Ví dụ 1
Innums = [3,6,9,1]
Out3
Ví dụ 2
Innums = [10]
Out0
Ý tưởng
Bucket sort / pigeonhole: gap max ≥ (max-min)/(n-1). Tạo n-1 bucket, mỗi bucket chỉ cần min/max. Max gap nằm giữa bucket liền kề hoặc trong bucket (chỉ cần biên).

int maximumGap(int* nums, int n) {
    if (n < 2) return 0;
    int lo = nums[0], hi = nums[0];
    for (int i = 1; i < n; i++) { if (nums[i] < lo) lo = nums[i]; if (nums[i] > hi) hi = nums[i]; }
    if (lo == hi) return 0;
    int gap = (hi - lo) / (n - 1); if (gap < 1) gap = 1;
    int* bmin = malloc((n-1)*sizeof(int)); int* bmax = malloc((n-1)*sizeof(int));
    for (int i = 0; i < n-1; i++) { bmin[i] = 2147483647; bmax[i] = -2147483647; }
    for (int i = 0; i < n; i++) {
        if (nums[i]==lo||nums[i]==hi) continue;
        int j = (nums[i]-lo)/gap;
        if (nums[i]<bmin[j]) bmin[j]=nums[i]; if (nums[i]>bmax[j]) bmax[j]=nums[i];
    }
    int prev=lo, ans=0;
    for (int i=0;i<n-1;i++) if (bmin[i]!=2147483647) { if (bmin[i]-prev>ans) ans=bmin[i]-prev; prev=bmax[i]; }
    if (hi-prev>ans) ans=hi-prev;
    free(bmin); free(bmax); return ans;
}
class Solution:
    def maximumGap(self, nums: List[int]) -> int:
        n = len(nums)
        if n < 2: return 0
        lo, hi = min(nums), max(nums)
        if lo == hi: return 0
        gap = (hi - lo) // (n - 1) or 1
        buckets = [[float('inf'), float('-inf')] for _ in range(n - 1)]
        for x in nums:
            if x == lo or x == hi: continue
            i = (x - lo) // gap
            buckets[i][0] = min(buckets[i][0], x)
            buckets[i][1] = max(buckets[i][1], x)
        prev, ans = lo, 0
        for b in buckets:
            if b[0] == float('inf'): continue
            ans = max(ans, b[0] - prev)
            prev = b[1]
        return max(ans, hi - prev)
class Solution {
public:
    int maximumGap(vector<int>& nums) {
        int n = nums.size();
        if (n < 2) return 0;
        int lo = *min_element(nums.begin(), nums.end());
        int hi = *max_element(nums.begin(), nums.end());
        if (lo == hi) return 0;
        int gap = max(1, (hi - lo) / (n - 1));
        vector<pair<int,int>> b(n - 1, {INT_MAX, INT_MIN});
        for (int x : nums) {
            if (x == lo || x == hi) continue;
            int i = (x - lo) / gap;
            b[i].first = min(b[i].first, x);
            b[i].second = max(b[i].second, x);
        }
        int prev = lo, ans = 0;
        for (auto& p : b) {
            if (p.first == INT_MAX) continue;
            ans = max(ans, p.first - prev);
            prev = p.second;
        }
        return max(ans, hi - prev);
    }
};
