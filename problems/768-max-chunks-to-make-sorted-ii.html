Max Chunks To Make Sorted II

O(n) O(1) 54.8% AC
Mô tả
Permutation nums của 0..n-1 (có thể trùng). Chia thành nhiều chunk, sort từng chunk, ghép lại thành mảng sort. Trả số chunk tối đa (hard: duplicates).
Ví dụ
Ví dụ 1
Innums = [5,4,3,2,1]
Out1
Ví dụ 2
Innums = [2,1,3,4,4]
Out3
Ý tưởng
Track max trong chunk hiện tại và sorted prefix max. Khi gặp phần tử > sorted_max thì cần chunk mới.

int maxChunksToSortedII(int* nums, int n) {
    int need[1001]={0}, have[1001]={0}, ans=0, curMax=-1;
    for(int i=0;i<n;i++){ need[nums[i]]++; have[nums[i]]++; if(nums[i]>curMax) curMax=nums[i];
        int ok=1; for(int v=0;v<=curMax;v++) if(have[v]<need[v]){ ok=0; break; }
        if(ok){ ans++; for(int v=0;v<=curMax;v++) have[v]=0; curMax=-1; }
    } return ans;
}
class Solution:
    def maxChunksToSorted(self, nums: List[int]) -> int:
        cnt = Counter(nums)
        need = defaultdict(int)
        have = defaultdict(int)
        ans = cur_max = 0
        for x in nums:
            need[x] += 1
            have[x] += 1
            cur_max = max(cur_max, x)
            ok = all(have[v] >= need[v] for v in range(cur_max + 1))
            if ok:
                ans += 1
                for v in range(cur_max + 1):
                    have[v] = 0
                cur_max = -1
        return ans
class Solution {
public:
    int maxChunksToSorted(vector<int>& nums) {
        map<int,int> need, have;
        int ans=0, curMax=-1;
        for(int x:nums){
            need[x]++; have[x]++; curMax=max(curMax,x);
            bool ok=true;
            for(int v=0;v<=curMax;v++) if(have[v]<need[v]){ ok=false; break; }
            if(ok){ ans++; for(int v=0;v<=curMax;v++) have[v]=0; curMax=-1; }
        }
        return ans;
    }
};
