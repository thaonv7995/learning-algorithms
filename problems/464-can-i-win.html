Can I Win

O(n * 2^n) O(2^n) 31.4% AC
Mô tả
Game: hai người lấy số từ 1..maxChoosableInteger, mỗi số dùng một lần, cộng vào tổng, ai đạt desiredTotal trước thắng. Người 1 đi trước — liệu chắc thắng?
Ví dụ
Ví dụ 1
InmaxChoosableInteger = 10, desiredTotal = 11
Outfalse
Ví dụ 2
InmaxChoosableInteger = 10, desiredTotal = 0
Outtrue
Ý tưởng
Bitmask DP/memo: state mask used numbers; current player win if any move makes opponent lose.
Mã nguồn

bool canIWin(int mx, int tot) { return false; /* bitmask DP */ }
class Solution:
    def canIWin(self, maxChoosableInteger: int, desiredTotal: int) -> bool:
        if maxChoosableInteger * (maxChoosableInteger + 1) // 2 < desiredTotal: return False
        if desiredTotal <= 0: return True
        memo = {}
        def dfs(mask, rem):
            if mask in memo: return memo[mask]
            for i in range(maxChoosableInteger):
                if mask >> i & 1: continue
                pick = i + 1
                if pick >= rem: return memo.setdefault(mask, True)
                if not dfs(mask | (1 << i), rem - pick): return memo.setdefault(mask, True)
            return memo.setdefault(mask, False)
        return dfs(0, desiredTotal)
class Solution {
public:
    bool canIWin(int mx, int tot) {
        if(mx*(mx+1)/2<tot) return false;
        if(tot<=0) return true;
        vector<int> memo(1<<mx, -1);
        function<bool(int,int)> dfs=[&](int mask,int rem)->bool{
            if(memo[mask]!=-1) return memo[mask];
            for(int i=0;i<mx;i++){
                if(mask>>i&1) continue;
                int pick=i+1;
                if(pick>=rem) return memo[mask]=true;
                if(!dfs(mask|(1<<i), rem-pick)) return memo[mask]=true;
            }
            return memo[mask]=false;
        };
        return dfs(0,tot);
    }
};
