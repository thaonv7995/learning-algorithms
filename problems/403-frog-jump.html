Mô tả
Ếch nhảy trên các đá stones (tọa độ tăng dần). Từ đá đầu chỉ nhảy 1 đơn vị. Từ đá trước đó đã nhảy k đơn vị, lần sau nhảy k-1, k hoặc k+1. Trả true nếu tới được đá cuối.
Ví dụ
Ví dụ 1
Instones = [0,1,3,5,6,8,12,17]
Outtrue
Ví dụ 2
Instones = [0,1,2,3,4,8,9,11]
Outfalse
Ví dụ 3
Instones = [0]
Outtrue
Ý tưởng
DP[i][k] = có thể tới stones[i] bằng bước k (k≤200). Map pos→index; từ mỗi (i,k) thử nhảy k-1,k,k+1 tới đá kế.

bool canCross(int* stones, int n) {
    int dp[2000][210] = {0};
    int idx[2001]; for(int i=0;i<2001;i++) idx[i]=-1;
    for(int i=0;i<n;i++) idx[stones[i]]=i;
    dp[0][1]=1;
    for(int i=0;i<n;i++) for(int k=1;k<=200;k++) if(dp[i][k]){
        int p=stones[i]+k; if(p>2000||idx[p]<0) continue;
        dp[idx[p]][k]=1; if(k>1) dp[idx[p]][k-1]=1; dp[idx[p]][k+1]=1;
    }
    for(int k=1;k<=200;k++) if(dp[n-1][k]) return true;
    return n==1;
}
class Solution:
    def canCross(self, stones: List[int]) -> bool:
        pos = {s: i for i, s in enumerate(stones)}
        n = len(stones)
        dp = [set() for _ in range(n)]
        dp[0].add(0)
        for i in range(n):
            for k in dp[i]:
                for nk in (k-1, k, k+1):
                    if nk <= 0: continue
                    j = stones[i] + nk
                    if j in pos:
                        dp[pos[j]].add(nk)
        return bool(dp[n-1])
class Solution {
public:
    bool canCross(vector<int>& stones) {
        unordered_map<int,int> pos;
        for (int i=0;i<(int)stones.size();i++) pos[stones[i]]=i;
        int n=stones.size();
        vector<unordered_set<int>> dp(n);
        dp[0].insert(0);
        for (int i=0;i<n;i++)
            for (int k : dp[i])
                for (int nk : {k-1,k,k+1})
                    if (nk>0 && pos.count(stones[i]+nk))
                        dp[pos[stones[i]+nk]].insert(nk);
        return !dp[n-1].empty();
    }
};
