Shortest Completing Word

O(n * L) O(1) 63.2% AC
Mô tả
Cho licensePlate và mảng words. Tìm word ngắn nhất trong words có thể hoàn thành licensePlate (bỏ qua khoảng/số; không phân biệt hoa thường). Tie → word đầu tiên theo thứ tự mảng.
Ví dụ
Ví dụ 1
InlicensePlate = "1s3 PSt", words = ["step","steps","stripe","stepple"]
Out"steps"
Ví dụ 2
InlicensePlate = "1s3 456", words = ["looks","pest","stew","show"]
Out"pest"
Ý tưởng
Đếm chữ cái licensePlate. Word hợp lệ nếu count chữ trong word >= count license. Chọn ngắn nhất.

char* shortestCompletingWord(char* licensePlate, char** words, int n) {
    int need[26]={0};
    for(char* p=licensePlate;*p;p++) if((*p>='A'&&*p<='Z')||(*p>='a'&&*p<='z')) need[tolower(*p)-'a']++;
    char* best=NULL; int bl=1e9;
    for(int i=0;i<n;i++){
        int have[26]={0}; for(char* p=words[i];*p;p++) if(isalpha(*p)) have[tolower(*p)-'a']++;
        int ok=1; for(int j=0;j<26;j++) if(have[j]<need[j]){ ok=0; break; }
        if(ok){ int L=strlen(words[i]); if(L<bl){ bl=L; best=words[i]; } }
    }
    return best?best:"";
}
class Solution:
    def shortestCompletingWord(self, licensePlate: str, words: List[str]) -> str:
        cnt = Counter(c.lower() for c in licensePlate if c.isalpha())
        best = ""
        for w in words:
            wc = Counter(w.lower())
            if all(wc[c] >= cnt[c] for c in cnt):
                if not best or len(w) < len(best): best = w
        return best
class Solution {
public:
    string shortestCompletingWord(string licensePlate, vector<string>& words) {
        int need[26]={};
        for(char c:licensePlate) if(isalpha(c)) need[tolower(c)-'a']++;
        string best;
        for(auto& w:words){
            int have[26]={};
            for(char c:w) have[tolower(c)-'a']++;
            bool ok=true;
            for(int i=0;i<26;i++) if(have[i]<need[i]){ ok=false; break; }
            if(ok && (best.empty()||w.size()<best.size())) best=w;
        }
        return best;
    }
};
