Contains Duplicate II

O(n) O(min(n,k)) 51.3% AC
Mô tả
Cho mảng nums và số nguyên k. Trả true nếu tồn tại hai chỉ số i≠j với |i-j|≤k và nums[i]==nums[j].
Ví dụ
Ví dụ 1
Innums = [1,2,3,1], k = 3
Outtrue
Ví dụ 2
Innums = [1,0,1,1], k = 1
Outtrue
Ví dụ 3
Innums = [1,2,3,1,2,3], k = 2
Outfalse
Ý tưởng
Sliding window hash map value→last index: nếu value đã có và i - last ≤ k → true; cập nhật index mỗi bước.

bool containsNearbyDuplicate(int* nums, int n, int k) {
    /* value -> last index; linear scan with small open hash or sort pairs */
    for (int i = 0; i < n; i++)
        for (int j = i + 1; j <= i + k && j < n; j++)
            if (nums[i] == nums[j]) return true;
    return false;
}
class Solution:
    def containsNearbyDuplicate(self, nums: List[int], k: int) -> bool:
        last = {}
        for i, x in enumerate(nums):
            if x in last and i - last[x] <= k:
                return True
            last[x] = i
        return False
class Solution {
public:
    bool containsNearbyDuplicate(vector<int>& nums, int k) {
        unordered_map<int,int> last;
        for (int i = 0; i < (int)nums.size(); i++) {
            if (last.count(nums[i]) && i - last[nums[i]] <= k) return true;
            last[nums[i]] = i;
        }
        return false;
    }
};
