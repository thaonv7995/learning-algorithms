class Solution:
    def maxSumOfThreeSubarrays(self, nums: List[int], k: int = 3) -> List[int]:
        n=len(nums); ps=[0]
        for x in nums: ps.append(ps[-1]+x)
        left=[0]*n; best=0; idx=0
        for i in range(n-k+1):
            s=ps[i+k]-ps[i]
            if s>best: best=s; idx=i
            left[i]=idx
        right=[0]*n; best=0; idx=0
        for i in range(n-k,-1,-1):
            s=ps[i+k]-ps[i]
            if s>=best: best=s; idx=i
            right[i]=idx
        ans=[0,0,0]; best_sum=-1
        for j in range(k,n-2*k+1):
            i=left[j-k]; l=right[j+k]
            s=ps[i+3]-ps[i]+ps[j+3]-ps[j]+ps[l+3]-ps[l]
            if s>best_sum: best_sum=s; ans=[i,j,l]
        return ans
