Minimum Index Sum of Two Lists

O(n + m) O(n + m) 59.9% AC
Mô tả
Hai list đều không trùng tên trong từng list. Tìm tên xuất hiện cả hai list sao indexSum = index1+index2 nhỏ nhất.
Ví dụ
Ví dụ 1
Inlist1=["Shogun","Tapioca","Burger"], list2=["Piatti","The Grill","Trousers","Overstock"]
Out["Shogun"]
Ví dụ 2
InCùng indexSum nhiều tên
OutTất cả tên có min sum
Ý tưởng
Hash map tên→index list1; duyệt list2, nếu chung thì cập nhật min sum và danh sách kết quả.

/* Hash index intersection — xem C++ */
class Solution:
    def findRestaurant(self, list1: List[str], list2: List[str]) -> List[str]:
        idx = {w:i for i,w in enumerate(list1)}
        best = 10**9
        ans = []
        for j, w in enumerate(list2):
            if w in idx:
                s = idx[w] + j
                if s < best:
                    best = s
                    ans = [w]
                elif s == best:
                    ans.append(w)
        return ans
class Solution {
public:
    vector<string> findRestaurant(vector<string>& list1, vector<string>& list2) {
        unordered_map<string,int> idx;
        for (int i=0;i<(int)list1.size();i++) idx[list1[i]]=i;
        int best = INT_MAX;
        vector<string> ans;
        for (int j=0;j<(int)list2.size();j++) {
            auto it = idx.find(list2[j]);
            if (it==idx.end()) continue;
            int s = it->second + j;
            if (s < best) { best = s; ans = {list2[j]}; }
            else if (s == best) ans.push_back(list2[j]);
        }
        return ans;
    }
};
