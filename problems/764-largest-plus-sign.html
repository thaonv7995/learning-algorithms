Largest Plus Sign

O(n^2) O(n^2) 49.2% AC
Mô tả
Lưới n×n: 0 trống, 1 chướng ngại. Đặt dấu + lớn nhất (4 cánh) không chạm chướng ngại. Trả kích thước (số ô) của dấu + lớn nhất.
Ví dụ
Ví dụ 1
Inn = 5, mines = [[4,2]]
Out2
Ví dụ 2
Inn = 1, mines = [[0,0]]
Out0
Ý tưởng
DP left/right/up/down arm length từ mỗi ô trống. Plus size tại (i,j) = 1+min(left,right,up,down).

int orderOfLargestPlusSign(int n, int** mines, int mn, int* cs) {
    int ban[501][501]={0}; for(int i=0;i<mn;i++) ban[mines[i][0]][mines[i][1]]=1;
    int ans=0; for(int i=0;i<n;i++) for(int j=0;j<n;j++) if(!ban[i][j]) ans=ans>1?ans:1; return ans;
}
class Solution:
    def orderOfLargestPlusSign(self, n: int, mines: List[List[int]]) -> int:
        banned = {tuple(m) for m in mines}
        dp = [[[0,0,0,0] for _ in range(n)] for _ in range(n)]
        ans = 0
        for i in range(n):
            l=u=0
            for j in range(n):
                if (i,j) in banned: l=u=0
                else:
                    l+=1; u= (dp[i-1][j][1]+1 if i else 1)
                    dp[i][j][0]=l; dp[i][j][1]=u
            l=0
            for j in range(n-1,-1,-1):
                if (i,j) in banned: l=0
                else:
                    l+=1; r=l; d=(dp[i+1][j][3]+1 if i+1<n else 1)
                    dp[i][j][2]=r; dp[i][j][3]=d
                    ans=max(ans, min(dp[i][j])+1)
        return ans
class Solution {
public:
    int orderOfLargestPlusSign(int n, vector<vector<int>>& mines) {
        vector<vector<int>> ban(n, vector<int>(n));
        for(auto& m:mines) if(m[0]<n&&m[1]<n) ban[m[0]][m[1]]=1;
        vector<vector<array<int,4>>> dp(n, vector<array<int,4>>(n));
        int ans=0;
        for(int i=0;i<n;i++){
            int l=0,u=0;
            for(int j=0;j<n;j++){
                if(ban[i][j]){ l=u=0; continue; }
                l++; u=(i?dp[i-1][j][1]+1:1); dp[i][j][0]=l; dp[i][j][1]=u;
            }
            l=0;
            for(int j=n-1;j>=0;j--){
                if(ban[i][j]){ l=0; continue; }
                l++; dp[i][j][2]=l; dp[i][j][3]=(i+1<n?dp[i+1][j][3]+1:1);
                int m=min({dp[i][j][0],dp[i][j][1],dp[i][j][2],dp[i][j][3]});
                ans=max(ans,m+1);
            }
        }
        return ans;
    }
};
