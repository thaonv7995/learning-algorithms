Out of Boundary Paths

O(m * n * N) O(m * n * N) 48.4% AC
Mô tả
Lưới m×n, bắt đầu (0,0), N bước di chuyển U/D/L/R. Đếm số path ra ngoài biên (out of boundary). Mod 10^9+7.
Ví dụ
Ví dụ 1
Inm=2, n=2, N=2, start=(0,0)
Out6
Ví dụ 2
Inm=1, n=3, N=3
Out12
Ý tưởng
DP[step][i][j] = số cách ở (i,j) sau step bước; chuyển 4 hướng, cộng out khi ra ngoài.

/* DP paths OOB — xem C++ */
class Solution:
    def findPaths(self, m: int, n: int, maxMove: int, startRow: int, startColumn: int) -> int:
        MOD = 10**9+7
        dp = [[[0]*n for _ in range(m)] for _ in range(maxMove+1)]
        dp[0][startRow][startColumn] = 1
        ans = 0
        dirs = [(1,0),(-1,0),(0,1),(0,-1)]
        for s in range(maxMove):
            for i in range(m):
                for j in range(n):
                    v = dp[s][i][j]
                    if not v: continue
                    for di,dj in dirs:
                        ni, nj = i+di, j+dj
                        if 0<=ni<m and 0<=nj<n:
                            dp[s+1][ni][nj] = (dp[s+1][ni][nj]+v)%MOD
                        else:
                            ans = (ans+v)%MOD
        return ans
class Solution {
public:
    int findPaths(int m, int n, int maxMove, int sr, int sc) {
        const int MOD = 1e9+7;
        vector<vector<vector<int>>> dp(maxMove+1, vector<vector<int>>(m, vector<int>(n)));
        dp[0][sr][sc] = 1;
        long long ans = 0;
        int d[4][2]={{1,0},{-1,0},{0,1},{0,-1}};
        for (int s=0;s<maxMove;s++)
            for (int i=0;i<m;i++)
                for (int j=0;j<n;j++) if (dp[s][i][j])
                    for (auto& dd:d) {
                        int ni=i+dd[0], nj=j+dd[1];
                        if (ni>=0&&ni<m&&nj>=0&&nj<n) dp[s+1][ni][nj]=(dp[s+1][ni][nj]+dp[s][i][j])%MOD;
                        else ans=(ans+dp[s][i][j])%MOD;
                    }
        return ans;
    }
};
