Wiggle Sort II

O(n) O(1) 37.4% AC
Mô tả
Sắp xếp lại mảng nums sao nums[0]<nums[1]>nums[2]<nums[3]... (wiggle). Yêu cầu O(n) thời gian, O(1) bộ nhớ phụ — nums[i] trong [0,5000].
Ví dụ
Ví dụ 1
Innums = [1,5,1,1,6,4]
Out[1,6,1,5,1,4]
Ví dụ 2
Innums = [1,3,2,2,3,1]
Out[2,3,1,3,1,2]
Ý tưởng
Virtual indexing + median: k=(n-1)/2; nth_element đưa median về vị trí k. Ba-way partition Dutch flag trên index ảo map về (1+(2*i)%(n|1)).

void wiggleSort(int* nums, int n) {
    int* tmp = malloc(n * sizeof(int));
    for (int i = 0; i < n; i++) tmp[i] = nums[i];
    for (int i = 0; i < n-1; i++)
        for (int j = i+1; j < n; j++)
            if (tmp[j] < tmp[i]) { int t=tmp[i]; tmp[i]=tmp[j]; tmp[j]=t; }
    int mid = (n-1)/2, j = mid;
    for (int i = 1; i < n; i += 2) nums[i] = tmp[j--];
    j = mid;
    for (int i = 0; i < n; i += 2) nums[i] = tmp[j--];
    free(tmp);
}
class Solution:
    def wiggleSort(self, nums: List[int]) -> None:
        n = len(nums)
        arr = sorted(nums)
        mid = (n - 1) // 2
        j = mid
        for i in range(1, n, 2):
            nums[i] = arr[j]; j -= 1
        j = mid
        for i in range(0, n, 2):
            nums[i] = arr[j]; j -= 1
class Solution {
public:
    void wiggleSort(vector<int>& nums) {
        int n = nums.size();
        auto mid = nums.begin() + (n - 1) / 2;
        nth_element(nums.begin(), mid, nums.end());
        vector<int> small(nums.begin(), mid), big(mid, nums.end());
        reverse(big.begin(), big.end());
        for (int i = 0, j = 0, k = 0; i < n; i++)
            nums[i] = (i % 2 == 0) ? small[j++] : big[k++];
    }
};
