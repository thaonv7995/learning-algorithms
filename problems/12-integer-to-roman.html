Integer to Roman

O(1) O(1) 71% AC
Mô tả
Chuyển số nguyên num (1–3999) sang chuỗi số La Mã.
Ví dụ
Ví dụ 1
Innum = 3749
Out"MMMDCCXLIX"
Ví dụ 2
Innum = 58
Out"LVIII"
Ví dụ 3
Innum = 1994
Out"MCMXCIV"
Ý tưởng
Greedy với bảng cặp (value, symbol) gồm cả subtractive (900→CM, 400→CD, …). Lặp từ lớn đến nhỏ: while num >= val { append symbol; num -= val }.
Mã nguồn
solution.c C99 
char* intToRoman(int num) {
    int vals[] = {1000,900,500,400,100,90,50,40,10,9,5,4,1};
    char* syms[] = {"M","CM","D","CD","C","XC","L","XL","X","IX","V","IV","I"};
    char* out = calloc(16, 1);
    for (int i = 0; i < 13; i++)
        while (num >= vals[i]) { strcat(out, syms[i]); num -= vals[i]; }
    return out;
}
class Solution:
    def intToRoman(self, num: int) -> str:
        pairs = [(1000,'M'),(900,'CM'),(500,'D'),(400,'CD'),(100,'C'),(90,'XC'),(50,'L'),(40,'XL'),(10,'X'),(9,'IX'),(5,'V'),(4,'IV'),(1,'I')]
        res = []
        for v, s in pairs:
            while num >= v:
                res.append(s); num -= v
        return ''.join(res)
class Solution {
public:
    string intToRoman(int num) {
        vector<pair<int,string>> p = {{1000,"M"},{900,"CM"},{500,"D"},{400,"CD"},{100,"C"},{90,"XC"},{50,"L"},{40,"XL"},{10,"X"},{9,"IX"},{5,"V"},{4,"IV"},{1,"I"}};
        string res;
        for (auto& [v,s] : p) while (num >= v) { res += s; num -= v; }
        return res;
    }
};
