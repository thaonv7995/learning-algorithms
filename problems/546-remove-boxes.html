Remove Boxes

O(n^4) O(n^3) 49.4% AC
Mô tả
Xóa hộp liên tiếp cùng màu thu điểm k*(k+1). Trả điểm tối đa xóa hết mảng boxes.
Ví dụ
Ví dụ 1
Inboxes = [1,3,5,8,8,3,4,1]
Out23
Ví dụ 2
Inboxes = [1,1,1]
Out9
Ý tưởng
DP[l][r][k]: max score đoạn [l,r] với k hộp cùng màu boxes[r] phía sau. Thử tách tại p hoặc gộp k+1 cùng màu.
Mã nguồn
solution.c C99 
/* 3D interval DP — xem C++ */
class Solution:
    def removeBoxes(self, boxes: List[int]) -> int:
        from functools import lru_cache
        @lru_cache(None)
        def dp(l, r, k):
            if l > r: return 0
            while r > l and boxes[r] == boxes[r-1]:
                r -= 1
                k += 1
            ans = dp(l, r-1, 0) + (k+1)*(k+1)
            for p in range(l, r):
                if boxes[p] == boxes[r]:
                    ans = max(ans, dp(l, p, k+1) + dp(p+1, r-1, 0))
            return ans
        return dp(0, len(boxes)-1, 0)
class Solution {
    vector<int> b;
    int dp[100][100][100];
    int solve(int l, int r, int k) {
        if (l > r) return 0;
        if (dp[l][r][k]) return dp[l][r][k];
        while (r > l && b[r] == b[r-1]) { r--; k++; }
        int ans = solve(l, r-1, 0) + (k+1)*(k+1);
        for (int p = l; p < r; p++)
            if (b[p] == b[r])
                ans = max(ans, solve(l, p, k+1) + solve(p+1, r-1, 0));
        return dp[l][r][k] = ans;
    }
public:
    int removeBoxes(vector<int>& boxes) {
        b = boxes;
        memset(dp, 0, sizeof dp);
        return solve(0, boxes.size()-1, 0);
    }
};
