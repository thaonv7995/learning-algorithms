Valid Number

O(N) O(1) 23% AC
Mô tả
Kiểm tra chuỗi có phải số hợp lệ không: integer, decimal, có/không exponent (e/E), dấu +/-.
Ví dụ
Ví dụ 1
Ins = "0"
Outtrue
Ví dụ 2

Outtrue
Ví dụ 3

Outfalse
Ý tưởng
Finite state machine: duyệt từng ký tự, chuyển trạng thái hợp lệ. Theo dõi seenDigit, seenDot, seenE.

/* isNumber — FSM một pass; xem C++ */
class Solution:
    def isNumber(self, s: str) -> bool:
        s = s.strip()
        if not s: return False
        seen_d = seen_dot = seen_e = False
        for i, c in enumerate(s):
            if c.isdigit(): seen_d = True
            elif c in '+-':
                if i and s[i-1] not in 'eE': return False
            elif c in 'eE':
                if seen_e or not seen_d: return False
                seen_e = seen_d = False; seen_dot = True
            elif c == '.':
                if seen_dot or seen_e: return False
                seen_dot = True
            else: return False
        return seen_d
class Solution {
public:
    bool isNumber(string s) {
        int i=0,n=s.size(); while(i<n&&s[i]==' ')i++; while(n>i&&s[n-1]==' ')n--;
        if(i>=n) return false;
        bool d=false, dot=false, e=false;
        for(;i<n;i++){
            if(isdigit(s[i])) d=true;
            else if(s[i]=='+'||s[i]=='-'){ if(i&&s[i-1]!='e'&&s[i-1]!='E') return false; }
            else if(s[i]=='e'||s[i]=='E'){ if(e||!d) return false; e=d=true; d=dot=false; }
            else if(s[i]=='.'){ if(dot||e) return false; dot=true; }
            else return false;
        }
        return d;
    }
};
