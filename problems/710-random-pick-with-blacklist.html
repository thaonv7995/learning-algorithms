Random Pick with Blacklist

O(n + b log b) O(b) 35% AC
Mô tả
Có n số [0,n-1] và blacklist. pick() trả số ngẫu nhiên đều từ tập hợp hợp lệ (không nằm blacklist). Mỗi số hợp lệ xác suất bằng nhau.
Ví dụ
Ví dụ 1
In["Solution","pick","pick","pick"], [[1,[]],[],[],[]]
Out[null,0,0,0]
Ví dụ 2
In["Solution","pick","pick","pick"], [[2,[1]],[],[],[]]
Out[null,0,0,1]
Ý tưởng
Map blacklist nhỏ vào cuối [0,m): hoán vị sao mọi phần tử blacklist nằm ngoài vùng pick [0,m) với m = n - |blacklist trong range|.

class Solution:
    def __init__(self, n: int, blacklist: List[int]):
        self.m = n - len(blacklist)
        self.mp = {}
        bl = set(blacklist)
        for x in blacklist:
            if x < self.m:
                while self.m in bl or self.m in self.mp:
                    self.m += 1
                self.mp[x] = self.m
                self.m += 1
    def pick(self) -> int:
        x = random.randint(0, self.m - 1)
        return self.mp.get(x, x)
class Solution {
    int m;
    unordered_map<int,int> mp;
public:
    Solution(int n, vector<int>& blacklist) {
        unordered_set<int> bl(blacklist.begin(), blacklist.end());
        m = n - (int)blacklist.size();
        for (int x : blacklist) {
            if (x < m) {
                while (bl.count(m) || mp.count(m)) m++;
                mp[x] = m++;
            }
        }
    }
    int pick() {
        int x = rand() % m;
        return mp.count(x) ? mp[x] : x;
    }
};
