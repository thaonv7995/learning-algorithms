Number of Longest Increasing Subsequence

Mô tả
Cho mảng nums. Đếm số dãy con tăng dài nhất (LIS) có độ dài bằng độ dài LIS tối đa của mảng.
Ví dụ
Ví dụ 1
Innums=[1,3,5,4,7]
Out2
Ví dụ 2
Innums=[2,2,2,2,2]
Out5
Ý tưởng
Patience sorting + count paths: dpLen[i], dpCnt[i] update on predecessors.

class Solution:
    def findNumberOfLIS(self, nums: List[int]) -> int:
        n=len(nums); ln=[1]*n; cnt=[1]*n
        for i in range(n):
            for j in range(i):
                if nums[j]<nums[i]:
                    if ln[j]+1>ln[i]: ln[i]=ln[j]+1; cnt[i]=cnt[j]
                    elif ln[j]+1==ln[i]: cnt[i]+=cnt[j]
        mx=max(ln); return sum(c for l,c in zip(ln,cnt) if l==mx)
class Solution {
public:
    int findNumberOfLIS(vector<int>& a) {
        int n=a.size(); vector<int> ln(n,1), cnt(n,1);
        for(int i=0;i<n;i++) for(int j=0;j<i;j++) if(a[j]<a[i]){
            if(ln[j]+1>ln[i]){ ln[i]=ln[j]+1; cnt[i]=cnt[j]; }
            else if(ln[j]+1==ln[i]) cnt[i]+=cnt[j];
        }
        int mx=*max_element(ln.begin(),ln.end());
        int ans=0; for(int i=0;i<n;i++) if(ln[i]==mx) ans+=cnt[i];
        return ans;
    }
};
