Longest Common Subsequence

O(N * M) (N, M là độ dài 2 chuỗi) O(N * M) 59.2% AC
Mô tả
Cho hai chuỗi text1 và text2, hãy tìm độ dài của chuỗi con chung dài nhất (LCS) giữa chúng.
Ví dụ
Ví dụ 1
Intext1 = "abcde", text2 = "ace"
Out3 (Chuỗi chung dài nhất là "ace")
Ý tưởng
Áp dụng Quy hoạch động lập bảng Bottom-Up 2D. Gọi `dp[i][j]` là độ dài LCS của chuỗi con `text1[0...i-1]` và `text2[0...j-1]`. Công thức chuyển trạng thái:
- Nếu `text1[i-1] == text2[j-1]`: hai ký tự khớp nhau, `dp[i][j] = 1 + dp[i-1][j-1]` (tăng độ dài + đi chéo lên trên bên trái).
- Nếu lệch: `dp[i][j] = max(dp[i-1][j], dp[i][j-1])` (mượn giá trị tối ưu lớn nhất từ ô trên hoặc ô trái).
Mã nguồn
solution.c C99 
// Solution in C (Longest Common Subsequence)
int longestCommonSubsequence(char* text1, char* text2) {
    int len1 = strlen(text1);
    int len2 = strlen(text2);
    int dp[len1 + 1][len2 + 1];
    for (int i = 0; i <= len1; i++) dp[i][0] = 0;
    for (int j = 0; j <= len2; j++) dp[0][j] = 0;
    for (int i = 1; i <= len1; i++) {
        for (int j = 1; j <= len2; j++) {
            if (text1[i - 1] == text2[j - 1]) {
                dp[i][j] = dp[i - 1][j - 1] + 1;
            } else {
                int max = dp[i - 1][j] > dp[i][j - 1] ? dp[i - 1][j] : dp[i][j - 1];
                dp[i][j] = max;
            }
        }
    }
    return dp[len1][len2];
}
# Solution in Python (Longest Common Subsequence)
class Solution:
    def longestCommonSubsequence(self, text1: str, text2: str) -> int:
        dp = [[0] * (len(text2) + 1) for _ in range(len(text1) + 1)]
        for i in range(1, len(text1) + 1):
            for j in range(1, len(text2) + 1):
                if text1[i - 1] == text2[j - 1]:
                    dp[i][j] = dp[i - 1][j - 1] + 1
                else:
                    dp[i][j] = max(dp[i - 1][j], dp[i][j - 1])
        return dp[-1][-1]
class Solution {
public:
    int longestCommonSubsequence(string text1, string text2) {
        int n = text1.length(), m = text2.length();
        vector<vector<int>> dp(n + 1, vector<int>(m + 1, 0));
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= m; j++) {
                if (text1[i - 1] == text2[j - 1]) {
                    dp[i][j] = 1 + dp[i - 1][j - 1];
                } else {
                    dp[i][j] = max(dp[i - 1][j], dp[i][j - 1]);
                }
            }
        }
        return dp[n][m];
    }
};
