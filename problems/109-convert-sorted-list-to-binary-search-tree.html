Convert Sorted List to Binary Search Tree

O(n) O(log n) 66.6% AC
Mô tả
Cho linked list đã sắp xếp tăng dần (giá trị duy nhất). Dựng BST cân bằng cao nhất có thể.
Ví dụ
Ví dụ 1
Inhead = [-10,-3,0,5,9]
Out[0,-3,9,-10,null,5]
Ví dụ 2
Inhead = []
Out[]
Ý tưởng
Đếm n nút. Đệ quy build(l,r): tại mid, lấy nút head hiện tại làm root (inorder simulation), head tiến 1 bước. Trái build trước, phải sau.
Mã nguồn
solution.c C99 
/* Inorder simulation từ list — xem C++ */
class Solution:
    def sortedListToBST(self, head: Optional[ListNode]) -> Optional[TreeNode]:
        def size(h):
            n = 0
            while h:
                n += 1
                h = h.next
            return n
        def build(l, r):
            nonlocal head
            if l > r: return None
            m = (l + r) // 2
            left = build(l, m - 1)
            root = TreeNode(head.val)
            head = head.next
            root.left = left
            root.right = build(m + 1, r)
            return root
        return build(0, size(head) - 1) if head else None
class Solution {
    ListNode* cur;
    int count(ListNode* h) {
        int n = 0;
        while (h) { n++; h = h->next; }
        return n;
    }
    TreeNode* build(int l, int r) {
        if (l > r) return nullptr;
        int m = l + (r - l) / 2;
        TreeNode* left = build(l, m - 1);
        TreeNode* root = new TreeNode(cur->val);
        cur = cur->next;
        root->left = left;
        root->right = build(m + 1, r);
        return root;
    }
public:
    TreeNode* sortedListToBST(ListNode* head) {
        if (!head) return nullptr;
        cur = head;
        return build(0, count(head) - 1);
    }
};
