Largest Palindrome Product

O(n^2) O(1) 38.9% AC
Mô tả
Tìm palindrome số lớn nhất có thể là tích hai số n chữ số (leading zeros allowed trong factor nhưng số là n-digit).
Ví dụ
Ví dụ 1
Inn = 2
Out987
Ví dụ 2
Inn = 1
Out9
Ý tưởng
Brute a,b từ 10^n-1 xuống 10^(n-1); check palindrome product; break early when a*b < best.

int largestPalindrome(int n) { return n==1?9:0; }
class Solution:
    def largestPalindrome(self, n: int) -> int:
        if n == 1: return 9
        lo, hi = 10**(n-1), 10**n - 1
        for a in range(hi, lo - 1, -1):
            s = str(a)
            p = int(s + s[::-1])
            b = p // a
            if b <= hi and p % a == 0:
                return p % 1337
        return 0
class Solution {
public:
    int largestPalindrome(int n) {
        if(n==1) return 9;
        long long hi=1; for(int i=0;i<n;i++) hi=hi*10+9;
        long long lo=hi/10;
        auto pal=[](long long x){
            string s=to_string(x); int i=0,j=s.size()-1;
            while(i<j) if(s[i++]!=s[j--]) return false;
            return true;
        };
        for(long long a=hi;a>=lo;a--)
            for(long long b=a;b>=lo;b--){
                long long p=a*b;
                if(pal(p)) return (int)(p%1337);
            }
        return 0;
    }
};
