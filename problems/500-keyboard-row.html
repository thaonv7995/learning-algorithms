char** findWords(char** words, int n, int* rs) {
    const char* rows[3]={"qwertyuiop","asdfghjkl","zxcvbnm"};
    int mp[256]; for(int i=0;i<256;i++) mp[i]=-1;
    for(int i=0;i<3;i++) for(int j=0;rows[i][j];j++) mp[(unsigned char)rows[i][j]]=i;
    char** ans=malloc(n*sizeof(char*)); *rs=0;
    for(int i=0;i<n;i++){
        int r=mp[(unsigned char)tolower(words[i][0])]; int ok=1;
        for(char* p=words[i];*p;p++) if(mp[(unsigned char)tolower(*p)]!=r){ ok=0; break; }
        if(ok) ans[(*rs)++]=words[i];
    }
    return ans;
}
class Solution:
    def findWords(self, words: List[str]) -> List[str]:
        rows = ["qwertyuiop","asdfghjkl","zxcvbnm"]
        mp = {c: i for i, r in enumerate(rows) for c in r}
        ans = []
        for w in words:
            r0 = mp[w[0].lower()]
            if all(mp[c.lower()] == r0 for c in w):
                ans.append(w)
        return ans
class Solution {
public:
    vector<string> findWords(vector<string>& words) {
        string rows[3]={"qwertyuiop","asdfghjkl","zxcvbnm"};
        int mp[256]; memset(mp,-1,sizeof mp);
        for(int i=0;i<3;i++) for(char c:rows[i]) mp[c]=i;
        vector<string> ans;
        for(auto& w:words){
            int r=mp[tolower(w[0])]; bool ok=1;
            for(char c:w) if(mp[tolower(c)]!=r){ ok=0; break; }
            if(ok) ans.push_back(w);
        }
        return ans;
    }
};
