Number of Atoms

O(n) O(n) 65.1% AC
Mô tả
Cho công thức hóa học dạng chuỗi (nguyên tố, số lượng, ngoặc lồng). Trả chuỗi đếm số nguyên tử dạng sort alphabet: NameCount, Count=1 thì bỏ số.
Ví dụ
Ví dụ 1
Informula = "H2O"
Out"H2O"
Ví dụ 2
Informula = "Mg(OH)2"
Out"H2MgO2"
Ví dụ 3
Informula = "K4(ON(SO3)2)2"
Out"K4N2O14S4"
Ý tưởng
Stack of map element→count. Parse token; ( push map mới; ) pop merge vào top nhân multiplier.
Mã nguồn

char* countOfAtoms(char* formula) {
    int cnt[128]={0};
    for(int i=0; formula[i]; ){
        if(formula[i]=='('){ i++; continue; }
        if(formula[i]==')'){ i++; while(formula[i]>='0'&&formula[i]<'9') i++; continue; }
        char name[8]; int p=0; name[p++]=formula[i++];
        while(formula[i]>='a'&&formula[i]<'z') name[p++]=formula[i++]; name[p]=0;
        int mul=1; if(formula[i]>='0'&&formula[i]<'9'){ mul=0; while(formula[i]>='0'&&formula[i]<'9') mul=mul*10+formula[i++]-'0'; }
        cnt[(unsigned char)name[0]]+=mul;
    }
    char* out=malloc(512); int o=0;
    for(int c='A'; c<='Z'; c++) if(cnt[c]){ out[o++]=c; if(cnt[c]>1){ char buf[16]; sprintf(buf,"%d",cnt[c]); strcat(out+o,buf); o=strlen(out); } }
    return out;
}
class Solution:
    def countOfAtoms(self, formula: str) -> str:
        import re
        def parse(i):
            st = [Counter()]
            while i < len(formula):
                if formula[i] == '(':
                    st.append(Counter()); i += 1
                elif formula[i] == ')':
                    top = st.pop(); i += 1
                    m = 1
                    if i < len(formula) and formula[i].isdigit():
                        j = i
                        while j < len(formula) and formula[j].isdigit(): j += 1
                        m = int(formula[i:j]); i = j
                    for k, v in top.items(): st[-1][k] += v * m
                else:
                    j = i + 1
                    while j < len(formula) and formula[j].islower(): j += 1
                    name = formula[i:j]; i = j
                    cnt = 1
                    if i < len(formula) and formula[i].isdigit():
                        k = i
                        while k < len(formula) and formula[k].isdigit(): k += 1
                        cnt = int(formula[i:k]); i = k
                    st[-1][name] += cnt
            return st[-1]
        cnt = parse(0)
        return "".join(name + (str(v) if v > 1 else "") for name, v in sorted(cnt.items()))
class Solution {
public:
    string countOfAtoms(string formula) {
        vector<map<string,int>> st(1);
        int i=0, n=formula.size();
        while(i<n){
            if(formula[i]=='('){ st.emplace_back(); i++; }
            else if(formula[i]==')'){
                auto top=st.back(); st.pop_back(); i++;
                int mul=1;
                if(i<n && isdigit(formula[i])){ mul=0; while(i<n&&isdigit(formula[i])) mul=mul*10+formula[i++]-'0'; }
                for(auto& p: top) st.back()[p.first]+=p.second*mul;
            } else {
                int j=i+1; while(j<n&&islower(formula[j])) j++;
                string name=formula.substr(i,j-i); i=j;
                int cnt=1;
                if(i<n&&isdigit(formula[i])){ cnt=0; while(i<n&&isdigit(formula[i])) cnt=cnt*10+formula[i++]-'0'; }
                st.back()[name]+=cnt;
            }
        }
        string ans;
        for(auto& p: st[0]){ ans+=p.first; if(p.second>1) ans+=to_string(p.second); }
        return ans;
    }
};
