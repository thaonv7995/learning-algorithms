Design Twitter

O(k log k) getNewsFeed O(n) 44.7% AC
Mô tả
Thiết kế mạng xã hội mini: postTweet, getNewsFeed (10 tweet mới nhất của user + followees), follow, unfollow.
Ví dụ
Ví dụ 1
InpostTweet(1,5), getNewsFeed(1), follow(1,2), postTweet(2,6), getNewsFeed(1)
Out[5], then [6,5]
Ví dụ 2
InpostTweet(1,10), unfollow(1,2), getNewsFeed(1)
Out[10]
Ý tưởng
Map user→danh sách tweet (timestamp, id); follow graph; getNewsFeed merge k tweet lists bằng max-heap lấy 10 mới nhất.

class Twitter:
    def __init__(self):
        self.time = 0
        self.tweets = defaultdict(list)
        self.follows = defaultdict(set)
    def postTweet(self, userId: int, tweetId: int) -> None:
        self.tweets[userId].append((self.time, tweetId)); self.time -= 1
    def getNewsFeed(self, userId: int) -> List[int]:
        heap = []
        users = self.follows[userId] | {userId}
        for u in users:
            for t in self.tweets[u][-10:]:
                heapq.heappush(heap, t)
                if len(heap) > 10: heapq.heappop(heap)
        return [t for _, t in sorted(heap)]
class Twitter {
    int timer = 0;
    unordered_map<int, vector<pair<int,int>>> tweets;
    unordered_map<int, unordered_set<int>> follows;
public:
    Twitter() {}
    void postTweet(int userId, int tweetId) { tweets[userId].push_back({timer++, tweetId}); }
    vector<int> getNewsFeed(int userId) {
        auto cmp = [](pair<int,int>& a, pair<int,int>& b){ return a.first > b.first; };
        priority_queue<pair<int,int>, vector<pair<int,int>>, decltype(cmp)> pq(cmp);
        follows[userId].insert(userId);
        for (int u : follows[userId]) {
            auto& v = tweets[u];
            for (int i = max(0, (int)v.size()-10); i < (int)v.size(); i++) pq.push(v[i]);
        }
        vector<int> ans;
        while (!pq.empty() && ans.size() < 10) { ans.push_back(pq.top().second); pq.pop(); }
        return ans;
    }
    void follow(int a, int b) { follows[a].insert(b); }
    void unfollow(int a, int b) { follows[a].erase(b); }
};
