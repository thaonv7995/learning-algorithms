Kth Smallest Element in a Sorted Matrix

O(n log(max-min)) O(1) 64.6% AC
Mô tả
Ma trận n×n sort theo hàng và cột. Trả phần tử nhỏ thứ k (1-indexed).
Ví dụ
Ví dụ 1
Inmatrix = [[1,5,9],[10,11,13],[12,13,15]], k = 8
Out13
Ví dụ 2
Inmatrix = [[-5]], k = 1
Out-5
Ý tưởng
Binary search trên giá trị mid: đếm số phần tử ≤ mid bằng walk từ góc trên-phải hoặc dưới-trái O(n).

int kthSmallest(int** matrix, int n, int* cs, int k) {
    int lo = matrix[0][0], hi = matrix[n-1][cs[n-1]-1];
    while (lo < hi) {
        int mid = lo + (hi - lo) / 2, c = 0, i = 0, j = n - 1;
        while (i < n && j >= 0) {
            if (matrix[i][j] <= mid) { c += j + 1; i++; }
            else j--;
        }
        if (c >= k) hi = mid; else lo = mid + 1;
    }
    return lo;
}
class Solution:
    def kthSmallest(self, matrix: List[List[int]], k: int) -> int:
        n = len(matrix)
        lo, hi = matrix[0][0], matrix[-1][-1]
        def count(x):
            i, j, c = 0, n-1, 0
            while i < n and j >= 0:
                if matrix[i][j] <= x:
                    c += j + 1; i += 1
                else:
                    j -= 1
            return c
        while lo < hi:
            mid = (lo + hi) // 2
            if count(mid) >= k: hi = mid
            else: lo = mid + 1
        return lo
