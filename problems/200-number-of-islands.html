Number of Islands

O(R * C) (R là hàng, C là cột) O(R * C) (Trong trường hợp xấu nhất toàn bộ grid là đất liền)
Mô tả
Cho một lưới 2D grid kích thước R x C chứa các ký tự '1' (đất liền) và '0' (nước), hãy đếm số lượng hòn đảo biệt lập (các vùng đất '1' kết nối với nhau theo 4 hướng ngang dọc).
Ví dụ
Ví dụ 1
Ingrid = [['1','1','0'],['1','1','0'],['0','0','1']]
Out2
Ý tưởng
Duyệt qua từng ô trong lưới 2D. Nếu phát hiện đất liền '1', ta bắt đầu loang đệ quy theo chiều sâu (DFS) sang các ô lân cận để đánh dấu toàn bộ hòn đảo đã được duyệt (để tránh lặp lại, ta đổi giá trị đất liền '1' đã duyệt thành nước '0'). Mỗi lần kích hoạt DFS từ vòng lặp ngoài đại diện cho một hòn đảo mới được tìm thấy.

// Solution in C (Number of Islands)
void dfs(char** grid, int gridSize, int* gridColSize, int r, int c) {
    if (r < 0 || c < 0 || r >= gridSize || c >= gridColSize[r] || grid[r][c] == '0') return;
    grid[r][c] = '0';
    dfs(grid, gridSize, gridColSize, r + 1, c);
    dfs(grid, gridSize, gridColSize, r - 1, c);
    dfs(grid, gridSize, gridColSize, r, c + 1);
    dfs(grid, gridSize, gridColSize, r, c - 1);
}

int numIslands(char** grid, int gridMatrixSize, int* gridColSize) {
    int count = 0;
    for (int r = 0; r < gridMatrixSize; r++) {
        for (int c = 0; c < gridColSize[r]; c++) {
            if (grid[r][c] == '1') {
                count++;
                dfs(grid, gridMatrixSize, gridColSize, r, c);
            }
        }
    }
    return count;
}
# Solution in Python (Number of Islands)
class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        if not grid: return 0
        count = 0
        rows, cols = len(grid), len(grid[0])
        def dfs(r, c):
            if r < 0 or c < 0 or r >= rows or c >= cols or grid[r][c] == '0':
                return
            grid[r][c] = '0'
            dfs(r + 1, c)
            dfs(r - 1, c)
            dfs(r, c + 1)
            dfs(r, c - 1)
        for r in range(rows):
            for c in range(cols):
                if grid[r][c] == '1':
                    count += 1
                    dfs(r, c)
        return count
class Solution {
    void dfs(vector<vector<char>>& grid, int r, int c) {
        if (r < 0 || c < 0 || r >= grid.size() || c >= grid[0].size() || grid[r][c] == '0') {
            return;
        }
        grid[r][c] = '0'; // Danh dau da duyet
        dfs(grid, r - 1, c);
        dfs(grid, r + 1, c);
        dfs(grid, r, c - 1);
        dfs(grid, r, c + 1);
    }
public:
    int numIslands(vector<vector<char>>& grid) {
        int count = 0;
        for (int r = 0; r < grid.size(); r++) {
            for (int c = 0; c < grid[0].size(); c++) {
                if (grid[r][c] == '1') {
                    count++;
                    dfs(grid, r, c); // Loang danh dau
                }
            }
        }
        return count;
    }
};
