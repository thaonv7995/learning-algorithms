Count Binary Substrings

O(n) O(1) 70.4% AC
Mô tả
Đếm substring có số bit 0 và 1 bằng nhau (chỉ 0/1).
Ví dụ
Ví dụ 1
Ins="00110011"
Out6
Ví dụ 2
Ins="10101"
Out4
Ý tưởng
Convert to +/-1; prefix sum + hash count equal zeros/ones.

class Solution:
    def countBinarySubstrings(self, s: str) -> int:
        groups=[]; i=0
        while i<len(s):
            j=i
            while j<len(s) and s[j]==s[i]: j+=1
            groups.append(j-i); i=j
        ans=0
        for i in range(1,len(groups)): ans+=min(groups[i-1],groups[i])
        return ans
class Solution {
public:
    int countBinarySubstrings(string s) {
        vector<int> g; int i=0;
        while(i<(int)s.size()){
            int j=i; while(j<(int)s.size()&&s[j]==s[i]) j++;
            g.push_back(j-i); i=j;
        }
        int ans=0; for(int k=1;k<(int)g.size();k++) ans+=min(g[k-1],g[k]);
        return ans;
    }
};
