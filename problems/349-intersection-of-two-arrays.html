Intersection of Two Arrays

O(n+m) O(n) 77.8% AC
Mô tả
Cho hai mảng nums1 và nums2. Trả giao các phần tử distinct (mỗi giá trị xuất hiện tối đa một lần trong kết quả).
Ví dụ
Ví dụ 1
Innums1 = [1,2,2,1], nums2 = [2,2]
Out[2]
Ví dụ 2
Innums1 = [4,9,5], nums2 = [9,4]
Out[9,4]
Ý tưởng
Hash set từ nums1; duyệt nums2, nếu x trong set thì thêm vào kết quả và xóa khỏi set (distinct).
Mã nguồn

int* intersection(int* nums1, int n1, int* nums2, int n2, int* retSize) {
    int* set = calloc(2001, sizeof(int));
    for (int i=0;i<n1;i++) if(nums1[i]>=0&&nums1[i]<=2000) set[nums1[i]]=1;
    int* ans = malloc((n1<n2?n1:n2)*sizeof(int)); *retSize=0;
    int* used = calloc(2001,sizeof(int));
    for (int i=0;i<n2;i++) if(nums2[i]>=0&&nums2[i]<=2000&&set[nums2[i]]&&!used[nums2[i]]){
        ans[(*retSize)++]=nums2[i]; used[nums2[i]]=1;
    }
    free(set); free(used); return ans;
}
class Solution:
    def intersection(self, nums1: List[int], nums2: List[int]) -> List[int]:
        s = set(nums1)
        return [x for x in set(nums2) if x in s]
class Solution {
public:
    vector<int> intersection(vector<int>& nums1, vector<int>& nums2) {
        unordered_set<int> s(nums1.begin(), nums1.end());
        unordered_set<int> out;
        for (int x : nums2) if (s.count(x)) out.insert(x);
        return vector<int>(out.begin(), out.end());
    }
};
