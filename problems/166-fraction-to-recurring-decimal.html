Fraction to Recurring Decimal

O(d) O(d) 30.9% AC
Mô tả
Cho hai số nguyên numerator và denominator. Trả chuỗi biểu diễn thập phân; phần lặp đặt trong ngoặc (). Trả "0" nếu bằng 0, "Error" nếu chia cho 0.
Ví dụ
Ví dụ 1
Innumerator = 1, denominator = 2
Out"0.5"
Ví dụ 2
Innumerator = 4, denominator = 333
Out"0.(012)"
Ví dụ 3
Innumerator = 2, denominator = 1
Out"2"
Ý tưởng
Long division: xử lý dấu, phần nguyên, rồi lặp remainder = (rem*10)%den. Hash map remainder→vị trí trong chuỗi thập phân để chèn '(' khi gặp lại.

char* fractionToDecimal(int n, int d) {
    if (d == 0) return strdup("Error");
    char buf[4096]; int k = 0;
    if ((long long)n * d < 0) buf[k++] = '-';
    long long a = llabs((long long)n), b = llabs((long long)d);
    k += sprintf(buf + k, "%lld", a / b);
    long long r = a % b;
    if (!r) return strdup(buf);
    buf[k++] = '.';
    long long rem[4096]; int idx[4096], len = 0;
    while (r) {
        for (int i = 0; i < len; i++) if (rem[i] == r) {
            buf[k] = 0;
            char* out = malloc(strlen(buf) + 64);
            sprintf(out, "%.*s(%s)", k - idx[i], buf, buf + idx[i]);
            return out;
        }
        rem[len] = r; idx[len++] = k;
        buf[k++] = '0' + (char)(r * 10 / b);
        r = r * 10 % b;
    }
    buf[k] = 0; return strdup(buf);
}
class Solution:
    def fractionToDecimal(self, n: int, d: int) -> str:
        if d == 0: return "Error"
        sign = '-' if (n < 0) ^ (d < 0) else ''
        n, d = abs(n), abs(d)
        q, r = divmod(n, d)
        res = sign + str(q)
        if r == 0: return res
        res += '.'
        pos = {}
        frac = []
        while r:
            if r in pos:
                i = pos[r]
                return res + ''.join(frac[:i]) + '(' + ''.join(frac[i:]) + ')'
            pos[r] = len(frac)
            frac.append(str(r * 10 // d))
            r = r * 10 % d
        return res + ''.join(frac)
class Solution {
public:
    string fractionToDecimal(int n, int d) {
        if (d == 0) return "Error";
        string res;
        if ((long long)n * d < 0) res += '-';
        long long a = abs((long long)n), b = abs((long long)d);
        res += to_string(a / b);
        long long r = a % b;
        if (!r) return res;
        res += '.';
        unordered_map<long long,int> pos;
        string frac;
        while (r) {
            if (pos.count(r)) {
                return res + frac.substr(0, pos[r]) + "(" + frac.substr(pos[r]) + ")";
            }
            pos[r] = frac.size();
            frac += char('0' + r * 10 / b);
            r = r * 10 % b;
        }
        return res + frac;
    }
};
