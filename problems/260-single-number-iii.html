Single Number III

O(n) O(1) 70.3% AC
Mô tả
Mảng nums: mọi phần tử xuất hiện đúng hai lần trừ đúng hai phần tử distinct xuất hiện một lần. Trả hai số đó (thứ tự bất kỳ).
Ví dụ
Ví dụ 1
Innums = [1,2,1,3,2,5]
Out[3,5]
Ví dụ 2
Innums = [-1,0]
Out[-1,0]
Ý tưởng
XOR cả mảng → xor=a^b; bit phân tách (xor & -xor); chia nums theo bit đó XOR riêng → a và b.

int* singleNumber260(int* nums, int n, int* rs) {
    long long xorAll = 0;
    for (int i = 0; i < n; i++) xorAll ^= nums[i];
    long long bit = xorAll & -xorAll;
    int a = 0, b = 0;
    for (int i = 0; i < n; i++)
        (nums[i] & bit) ? (a ^= nums[i]) : (b ^= nums[i]);
    int* out = malloc(2 * sizeof(int));
    out[0] = a; out[1] = b; *rs = 2;
    return out;
}
class Solution:
    def singleNumber(self, nums: List[int]) -> List[int]:
        xor = 0
        for x in nums:
            xor ^= x
        bit = xor & -xor
        a = b = 0
        for x in nums:
            if x & bit:
                a ^= x
            else:
                b ^= x
        return [a, b]
class Solution {
public:
    vector<int> singleNumber(vector<int>& nums) {
        long long xorAll = 0;
        for (int x : nums) xorAll ^= x;
        long long bit = xorAll & -xorAll;
        int a = 0, b = 0;
        for (int x : nums)
            (x & bit) ? a ^= x : b ^= x;
        return {a, b};
    }
};
