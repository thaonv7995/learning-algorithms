Sort Colors

O(n) O(1) 69.6% AC
Mô tả
Mảng nums gồm 0, 1, 2 (đỏ, trắng, xanh). Sắp xếp in-place theo thứ tự 0→1→2, một lần duyệt, không dùng sort thư viện.
Ví dụ
Ví dụ 1
Innums = [2,0,2,1,1,0]
Out[0,0,1,1,2,2]
Ví dụ 2
Innums = [2,0,1]
Out[0,1,2]
Ý tưởng
Dutch National Flag: con trỏ lo, mid, hi. nums[mid]==0 swap lo++; ==1 mid++; ==2 swap hi-- (không tăng mid).
Mã nguồn
solution.c C99 
void sortColors(int* nums, int n) {
    int lo=0, mid=0, hi=n-1, t;
    while (mid<=hi) {
        if (nums[mid]==0) { t=nums[lo]; nums[lo]=nums[mid]; nums[mid]=t; lo++; mid++; }
        else if (nums[mid]==1) mid++;
        else { t=nums[mid]; nums[mid]=nums[hi]; nums[hi]=t; hi--; }
    }
}
class Solution:
    def sortColors(self, nums: List[int]) -> None:
        lo = mid = 0; hi = len(nums) - 1
        while mid <= hi:
            if nums[mid] == 0:
                nums[lo], nums[mid] = nums[mid], nums[lo]
                lo += 1; mid += 1
            elif nums[mid] == 1:
                mid += 1
            else:
                nums[mid], nums[hi] = nums[hi], nums[mid]
                hi -= 1
class Solution {
public:
    void sortColors(vector<int>& a) {
        int lo=0, mid=0, hi=(int)a.size()-1;
        while (mid<=hi) {
            if (a[mid]==0) swap(a[lo++], a[mid++]);
            else if (a[mid]==1) mid++;
            else swap(a[mid], a[hi--]);
        }
    }
};
