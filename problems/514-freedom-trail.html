Mô tả
Ring chứa key (có thể lặp ký tự). Bắt đầu tại 'A', di chuyển tối thiểu trên ring để gõ key. Trả tổng bước (di chuyển + bấm phím).
Ví dụ
Ví dụ 1
Inring = "godding", key = "gd"
Out4
Ví dụ 2
Inring = "godding", key = "godding"
Out13
Ý tưởng
DP[i][j]: min cost gõ key[0..i) khi tay ở vị trí j trên ring. Precompute khoảng cách vòng giữa mọi cặp vị trí cùng ký tự.

/* DP ring + key — xem C++ */
class Solution:
    def findRotateSteps(self, ring: str, key: str) -> int:
        from collections import defaultdict
        pos = defaultdict(list)
        for i, c in enumerate(ring):
            pos[c].append(i)
        n, m = len(ring), len(key)
        INF = 10**9
        dp = [[INF] * n for _ in range(len(key) + 1)]
        dp[0][0] = 0
        for i in range(len(key)):
            for j in range(n):
                if dp[i][j] == INF:
                    continue
                for k in pos[key[i]]:
                    step = min(abs(k - j), n - abs(k - j)) + 1
                    dp[i + 1][k] = min(dp[i + 1][k], dp[i][j] + step)
        return min(dp[len(key)])

class Solution {
public:
    int findRotateSteps(string ring, string key) {
        int n = ring.size();
        vector<vector<int>> pos(26);
        for (int i = 0; i < n; i++) pos[ring[i]-'a'].push_back(i);
        int m = key.size();
        const int INF = 1e9;
        vector<vector<int>> dp(m+1, vector<int>(n, INF));
        dp[0][0] = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) if (dp[i][j] < INF) {
                for (int k : pos[key[i]-'a']) {
                    int step = min(abs(k-j), n-abs(k-j)) + 1;
                    dp[i+1][k] = min(dp[i+1][k], dp[i][j] + step);
                }
            }
        }
        return *min_element(dp[m].begin(), dp[m].end());
    }
};
