The Skyline Problem

O(n log n) O(n) 45.4% AC
Mô tả
Danh sách buildings[i]=[left, right, height]. Trả key points [x, height] mô tả đường skyline khi quét từ trái sang phải (height thay đổi mới ghi).
Ví dụ
Ví dụ 1
Inbuildings = [[2,9,10],[3,7,15],[5,12,12],[15,20,10],[19,24,8]]
Out[[2,10],[3,15],[7,12],[12,0],[15,10],[20,8],[24,0]]
Ví dụ 2
Inbuildings = [[0,2,3],[2,5,3]]
Out[[0,3],[5,0]]
Ý tưởng
Sweep line: events (left, -h), (right, h); sort; multiset max height; khi max đổi push [x, newMax].

typedef struct { int x, h; } Ev;
static int cmpEv(const void* a, const void* b) {
    Ev* ea = (Ev*)a; Ev* eb = (Ev*)b;
    if (ea->x != eb->x) return ea->x - eb->x;
    return ea->h - eb->h;
}
int** getSkyline(int** b, int bn, int* bc, int* rs, int* rc) {
    Ev* ev = malloc(bn * 2 * sizeof(Ev));
    int ec = 0;
    for (int i = 0; i < bn; i++) {
        ev[ec++] = (Ev){b[i][0], -b[i][2]};
        ev[ec++] = (Ev){b[i][1], b[i][2]};
    }
    qsort(ev, ec, sizeof(Ev), cmpEv);
    int* hs = calloc(bn * 2 + 1, sizeof(int)); hs[0] = 0; int hsz = 1;
    int** out = malloc(bn * 2 * sizeof(int*)); *rs = 0; *rc = 2;
    for (int i = 0; i < ec; i++) {
        int h = ev[i].h;
        if (h < 0) { hs[hsz++] = -h; } else {
            for (int j = 0; j < hsz; j++) if (hs[j] == h) { hs[j] = hs[--hsz]; break; }
        }
        int cur = 0; for (int j = 0; j < hsz; j++) if (hs[j] > cur) cur = hs[j];
        if (*rs == 0 || out[*rs - 1][1] != cur) {
            out[*rs] = malloc(2 * sizeof(int));
            out[*rs][0] = ev[i].x; out[*rs][1] = cur; (*rs)++;
        }
    }
    free(ev); free(hs); return out;
}
class Solution:
    def getSkyline(self, buildings: List[List[int]]) -> List[List[int]]:
        events = []
        for l, r, h in buildings:
            events.append((l, -h))
            events.append((r, h))
        events.sort()
        res, heights = [], [0]
        import bisect
        for x, h in events:
            if h < 0:
                bisect.insort(heights, -h)
            else:
                heights.remove(h)
            cur = heights[-1]
            if not res or res[-1][1] != cur:
                res.append([x, cur])
        return res
class Solution {
public:
    vector<vector<int>> getSkyline(vector<vector<int>>& b) {
        vector<pair<int,int>> ev;
        for (auto& x : b) { ev.push_back({x[0], -x[2]}); ev.push_back({x[1], x[2]}); }
        sort(ev.begin(), ev.end());
        multiset<int> hs = {0};
        vector<vector<int>> res;
        for (auto& [x, h] : ev) {
            if (h < 0) hs.insert(-h);
            else hs.erase(hs.find(h));
            int cur = *hs.rbegin();
            if (res.empty() || res.back()[1] != cur)
                res.push_back({x, cur});
        }
        return res;
    }
};
