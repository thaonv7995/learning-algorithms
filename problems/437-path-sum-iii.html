Mô tả
Đếm số path trong cây nhị phân (không nhất thiết qua root) có tổng bằng targetSum.

Ý tưởng
Prefix sum trên path root→node: DFS với map count prefix; tại node cộng map[curr-target].

int pathSum(struct TreeNode* root, int targetSum) { return 0; }
class Solution:
    def pathSum(self, root: Optional[TreeNode], targetSum: int) -> int:
        cnt = defaultdict(int); cnt[0] = 1
        def dfs(node, s):
            if not node: return 0
            s += node.val
            ans = cnt[s - targetSum]
            cnt[s] += 1
            ans += dfs(node.left, s) + dfs(node.right, s)
            cnt[s] -= 1
            return ans
        return dfs(root, 0)
class Solution {
public:
    int pathSum(TreeNode* root, int t) {
        unordered_map<long long,int> cnt; cnt[0]=1;
        function<int(TreeNode*,long long)> dfs=[&](TreeNode* n,long long s)->int{
            if(!n) return 0;
            s+=n->val; int ans=cnt[s-t];
            cnt[s]++; ans+=dfs(n->left,s)+dfs(n->right,s); cnt[s]--;
            return ans;
        };
        return dfs(root,0);
    }
};
