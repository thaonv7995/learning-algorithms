class Solution:
    def validPalindrome(self, s: str) -> bool:
        def ok(l,r,used):
            while l<r:
                if s[l]!=s[r]:
                    if used: return False
                    return ok(l+1,r,True) or ok(l,r-1,True)
                l+=1; r-=1
            return True
        return ok(0,len(s)-1,False)
class Solution {
    bool helper(string& s,int l,int r,bool used){
        while(l<r){
            if(s[l]!=s[r]){
                if(used) return false;
                return helper(s,l+1,r,true)||helper(s,l,r-1,true);
            }
            l++; r--;
        }
        return true;
    }
public:
    bool validPalindrome(string s){ return helper(s,0,s.size()-1,false); }
};
