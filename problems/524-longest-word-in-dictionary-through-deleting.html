Longest Word in Dictionary through Deleting

O(n * L^2) O(1) 52.7% AC
Mô tả
Cho s và mảng dictionary. Trả từ dài nhất trong dictionary là subsequence của s (xóa ký tự không cần); nếu nhiều từ cùng dài nhất chọn lex nhỏ nhất.
Ví dụ
Ví dụ 1
Ins = "abpcplea", dictionary = ["ale","apple","monkey","plea"]
Out"apple"
Ví dụ 2
Ins = "abpcplea", dictionary = ["a","b","c"]
Out"a"
Ý tưởng
Sort dictionary: length desc, lex asc. Trả từ đầu tiên là subsequence của s.
Mã nguồn
solution.c C99 
/* Sort dict + subseq — xem C++ */
class Solution:
    def findLongestWord(self, s: str, dictionary: List[str]) -> str:
        def subseq(w):
            i = 0
            for c in s:
                if i < len(w) and w[i] == c:
                    i += 1
            return i == len(w)
        for w in sorted(dictionary, key=lambda x: (-len(x), x)):
            if subseq(w):
                return w
        return ""
class Solution {
    bool subseq(const string& w, const string& s) {
        int i = 0;
        for (char c : s) if (i < (int)w.size() && w[i] == c) i++;
        return i == (int)w.size();
    }
public:
    string findLongestWord(string s, vector<string>& d) {
        sort(d.begin(), d.end(), [](auto& a, auto& b){
            if (a.size() != b.size()) return a.size() > b.size();
            return a < b;
        });
        for (auto& w : d) if (subseq(w, s)) return w;
        return "";
    }
};
