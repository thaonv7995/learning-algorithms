Network Delay Time

O(E log V) O(V+E) 60.4% AC
Mô tả
Có n node, times[i]=[u,v,w] thời gian tín hiệu u→v. Gửi từ node k. Trả thời gian tất cả node nhận được; -1 nếu có node không tới.
Ví dụ
Ví dụ 1
Intimes = [[2,1,1],[2,0,1],[0,1,1]], n = 3, k = 2
Out2
Ví dụ 2
Intimes = [[1,2,1],[2,1,3]], n = 2, k = 2
Out-1
Ý tưởng
Dijkstra từ k: dist[k]=0, min-heap relax cạnh. Trả max dist nếu mọi node finite else -1.

int networkDelayTime(int** times, int tn, int* cs, int n, int k) {
    const int INF = 1000000000;
    int dist[101]; for(int i=1;i<=n;i++) dist[i]=INF; dist[k]=0;
    for(int r=0;r<n;r++) for(int i=0;i<tn;i++){
        int u=times[i][0], v=times[i][1], w=times[i][2];
        if(dist[u]+w<dist[v]) dist[v]=dist[u]+w;
    }
    int ans=0; for(int i=1;i<=n;i++){ if(dist[i]==INF) return -1; if(dist[i]>ans) ans=dist[i]; }
    return ans;
}
class Solution:
    def networkDelayTime(self, times: List[List[int]], n: int, k: int) -> int:
        g = defaultdict(list)
        for u,v,w in times: g[u].append((v,w))
        dist = {i: float("inf") for i in range(1,n+1)}
        dist[k]=0
        pq=[(0,k)]
        while pq:
            d,u=heappop(pq)
            if d>dist[u]: continue
            for v,w in g[u]:
                if d+w<dist[v]: dist[v]=d+w; heappush(pq,(dist[v],v))
        ans=max(dist.values())
        return ans if ans<float("inf") else -1
class Solution {
public:
    int networkDelayTime(vector<vector<int>>& times, int n, int k) {
        vector<vector<pair<int,int>>> g(n+1);
        for(auto& t:times) g[t[0]].push_back({t[1],t[2]});
        const int INF=1e9;
        vector<int> dist(n+1,INF); dist[k]=0;
        priority_queue<pair<int,int>, vector<pair<int,int>>, greater<>> pq;
        pq.push({0,k});
        while(!pq.empty()){
            auto [d,u]=pq.top(); pq.pop();
            if(d>dist[u]) continue;
            for(auto [v,w]:g[u]) if(d+w<dist[v]){ dist[v]=d+w; pq.push({dist[v],v}); }
        }
        int ans=*max_element(dist.begin()+1,dist.end());
        return ans==INF?-1:ans;
    }
};
