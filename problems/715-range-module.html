Range Module

O(n) add/query O(n) 45% AC
Mô tả
Thiết kế cấu trúc Range Module: addRange(left,right) thêm đoạn [left,right); queryRange(left,right) kiểm tra [left,right) nằm trọn trong các đoạn đã thêm; removeRange(left,right) xóa phần giao với [left,right).
Ví dụ
Ví dụ 1
In["RangeModule","addRange","removeRange","queryRange","queryRange"], [[],[10,20],[14,16],[10,14],[13,15]]
Out[null,null,null,true,false]
Ví dụ 2
In["RangeModule","addRange","addRange","addRange","queryRange","queryRange"], [[],[10,20],[20,30],[10,40],[10,15],[20,25]]
Out[null,null,null,null,true,true]
Ý tưởng
Giữ danh sách đoạn không giao nhau đã merge sort theo left. add merge overlap; remove split/cắt đoạn; query kiểm tra một đoạn bao phủ.

void rangeModuleAddRange(int left,int right){ Seg nxt[10000]; int m=0; for(int i=0;i<tn;i++){ if(tr[i].r<left||tr[i].l>right) nxt[m++]=tr[i]; else { if(tr[i].l<left) left=tr[i].l; if(tr[i].r>right) right=tr[i].r; } } nxt[m].l=left; nxt[m++].r=right; tn=m; for(int i=0;i<tn;i++) tr[i]=nxt[i]; }
bool rangeModuleQueryRange(int left,int right){ for(int i=0;i<tn;i++){ if(tr[i].l<=left&&tr[i].r>=right) return true; if(tr[i].l>left) break; } return false; }
class RangeModule:
    def __init__(self):
        self.tracks = []
    def addRange(self, left: int, right: int) -> None:
        res = []
        for l, r in self.tracks:
            if r < left or l > right:
                res.append([l, r])
            else:
                left = min(left, l); right = max(right, r)
        res.append([left, right])
        self.tracks = sorted(res)
    def queryRange(self, left: int, right: int) -> bool:
        for l, r in self.tracks:
            if l <= left and r >= right: return True
            if l > left: break
        return False
    def removeRange(self, left: int, right: int) -> None:
        res = []
        for l, r in self.tracks:
            if r <= left or l >= right:
                res.append([l, r])
            else:
                if l < left: res.append([l, left])
                if r > right: res.append([right, r])
        self.tracks = res
class RangeModule {
    vector<pair<int,int>> t;
public:
    void addRange(int left, int right) {
        vector<pair<int,int>> nxt;
        for (auto [l,r] : t) {
            if (r < left || l > right) nxt.push_back({l,r});
            else { left = min(left,l); right = max(right,r); }
        }
        nxt.push_back({left,right});
        sort(nxt.begin(), nxt.end());
        t = move(nxt);
    }
    bool queryRange(int left, int right) {
        for (auto [l,r] : t) {
            if (l <= left && r >= right) return true;
            if (l > left) break;
        }
        return false;
    }
    void removeRange(int left, int right) {
        vector<pair<int,int>> nxt;
        for (auto [l,r] : t) {
            if (r <= left || l >= right) nxt.push_back({l,r});
            else {
                if (l < left) nxt.push_back({l,left});
                if (r > right) nxt.push_back({right,r});
            }
        }
        t = move(nxt);
    }
};
