Sliding Window Median

O(n log k) O(k) 39% AC
Mô tả
Sliding window median: với mỗi window size k trong nums, trả median (trung bình hai giữa nếu k chẵn).
Ví dụ
Ví dụ 1
Innums = [1,3,-1,-3,5,3,6,7], k = 3
Out[1.0,-1.0,-1.0,3.0,5.0,6.0]
Ví dụ 2
Innums = [1,2,3,4,2,3,1,4,2], k = 3
Out[2.0,3.0,3.0,3.0,2.0,3.0,2.0]
Ý tưởng
Two multiset heaps balance halves; lazy delete khi slide; median từ top left/right.

class Solution:
    def medianSlidingWindow(self, nums: List[int], k: int) -> List[float]:
        from sortedcontainers import SortedList
        window = SortedList(nums[:k])
        ans = []
        def med():
            return (window[(k-1)//2] + window[k//2]) / 2
        ans.append(med())
        for i in range(k, len(nums)):
            window.remove(nums[i-k])
            window.add(nums[i])
            ans.append(med())
        return ans
class Solution {
public:
    vector<double> medianSlidingWindow(vector<int>& nums, int k) {
        multiset<int> left, right; vector<double> ans;
        auto balance=[&](){
            while((int)left.size()>(k+1)/2){ right.insert(*left.rbegin()); left.erase(prev(left.end())); }
            while((int)left.size()<(k+1)/2){ left.insert(*right.begin()); right.erase(right.begin()); }
        };
        for(int i=0;i<(int)nums.size();i++){
            if(left.empty()||nums[i]<=*left.rbegin()) left.insert(nums[i]); else right.insert(nums[i]);
            balance();
            if(i>=k){
                int out=nums[i-k];
                auto er=[&](multiset<int>& s,int v){ auto it=s.find(v); if(it!=s.end()) s.erase(it); };
                er(left,out); er(right,out); balance();
            }
            if(i>=k-1){
                if(k&1) ans.push_back(*left.rbegin());
                else ans.push_back(((double)*left.rbegin()+*right.begin())/2);
            }
        }
        return ans;
    }
};
