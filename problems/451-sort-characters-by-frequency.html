Sort Characters By Frequency

O(n) O(k) 75.3% AC
Mô tả
Sắp xếp ký tự trong s theo tần suất giảm dần; cùng tần suất giữ thứ tự bất kỳ.
Ví dụ
Ví dụ 1
Ins = "tree"
Out"eert"
Ví dụ 2
Ins = "cccaaa"
Out"aaaccc"
Ví dụ 3
Ins = "Aabb"
Out"bbAa"
Ý tưởng
Đếm freq; bucket sort: bucket[i] chứa chars freq i; ghép từ freq cao xuống.

char* frequencySort(char* s) {
    int cnt[128]={0}; for(int i=0;s[i];i++) cnt[(unsigned char)s[i]]++;
    char* ans=malloc(strlen(s)+1); int p=0, n=strlen(s);
    for(int f=n;f>0;f--) for(int c=0;c<128;c++) if(cnt[c]==f)
        for(int k=0;k<f;k++) ans[p++]=(char)c;
    ans[p]=0; return ans;
}
class Solution:
    def frequencySort(self, s: str) -> str:
        cnt = Counter(s)
        buckets = [[] for _ in range(len(s)+1)]
        for ch, f in cnt.items(): buckets[f].append(ch)
        ans = []
        for f in range(len(s), 0, -1):
            for ch in buckets[f]: ans.append(ch * f)
        return "".join(ans)
class Solution {
public:
    string frequencySort(string s) {
        int cnt[128]={};
        for(char c:s) cnt[c]++;
        vector<string> b(s.size()+1);
        for(int i=0;i<128;i++) if(cnt[i]) b[cnt[i]]+=char(i);
        string ans;
        for(int f=s.size();f;f--) for(char c:b[f]) ans+=string(cnt[c],c);
        return ans;
    }
};
