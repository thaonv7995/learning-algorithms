class Solution:
    def maxAreaOfIsland(self, grid: List[List[int]]) -> int:
        m,n=len(grid),len(grid[0]); ans=0
        def dfs(i,j):
            if i<0 or j<0 or i>=m or j>=n or not grid[i][j]: return 0
            grid[i][j]=0; return 1+dfs(i+1,j)+dfs(i-1,j)+dfs(i,j+1)+dfs(i,j-1)
        for i in range(m):
            for j in range(n): ans=max(ans,dfs(i,j))
        return ans
class Solution {
    int dfs(vector<vector<int>>& g,int i,int j){
        if(i<0||j<0||i>=g.size()||j>=g[0].size()||!g[i][j]) return 0;
        g[i][j]=0; return 1+dfs(g,i+1,j)+dfs(g,i-1,j)+dfs(g,i,j+1)+dfs(g,i,j-1);
    }
public:
    int maxAreaOfIsland(vector<vector<int>>& g) {
        int ans=0;
        for(int i=0;i<(int)g.size();i++) for(int j=0;j<(int)g[0].size();j++) ans=max(ans,dfs(g,i,j));
        return ans;
    }
