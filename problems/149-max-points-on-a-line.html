Max Points on a Line

O(n²) O(n) 30.7% AC
Mô tả
Cho mảng points[i]=[xi,yi]. Tìm số điểm tối đa nằm trên cùng một đường thẳng.
Ví dụ
Ví dụ 1
Inpoints = [[1,1],[2,2],[3,3]]
Out3
Ví dụ 2
Inpoints = [[1,1],[3,2],[5,3],[4,1],[2,3],[1,4]]
Out4
Ý tưởng
Với mỗi điểm gốc, đếm số điểm cùng slope (dy,dx) chuẩn hóa bằng gcd; dùng pair hoặc string key. Trùng tọa độ tính riêng. O(n²) cặp.
Mã nguồn
solution.c C99 
/* Mỗi anchor đếm slope gcd — xem C++ */
from math import gcd

class Solution:
    def maxPoints(self, points: List[List[int]]) -> int:
        n = len(points)
        if n < 3:
            return n
        ans = 0
        for i in range(n):
            x1, y1 = points[i]
            slope = {}
            same = 1
            for j in range(i + 1, n):
                x2, y2 = points[j]
                dx, dy = x2 - x1, y2 - y1
                if dx == 0 and dy == 0:
                    same += 1
                    continue
                g = gcd(dx, dy)
                dx //= g
                dy //= g
                if dx < 0:
                    dx, dy = -dx, -dy
                key = (dx, dy)
                slope[key] = slope.get(key, 0) + 1
            local = same
            for cnt in slope.values():
                local = max(local, cnt + same)
            ans = max(ans, local)
        return ans
class Solution {
    static long long gcdll(long long a, long long b) {
        a = llabs(a); b = llabs(b);
        while (b) { long long t = a % b; a = b; b = t; }
        return a;
    }
public:
    int maxPoints(vector<vector<int>>& points) {
        int n = points.size();
        if (n < 3) return n;
        int ans = 0;
        for (int i = 0; i < n; ++i) {
            map<pair<long long,long long>, int> slope;
            int same = 1;
            for (int j = i + 1; j < n; ++j) {
                long long dx = points[j][0] - points[i][0];
                long long dy = points[j][1] - points[i][1];
                if (dx == 0 && dy == 0) { ++same; continue; }
                long long g = gcdll(dx, dy);
                dx /= g; dy /= g;
                if (dx < 0) { dx = -dx; dy = -dy; }
                ++slope[{dx, dy}];
            }
            int local = same;
            for (auto& p : slope)
                local = max(local, p.second + same);
            ans = max(ans, local);
        }
        return ans;
    }
};
