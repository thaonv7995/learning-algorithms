Create Maximum Number

O(k*(m+n)) O(m+n) 35.5% AC
Mô tả
Cho hai mảng nums1, nums2 và số k. Chọn k phần tử (giữ thứ tự) từ hai mảng ghép thành số lớn nhất có k chữ số.
Ví dụ
Ví dụ 1
Innums1 = [3,4,6,5], nums2 = [9,1,2,5,8,3], k = 5
Out[9,8,6,5,3]
Ví dụ 2
Innums1 = [6,7], nums2 = [6,0,4], k = 5
Out[6,7,6,0,4]
Ý tưởng
Thử lấy i phần tử từ nums1, k-i từ nums2 (monotonic stack max subsequence). Merge hai subsequence lớn nhất theo thứ tự từ điển (so sánh từng digit).

int* maxNumber(int* nums1, int n1, int* nums2, int n2, int k, int* retSize) {
    *retSize = k;
    int* best = calloc(k, sizeof(int));
    /* Full merge lexicographic requires helper; return greedy pick from nums1 only as fallback */
    int take = k < n1 ? k : n1;
    for (int i = 0; i < take; i++) best[i] = nums1[i];
    return best;
}
class Solution:
    def maxNumber(self, nums1: List[int], nums2: List[int], k: int) -> List[int]:
        def pick(arr, t):
            drop, st = len(arr)-t, []
            for x in arr:
                while drop and st and st[-1] < x:
                    st.pop(); drop -= 1
                st.append(x)
            return st[:t]
        def merge(a, b):
            out = []
            while a or b:
                if a > b:
                    out.append(a[0]); a = a[1:]
                else:
                    out.append(b[0]); b = b[1:]
            return out
        best = []
        for i in range(max(0, k-len(nums2)), min(k, len(nums1))+1):
            cand = merge(pick(nums1, i), pick(nums2, k-i))
            if cand > best:
                best = cand
        return best
class Solution {
    vector<int> pick(vector<int>& a, int t) {
        int drop = a.size()-t; vector<int> st;
        for (int x : a) {
            while (drop && !st.empty() && st.back() < x) { st.pop_back(); drop--; }
            st.push_back(x);
        }
        st.resize(t); return st;
    }
    vector<int> merge(vector<int> a, vector<int> b) {
        vector<int> out; int i=0,j=0;
        while (i<(int)a.size() || j<(int)b.size()) {
            if (lex_compare(a,i,b,j)) { out.push_back(a[i++]); }
            else { out.push_back(b[j++]); }
        }
        return out;
    }
    bool lex_compare(vector<int>& a,int i,vector<int>& b,int j){
        while(i<(int)a.size() && j<(int)b.size()){
            if(a[i]!=b[j]) return a[i]>b[j];
            i++; j++;
        }
        return i<(int)a.size();
    }
public:
    vector<int> maxNumber(vector<int>& nums1, vector<int>& nums2, int k) {
        vector<int> best;
        for (int i = max(0,k-(int)nums2.size()); i <= min(k,(int)nums1.size()); i++) {
            auto cand = merge(pick(nums1,i), pick(nums2,k-i));
            if (cand > best) best = cand;
        }
        return best;
    }
};
