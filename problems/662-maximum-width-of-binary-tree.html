int widthOfBinaryTree662(struct TreeNode* r) {
    if(!r) return 0;
    struct TreeNode** q=malloc(10000*sizeof(*q)); unsigned long long* idx=malloc(10000*8);
    int ql=0,qr=0,ans=0; q[qr]=r; idx[qr]=0; qr++;
    while(ql<qr){
        int sz=qr-ql;
        unsigned long long w=idx[qr-1]-idx[ql]+1; if((int)w>ans) ans=(int)w;
        for(int i=0;i<sz;i++){
            struct TreeNode* n=q[ql]; unsigned long long id=idx[ql]; ql++;
            if(n->left){ q[qr]=n->left; idx[qr]=id*2; qr++; }
            if(n->right){ q[qr]=n->right; idx[qr]=id*2+1; qr++; }
        }
    }
    free(q); free(idx); return ans;
}
class Solution:
    def widthOfBinaryTree(self, root: Optional[TreeNode]) -> int:
        if not root: return 0
        q=[(root,0)]; ans=0
        while q:
            ans=max(ans,q[-1][1]-q[0][1]+1)
            nxt=[]
            for node,i in q:
                if node.left: nxt.append((node.left,2*i))
                if node.right: nxt.append((node.right,2*i+1))
            q=nxt
        return ans
class Solution {
public:
    int widthOfBinaryTree(TreeNode* root) {
        if(!root) return 0;
        int ans=0; queue<pair<TreeNode*,unsigned long long>> q;
        q.push({root,0});
        while(!q.empty()){
            int sz=q.size(); ans=max(ans,(int)(q.back().second-q.front().second+1));
            for(int i=0;i<sz;i++){
                auto [n,idx]=q.front(); q.pop();
                if(n->left) q.push({n->left,2*idx});
                if(n->right) q.push({n->right,2*idx+1});
            }
        }
        return ans;
    }
};
