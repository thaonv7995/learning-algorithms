Pacific Atlantic Water Flow

O(mn) O(mn) 61% AC
Mô tả
Ma trận heights cho độ cao. Ô có thể chảy tới Pacific (biên trái/trên) và Atlantic (biên phải/dưới). Trả tọa độ các ô chảu được cả hai đại dương.
Ví dụ
Ví dụ 1
Inheights = [[1,2,2,3,5],[3,2,3,4,4],[2,4,5,3,1],[6,7,1,4,5],[5,1,1,2,4]]
Out[[0,4],[1,3],[1,4],[2,2],[3,0],[3,1],[4,0]]
Ví dụ 2
Inheights = [[2,1],[1,2]]
Out[[0,0],[0,1],[1,0],[1,1]]
Ý tưởng
DFS/BFS từ mọi ô biên Pacific và Atlantic (chỉ đi sang ô height ≥ current). Giao hai tập visited.

class Solution:
    def pacificAtlantic(self, heights: List[List[int]]) -> List[List[int]]:
        if not heights: return []
        m, n = len(heights), len(heights[0])
        pac, atl = set(), set()
        def dfs(r, c, seen):
            seen.add((r,c))
            for dr,dc in ((1,0),(-1,0),(0,1),(0,-1)):
                nr, nc = r+dr, c+dc
                if 0<=nr<m and 0<=nc<n and (nr,nc) not in seen and heights[nr][nc]>=heights[r][c]:
                    dfs(nr, nc, seen)
        for i in range(m):
            dfs(i, 0, pac); dfs(i, n-1, atl)
        for j in range(n):
            dfs(0, j, pac); dfs(m-1, j, atl)
        return [[r,c] for r,c in pac & atl]
class Solution {
public:
    vector<vector<int>> pacificAtlantic(vector<vector<int>>& h) {
        int m=h.size(), n=h[0].size();
        vector<vector<char>> p(m,vector<char>(n)), a(m,vector<char>(n));
        function<void(int,int,vector<vector<char>>&)> dfs=[&](int r,int c,vector<vector<char>>& vis){
            vis[r][c]=1;
            int d[4][2]={{1,0},{-1,0},{0,1},{0,-1}};
            for(auto& dd:d){ int nr=r+dd[0], nc=c+dd[1];
                if(nr>=0&&nr<m&&nc>=0&&nc<n&&!vis[nr][nc]&&h[nr][nc]>=h[r][c]) dfs(nr,nc,vis);
            }
        };
        for(int i=0;i<m;i++){ dfs(i,0,p); dfs(i,n-1,a); }
        for(int j=0;j<n;j++){ dfs(0,j,p); dfs(m-1,j,a); }
        vector<vector<int>> ans;
        for(int i=0;i<m;i++) for(int j=0;j<n;j++) if(p[i][j]&&a[i][j]) ans.push_back({i,j});
        return ans;
    }
};
