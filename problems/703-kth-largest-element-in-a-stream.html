Kth Largest Element in a Stream

O(log k) mỗi add O(k) 61% AC
Mô tả
Thiết kế lớp lưu dòng số và trả phần tử lớn thứ k sau mỗi lần thêm. KthLargest(k, nums) khởi tạo; add(val) thêm số và trả k-th largest hiện tại.
Ví dụ
Ví dụ 1
In["KthLargest","add","add","add","add","add","add"], [[3,[4,5,8,2]],[3],[5],[10],[9],[4]]
Out[null,4,5,5,8,8]
Ví dụ 2
In["KthLargest","add","add","add","add"], [[1,[1]],[1],[1],[2],[3]]
Out[null,1,1,1,2]
Ý tưởng
Min-heap size k: push val, nếu size>k pop min. Top heap là k-th largest (k phần tử lớn nhất, nhỏ nhất trong đó).
Mã nguồn
solution.c C99 
typedef struct { int k; int* a; int n, cap; } KthLargest;
static void hpush(int* h, int* sz, int x){ h[(*sz)++]=x; int i=*sz-1; while(i>0){ int p=(i-1)/2; if(h[p]<=h[i]) break; int t=h[p]; h[p]=h[i]; h[i]=t; i=p; } }
static void hpop(int* h, int* sz){ (*sz)--; h[0]=h[*sz]; int i=0; for(;;){ int l=2*i+1,r=l+1,m=i; if(l<*sz&&h[l]<h[m]) m=l; if(r<*sz&&h[r]<h[m]) m=r; if(m==i) break; int t=h[i]; h[i]=h[m]; h[m]=t; i=m; } }
KthLargest* kthLargestCreate(int k, int* nums, int n){ KthLargest* o=malloc(sizeof(KthLargest)); o->k=k; o->cap=1024; o->a=malloc(o->cap*4); o->n=0; for(int i=0;i<n;i++){ hpush(o->a,&o->n,nums[i]); if(o->n>k) hpop(o->a,&o->n);} return o; }
int kthLargestAdd(KthLargest* o, int val){ hpush(o->a,&o->n,val); if(o->n>o->k) hpop(o->a,&o->n); return o->a[0]; }
class KthLargest:
    def __init__(self, k: int, nums: List[int]):
        self.k = k
        self.h = nums
        heapify(self.h)
        while len(self.h) > k:
            heappop(self.h)
    def add(self, val: int) -> int:
        heappush(self.h, val)
        if len(self.h) > self.k:
            heappop(self.h)
        return self.h[0]
class KthLargest {
    int k;
    priority_queue<int, vector<int>, greater<int>> pq;
public:
    KthLargest(int k, vector<int>& nums) : k(k) {
        for (int x : nums) { pq.push(x); if ((int)pq.size() > k) pq.pop(); }
    }
    int add(int val) {
        pq.push(val);
        if ((int)pq.size() > k) pq.pop();
        return pq.top();
    }
};

  2. 2Duyệt chính — Top heap là k-th largest .

Heap giữ đúng k phần tử lớn nhất stream; min của chúng là thứ k.
