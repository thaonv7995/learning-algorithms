Prefix and Suffix Search

O(n * L) O(n * L) 40.9% AC
Mô tả
Thiết kế WordFilter: WordFilter(words) và f(prefix, suffix). f trả index lớn nhất của word vừa có prefix vừa có suffix (word phải chứa suffix).
Ví dụ
Ví dụ 1
In["WordFilter","f","f","f","f","f"], [[["apple"]],["a","e"],["b","e"],["a","le"],["apple","apple"]]
Out[null,0,-1,-1,0]
Ví dụ 2
In["WordFilter","f"], [[["abb","ba","ab","a"]],["a","a"]]
Out[null,3]
Ý tưởng
Trie: mỗi node lưu max index word đi qua. Insert word với mọi suffix (word#word) hoặc paired trie prefix+suffix.
Mã nguồn
solution.c C99 
typedef struct { char key[256]; int idx; } WFE;
static WFE wf[100000]; static int wfn;
WordFilter* wordFilterCreate(char** words, int n) {
    WordFilter* o=malloc(sizeof(int)); wfn=0;
    for(int i=0;i<n;i++){ int L=strlen(words[i]); for(int p=0;p<=L;p++) for(int s=0;s<=L;s++){ snprintf(wf[wfn].key,256,"%.*s#%s",p,words[i],words[i]+L-s); wf[wfn++].idx=i; } }
    return o;
}
int wordFilterF(WordFilter* o, char* prefix, char* suffix) {
    char k[256]; snprintf(k,256,"%s#%s",prefix,suffix);
    for(int i=0;i<wfn;i++) if(strcmp(wf[i].key,k)==0) return wf[i].idx;
    return -1;
}
class WordFilter:
    def __init__(self, words: List[str]):
        self.d = {}
        for i, w in enumerate(words):
            for p in range(len(w)+1):
                for s in range(len(w)+1):
                    self.d[w[:p]+"#"+w[len(w)-s:]] = i
    def f(self, prefix: str, suffix: str) -> int:
        return self.d.get(prefix+"#"+suffix, -1)
class WordFilter {
    unordered_map<string,int> mp;
public:
    WordFilter(vector<string>& words) {
        for (int i=0;i<(int)words.size();i++) {
            string& w=words[i];
            for (int p=0;p<=(int)w.size();p++)
                for (int s=0;s<=(int)w.size();s++)
                    mp[w.substr(0,p)+"#"+w.substr(w.size()-s)]=i;
        }
    }
    int f(string prefix, string suffix) {
        auto it=mp.find(prefix+"#"+suffix);
        return it==mp.end()?-1:it->second;
    }
};
