Partition Equal Subset Sum

O(n*sum) O(sum) 49.5% AC
Mô tả
Cho mảng nums số nguyên dương. Kiểm tra có thể chia thành hai tập con có tổng bằng nhau không.
Ví dụ
Ví dụ 1
Innums = [1,5,11,5]
Outtrue
Ví dụ 2
Innums = [1,2,3,5]
Outfalse
Ví dụ 3
Innums = [2,2,1,1]
Outtrue
Ý tưởng
Target = sum/2; 0/1 knapsack dp[s] = có subset sum s. Duyệt từng x cập nhật dp từ target xuống 0.

bool canPartition(int* nums, int n) {
    long s=0; for(int i=0;i<n;i++) s+=nums[i];
    if(s&1) return false;
    int t=(int)(s/2);
    char* dp=calloc(t+1,1); dp[0]=1;
    for(int i=0;i<n;i++) for(int j=t;j>=nums[i];j--) dp[j]|=dp[j-nums[i]];
    bool ok=dp[t]; free(dp); return ok;
}
class Solution:
    def canPartition(self, nums: List[int]) -> bool:
        s = sum(nums)
        if s & 1: return False
        t = s // 2
        dp = [False] * (t + 1)
        dp[0] = True
        for x in nums:
            for j in range(t, x - 1, -1):
                dp[j] |= dp[j - x]
        return dp[t]
class Solution {
public:
    bool canPartition(vector<int>& nums) {
        int s=accumulate(nums.begin(),nums.end(),0);
        if(s&1) return false;
        int t=s/2; vector<char> dp(t+1); dp[0]=1;
        for(int x:nums)
            for(int j=t;j>=x;j--) dp[j]|=dp[j-x];
        return dp[t];
    }
};
