class Solution:
    def cutOffTree(self, forest: List[List[int]]) -> int:
        trees=sorted((h,i,j) for i,row in enumerate(forest) for j,h in enumerate(row) if h>1)
        def bfs(si,sj,ti,tj):
            q=[(si,sj,0)]; vis={(si,sj)}
            while q:
                i,j,d=q.pop(0)
                if i==ti and j==tj: return d
                for di,dj in ((1,0),(-1,0),(0,1),(0,-1)):
                    ni,nj=i+di,j+dj
                    if 0<=ni<len(forest) and 0<=nj<len(forest[0]) and forest[ni][nj] and (ni,nj) not in vis:
                        vis.add((ni,nj)); q.append((ni,nj,d+1))
            return -1
        steps=0; ci=cj=0
        for _,ti,tj in trees:
            d=bfs(ci,cj,ti,tj)
            if d<0: return -1
            steps+=d; ci,cj=ti,tj
        return steps
