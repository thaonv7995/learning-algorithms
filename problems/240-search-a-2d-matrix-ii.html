Mô tả
Ma trận m×n: mỗi hàng tăng dần trái→phải, mỗi cột tăng dần trên→dưới. Kiểm tra target có trong matrix không.

Inmatrix = [[1,4,7,11,15],[2,5,8,12,19],[3,6,9,16,22],[10,13,14,17,24],[18,21,23,26,30]], target = 5
Outtrue
Ví dụ 2
Inmatrix = [[1,4,7,11,15],[2,5,8,12,19],[3,6,9,16,22],[10,13,14,17,24],[18,21,23,26,30]], target = 20
Outfalse
Ý tưởng
Bắt đầu góc trên-phải (hoặc dưới-trái): nếu val > target đi trái; nếu < target đi xuống; bằng → true.

bool searchMatrix240(int** matrix, int m, int* cs, int target) {
    if (!m) return false;
    int n = cs[0], r = 0, c = n - 1;
    while (r < m && c >= 0) {
        int v = matrix[r][c];
        if (v == target) return true;
        if (v > target) c--;
        else r++;
    }
    return false;
}
class Solution:
    def searchMatrix(self, matrix: List[List[int]], target: int) -> bool:
        if not matrix:
            return False
        m, n = len(matrix), len(matrix[0])
        r, c = 0, n - 1
        while r < m and c >= 0:
            v = matrix[r][c]
            if v == target:
                return True
            if v > target:
                c -= 1
            else:
                r += 1
        return False
class Solution {
public:
    bool searchMatrix(vector<vector<int>>& matrix, int target) {
        if (matrix.empty()) return false;
        int m = matrix.size(), n = matrix[0].size();
        int r = 0, c = n - 1;
        while (r < m && c >= 0) {
            if (matrix[r][c] == target) return true;
            if (matrix[r][c] > target) c--;
            else r++;
        }
        return false;
    }
};
