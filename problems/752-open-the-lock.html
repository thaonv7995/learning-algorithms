Open the Lock

O(10^4) O(10^4) 61.3% AC
Mô tả
Khóa 4 chữ số "0000" đến "9999". deadends là tổ hợp chết. Mỗi bước xoay một bánh răng ±1. Trả số bước mở "0000"→target; -1 nếu không mở được.
Ví dụ
Ví dụ 1
Indeadends = ["0201","0101","0102","1212","2002"], target = "0202"
Out6
Ví dụ 2
Indeadends = ["8888"], target = "0009"
Out1
Ví dụ 3
Indeadends = ["8887","8889","8878","8898","8788","8988","7888","9888"], target = "8888"
Out-1
Ý tưởng
BFS từ "0000": mỗi state 4 dial ±1 mod 10; skip deadends và visited.

class Solution:
    def openLock(self, deadends: List[str], target: str) -> int:
        dead = set(deadends)
        if "0000" in dead: return -1
        if target == "0000": return 0
        q = deque([("0000", 0)])
        seen = {"0000"}
        while q:
            s, d = q.popleft()
            for i in range(4):
                for diff in (-1, 1):
                    ns = s[:i] + str((int(s[i])+diff)%10) + s[i+1:]
                    if ns in dead or ns in seen: continue
                    if ns == target: return d + 1
                    seen.add(ns); q.append((ns, d+1))
        return -1
class Solution {
public:
    int openLock(vector<string>& deadends, string target) {
        unordered_set<string> dead(deadends.begin(), deadends.end());
        if(dead.count("0000")) return -1;
        if(target=="0000") return 0;
        queue<pair<string,int>> q; q.push({"0000",0});
        unordered_set<string> seen={"0000"};
        while(!q.empty()){
            auto [s,d]=q.front(); q.pop();
            for(int i=0;i<4;i++) for(int diff:-1){
                string ns=s; ns[i]=char('0'+(ns[i]-'0'+diff+10)%10);
                if(dead.count(ns)||seen.count(ns)) continue;
                if(ns==target) return d+1;
                seen.insert(ns); q.push({ns,d+1});
            }
        }
        return -1;
    }
};
