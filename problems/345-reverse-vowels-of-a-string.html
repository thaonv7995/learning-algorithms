Reverse Vowels of a String

O(n) O(n) 61.3% AC
Mô tả
Đảo vị trí các nguyên âm trong chuỗi s, giữ nguyên phụ âm và thứ tự tương đối của chúng.
Ví dụ
Ví dụ 1
Ins = "hello"
Out"holle"
Ví dụ 2
Ins = "leetcode"
Out"leotcede"
Ý tưởng
Two pointers trên mảng char: tìm vowel từ trái/phải, swap, tiếp tục. isVowel check a,e,i,o,u hoa thường.

    return c=='a'||c=='e'||c=='i'||c=='o'||c=='u'||c=='A'||c=='E'||c=='I'||c=='O'||c=='U';
}
char* reverseVowels(char* s) {
    int l=0, r=strlen(s)-1;
    while (l<r) {
        while (l<r && !isV(s[l])) l++;
        while (l<r && !isV(s[r])) r--;
        char t=s[l]; s[l]=s[r]; s[r]=t; l++; r--;
    }
    return s;
}
class Solution:
    def reverseVowels(self, s: str) -> str:
        v = set('aeiouAEIOU')
        arr = list(s)
        l, r = 0, len(arr) - 1
        while l < r:
            while l < r and arr[l] not in v: l += 1
            while l < r and arr[r] not in v: r -= 1
            arr[l], arr[r] = arr[r], arr[l]
            l += 1; r -= 1
        return ''.join(arr)
class Solution {
    bool isV(char c) {
        return c=='a'||c=='e'||c=='i'||c=='o'||c=='u'||c=='A'||c=='E'||c=='I'||c=='O'||c=='U';
    }
public:
    string reverseVowels(string s) {
        int l = 0, r = s.size()-1;
        while (l < r) {
            while (l < r && !isV(s[l])) l++;
            while (l < r && !isV(s[r])) r--;
            swap(s[l++], s[r--]);
        }
        return s;
    }
};
