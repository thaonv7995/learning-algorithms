Redundant Connection II

O(n) O(n) 36.2% AC
Mô tả
Directed tree n nodes one wrong edge. Tìm edge cần sửa để thành valid rooted tree.
Ví dụ
Ví dụ 1
Inedges=[[1,2],[1,3],[4,2]]
Out[4,2]
Ví dụ 2
Inedges=[[1,2],[2,3],[3,4],[4,1],[1,5]]
Out[4,1]
Ý tưởng
Find node with 2 parents (candidate child), UF cycle in undirected view.

int* findRedundantDirectedConnection685(int** e,int n,int* rs) {
    int par[1001]={0}, cand[4]={0}, hasCand=0;
    for(int i=0;i<n;i++){ int u=e[i][0],v=e[i][1];
        if(par[v]){ cand[0]=par[v]; cand[1]=v; cand[2]=u; cand[3]=v; hasCand=1; }
        else par[v]=u;
    }
    int cyc(int s,int ban){ int vis[1001]={0},x=s; while(x&&!vis[x]){ if(x==ban)return 0; vis[x]=1; x=par[x]; } return x!=0; }
    static int ans[2];
    if(hasCand){ if(cyc(cand[2],cand[3])){ ans[0]=cand[0]; ans[1]=cand[1]; } else { ans[0]=cand[2]; ans[1]=cand[3]; } *rs=2; return ans; }
    for(int i=n-1;i>=0;i--){ int u=e[i][0],v=e[i][1],old=par[v]; par[v]=0;
        if(!cyc(u,0)){ ans[0]=u; ans[1]=v; *rs=2; return ans; } par[v]=old; }
    *rs=0; return NULL;
}
class Solution:
    def findRedundantDirectedConnection(self, edges: List[List[int]]) -> List[int]:
        n=len(edges); par=[0]*(n+1); cand=[]
        for u,v in edges:
            if par[v]: cand=[par[v],v,u,v]
            else: par[v]=u
        root=1
        def find(x, skip=-1):
            vis=set()
            while x not in vis:
                if x==skip: break
                vis.add(x); x=par[x]
            return x in vis
        if not cand: u,v=edges[-1]
        else:
            a,b,c,d=cand
            if find(c,d): return [a,b]
            return [c,d]
        for i in range(n-1,-1,-1):
            u,v=edges[i]
            par[v]=0
            if not find(u): return [u,v]
            par[v]=u
        return []
