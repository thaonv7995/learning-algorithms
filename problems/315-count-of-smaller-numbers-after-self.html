Count of Smaller Numbers After Self

O(n log n) O(n) 43.6% AC
Mô tả
Cho mảng nums. Với mỗi nums[i], đếm số phần tử bên phải i nhỏ hơn nums[i]. Trả mảng kết quả cùng độ dài.
Ví dụ
Ví dụ 1
Innums = [5,2,6,1]
Out[2,1,1,0]
Ví dụ 2
Innums = [-1]
Out[0]
Ý tưởng
Merge sort đếm nghịch: duyệt từ phải, merge vào sorted; với mỗi x tìm vị trí lower_bound trong sorted → số nhỏ hơn bên phải.

typedef struct { int v, i; } Pair;
static void mergePairs(Pair* a, int l, int m, int r, int* ans) {
    Pair* tmp = malloc((r-l)*sizeof(Pair));
    int i=l, j=m, k=0;
    while (i<m && j<r) {
        if (a[i].v <= a[j].v) tmp[k++] = a[i++];
        else { ans[a[i].i] += r-j; tmp[k++] = a[j++]; }
    }
    while (i<m) tmp[k++] = a[i++];
    while (j<r) tmp[k++] = a[j++];
    for (int t=0;t<r-l;t++) a[l+t]=tmp[t];
    free(tmp);
}
static void sortPairs(Pair* a, int l, int r, int* ans) {
    if (r-l<=1) return;
    int m=(l+r)/2; sortPairs(a,l,m,ans); sortPairs(a,m,r,ans); mergePairs(a,l,m,r,ans);
}
int* countSmaller(int* nums, int n, int* retSize) {
    *retSize = n; int* ans = calloc(n,sizeof(int));
    Pair* a = malloc(n*sizeof(Pair));
    for (int i=0;i<n;i++){a[i].v=nums[i];a[i].i=i;}
    sortPairs(a,0,n,ans); free(a); return ans;
}
class Solution:
    def countSmaller(self, nums: List[int]) -> List[int]:
        ans = [0] * len(nums)
        arr = [(v, i) for i, v in enumerate(nums)]
        def merge(a):
            if len(a) <= 1:
                return a
            m = len(a) // 2
            left, right = merge(a[:m]), merge(a[m:])
            out, i, j = [], 0, 0
            while i < len(left) and j < len(right):
                if left[i][0] <= right[j][0]:
                    out.append(left[i]); i += 1
                else:
                    ans[left[i][1]] += len(right) - j
                    out.append(right[j]); j += 1
            out.extend(left[i:]); out.extend(right[j:])
            return out
        merge(arr)
        return ans
class Solution {
    vector<int> ans;
    void merge(vector<pair<int,int>>& a, int l, int m, int r) {
        vector<pair<int,int>> tmp;
        int i = l, j = m;
        while (i < m && j < r) {
            if (a[i].second <= a[j].second) tmp.push_back(a[i++]);
            else { ans[a[i].first] += r - j; tmp.push_back(a[j++]); }
        }
        while (i < m) tmp.push_back(a[i++]);
        while (j < r) tmp.push_back(a[j++]);
        for (int k = l; k < r; k++) a[k] = tmp[k-l];
    }
    void sort(vector<pair<int,int>>& a, int l, int r) {
        if (r - l <= 1) return;
        int m = (l + r) / 2;
        sort(a, l, m); sort(a, m, r); merge(a, l, m, r);
    }
public:
    vector<int> countSmaller(vector<int>& nums) {
        ans.assign(nums.size(), 0);
        vector<pair<int,int>> a;
        for (int i = 0; i < (int)nums.size(); i++) a.push_back({i, nums[i]});
        sort(a, 0, a.size());
        return ans;
    }
};
