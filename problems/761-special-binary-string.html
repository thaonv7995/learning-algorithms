Special Binary String

O(n) O(n) 79.3% AC
Mô tả
Chuỗi nhị phân special: 1...0 với số 1 = số 0 và mọi prefix có #0 >= #1. Cho special string s. Trả chuỗi special lớn nhất lex (cùng length).
Ví dụ
Ví dụ 1
Ins = "11011000"
Out"11100100"
Ví dụ 2
Ins = "10"
Out"10"
Ý tưởng
Recursive divide: special = "1" + A + "0" + B với A,B special; sort các block con theo quy tắc đảo lex special.

class Solution:
    def makeLargestSpecial(self, s: str) -> str:
        if not s: return ""
        cnt = 0; parts = []; i = 0
        for j, ch in enumerate(s):
            cnt += 1 if ch == "1" else -1
            if cnt == 0:
                parts.append("1" + self.makeLargestSpecial(s[i+1:j]) + "0")
                i = j + 1
        parts.sort(reverse=True)
        return "".join(parts)
class Solution {
public:
    string makeLargestSpecial(string s) {
        if(s.empty()) return "";
        vector<string> parts; int cnt=0, i=0;
        for(int j=0;j<(int)s.size();j++){
            cnt += s[j]=='1'?1:-1;
            if(cnt==0){
                parts.push_back("1"+makeLargestSpecial(s.substr(i+1,j-i-1))+"0");
                i=j+1;
            }
        }
        sort(parts.begin(), parts.end(), greater<string>());
        string ans;
        for(auto& p: parts) ans+=p;
        return ans;
    }
};
