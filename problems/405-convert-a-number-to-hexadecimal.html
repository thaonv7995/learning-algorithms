Convert a Number to Hexadecimal

O(1) O(1) 54.1% AC
Mô tả
Cho số nguyên num (32-bit signed). Trả chuỗi biểu diễn hexadecimal của num coi như unsigned 32-bit (chữ thường a–f).
Ví dụ
Ví dụ 1
Innum = 26
Out"1a"
Ví dụ 2
Innum = -1
Out"ffffffff"
Ví dụ 3
Innum = 0
Out"0"
Ý tưởng
Ép unsigned: u = num & 0xFFFFFFFF. Lặp lấy 4 bit cuối, map 0–15 → hex, đảo chuỗi.
Mã nguồn
solution.c C99 
char* toHex(int num) {
    if (num == 0) { char* r=malloc(2); r[0]='0'; r[1]=0; return r; }
    unsigned u = (unsigned)num;
    char* ans = malloc(9); int p=0;
    const char* d = "0123456789abcdef";
    while (u) { ans[p++] = d[u & 15]; u >>= 4; }
    ans[p]=0; for(int i=0,j=p-1;i<j;i++,j--){ char t=ans[i]; ans[i]=ans[j]; ans[j]=t; }
    return ans;
}
class Solution:
    def toHex(self, num: int) -> str:
        if num == 0: return "0"
        u = num & 0xFFFFFFFF
        digits = "0123456789abcdef"
        ans = []
        while u:
            ans.append(digits[u & 15])
            u >>= 4
        return "".join(reversed(ans))
class Solution {
public:
    string toHex(int num) {
        if (num == 0) return "0";
        unsigned u = num;
        string d = "0123456789abcdef", ans;
        while (u) { ans.push_back(d[u & 15]); u >>= 4; }
        reverse(ans.begin(), ans.end());
        return ans;
    }
};
