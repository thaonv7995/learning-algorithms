Reverse Nodes in k-Group

O(N) O(1) 66.1% AC
Mô tả
Đảo ngược từng nhóm k nút liên tiếp trong linked list. Nếu số nút còn lại < k thì giữ nguyên.
Ví dụ
Ví dụ 1
Inhead = [1,2,3,4,5], k = 2
Out[2,1,4,3,5]
Ví dụ 2
Inhead = [1,2,3,4,5], k = 3
Out[3,2,1,4,5]
Ví dụ 3
Inhead = [1], k = 1
Out[1]
Ý tưởng
Đếm k nút từ head. Nếu đủ k: reverse đoạn [head..k] in-place, nối prev→newHead, tail→reverseKGroup(node k+1).
Mã nguồn
solution.c C99 
/* reverseKGroup — reverse segment + đệ quy; xem C++ */
class Solution:
    def reverseKGroup(self, head, k):
        cur = head; count = 0
        while cur and count < k:
            cur = cur.next; count += 1
        if count == k:
            rev = self._rev(head, k)
            head.next = self.reverseKGroup(cur, k)
            return rev
        return head
    def _rev(self, head, k):
        prev = None; cur = head
        for _ in range(k):
            nxt = cur.next; cur.next = prev; prev = cur; cur = nxt
        return prev
class Solution {
    ListNode* rev(ListNode* a, ListNode* b) {
        ListNode* prev = b;
        while (a != b) { auto* n = a->next; a->next = prev; prev = a; a = n; }
        return prev;
    }
public:
    ListNode* reverseKGroup(ListNode* head, int k) {
        ListNode* cur = head;
        for (int i = 0; i < k; i++) if (!cur) return head;
        cur = cur->next;
        head->next = reverseKGroup(cur, k);
        return rev(head, cur);
    }
};
