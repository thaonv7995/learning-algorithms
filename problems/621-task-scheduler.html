Task Scheduler

O(n) O(1) 63.1% AC
Mô tả
Cho tasks (A-Z) và cooldown n. Mỗi đơn vị thời gian làm một task hoặc idle; cùng task cách nhau ít nhất n. Tìm thời gian tối thiểu.
Ví dụ
Ví dụ 1
Intasks = ["A","A","A","B","B","B"], n = 2
Out8
Ví dụ 2
Intasks = ["A","A","A","B","B","B"], n = 0
Out6
Ý tưởng
Đếm tần suất; task nhiều nhất quyết idle slots: (maxCount-1)*(n+1)+số task có freq max.
Mã nguồn
solution.c C99 
int leastInterval(char* tasks, int len, int n) {
    int c[26]={0}; for(int i=0;i<len;i++) c[tasks[i]-'A']++;
    int mx=0; for(int i=0;i<26;i++) if(c[i]>mx) mx=c[i];
    int num=0; for(int i=0;i<26;i++) if(c[i]==mx) num++;
    int frame=(mx-1)*(n+1)+num;
    return frame>len?frame:len;
}
class Solution:
    def leastInterval(self, tasks: List[str], n: int) -> int:
        cnt = Counter(tasks); m = max(cnt.values())
        num = sum(1 for v in cnt.values() if v == m)
        return max(len(tasks), (m-1)*(n+1)+num)
class Solution {
public:
    int leastInterval(vector<char>& tasks, int n) {
        int c[26]={}; for(char t:tasks) c[t-'A']++;
        int mx=*max_element(c,c+26), num=0;
        for(int x:c) if(x==mx) num++;
        return max((int)tasks.size(), (mx-1)*(n+1)+num);
    }
};
