/* Stack tag parser — xem C++ */
class Solution:
    def isValid(self, code: str) -> bool:
        if not code.startswith("<") or code[0] != "<" or not code.endswith(">"):
            return False
        stack = []
        i = 0
        while i < len(code):
            if stack and code[i] not in "<":
                i += 1
                continue
            if code.startswith("<![CDATA[", i):
                j = code.find("]]>", i)
                if j == -1 or not stack: return False
                i = j + 3
            elif code.startswith("</", i):
                j = code.find(">", i)
                if j == -1: return False
                tag = code[i+2:j]
                if not stack or stack.pop() != tag: return False
                i = j + 1
                if not stack and i != len(code): return False
            elif code.startswith("<", i):
                j = code.find(">", i)
                if j == -1: return False
                tag = code[i+1:j]
                if not tag or len(tag) > 9 or not tag[0].isupper() or not tag.isalpha():
                    return False
                stack.append(tag)
                i = j + 1
            else:
                return False
        return not stack
class Solution {
public:
    bool isValid(string code) {
        if (code.empty() || code[0] != '<' || code.back() != '>') return false;
        vector<string> st;
        for (int i = 0; i < (int)code.size(); ) {
            if (!st.empty() && code[i] != '<') { i++; continue; }
            if (code.compare(i, 9, "<![CDATA[") == 0) {
                size_t j = code.find("]]>", i);
                if (j == string::npos || st.empty()) return false;
                i = j + 3;
            } else if (code.compare(i, 2, "</") == 0) {
                size_t j = code.find('>', i);
                if (j == string::npos) return false;
                string tag = code.substr(i+2, j-i-2);
                if (st.empty() || st.back() != tag) return false;
                st.pop_back();
                i = j + 1;
                if (st.empty() && i != (int)code.size()) return false;
            } else if (code[i] == '<') {
                size_t j = code.find('>', i);
                if (j == string::npos) return false;
                string tag = code.substr(i+1, j-i-1);
                if (tag.empty() || tag.size()>9 || !isupper(tag[0])) return false;
                for (char c : tag) if (!isalpha(c)) return false;
                st.push_back(tag);
                i = j + 1;
            } else return false;
        }
        return st.empty();
    }
};
