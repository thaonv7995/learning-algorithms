Mô tả
Trả về giá trị các nút theo từng tầng, nhưng thứ tự tầng từ dưới lên trên (lá → gốc). Mỗi tầng trái → phải.
Ví dụ
Ví dụ 1
Inroot = [3,9,20,null,null,15,7]
Out[[15,7],[9,20],[3]]
Ví dụ 2
Inroot = [1]
Out[[1]]
Ví dụ 3
Inroot = []
Out[]
Ý tưởng
BFS giống LC 102: queue, mỗi vòng xử lý đúng size hiện tại = một tầng. Cuối cùng đảo ngược mảng kết quả.

/* BFS tầng rồi đảo — xem C++ */
class Solution:
    def levelOrderBottom(self, root: Optional[TreeNode]) -> List[List[int]]:
        if not root: return []
        res, q = [], deque([root])
        while q:
            row, n = [], len(q)
            for _ in range(n):
                node = q.popleft()
                row.append(node.val)
                if node.left: q.append(node.left)
                if node.right: q.append(node.right)
            res.append(row)
        return res[::-1]
class Solution {
public:
    vector<vector<int>> levelOrderBottom(TreeNode* root) {
        vector<vector<int>> res;
        if (!root) return res;
        queue<TreeNode*> q; q.push(root);
        while (!q.empty()) {
            int n = q.size(); vector<int> row;
            for (int i = 0; i < n; i++) {
                auto* nd = q.front(); q.pop();
                row.push_back(nd->val);
                if (nd->left) q.push(nd->left);
                if (nd->right) q.push(nd->right);
            }
            res.push_back(row);
        }
        reverse(res.begin(), res.end());
        return res;
    }
};
