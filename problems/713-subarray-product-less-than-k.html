Subarray Product Less Than K

O(n) O(1) 54.3% AC
Mô tả
Cho mảng nums số nguyên dương và k. Đếm số đoạn con liên tiếp có tích các phần tử < k.
Ví dụ
Ví dụ 1
Innums = [10,5,2,6], k = 100
Out8
Ví dụ 2
Innums = [1,1,1], k = 2
Out0
Ví dụ 3
Innums = [1,2,3], k = 10
Out5
Ý tưởng
Sliding window: nhân vào prod; khi prod>=k chia nums[left++] cho prod. Mỗi vị trí right, thêm (right-left+1) đoạn kết thúc tại right.
Mã nguồn
solution.c C99 
int numSubarrayProductLessThanK(int* nums, int n, int k) {
    if (k <= 1) return 0;
    long long prod = 1;
    int ans = 0, left = 0;
    for (int right = 0; right < n; right++) {
        prod *= nums[right];
        while (prod >= k) prod /= nums[left++];
        ans += right - left + 1;
    }
    return ans;
}
class Solution:
    def numSubarrayProductLessThanK(self, nums: List[int], k: int) -> int:
        if k <= 1: return 0
        ans = left = 0
        prod = 1
        for right, x in enumerate(nums):
            prod *= x
            while prod >= k:
                prod //= nums[left]
                left += 1
            ans += right - left + 1
        return ans
class Solution {
public:
    int numSubarrayProductLessThanK(vector<int>& nums, int k) {
        if (k <= 1) return 0;
        long long prod = 1;
        int ans = 0, left = 0;
        for (int right = 0; right < (int)nums.size(); right++) {
            prod *= nums[right];
            while (prod >= k) prod /= nums[left++];
            ans += right - left + 1;
        }
        return ans;
    }
};
