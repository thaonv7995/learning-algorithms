Super Ugly Number

O(n×k) O(n+k) 46.1% AC
Mô tả
Số xấu xí chỉ có thừa số nguyên tố 2,3,5. Cho mảng primes và n, trả số xấu xí thứ n (super ugly) chỉ dùng thừa số trong primes.
Ví dụ
Ví dụ 1
Inn = 12, primes = [2,7,13,19]
Out32
Ví dụ 2
Inn = 1, primes = [2,3,5]
Out1
Ý tưởng
DP giống Ugly Number II: ugly[0]=1; mỗi prime giữ con trỏ idx[i]; ugly[j]=min(ugly[idx[i]]*primes[i]); tăng mọi idx[i] đạt min.

int nthSuperUglyNumber(int n, int* primes, int k) {
    long long* ugly = malloc(n * sizeof(long long));
    int* idx = calloc(k, sizeof(int));
    ugly[0] = 1; int sz = 1;
    while (sz < n) {
        long long nxt = LLONG_MAX;
        for (int i = 0; i < k; i++) {
            long long v = ugly[idx[i]] * primes[i];
            if (v < nxt) nxt = v;
        }
        ugly[sz++] = nxt;
        for (int i = 0; i < k; i++)
            if (ugly[idx[i]] * primes[i] == nxt) idx[i]++;
    }
    int ans = (int)ugly[n-1];
    free(ugly); free(idx);
    return ans;
}
class Solution:
    def nthSuperUglyNumber(self, n: int, primes: List[int]) -> int:
        ugly = [1]
        idx = [0]*len(primes)
        while len(ugly) < n:
            nxt = min(ugly[idx[i]]*primes[i] for i in range(len(primes)))
            ugly.append(nxt)
            for i in range(len(primes)):
                if ugly[idx[i]]*primes[i] == nxt:
                    idx[i] += 1
        return ugly[-1]
class Solution {
public:
    int nthSuperUglyNumber(int n, vector<int>& primes) {
        vector<long long> ugly = {1};
        vector<int> idx(primes.size(), 0);
        while ((int)ugly.size() < n) {
            long long nxt = LLONG_MAX;
            for (int i = 0; i < (int)primes.size(); i++)
                nxt = min(nxt, ugly[idx[i]] * (long long)primes[i]);
            ugly.push_back(nxt);
            for (int i = 0; i < (int)primes.size(); i++)
                if (ugly[idx[i]] * (long long)primes[i] == nxt) idx[i]++;
        }
        return (int)ugly.back();
    }
};
