Cherry Pickup

O(n^3) O(n^2) 39.5% AC
Mô tả
Lưới grid: 1 là ô trống, -1 là chướng ngại. Bắt đầu (0,0), đích (n-1,n-1), chỉ đi xuống/phải. Thu cherry trên đường (mỗi ô tối đa 1). Trả số cherry tối đa.
Ví dụ
Ví dụ 1
Ingrid = [[0,1,-1],[1,0,-1],[1,1,1]]
Out5
Ví dụ 2
Ingrid = [[1,1,-1],[1,-1,1],[-1,1,1]]
Out0
Ý tưởng
DP 3 chiều: dp[r1][c1][r2] hai người đi song song từ start; c1=r1+steps-r2, c2=c1+(c1-r1). Cộng cherry nếu ô hợp lệ, trừ trùng ô.

int cherryPickup(int** grid, int n) {
    int dp[25][25][25]; for(int i=0;i<25;i++) for(int j=0;j<25;j++) for(int k=0;k<25;k++) dp[i][j][k]=-1;
    dp[0][0][0]=grid[0][0];
    for(int s=1;s<2*n-1;s++) for(int r1=0;r1<n;r1++) for(int r2=0;r2<n;r2++){
        int c1=s-r2; if(r1>s||c1<0||c1>=n||grid[r1][c1]<0) continue;
        int c2=c1+(r2-r1); if(c2<0||c2>=n||grid[r2][c2]<0) continue;
        int best=-1;
        for(int pr1=r1-1;pr1<=r1;pr1++) for(int pr2=r2-1;pr2<=r2;pr2++){
            if(pr1<0||pr2<0) continue; int pc1=c1-(r1-pr1); if(dp[pr1][pc1][pr2]>=0 && dp[pr1][pc1][pr2]>best) best=dp[pr1][pc1][pr2];
        }
        if(best>=0){ int add=grid[r1][c1]+((r1!=r2||c1!=c2)?grid[r2][c2]:0); dp[r1][c1][r2]=best+add; }
    }
    int v=dp[n-1][n-1][n-1]; return v<0?0:v;
}
class Solution:
    def cherryPickup(self, grid: List[List[int]]) -> int:
        n = len(grid)
        dp = [[[-1]*n for _ in range(n)] for _ in range(n)]
        dp[0][0][0] = grid[0][0]
        for steps in range(1, 2*n-1):
            nd = [[-1]*n for _ in range(n)]
            for r1 in range(n):
                for r2 in range(n):
                    c1 = steps - r2
                    if r1 > steps or c1 >= n or c1 < 0: continue
                    if grid[r1][c1] == -1: continue
                    c2 = c1 + (r2 - r1)
                    if c2 >= n or c2 < 0 or grid[r2][c2] == -1: continue
                    best = -1
                    for pr1 in (r1-1, r1):
                        for pr2 in (r2-1, r2):
                            if pr1 >= 0 and pr2 >= 0 and dp[pr1][c1-(r1-pr1)][pr2] >= 0:
                                best = max(best, dp[pr1][c1-(r1-pr1)][pr2])
                    if best >= 0:
                        gain = grid[r1][c1] + (grid[r2][c2] if (r1,c1)!=(r2,c2) else 0)
                        nd[r1][r2] = max(nd[r1][r2], best + gain)
            dp = nd
        return max(0, dp[n-1][n-1][n-1])
class Solution {
public:
    int cherryPickup(vector<vector<int>>& g) {
        int n=g.size();
        vector<vector<vector<int>>> dp(n, vector<vector<int>>(n, vector<int>(n, -1)));
        dp[0][0][0]=g[0][0];
        for(int s=1;s<2*n-1;s++){
            vector<vector<vector<int>>> nd(n, vector<vector<int>>(n, vector<int>(n,-1)));
            for(int r1=0;r1<n;r1++) for(int r2=0;r2<n;r2++){
                int c1=s-r2; if(r1>s||c1<0||c1>=n||g[r1][c1]<0) continue;
                int c2=c1+(r2-r1); if(c2<0||c2>=n||g[r2][c2]<0) continue;
                int best=-1;
                for(int a=r1-1;a<=r1;a++) for(int b=r2-1;b<=r2;b++){
                    if(a>=0&&b>=0&&dp[a][c1-(r1-a)][b]>=0) best=max(best,dp[a][c1-(r1-a)][b]);
                }
                if(best>=0){
                    int add=g[r1][c1]+((r1!=r2||c1!=c2)?g[r2][c2]:0);
                    nd[r1][r2]=max(nd[r1][r2], best+add);
                }
            }
            dp=move(nd);
        }
        return max(0, dp[n-1][n-1][n-1]);
    }
};
