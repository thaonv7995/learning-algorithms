My Calendar I

O(n) per op O(n) 58.3% AC
Mô tả
Thiết kế lịch: book(start,end) thêm [start,end); nếu không trùng interval đã book thì true, ngược lại false. Half-open interval.
Ví dụ
Ví dụ 1
In["MyCalendar","book","book","book"], [[],[10,20],[15,25],[20,30]]
Out[null,true,false,true]
Ví dụ 2
In["MyCalendar","book","book","book","book","book","book"], [[],[47,48],[24,43],[36,39],[46,49],[33,42],[12,35],[27,38]]
Out[null,true,true,false,true,false,false]
Ý tưởng
Giữ list intervals đã book. book kiểm tra không overlap: start < other.end && end > other.start.
Mã nguồn
solution.c C99 
typedef struct { int s, e; } Book;
static Book cal[1000]; static int cn;
bool myCalendarBook(int start, int end) {
    for (int i = 0; i < cn; i++)
        if (start < cal[i].e && end > cal[i].s) return false;
    cal[cn].s = start; cal[cn++].e = end;
    return true;
}
class MyCalendar:
    def __init__(self):
        self.bookings = []
    def book(self, start: int, end: int) -> bool:
        for s, e in self.bookings:
            if start < e and end > s: return False
        self.bookings.append((start, end))
        return True
class MyCalendar {
    vector<pair<int,int>> b;
public:
    bool book(int start, int end) {
        for (auto [s,e] : b)
            if (start < e && end > s) return false;
        b.push_back({start,end});
        return true;
    }
};
