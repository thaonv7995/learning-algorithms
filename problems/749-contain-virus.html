class Solution:
    def containVirus(self, grid: List[List[int]]) -> int:
        m,n=len(grid),len(grid[0])
        def regions():
            vis=set(); regs=[]
            for i in range(m):
                for j in range(n):
                    if grid[i][j]!=1 or (i,j) in vis: continue
                    q=[(i,j)]; vis.add((i,j)); cells=[(i,j)]; per=0
                    while q:
                        r,c=q.pop()
                        for dr,dc in ((1,0),(-1,0),(0,1),(0,-1)):
                            nr,nc=r+dr,c+dc
                            if nr<0 or nc<0 or nr>=m or nc>=n: per+=1
                            elif grid[nr][nc]==0: per+=1
                            elif (nr,nc) not in vis: vis.add((nr,nc)); q.append((nr,nc)); cells.append((nr,nc))
                    regs.append((len(cells),-per,-cells[0][0],-cells[0][1],cells))
            return regs
        ans=0
        while True:
            regs=regions()
            if not regs: break
            regs.sort(reverse=True)
            _,_,_,_,wall=regs[0]
            for r,c in wall: grid[r][c]=-1
            for _,_,_,_,cells in regs[1:]:
                for r,c in cells:
                    for dr,dc in ((1,0),(-1,0),(0,1),(0,-1)):
                        nr,nc=r+dr,c+dc
                        if 0<=nr<m and 0<=nc<n and grid[nr][nc]==0: grid[nr][nc]=1
            ans+=sum(grid[r][c]==1 for r in range(m) for c in range(n))
        return ans
class Solution {
public:
    int containVirus(vector<vector<int>>& g) {
        int m=g.size(), n=g[0].size(), ans=0;
        auto dirs=vector<pair<int,int>>{{1,0},{-1,0},{0,1},{0,-1}};
        while(true){
            vector<vector<int>> vis(m, vector<int>(n));
            vector<tuple<int,int,int,int,vector<pair<int,int>>>> regs;
            for(int i=0;i<m;i++) for(int j=0;j<n;j++) if(g[i][j]==1&&!vis[i][j]){
                queue<pair<int,int>> q; q.push({i,j}); vis[i][j]=1;
                vector<pair<int,int>> cells={{i,j}}; int per=0;
                while(!q.empty()){
                    auto [r,c]=q.front(); q.pop();
                    for(auto [dr,dc]:dirs){
                        int nr=r+dr, nc=c+dc;
                        if(nr<0||nc<0||nr>=m||nc>=n||g[nr][nc]==0) per++;
                        else if(!vis[nr][nc]){ vis[nr][nc]=1; q.push({nr,nc}); cells.push_back({nr,nc}); }
                    }
                }
                regs.push_back({(int)cells.size(),-per,-i,-j,cells});
            }
            if(regs.empty()) break;
            sort(regs.begin(), regs.end(), greater<>());
            for(auto& [a,b,c,d,cells]: regs[0]|std::ignore) {}
            auto wall=get<4>(regs[0]);
            for(auto [r,c]: wall) g[r][c]=-1;
            for(int t=1;t<(int)regs.size();t++){
                for(auto [r,c]: get<4>(regs[t]))
                    for(auto [dr,dc]:dirs){
                        int nr=r+dr, nc=c+dc;
                        if(nr>=0&&nc>=0&&nr<m&&nc<n&&g[nr][nc]==0) g[nr][nc]=1;
                    }
            }
            for(int i=0;i<m;i++) for(int j=0;j<n;j++) if(g[i][j]==1) ans++;
        }
        return ans;
    }
};
