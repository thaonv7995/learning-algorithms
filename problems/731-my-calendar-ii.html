My Calendar II

O(n) per op O(n) 63.1% AC
Mô tả
My Calendar II: book(start,end) cho phép tối đa 2 booking overlap tại một thời điểm (triple overlap forbidden). Trả false nếu book tạo triple overlap.
Ví dụ
Ví dụ 1
In["MyCalendarTwo","book","book","book","book","book","book"], [[],[10,20],[50,60],[10,40],[5,15],[5,10],[25,55]]
Out[null,true,true,true,false,true,true]
Ví dụ 2
In["MyCalendarTwo","book","book","book","book"], [[],[26,35],[26,32],[25,32],[18,26]]
Out[null,true,true,false,true]
Ý tưởng
Giữ bookings và overlaps (double-booked intervals). book kiểm tra overlap với overlaps → triple; else merge overlap vào overlaps, book vào bookings.
Mã nguồn
solution.c C99 
typedef struct { int s,e; } Iv;
static Iv bk2[1000], ov2[1000]; static int nb2, no2;
static int xov(int a,int b,int c,int d){ return a<d && b>c; }
bool myCalendarTwoBook(int start,int end){
    for(int i=0;i<no2;i++) if(xov(start,end,ov2[i].s,ov2[i].e)) return false;
    Iv nb[1000]; int m=0;
    for(int i=0;i<nb2;i++){
        if(xov(start,end,bk2[i].s,bk2[i].e)){
            ov2[no2].s=start>bk2[i].s?start:bk2[i].s;
            ov2[no2].e=end<bk2[i].e?end:bk2[i].e; no2++;
        } else nb[m++]=bk2[i];
    }
    nb[m].s=start; nb[m++].e=end; nb2=m; for(int i=0;i<m;i++) bk2[i]=nb[i];
    return true;
}
class MyCalendarTwo:
    def __init__(self):
        self.bookings = []
        self.overlaps = []
    def book(self, start: int, end: int) -> bool:
        for s, e in self.overlaps:
            if start < e and end > s: return False
        new_over = []
        for s, e in self.bookings:
            if start < e and end > s:
                self.overlaps.append((max(start,s), min(end,e)))
            else:
                new_over.append((s,e))
        self.bookings = new_over + [(start,end)]
        return True
class MyCalendarTwo {
    vector<pair<int,int>> bk, ov;
    static bool cross(int a,int b,int c,int d){ return a<d && b>c; }
public:
    bool book(int start, int end) {
        for (auto [s,e]: ov) if (cross(start,end,s,e)) return false;
        vector<pair<int,int>> nb;
        for (auto [s,e]: bk) {
            if (cross(start,end,s,e)) ov.push_back({max(start,s), min(end,e)});
            else nb.push_back({s,e});
        }
        nb.push_back({start,end});
        bk = move(nb);
        return true;
    }
};
