Zuma Game

O(states) O(states) 29.7% AC
Mô tả
Zuma: chèn bóng từ hand vào board để gom ≥3 cùng màu liên tiếp (cascade). Min số bóng để clear board, -1 nếu không.
Ví dụ
Ví dụ 1
Inboard = "WRRBBW", hand = "RB"
Out-1
Ví dụ 2
Inboard = "WWRRBBWW", hand = "WRBRW"
Out2
Ý tưởng
BFS (board, hand): thử insert từng màu tại mọi vị trí hợp lệ; collapse ≥3 lặp đến ổn.

class Solution:
    def findMinStep(self, board: str, hand: str) -> int:
        def clean(s):
            i = 0
            while i < len(s):
                j = i
                while j < len(s) and s[j] == s[i]: j += 1
                if j - i >= 3: return clean(s[:i] + s[j:])
                i = j
            return s
        hand = "".join(sorted(hand))
        q = deque([(board, hand, 0)])
        seen = {(board, hand)}
        while q:
            b, h, steps = q.popleft()
            if not b: return steps
            if not h: continue
            for i in range(len(b)+1):
                for j, c in enumerate(h):
                    if i > 0 and b[i-1] == c: continue
                    if i < len(b) and b[i] == c: continue
                    nb = clean(b[:i]+c+b[i:])
                    nh = h[:j]+h[j+1:]
                    key = (nb, nh)
                    if key not in seen:
                        seen.add(key); q.append((nb, nh, steps+1))
        return -1
class Solution {
public:
    int findMinStep(string board, string hand) {
        sort(hand.begin(), hand.end());
        queue<tuple<string,string,int>> q;
        set<pair<string,string>> seen;
        auto clean=[](string s){
            bool ch=1;
            while(ch){
                ch=0;
                for(int i=0;i<(int)s.size();){
                    int j=i; while(j<(int)s.size()&&s[j]==s[i]) j++;
                    if(j-i>=3){ s=s.substr(0,i)+s.substr(j); ch=1; break; }
                    i=j;
                }
            }
            return s;
        };
        q.push({board,hand,0}); seen.insert({board,hand});
        while(!q.empty()){
            auto [b,h,steps]=q.front(); q.pop();
            if(b.empty()) return steps;
            if(h.empty()) continue;
            for(int i=0;i<=(int)b.size();i++)
                for(int j=0;j<(int)h.size();j++){
                    if(i&&b[i-1]==h[j]) continue;
                    if(i<(int)b.size()&&b[i]==h[j]) continue;
                    string nb=clean(b.substr(0,i)+h[j]+b.substr(i));
                    string nh=h; nh.erase(nh.begin()+j);
                    if(!seen.count({nb,nh})){ seen.insert({nb,nh}); q.push({nb,nh,steps+1}); }
                }
        }
        return -1;
    }
};
