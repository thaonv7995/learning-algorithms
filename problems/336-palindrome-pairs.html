Palindrome Pairs

O(n×k²) O(n×k) 37.2% AC
Mô tả
Cho mảng các chuỗi words không rỗng. Trả mọi cặp (i,j) sao words[i]+words[j] là palindrome.
Ví dụ
Ví dụ 1
Inwords = ["abcd","dcba","lls","s","sssll"]
Out[[0,1],[1,0],[3,2],[2,4]]
Ví dụ 2
Inwords = ["bat","tab","cat"]
Out[[0,1],[1,0]]
Ý tưởng
Hash map word→index. Với mỗi word, thử mọi cắt l/r: nếu l palindrome và reverse(r) trong map → pair; nếu r palindrome và reverse(l) trong map → pair.

class Solution:
    def palindromePairs(self, words: List[str]) -> List[List[int]]:
        idx = {w: i for i, w in enumerate(words)}
        ans = []
        def is_pal(s, l, r):
            while l < r:
                if s[l] != s[r]: return False
                l += 1; r -= 1
            return True
        for i, w in enumerate(words):
            for j in range(len(w)+1):
                left, right = w[:j], w[j:]
                rev_l, rev_r = left[::-1], right[::-1]
                if is_pal(w, 0, j-1) and rev_r in idx and idx[rev_r] != i:
                    ans.append([i, idx[rev_r]])
                if j != len(w) and is_pal(w, j, len(w)-1) and rev_l in idx and idx[rev_l] != i:
                    ans.append([idx[rev_l], i])
        return ans
class Solution {
    bool pal(const string& s, int l, int r) {
        while (l < r) if (s[l++] != s[r--]) return false;
        return true;
    }
public:
    vector<vector<int>> palindromePairs(vector<string>& words) {
        unordered_map<string,int> idx;
        for (int i = 0; i < (int)words.size(); i++) idx[words[i]] = i;
        vector<vector<int>> ans;
        for (int i = 0; i < (int)words.size(); i++) {
            string& w = words[i];
            for (int j = 0; j <= (int)w.size(); j++) {
                string left = w.substr(0,j), right = w.substr(j);
                string revL(left.rbegin(), left.rend()), revR(right.rbegin(), right.rend());
                if (pal(w,0,j-1) && idx.count(revR) && idx[revR]!=i) ans.push_back({i, idx[revR]});
                if (j!=(int)w.size() && pal(w,j,w.size()-1) && idx.count(revL) && idx[revL]!=i) ans.push_back({idx[revL], i});
            }
        }
        return ans;
    }
};
