Construct Binary Tree from Preorder and Inorder Traversal

O(n) O(n) 68.9% AC
Mô tả
Cho preorder và inorder (không trùng). Dựng lại cây nhị phân.
Ví dụ
Ví dụ 1
Inpreorder = [3,9,20,15,7], inorder = [9,3,15,20,7]
Out[3,9,20,null,null,15,7]
Ý tưởng
pre[0] là root. Tìm root trong inorder → cắt trái/phải. Đệ quy với pre[1:1+len(left)] và in_left, tương tự phải. Dùng hash map vị trí inorder.
Mã nguồn
solution.c C99 
/* Map inorder + đệ quy — xem C++ */
class Solution:
    def buildTree(self, preorder: List[int], inorder: List[int]) -> Optional[TreeNode]:
        pos = {v: i for i, v in enumerate(inorder)}
        def build(pre, ino):
            if not pre: return None
            root = TreeNode(pre[0])
            k = pos[pre[0]]
            root.left = build(pre[1:k+1], ino[:k])
            root.right = build(pre[k+1:], ino[k+1:])
            return root
        return build(preorder, inorder)
class Solution {
    unordered_map<int,int> pos;
    int pi = 0;
    TreeNode* build(int l, int r, vector<int>& pre) {
        if (l > r) return nullptr;
        int v = pre[pi++];
        auto* nd = new TreeNode(v);
        nd->left = build(l, pos[v]-1, pre);
        nd->right = build(pos[v]+1, r, pre);
        return nd;
    }
public:
    TreeNode* buildTree(vector<int>& pre, vector<int>& in) {
        for (int i = 0; i < (int)in.size(); i++) pos[in[i]] = i;
        pi = 0;
        return build(0, in.size()-1, pre);
    }
};

  2. 2Duyệt chính — Tìm root trong inorder → cắt trái/phải.
  3. 3Cập nhật — Đệ quy với pre[1:1+len] và in_left, tương tự phải.
  4. 4Kết luận — Dùng hash map vị trí inorder.

Preorder xác định root; inorder xác định phân hoạch L/R.
