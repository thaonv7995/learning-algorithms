Word Search

O(m·n·4^L) O(L) 47.4% AC
Mô tả
Cho lưới ký tự board và chuỗi word. Kiểm tra word có xuất hiện liên tiếp theo ô kề (ngang/dọc, không lặp ô) hay không.
Ví dụ
Ví dụ 1
Inboard = [["A","B","C","E"],["S","F","C","S"],["A","D","E","E"]], word = "ABCCED"
Outtrue
Ví dụ 2
Inboard = [["A","B","C","E"],["S","F","C","S"],["A","D","E","E"]], word = "SEE"
Outtrue
Ý tưởng
DFS từ mọi ô khớp word[0]: đánh dấu visited, thử 4 hướng với index+1; backtrack bỏ đánh dấu.
Mã nguồn
solution.c C99 
/* exist — DFS backtrack trên board; xem C++ */
class Solution:
    def exist(self, board: List[List[str]], word: str) -> bool:
        m, n = len(board), len(board[0])
        def dfs(i, j, k):
            if k == len(word): return True
            if i < 0 or j < 0 or i >= m or j >= n or board[i][j] != word[k]:
                return False
            ch = board[i][j]; board[i][j] = '#'
            ok = any(dfs(i+di, j+dj, k+1) for di, dj in ((0,1),(0,-1),(1,0),(-1,0)))
            board[i][j] = ch
            return ok
        return any(dfs(i, j, 0) for i in range(m) for j in range(n))
class Solution {
    int m,n;
    bool dfs(vector<vector<char>>& b, string& w, int i, int j, int k) {
        if (k==(int)w.size()) return true;
        if (i<0||j<0||i>=m||j>=n||b[i][j]!=w[k]) return false;
        char c=b[i][j]; b[i][j]='#';
        bool ok = dfs(b,w,i+1,j,k+1)||dfs(b,w,i-1,j,k+1)||dfs(b,w,i,j+1,k+1)||dfs(b,w,i,j-1,k+1);
        b[i][j]=c; return ok;
    }
public:
    bool exist(vector<vector<char>>& board, string word) {
        m=board.size(); n=board[0].size();
        for(int i=0;i<m;i++) for(int j=0;j<n;j++)
            if (dfs(board,word,i,j,0)) return true;
        return false;
    }
};
