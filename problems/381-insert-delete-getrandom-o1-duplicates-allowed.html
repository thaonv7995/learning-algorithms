Insert Delete GetRandom O(1) - Duplicates allowed

O(1) amortized O(n) 36.6% AC
Mô tả
RandomizedCollection cho phép trùng: insert/remove một instance, getRandom đều O(1) trung bình.
Ví dụ
Ví dụ 1
Ininsert(1), insert(1), insert(2), getRandom(), remove(1), getRandom()
Outtrue,true,true,1 or 1 or 2,true,2
Ví dụ 2
Ininsert(1), remove(1), insert(2), getRandom()
Outtrue,true,true,2
Ý tưởng
Vector + map val→unordered_set indices; remove: chọn một index từ set, swap với cuối vector, cập nhật cả hai index trong map.

RandomizedCollection* randomizedCollectionCreate(void) { return calloc(1,sizeof(RandomizedCollection)); }
bool randomizedCollectionInsert(RandomizedCollection* o, int val) { return true; }
bool randomizedCollectionRemove(RandomizedCollection* o, int val) { return false; }
int randomizedCollectionGetRandom(RandomizedCollection* o) { return 0; }
class RandomizedCollection:
    def __init__(self):
        self.arr = []
        self.idx = defaultdict(set)
    def insert(self, val: int) -> bool:
        self.idx[val].add(len(self.arr))
        self.arr.append(val)
        return len(self.idx[val]) == 1
    def remove(self, val: int) -> bool:
        if val not in self.idx: return False
        i = self.idx[val].pop()
        if i != len(self.arr) - 1:
            last = self.arr[-1]
            self.arr[i] = last
            self.idx[last].remove(len(self.arr)-1)
            self.idx[last].add(i)
        self.arr.pop()
        if not self.idx[val]: del self.idx[val]
        return True
    def getRandom(self) -> int:
        return random.choice(self.arr)
class RandomizedCollection {
    vector<pair<int,int>> arr;
    unordered_map<int, unordered_set<int>> idx;
public:
    RandomizedCollection() {}
    bool insert(int val) {
        idx[val].insert(arr.size());
        arr.push_back({val, (int)idx[val].size()-1});
        return idx[val].size() == 1;
    }
    bool remove(int val) {
        if (!idx.count(val)) return false;
        int i = *idx[val].begin(); idx[val].erase(i);
        if (idx[val].empty()) idx.erase(val);
        if (i != (int)arr.size()-1) {
            swap(arr[i], arr.back());
            idx[arr[i].first].erase(arr.size()-1);
            idx[arr[i].first].insert(i);
        }
        arr.pop_back();
        return true;
    }
    int getRandom() { return arr[rand() % arr.size()].first; }
};
