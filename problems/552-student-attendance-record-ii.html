Mô tả
Đếm số chuỗi attendance hợp lệ độ dài n (P/L/A) với tối đa 1 A và không LLL. Trả mod 10^9+7.

/* DP absent/late — xem C++ */
class Solution:
    def checkRecord(self, n: int) -> int:
        MOD = 10**9+7
        dp = [[0]*3 for _ in range(2)]
        dp[0][0] = 1
        for _ in range(n):
            ndp = [[0]*3 for _ in range(2)]
            for a in range(2):
                for l in range(3):
                    v = dp[a][l]
                    if not v: continue
                    ndp[a][0] = (ndp[a][0]+v)%MOD
                    if l+1 < 3: ndp[a][l+1] = (ndp[a][l+1]+v)%MOD
                    if a == 0: ndp[1][0] = (ndp[1][0]+v)%MOD
            dp = ndp
        return sum(sum(row) for row in dp) % MOD
class Solution {
public:
    int checkRecord(int n) {
        const int MOD = 1e9+7;
        long long dp[2][3] = {};
        dp[0][0] = 1;
        while (n--) {
            long long ndp[2][3] = {};
            for (int a = 0; a < 2; a++)
                for (int l = 0; l < 3; l++) if (dp[a][l]) {
                    ndp[a][0] = (ndp[a][0]+dp[a][l])%MOD;
                    if (l+1<3) ndp[a][l+1]=(ndp[a][l+1]+dp[a][l])%MOD;
                    if (!a) ndp[1][0]=(ndp[1][0]+dp[a][l])%MOD;
                }
            memcpy(dp, ndp, sizeof dp);
        }
        long long ans=0;
        for (int a=0;a<2;a++) for(int l=0;l<3;l++) ans=(ans+dp[a][l])%MOD;
        return ans;
    }
};
