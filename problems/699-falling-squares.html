Mô tả
Falling squares: squares rơi theo trục x side length. Trả [heights] sau mỗi drop max height profile.

int* fallingSquares699(int** pos,int n,int* rs) {
    int* h=malloc(n*4); int* ans=malloc(n*4); int cur=0;
    for(int i=0;i<n;i++){
        int left=pos[i][0], side=pos[i][1], base=0;
        for(int j=0;j<i;j++){
            int l2=pos[j][0], s2=pos[j][1];
            if(left+side-1>=l2 && l2+s2-1>=left && h[j]>base) base=h[j];
        }
        h[i]=base+side; if(h[i]>cur) cur=h[i]; ans[i]=cur;
    }
    free(h); *rs=n; return ans;
}
class Solution:
    def fallingSquares(self, positions: List[List[int]]) -> List[int]:
        xs=sorted({x for l,s in positions for x in (l,l+s-1)})
        idx={x:i for i,x in enumerate(xs)}
        tree=[0]*(4*len(xs)); ans=[]; cur=0
        def upd(p,l,r,v,i=1):
            if l==r: tree[i]=max(tree[i],v); return
            m=(l+r)//2
            if p<=m: upd(p,l,m,i*2,v)
            else: upd(p,m+1,r,i*2+1,v)
            tree[i]=max(tree[i*2],tree[i*2+1])
        def qry(L,R,l=0,r=len(xs)-1,i=1):
            if R<l or r<L: return 0
            if L<=l and r<=R: return tree[i]
            m=(l+r)//2
            return max(qry(L,R,l,m,i*2), qry(L,R,m+1,r,i*2+1))
        for left,side in positions:
            L,R=idx[left],idx[left+side-1]
            h=qry(L,R)+side
            cur=max(cur,h); ans.append(cur)
            for p in range(L,R+1): upd(p,0,len(xs)-1,h)
        return ans
