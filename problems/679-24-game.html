static double eval679(double a,double b,char op){ if(op=='+')return a+b; if(op=='-')return a-b; if(op=='*')return a*b; return b?a/b:0; }
static int solve679(double* v,int n){
    if(n==1) return fabs(v[0]-24.0)<1e-6;
    for(int i=0;i<n;i++) for(int j=0;j<n;j++) if(i!=j){
        double nv[4]; int nn=0;
        for(int k=0;k<n;k++) if(k!=i&&k!=j) nv[nn++]=v[k];
        char ops[]="+-*/";
        for(int o=0;o<4;o++){
            nv[nn]=eval679(v[i],v[j],ops[o]); if(solve679(nv,nn+1)) return 1;
            if(o<2){ nv[nn]=eval679(v[j],v[i],ops[o]); if(solve679(nv,nn+1)) return 1; }
        }
    }
    return 0;
}
bool judgePoint24679(int* c,int n){ double v[4]; for(int i=0;i<4;i++) v[i]=c[i]; return solve679(v,4); }
class Solution:
    def judgePoint24(self, cards: List[int]) -> bool:
        if len(cards)==1: return abs(cards[0]-24)<1e-6
        for i in range(len(cards)):
            for j in range(len(cards)):
                if i==j: continue
                nxt=[cards[k] for k in range(len(cards)) if k!=i and k!=j]
                for op in "+-*/":
                    if op in "+-" or cards[j]!=0:
                        a=cards[i]; b=cards[j]
                        v=a+b if op=="+" else a-b if op=="-" else a*b if op=="*" else a/b
                        if self.judgePoint24(nxt+[v]): return True
        return False
class Solution {
    bool dfs(vector<double>& a){
        if(a.size()==1) return abs(a[0]-24)<1e-6;
        for(int i=0;i<(int)a.size();i++) for(int j=0;j<(int)a.size();j++) if(i!=j){
            vector<double> nxt;
            for(int k=0;k<(int)a.size();k++) if(k!=i&&k!=j) nxt.push_back(a[k]);
            double x=a[i],y=a[j];
            if(dfs(nxt,x+y)||dfs(nxt,x-y)||dfs(nxt,x*y)||(y&&dfs(nxt,x/y))) return true;
        }
        return false;
    }
public:
    bool judgePoint24(vector<int>& c){ vector<double> a(c.begin(),c.end()); return dfs(a); }
};
